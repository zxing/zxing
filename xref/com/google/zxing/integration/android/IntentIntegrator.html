38  /**
39   * <p>A utility class which helps ease integration with Barcode Scanner via {@link Intent}s. This is a simple
40   * way to invoke barcode scanning and receive the result, without any need to integrate, modify, or learn the
41   * project's source code.</p>
42   *
43   * <h2>Initiating a barcode scan</h2>
44   *
45   * <p>To integrate, create an instance of {@code IntentIntegrator} and call {@link #initiateScan()} and wait
46   * for the result in your app.</p>
47   *
48   * <p>It does require that the Barcode Scanner (or work-alike) application is installed. The
49   * {@link #initiateScan()} method will prompt the user to download the application, if needed.</p>
50   *
51   * <p>There are a few steps to using this integration. First, your {@link Activity} must implement
52   * the method {@link Activity#onActivityResult(int, int, Intent)} and include a line of code like this:</p>
53   *
54   * <pre>{@code
55   * public void onActivityResult(int requestCode, int resultCode, Intent intent) {
56   *   IntentResult scanResult = IntentIntegrator.parseActivityResult(requestCode, resultCode, intent);
57   *   if (scanResult != null) {
58   *     // handle scan result
59   *   }
60   *   // else continue with any other code you need in the method
61   *   ...
62   * }
63   * }</pre>
64   *
65   * <p>This is where you will handle a scan result.</p>
66   *
67   * <p>Second, just call this in response to a user action somewhere to begin the scan process:</p>
68   *
69   * <pre>{@code
70   * IntentIntegrator integrator = new IntentIntegrator(yourActivity);
71   * integrator.initiateScan();
72   * }</pre>
73   *
74   * <p>Note that {@link #initiateScan()} returns an {@link AlertDialog} which is non-null if the
75   * user was prompted to download the application. This lets the calling app potentially manage the dialog.
76   * In particular, ideally, the app dismisses the dialog if it's still active in its {@link Activity#onPause()}
77   * method.</p>
78   * 
79   * <p>You can use {@link #setTitle(String)} to customize the title of this download prompt dialog (or, use
80   * {@link #setTitleByID(int)} to set the title by string resource ID.) Likewise, the prompt message, and
81   * yes/no button labels can be changed.</p>
82   *
83   * <p>Finally, you can use {@link #addExtra(String, Object)} to add more parameters to the Intent used
84   * to invoke the scanner. This can be used to set additional options not directly exposed by this
85   * simplified API.</p>
86   * 
87   * <p>By default, this will only allow applications that are known to respond to this intent correctly
88   * do so. The apps that are allowed to response can be set with {@link #setTargetApplications(List)}.
89   * For example, set to {@link #TARGET_BARCODE_SCANNER_ONLY} to only target the Barcode Scanner app itself.</p>
90   *
91   * <h2>Sharing text via barcode</h2>
92   *
93   * <p>To share text, encoded as a QR Code on-screen, similarly, see {@link #shareText(CharSequence)}.</p>
94   *
95   * <p>Some code, particularly download integration, was contributed from the Anobiit application.</p>
96   *
97   * <h2>Enabling experimental barcode formats</h2>
98   *
99   * <p>Some formats are not enabled by default even when scanning with {@link #ALL_CODE_TYPES}, such as
100  * PDF417. Use {@link #initiateScan(java.util.Collection)} with
101  * a collection containing the names of formats to scan for explicitly, like "PDF_417", to use such
102  * formats.</p>
103  *
104  * @author Sean Owen
105  * @author Fred Lin
106  * @author Isaac Potoczny-Jones
107  * @author Brad Drehmer
108  * @author gcstang
109  */

261   /**
262    * Initiates a scan only for a certain set of barcode types, given as strings corresponding
263    * to their names in ZXing's {@code BarcodeFormat} class like "UPC_A". You can supply constants
264    * like {@link #PRODUCT_CODE_TYPES} for example.
265    *
266    * @param desiredBarcodeFormats names of {@code BarcodeFormat}s to scan for
267    * @return the {@link AlertDialog} that was shown to the user prompting them to download the app
268    *   if a prompt was needed, or null otherwise
269    */
270   public final AlertDialog initiateScan(Collection<String> desiredBarcodeFormats) {
271     Intent intentScan = new Intent(BS_PACKAGE + ".SCAN");
272     intentScan.addCategory(Intent.CATEGORY_DEFAULT);
273 
274     // check which types of codes to scan for
275     if (desiredBarcodeFormats != null) {
276       // set the desired barcode types
277       StringBuilder joinedByComma = new StringBuilder();
278       for (String format : desiredBarcodeFormats) {
279         if (joinedByComma.length() > 0) {
280           joinedByComma.append(',');
281         }
282         joinedByComma.append(format);
283       }
284       intentScan.putExtra("SCAN_FORMATS", joinedByComma.toString());
285     }
286 
287     String targetAppPackage = findTargetAppPackage(intentScan);
288     if (targetAppPackage == null) {
289       return showDownloadDialog();
290     }
291     intentScan.setPackage(targetAppPackage);
292     intentScan.addFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP);
293     intentScan.addFlags(Intent.FLAG_ACTIVITY_CLEAR_WHEN_TASK_RESET);
294     attachMoreExtras(intentScan);
295     startActivityForResult(intentScan, REQUEST_CODE);
296     return null;
297   }

299   /**
300    * Start an activity. This method is defined to allow different methods of activity starting for
301    * newer versions of Android and for compatibility library.
302    *
303    * @param intent Intent to start.
304    * @param code Request code for the activity
305    * @see android.app.Activity#startActivityForResult(Intent, int)
306    * @see android.app.Fragment#startActivityForResult(Intent, int)
307    */
308   protected void startActivityForResult(Intent intent, int code) {
309     if (fragment == null) {
310       activity.startActivityForResult(intent, code);
311     } else {
312       fragment.startActivityForResult(intent, code);
313     }
314   }

316   private String findTargetAppPackage(Intent intent) {
317     PackageManager pm = activity.getPackageManager();
318     List<ResolveInfo> availableApps = pm.queryIntentActivities(intent, PackageManager.MATCH_DEFAULT_ONLY);
319     if (availableApps != null) {
320       for (String targetApp : targetApplications) {
321         if (contains(availableApps, targetApp)) {
322           return targetApp;
323         }
324       }
325     }
326     return null;
327   }
328   
329   private static boolean contains(Iterable<ResolveInfo> availableApps, String targetApp) {
330     for (ResolveInfo availableApp : availableApps) {
331       String packageName = availableApp.activityInfo.packageName;
332       if (targetApp.equals(packageName)) {
333         return true;
334       }
335     }
336     return false;
337   }

339   private AlertDialog showDownloadDialog() {
340     AlertDialog.Builder downloadDialog = new AlertDialog.Builder(activity);
341     downloadDialog.setTitle(title);
342     downloadDialog.setMessage(message);
343     downloadDialog.setPositiveButton(buttonYes, new DialogInterface.OnClickListener() {
344       @Override
345       public void onClick(DialogInterface dialogInterface, int i) {
346         String packageName;
347         if (targetApplications.contains(BS_PACKAGE)) {
348           // Prefer to suggest download of BS if it's anywhere in the list
349           packageName = BS_PACKAGE;
350         } else {
351           // Otherwise, first option:
352           packageName = targetApplications.get(0);
353         }
354         Uri uri = Uri.parse("market://details?id=" + packageName);
355         Intent intent = new Intent(Intent.ACTION_VIEW, uri);
356         try {
357           if (fragment == null) {
358             activity.startActivity(intent);
359           } else {
360             fragment.startActivity(intent);
361           }
362         } catch (ActivityNotFoundException anfe) {
363           // Hmm, market is not installed
364           Log.w(TAG, "Google Play is not installed; cannot install " + packageName);
365         }
366       }
367     });
368     downloadDialog.setNegativeButton(buttonNo, null);
369     downloadDialog.setCancelable(true);
370     return downloadDialog.show();
371   }

374   /**
375    * <p>Call this from your {@link Activity}'s
376    * {@link Activity#onActivityResult(int, int, Intent)} method.</p>
377    *
378    * @param requestCode request code from {@code onActivityResult()}
379    * @param resultCode result code from {@code onActivityResult()}
380    * @param intent {@link Intent} from {@code onActivityResult()}
381    * @return null if the event handled here was not related to this class, or
382    *  else an {@link IntentResult} containing the result of the scan. If the user cancelled scanning,
383    *  the fields will be null.
384    */
385   public static IntentResult parseActivityResult(int requestCode, int resultCode, Intent intent) {
386     if (requestCode == REQUEST_CODE) {
387       if (resultCode == Activity.RESULT_OK) {
388         String contents = intent.getStringExtra("SCAN_RESULT");
389         String formatName = intent.getStringExtra("SCAN_RESULT_FORMAT");
390         byte[] rawBytes = intent.getByteArrayExtra("SCAN_RESULT_BYTES");
391         int intentOrientation = intent.getIntExtra("SCAN_RESULT_ORIENTATION", Integer.MIN_VALUE);
392         Integer orientation = intentOrientation == Integer.MIN_VALUE ? null : intentOrientation;
393         String errorCorrectionLevel = intent.getStringExtra("SCAN_RESULT_ERROR_CORRECTION_LEVEL");
394         return new IntentResult(contents,
395                                 formatName,
396                                 rawBytes,
397                                 orientation,
398                                 errorCorrectionLevel);
399       }
400       return new IntentResult();
401     }
402     return null;
403   }

418   /**
419    * Shares the given text by encoding it as a barcode, such that another user can
420    * scan the text off the screen of the device.
421    *
422    * @param text the text string to encode as a barcode
423    * @param type type of data to encode. See {@code com.google.zxing.client.android.Contents.Type} constants.
424    * @return the {@link AlertDialog} that was shown to the user prompting them to download the app
425    *   if a prompt was needed, or null otherwise
426    */
427   public final AlertDialog shareText(CharSequence text, CharSequence type) {
428     Intent intent = new Intent();
429     intent.addCategory(Intent.CATEGORY_DEFAULT);
430     intent.setAction(BS_PACKAGE + ".ENCODE");
431     intent.putExtra("ENCODE_TYPE", type);
432     intent.putExtra("ENCODE_DATA", text);
433     String targetAppPackage = findTargetAppPackage(intent);
434     if (targetAppPackage == null) {
435       return showDownloadDialog();
436     }
437     intent.setPackage(targetAppPackage);
438     intent.addFlags(Intent.FLAG_ACTIVITY_CLEAR_TOP);
439     intent.addFlags(Intent.FLAG_ACTIVITY_CLEAR_WHEN_TASK_RESET);
440     attachMoreExtras(intent);
441     if (fragment == null) {
442       activity.startActivity(intent);
443     } else {
444       fragment.startActivity(intent);
445     }
446     return null;
447   }

449   private static List<String> list(String... values) {
450     return Collections.unmodifiableList(Arrays.asList(values));
451   }
452 
453   private void attachMoreExtras(Intent intent) {
454     for (Map.Entry<String,Object> entry : moreExtras.entrySet()) {
455       String key = entry.getKey();
456       Object value = entry.getValue();
457       // Kind of hacky
458       if (value instanceof Integer) {
459         intent.putExtra(key, (Integer) value);
460       } else if (value instanceof Long) {
461         intent.putExtra(key, (Long) value);
462       } else if (value instanceof Boolean) {
463         intent.putExtra(key, (Boolean) value);
464       } else if (value instanceof Double) {
465         intent.putExtra(key, (Double) value);
466       } else if (value instanceof Float) {
467         intent.putExtra(key, (Float) value);
468       } else if (value instanceof Bundle) {
469         intent.putExtra(key, (Bundle) value);
470       } else {
471         intent.putExtra(key, value.toString());
472       }
473     }
474   }
