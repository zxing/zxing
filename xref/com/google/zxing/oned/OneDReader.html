44    protected static final int INTEGER_MATH_SHIFT = 8;
45    protected static final int PATTERN_MATCH_RESULT_SCALE_FACTOR = 1 << INTEGER_MATH_SHIFT;

47    @Override
48    public Result decode(BinaryBitmap image) throws NotFoundException, FormatException {
49      return decode(image, null);

52    // Note that we don't try rotation without the try harder flag, even if rotation was supported.
53    @Override
54    public Result decode(BinaryBitmap image,
55                         Map<DecodeHintType,?> hints) throws NotFoundException, FormatException {
56      try {
57        return doDecode(image, hints);
58      } catch (NotFoundException nfe) {
59        boolean tryHarder = hints != null && hints.containsKey(DecodeHintType.TRY_HARDER);
60        if (tryHarder && image.isRotateSupported()) {
61          BinaryBitmap rotatedImage = image.rotateCounterClockwise();
62          Result result = doDecode(rotatedImage, hints);
63          // Record that we found it rotated 90 degrees CCW / 270 degrees CW
64          Map<ResultMetadataType,?> metadata = result.getResultMetadata();
65          int orientation = 270;
66          if (metadata != null && metadata.containsKey(ResultMetadataType.ORIENTATION)) {
67            // But if we found it reversed in doDecode(), add in that result here:
68            orientation = (orientation +
69                (Integer) metadata.get(ResultMetadataType.ORIENTATION)) % 360;
70          }
71          result.putMetadata(ResultMetadataType.ORIENTATION, orientation);
72          // Update result points
73          ResultPoint[] points = result.getResultPoints();
74          if (points != null) {
75            int height = rotatedImage.getHeight();
76            for (int i = 0; i < points.length; i++) {
77              points[i] = new ResultPoint(height - points[i].getY() - 1, points[i].getX());
78            }
79          }
80          return result;
81        } else {
82          throw nfe;
83        }
84      }
85    }

92    /**
93     * We're going to examine rows from the middle outward, searching alternately above and below the
94     * middle, and farther out each time. rowStep is the number of rows between each successive
95     * attempt above and below the middle. So we'd scan row middle, then middle - rowStep, then
96     * middle + rowStep, then middle - (2 * rowStep), etc.
97     * rowStep is bigger as the image is taller, but is always at least 1. We've somewhat arbitrarily
98     * decided that moving up and down by about 1/16 of the image is pretty good; we try more of the
99     * image if "trying harder".
100    *
101    * @param image The image to decode
102    * @param hints Any hints that were requested
103    * @return The contents of the decoded barcode
104    * @throws NotFoundException Any spontaneous errors which occur
105    */
106   private Result doDecode(BinaryBitmap image,
107                           Map<DecodeHintType,?> hints) throws NotFoundException {
108     int width = image.getWidth();
109     int height = image.getHeight();
110     BitArray row = new BitArray(width);
111 
112     int middle = height >> 1;
113     boolean tryHarder = hints != null && hints.containsKey(DecodeHintType.TRY_HARDER);
114     int rowStep = Math.max(1, height >> (tryHarder ? 8 : 5));
115     int maxLines;
116     if (tryHarder) {
117       maxLines = height; // Look at the whole image, not just the center
118     } else {
119       maxLines = 15; // 15 rows spaced 1/32 apart is roughly the middle half of the image
120     }
121 
122     for (int x = 0; x < maxLines; x++) {
123 
124       // Scanning from the middle out. Determine which row we're looking at next:
125       int rowStepsAboveOrBelow = (x + 1) >> 1;
126       boolean isAbove = (x & 0x01) == 0; // i.e. is x even?
127       int rowNumber = middle + rowStep * (isAbove ? rowStepsAboveOrBelow : -rowStepsAboveOrBelow);
128       if (rowNumber < 0 || rowNumber >= height) {
129         // Oops, if we run off the top or bottom, stop
130         break;
131       }
132 
133       // Estimate black point for this row and load it:
134       try {
135         row = image.getBlackRow(rowNumber, row);
136       } catch (NotFoundException ignored) {
137         continue;
138       }
139 
140       // While we have the image data in a BitArray, it's fairly cheap to reverse it in place to
141       // handle decoding upside down barcodes.
142       for (int attempt = 0; attempt < 2; attempt++) {
143         if (attempt == 1) { // trying again?
144           row.reverse(); // reverse the row and continue
145           // This means we will only ever draw result points *once* in the life of this method
146           // since we want to avoid drawing the wrong points after flipping the row, and,
147           // don't want to clutter with noise from every single row scan -- just the scans
148           // that start on the center line.
149           if (hints != null && hints.containsKey(DecodeHintType.NEED_RESULT_POINT_CALLBACK)) {
150             Map<DecodeHintType,Object> newHints = new EnumMap<>(DecodeHintType.class);
151             newHints.putAll(hints);
152             newHints.remove(DecodeHintType.NEED_RESULT_POINT_CALLBACK);
153             hints = newHints;
154           }
155         }
156         try {
157           // Look for a barcode
158           Result result = decodeRow(rowNumber, row, hints);
159           // We found our barcode
160           if (attempt == 1) {
161             // But it was upside down, so note that
162             result.putMetadata(ResultMetadataType.ORIENTATION, 180);
163             // And remember to flip the result points horizontally.
164             ResultPoint[] points = result.getResultPoints();
165             if (points != null) {
166               points[0] = new ResultPoint(width - points[0].getX() - 1, points[0].getY());
167               points[1] = new ResultPoint(width - points[1].getX() - 1, points[1].getY());
168             }
169           }
170           return result;
171         } catch (ReaderException re) {
172           // continue -- just couldn't decode this row
173         }
174       }
175     }
176 
177     throw NotFoundException.getNotFoundInstance();
178   }

180   /**
181    * Records the size of successive runs of white and black pixels in a row, starting at a given point.
182    * The values are recorded in the given array, and the number of runs recorded is equal to the size
183    * of the array. If the row starts on a white pixel at the given start point, then the first count
184    * recorded is the run of white pixels starting from that point; likewise it is the count of a run
185    * of black pixels if the row begin on a black pixels at that point.
186    *
187    * @param row row to count from
188    * @param start offset into row to start at
189    * @param counters array into which to record counts
190    * @throws NotFoundException if counters cannot be filled entirely from row before running out
191    *  of pixels
192    */
193   protected static void recordPattern(BitArray row,
194                                       int start,
195                                       int[] counters) throws NotFoundException {
196     int numCounters = counters.length;
197     Arrays.fill(counters, 0, numCounters, 0);
198     int end = row.getSize();
199     if (start >= end) {
200       throw NotFoundException.getNotFoundInstance();
201     }
202     boolean isWhite = !row.get(start);
203     int counterPosition = 0;
204     int i = start;
205     while (i < end) {
206       if (row.get(i) ^ isWhite) { // that is, exactly one is true
207         counters[counterPosition]++;
208       } else {
209         counterPosition++;
210         if (counterPosition == numCounters) {
211           break;
212         } else {
213           counters[counterPosition] = 1;
214           isWhite = !isWhite;
215         }
216       }
217       i++;
218     }
219     // If we read fully the last section of pixels and filled up our counters -- or filled
220     // the last counter but ran off the side of the image, OK. Otherwise, a problem.
221     if (!(counterPosition == numCounters || (counterPosition == numCounters - 1 && i == end))) {
222       throw NotFoundException.getNotFoundInstance();
223     }
224   }
225 
226   protected static void recordPatternInReverse(BitArray row, int start, int[] counters)
227       throws NotFoundException {
228     // This could be more efficient I guess
229     int numTransitionsLeft = counters.length;
230     boolean last = row.get(start);
231     while (start > 0 && numTransitionsLeft >= 0) {
232       if (row.get(--start) != last) {
233         numTransitionsLeft--;
234         last = !last;
235       }
236     }
237     if (numTransitionsLeft >= 0) {
238       throw NotFoundException.getNotFoundInstance();
239     }
240     recordPattern(row, start + 1, counters);
241   }

243   /**
244    * Determines how closely a set of observed counts of runs of black/white values matches a given
245    * target pattern. This is reported as the ratio of the total variance from the expected pattern
246    * proportions across all pattern elements, to the length of the pattern.
247    *
248    * @param counters observed counters
249    * @param pattern expected pattern
250    * @param maxIndividualVariance The most any counter can differ before we give up
251    * @return ratio of total variance between counters and pattern compared to total pattern size,
252    *  where the ratio has been multiplied by 256. So, 0 means no variance (perfect match); 256 means
253    *  the total variance between counters and patterns equals the pattern length, higher values mean
254    *  even more variance
255    */
256   protected static int patternMatchVariance(int[] counters,
257                                             int[] pattern,
258                                             int maxIndividualVariance) {
259     int numCounters = counters.length;
260     int total = 0;
261     int patternLength = 0;
262     for (int i = 0; i < numCounters; i++) {
263       total += counters[i];
264       patternLength += pattern[i];
265     }
266     if (total < patternLength) {
267       // If we don't even have one pixel per unit of bar width, assume this is too small
268       // to reliably match, so fail:
269       return Integer.MAX_VALUE;
270     }
271     // We're going to fake floating-point math in integers. We just need to use more bits.
272     // Scale up patternLength so that intermediate values below like scaledCounter will have
273     // more "significant digits"
274     int unitBarWidth = (total << INTEGER_MATH_SHIFT) / patternLength;
275     maxIndividualVariance = (maxIndividualVariance * unitBarWidth) >> INTEGER_MATH_SHIFT;
276 
277     int totalVariance = 0;
278     for (int x = 0; x < numCounters; x++) {
279       int counter = counters[x] << INTEGER_MATH_SHIFT;
280       int scaledPattern = pattern[x] * unitBarWidth;
281       int variance = counter > scaledPattern ? counter - scaledPattern : scaledPattern - counter;
282       if (variance > maxIndividualVariance) {
283         return Integer.MAX_VALUE;
284       }
285       totalVariance += variance;
286     }
287     return totalVariance / total;
288   }
