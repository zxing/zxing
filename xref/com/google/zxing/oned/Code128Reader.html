149   private static final float MAX_AVG_VARIANCE = 0.25f;
150   private static final float MAX_INDIVIDUAL_VARIANCE = 0.7f;
151 
152   private static final int CODE_SHIFT = 98;
153 
154   private static final int CODE_CODE_C = 99;
155   private static final int CODE_CODE_B = 100;
156   private static final int CODE_CODE_A = 101;
157 
158   private static final int CODE_FNC_1 = 102;
159   private static final int CODE_FNC_2 = 97;
160   private static final int CODE_FNC_3 = 96;
161   private static final int CODE_FNC_4_A = 101;
162   private static final int CODE_FNC_4_B = 100;
163 
164   private static final int CODE_START_A = 103;
165   private static final int CODE_START_B = 104;
166   private static final int CODE_START_C = 105;
167   private static final int CODE_STOP = 106;
168 
169   private static int[] findStartPattern(BitArray row) throws NotFoundException {
170     int width = row.getSize();
171     int rowOffset = row.getNextSet(0);
172 
173     int counterPosition = 0;
174     int[] counters = new int[6];
175     int patternStart = rowOffset;
176     boolean isWhite = false;
177     int patternLength = counters.length;
178 
179     for (int i = rowOffset; i < width; i++) {
180       if (row.get(i) ^ isWhite) {
181         counters[counterPosition]++;
182       } else {
183         if (counterPosition == patternLength - 1) {
184           float bestVariance = MAX_AVG_VARIANCE;
185           int bestMatch = -1;
186           for (int startCode = CODE_START_A; startCode <= CODE_START_C; startCode++) {
187             float variance = patternMatchVariance(counters, CODE_PATTERNS[startCode],
188                 MAX_INDIVIDUAL_VARIANCE);
189             if (variance < bestVariance) {
190               bestVariance = variance;
191               bestMatch = startCode;
192             }
193           }
194           // Look for whitespace before start pattern, >= 50% of width of start pattern
195           if (bestMatch >= 0 &&
196               row.isRange(Math.max(0, patternStart - (i - patternStart) / 2), patternStart, false)) {
197             return new int[]{patternStart, i, bestMatch};
198           }
199           patternStart += counters[0] + counters[1];
200           System.arraycopy(counters, 2, counters, 0, patternLength - 2);
201           counters[patternLength - 2] = 0;
202           counters[patternLength - 1] = 0;
203           counterPosition--;
204         } else {
205           counterPosition++;
206         }
207         counters[counterPosition] = 1;
208         isWhite = !isWhite;
209       }
210     }
211     throw NotFoundException.getNotFoundInstance();
212   }

214   private static int decodeCode(BitArray row, int[] counters, int rowOffset)
215       throws NotFoundException {
216     recordPattern(row, rowOffset, counters);
217     float bestVariance = MAX_AVG_VARIANCE; // worst variance we'll accept
218     int bestMatch = -1;
219     for (int d = 0; d < CODE_PATTERNS.length; d++) {
220       int[] pattern = CODE_PATTERNS[d];
221       float variance = patternMatchVariance(counters, pattern, MAX_INDIVIDUAL_VARIANCE);
222       if (variance < bestVariance) {
223         bestVariance = variance;
224         bestMatch = d;
225       }
226     }
227     // TODO We're overlooking the fact that the STOP pattern has 7 values, not 6.
228     if (bestMatch >= 0) {
229       return bestMatch;
230     } else {
231       throw NotFoundException.getNotFoundInstance();
232     }
233   }

235   @Override
236   public Result decodeRow(int rowNumber, BitArray row, Map<DecodeHintType,?> hints)
237       throws NotFoundException, FormatException, ChecksumException {
238 
239     boolean convertFNC1 = hints != null && hints.containsKey(DecodeHintType.ASSUME_GS1);
240 
241     int[] startPatternInfo = findStartPattern(row);
242     int startCode = startPatternInfo[2];
243 
244     List<Byte> rawCodes = new ArrayList<>(20);
245     rawCodes.add((byte) startCode);
246 
247     int codeSet;
248     switch (startCode) {
249       case CODE_START_A:
250         codeSet = CODE_CODE_A;
251         break;
252       case CODE_START_B:
253         codeSet = CODE_CODE_B;
254         break;
255       case CODE_START_C:
256         codeSet = CODE_CODE_C;
257         break;
258       default:
259         throw FormatException.getFormatInstance();
260     }
261 
262     boolean done = false;
263     boolean isNextShifted = false;
264 
265     StringBuilder result = new StringBuilder(20);
266 
267     int lastStart = startPatternInfo[0];
268     int nextStart = startPatternInfo[1];
269     int[] counters = new int[6];
270 
271     int lastCode = 0;
272     int code = 0;
273     int checksumTotal = startCode;
274     int multiplier = 0;
275     boolean lastCharacterWasPrintable = true;
276     boolean upperMode = false;
277     boolean shiftUpperMode = false;
278 
279     while (!done) {
280 
281       boolean unshift = isNextShifted;
282       isNextShifted = false;
283 
284       // Save off last code
285       lastCode = code;
286 
287       // Decode another code from image
288       code = decodeCode(row, counters, nextStart);
289 
290       rawCodes.add((byte) code);
291 
292       // Remember whether the last code was printable or not (excluding CODE_STOP)
293       if (code != CODE_STOP) {
294         lastCharacterWasPrintable = true;
295       }
296 
297       // Add to checksum computation (if not CODE_STOP of course)
298       if (code != CODE_STOP) {
299         multiplier++;
300         checksumTotal += multiplier * code;
301       }
302 
303       // Advance to where the next code will to start
304       lastStart = nextStart;
305       for (int counter : counters) {
306         nextStart += counter;
307       }
308 
309       // Take care of illegal start codes
310       switch (code) {
311         case CODE_START_A:
312         case CODE_START_B:
313         case CODE_START_C:
314           throw FormatException.getFormatInstance();
315       }
316 
317       switch (codeSet) {
318 
319         case CODE_CODE_A:
320           if (code < 64) {
321             if (shiftUpperMode == upperMode) {
322               result.append((char) (' ' + code));
323             } else {
324               result.append((char) (' ' + code + 128));
325             }
326             shiftUpperMode = false;
327           } else if (code < 96) {
328             if (shiftUpperMode == upperMode) {
329               result.append((char) (code - 64));
330             } else {
331               result.append((char) (code + 64));
332             }
333             shiftUpperMode = false;
334           } else {
335             // Don't let CODE_STOP, which always appears, affect whether whether we think the last
336             // code was printable or not.
337             if (code != CODE_STOP) {
338               lastCharacterWasPrintable = false;
339             }
340             switch (code) {
341               case CODE_FNC_1:
342                 if (convertFNC1) {
343                   if (result.length() == 0){
344                     // GS1 specification 5.4.3.7. and 5.4.6.4. If the first char after the start code
345                     // is FNC1 then this is GS1-128. We add the symbology identifier.
346                     result.append("]C1");
347                   } else {
348                     // GS1 specification 5.4.7.5. Every subsequent FNC1 is returned as ASCII 29 (GS)
349                     result.append((char) 29);
350                   }
351                 }
352                 break;
353               case CODE_FNC_2:
354               case CODE_FNC_3:
355                 // do nothing?
356                 break;
357               case CODE_FNC_4_A:
358                 if (!upperMode && shiftUpperMode) {
359                   upperMode = true;
360                   shiftUpperMode = false;
361                 } else if (upperMode && shiftUpperMode) {
362                   upperMode = false;
363                   shiftUpperMode = false;
364                 } else {
365                   shiftUpperMode = true;
366                 }
367                 break;
368               case CODE_SHIFT:
369                 isNextShifted = true;
370                 codeSet = CODE_CODE_B;
371                 break;
372               case CODE_CODE_B:
373                 codeSet = CODE_CODE_B;
374                 break;
375               case CODE_CODE_C:
376                 codeSet = CODE_CODE_C;
377                 break;
378               case CODE_STOP:
379                 done = true;
380                 break;
381             }
382           }
383           break;
384         case CODE_CODE_B:
385           if (code < 96) {
386             if (shiftUpperMode == upperMode) {
387               result.append((char) (' ' + code));
388             } else {
389               result.append((char) (' ' + code + 128));
390             }
391             shiftUpperMode = false;
392           } else {
393             if (code != CODE_STOP) {
394               lastCharacterWasPrintable = false;
395             }
396             switch (code) {
397               case CODE_FNC_1:
398                 if (convertFNC1) {
399                   if (result.length() == 0){
400                     // GS1 specification 5.4.3.7. and 5.4.6.4. If the first char after the start code
401                     // is FNC1 then this is GS1-128. We add the symbology identifier.
402                     result.append("]C1");
403                   } else {
404                     // GS1 specification 5.4.7.5. Every subsequent FNC1 is returned as ASCII 29 (GS)
405                     result.append((char) 29);
406                   }
407                 }
408                 break;
409               case CODE_FNC_2:
410               case CODE_FNC_3:
411                 // do nothing?
412                 break;
413               case CODE_FNC_4_B:
414                 if (!upperMode && shiftUpperMode) {
415                   upperMode = true;
416                   shiftUpperMode = false;
417                 } else if (upperMode && shiftUpperMode) {
418                   upperMode = false;
419                   shiftUpperMode = false;
420                 } else {
421                   shiftUpperMode = true;
422                 }
423                 break;
424               case CODE_SHIFT:
425                 isNextShifted = true;
426                 codeSet = CODE_CODE_A;
427                 break;
428               case CODE_CODE_A:
429                 codeSet = CODE_CODE_A;
430                 break;
431               case CODE_CODE_C:
432                 codeSet = CODE_CODE_C;
433                 break;
434               case CODE_STOP:
435                 done = true;
436                 break;
437             }
438           }
439           break;
440         case CODE_CODE_C:
441           if (code < 100) {
442             if (code < 10) {
443               result.append('0');
444             }
445             result.append(code);
446           } else {
447             if (code != CODE_STOP) {
448               lastCharacterWasPrintable = false;
449             }
450             switch (code) {
451               case CODE_FNC_1:
452                 if (convertFNC1) {
453                   if (result.length() == 0){
454                     // GS1 specification 5.4.3.7. and 5.4.6.4. If the first char after the start code
455                     // is FNC1 then this is GS1-128. We add the symbology identifier.
456                     result.append("]C1");
457                   } else {
458                     // GS1 specification 5.4.7.5. Every subsequent FNC1 is returned as ASCII 29 (GS)
459                     result.append((char) 29);
460                   }
461                 }
462                 break;
463               case CODE_CODE_A:
464                 codeSet = CODE_CODE_A;
465                 break;
466               case CODE_CODE_B:
467                 codeSet = CODE_CODE_B;
468                 break;
469               case CODE_STOP:
470                 done = true;
471                 break;
472             }
473           }
474           break;
475       }
476 
477       // Unshift back to another code set if we were shifted
478       if (unshift) {
479         codeSet = codeSet == CODE_CODE_A ? CODE_CODE_B : CODE_CODE_A;
480       }
481 
482     }
483 
484     int lastPatternSize = nextStart - lastStart;
485 
486     // Check for ample whitespace following pattern, but, to do this we first need to remember that
487     // we fudged decoding CODE_STOP since it actually has 7 bars, not 6. There is a black bar left
488     // to read off. Would be slightly better to properly read. Here we just skip it:
489     nextStart = row.getNextUnset(nextStart);
490     if (!row.isRange(nextStart,
491                      Math.min(row.getSize(), nextStart + (nextStart - lastStart) / 2),
492                      false)) {
493       throw NotFoundException.getNotFoundInstance();
494     }
495 
496     // Pull out from sum the value of the penultimate check code
497     checksumTotal -= multiplier * lastCode;
498     // lastCode is the checksum then:
499     if (checksumTotal % 103 != lastCode) {
500       throw ChecksumException.getChecksumInstance();
501     }
502 
503     // Need to pull out the check digits from string
504     int resultLength = result.length();
505     if (resultLength == 0) {
506       // false positive
507       throw NotFoundException.getNotFoundInstance();
508     }
509 
510     // Only bother if the result had at least one character, and if the checksum digit happened to
511     // be a printable character. If it was just interpreted as a control code, nothing to remove.
512     if (resultLength > 0 && lastCharacterWasPrintable) {
513       if (codeSet == CODE_CODE_C) {
514         result.delete(resultLength - 2, resultLength);
515       } else {
516         result.delete(resultLength - 1, resultLength);
517       }
518     }
519 
520     float left = (float) (startPatternInfo[1] + startPatternInfo[0]) / 2.0f;
521     float right = lastStart + lastPatternSize / 2.0f;
522 
523     int rawCodesSize = rawCodes.size();
524     byte[] rawBytes = new byte[rawCodesSize];
525     for (int i = 0; i < rawCodesSize; i++) {
526       rawBytes[i] = rawCodes.get(i);
527     }
528 
529     return new Result(
530         result.toString(),
531         rawBytes,
532         new ResultPoint[]{
533             new ResultPoint(left, (float) rowNumber),
534             new ResultPoint(right, (float) rowNumber)},
535         BarcodeFormat.CODE_128);
536 
537   }
