29  /**
30   * <p>Implements decoding of the ITF format, or Interleaved Two of Five.</p>
31   *
32   * <p>This Reader will scan ITF barcodes of certain lengths only.
33   * At the moment it reads length 6, 8, 10, 12, 14, 16, 18, 20, 24, and 44 as these have appeared "in the wild". Not all
34   * lengths are scanned, especially shorter ones, to avoid false positives. This in turn is due to a lack of
35   * required checksum function.</p>
36   *
37   * <p>The checksum is optional and is not applied by this Reader. The consumer of the decoded
38   * value will have to apply a checksum if required.</p>
39   *
40   * <p><a href="http://en.wikipedia.org/wiki/Interleaved_2_of_5">http://en.wikipedia.org/wiki/Interleaved_2_of_5</a>
41   * is a great reference for Interleaved 2 of 5 information.</p>
42   *
43   * @author kevin.osullivan@sita.aero, SITA Lab.
44   */
45  public final class ITFReader extends OneDReader {
46  
47    private static final int MAX_AVG_VARIANCE = (int) (PATTERN_MATCH_RESULT_SCALE_FACTOR * 0.42f);
48    private static final int MAX_INDIVIDUAL_VARIANCE = (int) (PATTERN_MATCH_RESULT_SCALE_FACTOR * 0.78f);
49  
50    private static final int W = 3; // Pixel width of a wide line
51    private static final int N = 1; // Pixed width of a narrow line
52  
53    /** Valid ITF lengths. Anything longer than the largest value is also allowed. */
54    private static final int[] DEFAULT_ALLOWED_LENGTHS = { 6, 8, 10, 12, 14 };
55  
56    // Stores the actual narrow line width of the image being decoded.
57    private int narrowLineWidth = -1;
58  
59    /**
60     * Start/end guard pattern.
61     *
62     * Note: The end pattern is reversed because the row is reversed before
63     * searching for the END_PATTERN
64     */
65    private static final int[] START_PATTERN = {N, N, N, N};
66    private static final int[] END_PATTERN_REVERSED = {N, N, W};
67  
68    /**
69     * Patterns of Wide / Narrow lines to indicate each digit
70     */
71    static final int[][] PATTERNS = {
72        {N, N, W, W, N}, // 0
73        {W, N, N, N, W}, // 1
74        {N, W, N, N, W}, // 2
75        {W, W, N, N, N}, // 3
76        {N, N, W, N, W}, // 4
77        {W, N, W, N, N}, // 5
78        {N, W, W, N, N}, // 6
79        {N, N, N, W, W}, // 7
80        {W, N, N, W, N}, // 8
81        {N, W, N, W, N}  // 9
82    };
83  
84    @Override
85    public Result decodeRow(int rowNumber, BitArray row, Map<DecodeHintType,?> hints)
86        throws FormatException, NotFoundException {
87  
88      // Find out where the Middle section (payload) starts & ends
89      int[] startRange = decodeStart(row);
90      int[] endRange = decodeEnd(row);
91  
92      StringBuilder result = new StringBuilder(20);
93      decodeMiddle(row, startRange[1], endRange[0], result);
94      String resultString = result.toString();
95  
96      int[] allowedLengths = null;
97      if (hints != null) {
98        allowedLengths = (int[]) hints.get(DecodeHintType.ALLOWED_LENGTHS);
99  
100     }
101     if (allowedLengths == null) {
102       allowedLengths = DEFAULT_ALLOWED_LENGTHS;
103     }
104 
105     // To avoid false positives with 2D barcodes (and other patterns), make
106     // an assumption that the decoded string must be a 'standard' length if it's short
107     int length = resultString.length();
108     boolean lengthOK = false;
109     int maxAllowedLength = 0;
110     for (int allowedLength : allowedLengths) {
111       if (length == allowedLength) {
112         lengthOK = true;
113         break;
114       }
115       if (allowedLength > maxAllowedLength) {
116         maxAllowedLength = allowedLength;
117       }
118     }
119     if (!lengthOK && length > maxAllowedLength) {
120       lengthOK = true;
121     }
122     if (!lengthOK) {
123       throw FormatException.getFormatInstance();
124     }
125 
126     return new Result(
127         resultString,
128         null, // no natural byte representation for these barcodes
129         new ResultPoint[] { new ResultPoint(startRange[1], (float) rowNumber),
130                             new ResultPoint(endRange[0], (float) rowNumber)},
131         BarcodeFormat.ITF);
132   }

134   /**
135    * @param row          row of black/white values to search
136    * @param payloadStart offset of start pattern
137    * @param resultString {@link StringBuilder} to append decoded chars to
138    * @throws NotFoundException if decoding could not complete successfully
139    */
140   private static void decodeMiddle(BitArray row,
141                                    int payloadStart,
142                                    int payloadEnd,
143                                    StringBuilder resultString) throws NotFoundException {
144 
145     // Digits are interleaved in pairs - 5 black lines for one digit, and the
146     // 5
147     // interleaved white lines for the second digit.
148     // Therefore, need to scan 10 lines and then
149     // split these into two arrays
150     int[] counterDigitPair = new int[10];
151     int[] counterBlack = new int[5];
152     int[] counterWhite = new int[5];
153 
154     while (payloadStart < payloadEnd) {
155 
156       // Get 10 runs of black/white.
157       recordPattern(row, payloadStart, counterDigitPair);
158       // Split them into each array
159       for (int k = 0; k < 5; k++) {
160         int twoK = k << 1;
161         counterBlack[k] = counterDigitPair[twoK];
162         counterWhite[k] = counterDigitPair[twoK + 1];
163       }
164 
165       int bestMatch = decodeDigit(counterBlack);
166       resultString.append((char) ('0' + bestMatch));
167       bestMatch = decodeDigit(counterWhite);
168       resultString.append((char) ('0' + bestMatch));
169 
170       for (int counterDigit : counterDigitPair) {
171         payloadStart += counterDigit;
172       }
173     }
174   }

176   /**
177    * Identify where the start of the middle / payload section starts.
178    *
179    * @param row row of black/white values to search
180    * @return Array, containing index of start of 'start block' and end of
181    *         'start block'
182    * @throws NotFoundException
183    */
184   int[] decodeStart(BitArray row) throws NotFoundException {
185     int endStart = skipWhiteSpace(row);
186     int[] startPattern = findGuardPattern(row, endStart, START_PATTERN);
187 
188     // Determine the width of a narrow line in pixels. We can do this by
189     // getting the width of the start pattern and dividing by 4 because its
190     // made up of 4 narrow lines.
191     this.narrowLineWidth = (startPattern[1] - startPattern[0]) >> 2;
192 
193     validateQuietZone(row, startPattern[0]);
194 
195     return startPattern;
196   }
197 
198   /**
199    * The start & end patterns must be pre/post fixed by a quiet zone. This
200    * zone must be at least 10 times the width of a narrow line.  Scan back until
201    * we either get to the start of the barcode or match the necessary number of
202    * quiet zone pixels.
203    *
204    * Note: Its assumed the row is reversed when using this method to find
205    * quiet zone after the end pattern.
206    *
207    * ref: http://www.barcode-1.net/i25code.html
208    *
209    * @param row bit array representing the scanned barcode.
210    * @param startPattern index into row of the start or end pattern.
211    * @throws NotFoundException if the quiet zone cannot be found, a ReaderException is thrown.
212    */
213   private void validateQuietZone(BitArray row, int startPattern) throws NotFoundException {
214 
215     int quietCount = this.narrowLineWidth * 10;  // expect to find this many pixels of quiet zone
216 
217     // if there are not so many pixel at all let's try as many as possible
218     quietCount = quietCount < startPattern ? quietCount : startPattern;
219 
220     for (int i = startPattern - 1; quietCount > 0 && i >= 0; i--) {
221       if (row.get(i)) {
222         break;
223       }
224       quietCount--;
225     }
226     if (quietCount != 0) {
227       // Unable to find the necessary number of quiet zone pixels.
228       throw NotFoundException.getNotFoundInstance();
229     }
230   }
231 
232   /**
233    * Skip all whitespace until we get to the first black line.
234    *
235    * @param row row of black/white values to search
236    * @return index of the first black line.
237    * @throws NotFoundException Throws exception if no black lines are found in the row
238    */
239   private static int skipWhiteSpace(BitArray row) throws NotFoundException {
240     int width = row.getSize();
241     int endStart = row.getNextSet(0);
242     if (endStart == width) {
243       throw NotFoundException.getNotFoundInstance();
244     }
245 
246     return endStart;
247   }
248 
249   /**
250    * Identify where the end of the middle / payload section ends.
251    *
252    * @param row row of black/white values to search
253    * @return Array, containing index of start of 'end block' and end of 'end
254    *         block'
255    * @throws NotFoundException
256    */
257   int[] decodeEnd(BitArray row) throws NotFoundException {
258 
259     // For convenience, reverse the row and then
260     // search from 'the start' for the end block
261     row.reverse();
262     try {
263       int endStart = skipWhiteSpace(row);
264       int[] endPattern = findGuardPattern(row, endStart, END_PATTERN_REVERSED);
265 
266       // The start & end patterns must be pre/post fixed by a quiet zone. This
267       // zone must be at least 10 times the width of a narrow line.
268       // ref: http://www.barcode-1.net/i25code.html
269       validateQuietZone(row, endPattern[0]);
270 
271       // Now recalculate the indices of where the 'endblock' starts & stops to
272       // accommodate
273       // the reversed nature of the search
274       int temp = endPattern[0];
275       endPattern[0] = row.getSize() - endPattern[1];
276       endPattern[1] = row.getSize() - temp;
277 
278       return endPattern;
279     } finally {
280       // Put the row back the right way.
281       row.reverse();
282     }
283   }

285   /**
286    * @param row       row of black/white values to search
287    * @param rowOffset position to start search
288    * @param pattern   pattern of counts of number of black and white pixels that are
289    *                  being searched for as a pattern
290    * @return start/end horizontal offset of guard pattern, as an array of two
291    *         ints
292    * @throws NotFoundException if pattern is not found
293    */
294   private static int[] findGuardPattern(BitArray row,
295                                         int rowOffset,
296                                         int[] pattern) throws NotFoundException {
297 
298     // TODO: This is very similar to implementation in UPCEANReader. Consider if they can be
299     // merged to a single method.
300     int patternLength = pattern.length;
301     int[] counters = new int[patternLength];
302     int width = row.getSize();
303     boolean isWhite = false;
304 
305     int counterPosition = 0;
306     int patternStart = rowOffset;
307     for (int x = rowOffset; x < width; x++) {
308       if (row.get(x) ^ isWhite) {
309         counters[counterPosition]++;
310       } else {
311         if (counterPosition == patternLength - 1) {
312           if (patternMatchVariance(counters, pattern, MAX_INDIVIDUAL_VARIANCE) < MAX_AVG_VARIANCE) {
313             return new int[]{patternStart, x};
314           }
315           patternStart += counters[0] + counters[1];
316           System.arraycopy(counters, 2, counters, 0, patternLength - 2);
317           counters[patternLength - 2] = 0;
318           counters[patternLength - 1] = 0;
319           counterPosition--;
320         } else {
321           counterPosition++;
322         }
323         counters[counterPosition] = 1;
324         isWhite = !isWhite;
325       }
326     }
327     throw NotFoundException.getNotFoundInstance();
328   }

330   /**
331    * Attempts to decode a sequence of ITF black/white lines into single
332    * digit.
333    *
334    * @param counters the counts of runs of observed black/white/black/... values
335    * @return The decoded digit
336    * @throws NotFoundException if digit cannot be decoded
337    */
338   private static int decodeDigit(int[] counters) throws NotFoundException {
339 
340     int bestVariance = MAX_AVG_VARIANCE; // worst variance we'll accept
341     int bestMatch = -1;
342     int max = PATTERNS.length;
343     for (int i = 0; i < max; i++) {
344       int[] pattern = PATTERNS[i];
345       int variance = patternMatchVariance(counters, pattern, MAX_INDIVIDUAL_VARIANCE);
346       if (variance < bestVariance) {
347         bestVariance = variance;
348         bestMatch = i;
349       }
350     }
351     if (bestMatch >= 0) {
352       return bestMatch;
353     } else {
354       throw NotFoundException.getNotFoundInstance();
355     }
356   }
