29  /**
30   * <p>Decodes Codabar barcodes.</p>
31   *
32   * @author Bas Vijfwinkel
33   * @author David Walker
34   */
35  public final class CodaBarReader extends OneDReader {
36  
37    // These values are critical for determining how permissive the decoding
38    // will be. All stripe sizes must be within the window these define, as
39    // compared to the average stripe size.
40    private static final float MAX_ACCEPTABLE = 2.0f;
41    private static final float PADDING = 1.5f;
42  
43    private static final String ALPHABET_STRING = "0123456789-$:/.+ABCD";
44    static final char[] ALPHABET = ALPHABET_STRING.toCharArray();
45  
46    /**
47     * These represent the encodings of characters, as patterns of wide and narrow bars. The 7 least-significant bits of
48     * each int correspond to the pattern of wide and narrow, with 1s representing "wide" and 0s representing narrow.
49     */
50    static final int[] CHARACTER_ENCODINGS = {
51        0x003, 0x006, 0x009, 0x060, 0x012, 0x042, 0x021, 0x024, 0x030, 0x048, // 0-9
52        0x00c, 0x018, 0x045, 0x051, 0x054, 0x015, 0x01A, 0x029, 0x00B, 0x00E, // -$:/.+ABCD
53    };
54  
55    // minimal number of characters that should be present (inclusing start and stop characters)
56    // under normal circumstances this should be set to 3, but can be set higher
57    // as a last-ditch attempt to reduce false positives.
58    private static final int MIN_CHARACTER_LENGTH = 3;
59  
60    // official start and end patterns
61    private static final char[] STARTEND_ENCODING = {'A', 'B', 'C', 'D'};
62    // some codabar generator allow the codabar string to be closed by every
63    // character. This will cause lots of false positives!
64  
65    // some industries use a checksum standard but this is not part of the original codabar standard
66    // for more information see : http://www.mecsw.com/specs/codabar.html
67  
68    // Keep some instance variables to avoid reallocations
69    private final StringBuilder decodeRowResult;
70    private int[] counters;
71    private int counterLength;
72  
73    public CodaBarReader() {
74      decodeRowResult = new StringBuilder(20);
75      counters = new int[80];
76      counterLength = 0;
77    }
78  
79    @Override
80    public Result decodeRow(int rowNumber, BitArray row, Map<DecodeHintType,?> hints) throws NotFoundException {
81  
82      Arrays.fill(counters, 0);
83      setCounters(row);
84      int startOffset = findStartPattern();
85      int nextStart = startOffset;
86  
87      decodeRowResult.setLength(0);
88      do {
89        int charOffset = toNarrowWidePattern(nextStart);
90        if (charOffset == -1) {
91          throw NotFoundException.getNotFoundInstance();
92        }
93        // Hack: We store the position in the alphabet table into a
94        // StringBuilder, so that we can access the decoded patterns in
95        // validatePattern. We'll translate to the actual characters later.
96        decodeRowResult.append((char)charOffset);
97        nextStart += 8;
98        // Stop as soon as we see the end character.
99        if (decodeRowResult.length() > 1 &&
100           arrayContains(STARTEND_ENCODING, ALPHABET[charOffset])) {
101         break;
102       }
103     } while (nextStart < counterLength); // no fixed end pattern so keep on reading while data is available
104 
105     // Look for whitespace after pattern:
106     int trailingWhitespace = counters[nextStart - 1];
107     int lastPatternSize = 0;
108     for (int i = -8; i < -1; i++) {
109       lastPatternSize += counters[nextStart + i];
110     }
111 
112     // We need to see whitespace equal to 50% of the last pattern size,
113     // otherwise this is probably a false positive. The exception is if we are
114     // at the end of the row. (I.e. the barcode barely fits.)
115     if (nextStart < counterLength && trailingWhitespace < lastPatternSize / 2) {
116       throw NotFoundException.getNotFoundInstance();
117     }
118 
119     validatePattern(startOffset);
120 
121     // Translate character table offsets to actual characters.
122     for (int i = 0; i < decodeRowResult.length(); i++) {
123       decodeRowResult.setCharAt(i, ALPHABET[decodeRowResult.charAt(i)]);
124     }
125     // Ensure a valid start and end character
126     char startchar = decodeRowResult.charAt(0);
127     if (!arrayContains(STARTEND_ENCODING, startchar)) {
128       throw NotFoundException.getNotFoundInstance();
129     }
130     char endchar = decodeRowResult.charAt(decodeRowResult.length() - 1);
131     if (!arrayContains(STARTEND_ENCODING, endchar)) {
132       throw NotFoundException.getNotFoundInstance();
133     }
134 
135     // remove stop/start characters character and check if a long enough string is contained
136     if (decodeRowResult.length() <= MIN_CHARACTER_LENGTH) {
137       // Almost surely a false positive ( start + stop + at least 1 character)
138       throw NotFoundException.getNotFoundInstance();
139     }
140 
141     if (hints == null || !hints.containsKey(DecodeHintType.RETURN_CODABAR_START_END)) {
142       decodeRowResult.deleteCharAt(decodeRowResult.length() - 1);
143       decodeRowResult.deleteCharAt(0);
144     }
145 
146     int runningCount = 0;
147     for (int i = 0; i < startOffset; i++) {
148       runningCount += counters[i];
149     }
150     float left = (float) runningCount;
151     for (int i = startOffset; i < nextStart - 1; i++) {
152       runningCount += counters[i];
153     }
154     float right = (float) runningCount;
155     return new Result(
156         decodeRowResult.toString(),
157         null,
158         new ResultPoint[]{
159             new ResultPoint(left, (float) rowNumber),
160             new ResultPoint(right, (float) rowNumber)},
161         BarcodeFormat.CODABAR);
162   }

164   void validatePattern(int start) throws NotFoundException {
165     // First, sum up the total size of our four categories of stripe sizes;
166     int[] sizes = {0, 0, 0, 0};
167     int[] counts = {0, 0, 0, 0};
168     int end = decodeRowResult.length() - 1;
169 
170     // We break out of this loop in the middle, in order to handle
171     // inter-character spaces properly.
172     int pos = start;
173     for (int i = 0; true; i++) {
174       int pattern = CHARACTER_ENCODINGS[decodeRowResult.charAt(i)];
175       for (int j = 6; j >= 0; j--) {
176         // Even j = bars, while odd j = spaces. Categories 2 and 3 are for
177         // long stripes, while 0 and 1 are for short stripes.
178         int category = (j & 1) + (pattern & 1) * 2;
179         sizes[category] += counters[pos + j];
180         counts[category]++;
181         pattern >>= 1;
182       }
183       if (i >= end) {
184         break;
185       }
186       // We ignore the inter-character space - it could be of any size.
187       pos += 8;
188     }
189 
190     // Calculate our allowable size thresholds using fixed-point math.
191     float[] maxes = new float[4];
192     float[] mins = new float[4];
193     // Define the threshold of acceptability to be the midpoint between the
194     // average small stripe and the average large stripe. No stripe lengths
195     // should be on the "wrong" side of that line.
196     for (int i = 0; i < 2; i++) {
197       mins[i] = 0.0f;  // Accept arbitrarily small "short" stripes.
198       mins[i + 2] = ((float) sizes[i] / counts[i] + (float) sizes[i + 2] / counts[i + 2]) / 2.0f;
199       maxes[i] = mins[i + 2];
200       maxes[i + 2] = (sizes[i + 2] * MAX_ACCEPTABLE + PADDING) / counts[i + 2];
201     }
202 
203     // Now verify that all of the stripes are within the thresholds.
204     pos = start;
205     for (int i = 0; true; i++) {
206       int pattern = CHARACTER_ENCODINGS[decodeRowResult.charAt(i)];
207       for (int j = 6; j >= 0; j--) {
208         // Even j = bars, while odd j = spaces. Categories 2 and 3 are for
209         // long stripes, while 0 and 1 are for short stripes.
210         int category = (j & 1) + (pattern & 1) * 2;
211         int size = counters[pos + j];
212         if (size < mins[category] || size > maxes[category]) {
213           throw NotFoundException.getNotFoundInstance();
214         }
215         pattern >>= 1;
216       }
217       if (i >= end) {
218         break;
219       }
220       pos += 8;
221     }
222   }

224   /**
225    * Records the size of all runs of white and black pixels, starting with white.
226    * This is just like recordPattern, except it records all the counters, and
227    * uses our builtin "counters" member for storage.
228    * @param row row to count from
229    */
230   private void setCounters(BitArray row) throws NotFoundException {
231     counterLength = 0;
232     // Start from the first white bit.
233     int i = row.getNextUnset(0);
234     int end = row.getSize();
235     if (i >= end) {
236       throw NotFoundException.getNotFoundInstance();
237     }
238     boolean isWhite = true;
239     int count = 0;
240     while (i < end) {
241       if (row.get(i) ^ isWhite) { // that is, exactly one is true
242         count++;
243       } else {
244         counterAppend(count);
245         count = 1;
246         isWhite = !isWhite;
247       }
248       i++;
249     }
250     counterAppend(count);
251   }
252 
253   private void counterAppend(int e) {
254     counters[counterLength] = e;
255     counterLength++;
256     if (counterLength >= counters.length) {
257       int[] temp = new int[counterLength * 2];
258       System.arraycopy(counters, 0, temp, 0, counterLength);
259       counters = temp;
260     }
261   }

263   private int findStartPattern() throws NotFoundException {
264     for (int i = 1; i < counterLength; i += 2) {
265       int charOffset = toNarrowWidePattern(i);
266       if (charOffset != -1 && arrayContains(STARTEND_ENCODING, ALPHABET[charOffset])) {
267         // Look for whitespace before start pattern, >= 50% of width of start pattern
268         // We make an exception if the whitespace is the first element.
269         int patternSize = 0;
270         for (int j = i; j < i + 7; j++) {
271           patternSize += counters[j];
272         }
273         if (i == 1 || counters[i-1] >= patternSize / 2) {
274           return i;
275         }
276       }
277     }
278     throw NotFoundException.getNotFoundInstance();
279   }

281   static boolean arrayContains(char[] array, char key) {
282     if (array != null) {
283       for (char c : array) {
284         if (c == key) {
285           return true;
286         }
287       }
288     }
289     return false;
290   }

292   // Assumes that counters[position] is a bar.
293   private int toNarrowWidePattern(int position) {
294     int end = position + 7;
295     if (end >= counterLength) {
296       return -1;
297     }
298 
299     int[] theCounters = counters;
300 
301     int maxBar = 0;
302     int minBar = Integer.MAX_VALUE;
303     for (int j = position; j < end; j += 2) {
304       int currentCounter = theCounters[j];
305       if (currentCounter < minBar) {
306         minBar = currentCounter;
307       }
308       if (currentCounter > maxBar) {
309         maxBar = currentCounter;
310       }
311     }
312     int thresholdBar = (minBar + maxBar) / 2;
313 
314     int maxSpace = 0;
315     int minSpace = Integer.MAX_VALUE;
316     for (int j = position + 1; j < end; j += 2) {
317       int currentCounter = theCounters[j];
318       if (currentCounter < minSpace) {
319         minSpace = currentCounter;
320       }
321       if (currentCounter > maxSpace) {
322         maxSpace = currentCounter;
323       }
324     }
325     int thresholdSpace = (minSpace + maxSpace) / 2;
326 
327     int bitmask = 1 << 7;
328     int pattern = 0;
329     for (int i = 0; i < 7; i++) {
330       int threshold = (i & 1) == 0 ? thresholdBar : thresholdSpace;
331       bitmask >>= 1;
332       if (theCounters[position + i] > threshold) {
333         pattern |= bitmask;
334       }
335     }
336 
337     for (int i = 0; i < CHARACTER_ENCODINGS.length; i++) {
338       if (CHARACTER_ENCODINGS[i] == pattern) {
339         return i;
340       }
341     }
342     return -1;
343   }
