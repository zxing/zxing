17  package com.google.zxing.oned;
18  
19  import com.google.zxing.BarcodeFormat;
20  import com.google.zxing.NotFoundException;
21  import com.google.zxing.Result;
22  import com.google.zxing.ResultMetadataType;
23  import com.google.zxing.ResultPoint;
24  import com.google.zxing.common.BitArray;
25  
26  import java.util.EnumMap;
27  import java.util.Map;
28  
29  /**
30   * @see UPCEANExtension5Support
31   */
32  final class UPCEANExtension2Support {
33  
34    private final int[] decodeMiddleCounters = new int[4];
35    private final StringBuilder decodeRowStringBuffer = new StringBuilder();
36  
37    Result decodeRow(int rowNumber, BitArray row, int[] extensionStartRange) throws NotFoundException {
38  
39      StringBuilder result = decodeRowStringBuffer;
40      result.setLength(0);
41      int end = decodeMiddle(row, extensionStartRange, result);
42  
43      String resultString = result.toString();
44      Map<ResultMetadataType,Object> extensionData = parseExtensionString(resultString);
45  
46      Result extensionResult =
47          new Result(resultString,
48                     null,
49                     new ResultPoint[] {
50                         new ResultPoint((extensionStartRange[0] + extensionStartRange[1]) / 2.0f, (float) rowNumber),
51                         new ResultPoint((float) end, (float) rowNumber),
52                     },
53                     BarcodeFormat.UPC_EAN_EXTENSION);
54      if (extensionData != null) {
55        extensionResult.putAllMetadata(extensionData);
56      }
57      return extensionResult;
58    }
59  
60    int decodeMiddle(BitArray row, int[] startRange, StringBuilder resultString) throws NotFoundException {
61      int[] counters = decodeMiddleCounters;
62      counters[0] = 0;
63      counters[1] = 0;
64      counters[2] = 0;
65      counters[3] = 0;
66      int end = row.getSize();
67      int rowOffset = startRange[1];
68  
69      int checkParity = 0;
70  
71      for (int x = 0; x < 2 && rowOffset < end; x++) {
72        int bestMatch = UPCEANReader.decodeDigit(row, counters, rowOffset, UPCEANReader.L_AND_G_PATTERNS);
73        resultString.append((char) ('0' + bestMatch % 10));
74        for (int counter : counters) {
75          rowOffset += counter;
76        }
77        if (bestMatch >= 10) {
78          checkParity |= 1 << (1 - x);
79        }
80        if (x != 1) {
81          // Read off separator if not last
82          rowOffset = row.getNextSet(rowOffset);
83          rowOffset = row.getNextUnset(rowOffset);
84        }
85      }
86  
87      if (resultString.length() != 2) {
88        throw NotFoundException.getNotFoundInstance();
89      }
90  
91      if (Integer.parseInt(resultString.toString()) % 4 != checkParity) {
92        throw NotFoundException.getNotFoundInstance();
93      }
94      
95      return rowOffset;
96    }
97  
98    /**
99     * @param raw raw content of extension
100    * @return formatted interpretation of raw content as a {@link Map} mapping
101    *  one {@link ResultMetadataType} to appropriate value, or {@code null} if not known
102    */
103   private static Map<ResultMetadataType,Object> parseExtensionString(String raw) {
104     if (raw.length() != 2) {
105       return null;
106     }
107     Map<ResultMetadataType,Object> result = new EnumMap<>(ResultMetadataType.class);
108     result.put(ResultMetadataType.ISSUE_NUMBER, Integer.valueOf(raw));
109     return result;
110   }
