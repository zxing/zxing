123   @Override
124   public Result decodeRow(int rowNumber,
125                           BitArray row,
126                           Map<DecodeHintType,?> hints) throws NotFoundException, FormatException {
127     // Rows can start with even pattern in case in prev rows there where odd number of patters.
128     // So lets try twice
129     this.pairs.clear();
130     this.startFromEven = false;
131     try {
132       List<ExpandedPair> pairs = decodeRow2pairs(rowNumber, row);
133       return constructResult(pairs);
134     } catch (NotFoundException e) {
135       // OK
136     }
137 
138     this.pairs.clear();
139     this.startFromEven = true;
140     List<ExpandedPair> pairs = decodeRow2pairs(rowNumber, row);
141     return constructResult(pairs);
142   }

150   // Not private for testing
151   List<ExpandedPair> decodeRow2pairs(int rowNumber, BitArray row) throws NotFoundException {
152     try {
153       while (true){
154         ExpandedPair nextPair = retrieveNextPair(row, this.pairs, rowNumber);
155         this.pairs.add(nextPair);
156         //System.out.println(this.pairs.size()+" pairs found so far on row "+rowNumber+": "+this.pairs);
157         // exit this loop when retrieveNextPair() fails and throws
158       }
159     } catch (NotFoundException nfe) {
160       if (this.pairs.isEmpty()) {
161         throw nfe;
162       }
163     }
164 
165     // TODO: verify sequence of finder patterns as in checkPairSequence()
166     if (checkChecksum()) {
167       return this.pairs;
168     }
169     
170     boolean tryStackedDecode = !this.rows.isEmpty();
171     boolean wasReversed = false; // TODO: deal with reversed rows
172     storeRow(rowNumber, wasReversed);
173     if (tryStackedDecode) {
174       // When the image is 180-rotated, then rows are sorted in wrong dirrection.
175       // Try twice with both the directions.
176       List<ExpandedPair> ps = checkRows(false);
177       if (ps != null) {
178         return ps;
179       }
180       ps = checkRows(true);
181       if (ps != null) {
182         return ps;
183       }
184     }
185     
186     throw NotFoundException.getNotFoundInstance();
187   }

189   private List<ExpandedPair> checkRows(boolean reverse) {
190     // Limit number of rows we are checking
191     // We use recursive algorithm with pure complexity and don't want it to take forever
192     // Stacked barcode can have up to 11 rows, so 25 seems resonable enough
193     if (this.rows.size() > 25) {
194       this.rows.clear();  // We will never have a chance to get result, so clear it
195       return null;
196     }
197 
198     this.pairs.clear();
199     if (reverse) {
200       Collections.reverse(this.rows);
201     }
202 
203     List<ExpandedPair> ps = null;
204     try {
205       ps = checkRows(new ArrayList<ExpandedRow>(), 0);
206     } catch (NotFoundException e) {
207       // OK
208     }
209 
210     if (reverse) {
211       Collections.reverse(this.rows);
212     }
213 
214     return ps;
215   }

217   // Try to construct a valid rows sequence
218   // Recursion is used to implement backtracking
219   private List<ExpandedPair> checkRows(List<ExpandedRow> collectedRows, int currentRow) throws NotFoundException {
220     for (int i = currentRow; i < rows.size(); i++) {
221       ExpandedRow row = rows.get(i);
222       this.pairs.clear();
223       int size = collectedRows.size();
224       for (int j = 0; j < size; j++) {
225         this.pairs.addAll(collectedRows.get(j).getPairs());
226       }
227       this.pairs.addAll(row.getPairs());
228 
229       if (!isValidSequence(this.pairs)) {
230         continue;
231       }
232 
233       if (checkChecksum()) {
234         return this.pairs;
235       }
236 
237       List<ExpandedRow> rs = new ArrayList<>();
238       rs.addAll(collectedRows);
239       rs.add(row);
240       try {
241         // Recursion: try to add more rows
242         return checkRows(rs, i + 1);
243       } catch (NotFoundException e) {
244         // We failed, try the next candidate
245       }
246     }
247 
248     throw NotFoundException.getNotFoundInstance();
249   }

251   // Whether the pairs form a valid find pattern seqience,
252   // either complete or a prefix
253   private static boolean isValidSequence(List<ExpandedPair> pairs) {
254     for (int[] sequence : FINDER_PATTERN_SEQUENCES) {
255       if (pairs.size() > sequence.length) {
256         continue;
257       }
258 
259       boolean stop = true;
260       for (int j = 0; j < pairs.size(); j++) {
261         if (pairs.get(j).getFinderPattern().getValue() != sequence[j]) {
262           stop = false;
263           break;
264         }
265       }
266 
267       if (stop) {
268         return true;
269       }
270     }
271 
272     return false;
273   }

275   private void storeRow(int rowNumber, boolean wasReversed) {
276     // Discard if duplicate above or below; otherwise insert in order by row number.
277     int insertPos = 0;
278     boolean prevIsSame = false;
279     boolean nextIsSame = false;
280     while (insertPos < this.rows.size()) {
281       ExpandedRow erow = this.rows.get(insertPos);
282       if (erow.getRowNumber() > rowNumber) {
283         nextIsSame = erow.isEquivalent(this.pairs);
284         break;
285       }
286       prevIsSame = erow.isEquivalent(this.pairs);
287       insertPos++;
288     }
289     if (nextIsSame || prevIsSame) {
290       return;
291     }
292 
293     // When the row was partially decoded (e.g. 2 pairs found instead of 3),
294     // it will prevent us from detecting the barcode.
295     // Try to merge partial rows
296 
297     // Check whether the row is part of an allready detected row
298     if (isPartialRow(this.pairs, this.rows)) {
299       return;
300     }
301 
302     this.rows.add(insertPos, new ExpandedRow(this.pairs, rowNumber, wasReversed));
303 
304     removePartialRows(this.pairs, this.rows);
305   }

307   // Remove all the rows that contains only specified pairs 
308   private static void removePartialRows(List<ExpandedPair> pairs, List<ExpandedRow> rows) {
309     for (Iterator<ExpandedRow> iterator = rows.iterator(); iterator.hasNext(); ) {
310       ExpandedRow r = iterator.next();
311       if (r.getPairs().size() == pairs.size()) {
312         continue;
313       }
314       boolean allFound = true;
315       for (ExpandedPair p : r.getPairs()) {
316         boolean found = false;
317         for (ExpandedPair pp : pairs) {
318           if (p.equals(pp)) {
319             found = true;
320             break;
321           }
322         }
323         if (!found) {
324           allFound = false;
325           break;
326         }
327       }
328       if (allFound) {
329         // 'pairs' contains all the pairs from the row 'r'
330         iterator.remove();
331       }
332     }
333   }

335   // Returns true when one of the rows already contains all the pairs
336   private static boolean isPartialRow(Iterable<ExpandedPair> pairs, Iterable<ExpandedRow> rows) {
337     for (ExpandedRow r : rows) {
338       boolean allFound = true;
339       for (ExpandedPair p : pairs) {
340         boolean found = false;
341         for (ExpandedPair pp : r.getPairs()) {
342           if (p.equals(pp)) {
343             found = true;
344             break;
345           }
346         }
347         if (!found) {
348           allFound = false;
349           break;
350         }
351       }
352       if (allFound) {
353         // the row 'r' contain all the pairs from 'pairs'
354         return true;
355       }
356     }
357     return false;
358   }

367     BitArray binary = BitArrayBuilder.buildBitArray(pairs);

369     AbstractExpandedDecoder decoder = AbstractExpandedDecoder.createDecoder(binary);
370     String resultingString = decoder.parseInformation();
371 
372     ResultPoint[] firstPoints = pairs.get(0).getFinderPattern().getResultPoints();
373     ResultPoint[] lastPoints  = pairs.get(pairs.size() - 1).getFinderPattern().getResultPoints();
374 
375     return new Result(
376           resultingString,
377           null,
378           new ResultPoint[]{firstPoints[0], firstPoints[1], lastPoints[0], lastPoints[1]},
379           BarcodeFormat.RSS_EXPANDED
380       );

383   private boolean checkChecksum() {
384     ExpandedPair firstPair = this.pairs.get(0);
385     DataCharacter checkCharacter = firstPair.getLeftChar();
386     DataCharacter firstCharacter = firstPair.getRightChar();
387 
388     if (firstCharacter == null) {
389       return false;
390     }
391 
392     int checksum = firstCharacter.getChecksumPortion();
393     int s = 2;
394 
395     for(int i = 1; i < this.pairs.size(); ++i){
396       ExpandedPair currentPair = this.pairs.get(i);
397       checksum += currentPair.getLeftChar().getChecksumPortion();
398       s++;
399       DataCharacter currentRightChar = currentPair.getRightChar();
400       if (currentRightChar != null) {
401         checksum += currentRightChar.getChecksumPortion();
402         s++;
403       }
404     }
405 
406     checksum %= 211;
407 
408     int checkCharacterValue = 211 * (s - 4) + checksum;
409 
410     return checkCharacterValue == checkCharacter.getValue();
411   }

413   private static int getNextSecondBar(BitArray row, int initialPos){
414     int currentPos;
415     if (row.get(initialPos)) {
416       currentPos = row.getNextUnset(initialPos);
417       currentPos = row.getNextSet(currentPos);
418     } else {
419       currentPos = row.getNextSet(initialPos);
420       currentPos = row.getNextUnset(currentPos);
421     }
422     return currentPos;
423   }

425   // not private for testing
426   ExpandedPair retrieveNextPair(BitArray row, List<ExpandedPair> previousPairs, int rowNumber)
427       throws NotFoundException {
428     boolean isOddPattern  = previousPairs.size() % 2 == 0;
429     if (startFromEven) {
430       isOddPattern = !isOddPattern;
431     }
432 
433     FinderPattern pattern;
434 
435     boolean keepFinding = true;
436     int forcedOffset = -1;
437     do{
438       this.findNextPair(row, previousPairs, forcedOffset);
439       pattern = parseFoundFinderPattern(row, rowNumber, isOddPattern);
440       if (pattern == null){
441         forcedOffset = getNextSecondBar(row, this.startEnd[0]);
442       } else {
443         keepFinding = false;
444       }
445     }while(keepFinding);
446 
447     // When stacked symbol is split over multiple rows, there's no way to guess if this pair can be last or not.
448     // boolean mayBeLast = checkPairSequence(previousPairs, pattern);
449 
450     DataCharacter leftChar  = this.decodeDataCharacter(row, pattern, isOddPattern, true);
451     
452     if (!previousPairs.isEmpty() && previousPairs.get(previousPairs.size()-1).mustBeLast()) {
453       throw NotFoundException.getNotFoundInstance();
454     }
455     
456     DataCharacter rightChar;
457     try {
458       rightChar = this.decodeDataCharacter(row, pattern, isOddPattern, false);
459     } catch(NotFoundException ignored) {
460       rightChar = null;
461     }
462     boolean mayBeLast = true;
463     return new ExpandedPair(leftChar, rightChar, pattern, mayBeLast);
464   }

466   private void findNextPair(BitArray row, List<ExpandedPair> previousPairs, int forcedOffset)
467       throws NotFoundException {
468     int[] counters = this.getDecodeFinderCounters();
469     counters[0] = 0;
470     counters[1] = 0;
471     counters[2] = 0;
472     counters[3] = 0;
473 
474     int width = row.getSize();
475 
476     int rowOffset;
477     if (forcedOffset >= 0) {
478       rowOffset = forcedOffset;
479     } else if (previousPairs.isEmpty()) {
480       rowOffset = 0;
481     } else{
482       ExpandedPair lastPair = previousPairs.get(previousPairs.size() - 1);
483       rowOffset = lastPair.getFinderPattern().getStartEnd()[1];
484     }
485     boolean searchingEvenPair = previousPairs.size() % 2 != 0;
486     if (startFromEven) {
487       searchingEvenPair = !searchingEvenPair;
488     }
489 
490     boolean isWhite = false;
491     while (rowOffset < width) {
492       isWhite = !row.get(rowOffset);
493       if (!isWhite) {
494         break;
495       }
496       rowOffset++;
497     }
498 
499     int counterPosition = 0;
500     int patternStart = rowOffset;
501     for (int x = rowOffset; x < width; x++) {
502       if (row.get(x) ^ isWhite) {
503         counters[counterPosition]++;
504       } else {
505         if (counterPosition == 3) {
506           if (searchingEvenPair) {
507             reverseCounters(counters);
508           }
509 
510           if (isFinderPattern(counters)){
511             this.startEnd[0] = patternStart;
512             this.startEnd[1] = x;
513             return;
514           }
515 
516           if (searchingEvenPair) {
517             reverseCounters(counters);
518           }
519 
520           patternStart += counters[0] + counters[1];
521           counters[0] = counters[2];
522           counters[1] = counters[3];
523           counters[2] = 0;
524           counters[3] = 0;
525           counterPosition--;
526         } else {
527           counterPosition++;
528         }
529         counters[counterPosition] = 1;
530         isWhite = !isWhite;
531       }
532     }
533     throw NotFoundException.getNotFoundInstance();
534   }

536   private static void reverseCounters(int [] counters){
537     int length = counters.length;
538     for(int i = 0; i < length / 2; ++i){
539       int tmp = counters[i];
540       counters[i] = counters[length - i - 1];
541       counters[length - i - 1] = tmp;
542     }
543   }

545   private FinderPattern parseFoundFinderPattern(BitArray row, int rowNumber, boolean oddPattern) {
546     // Actually we found elements 2-5.
547     int firstCounter;
548     int start;
549     int end;
550 
551     if(oddPattern){
552       // If pattern number is odd, we need to locate element 1 *before* the current block.
553 
554       int firstElementStart = this.startEnd[0] - 1;
555       // Locate element 1
556       while (firstElementStart >= 0 && !row.get(firstElementStart)) {
557         firstElementStart--;
558       }
559 
560       firstElementStart++;
561       firstCounter = this.startEnd[0] - firstElementStart;
562       start = firstElementStart;
563       end = this.startEnd[1];
564 
565     }else{
566       // If pattern number is even, the pattern is reversed, so we need to locate element 1 *after* the current block.
567 
568       start = this.startEnd[0];
569 
570       end = row.getNextUnset(this.startEnd[1] + 1);
571       firstCounter = end - this.startEnd[1];
572     }
573 
574     // Make 'counters' hold 1-4
575     int [] counters = this.getDecodeFinderCounters();
576     System.arraycopy(counters, 0, counters, 1, counters.length - 1);
577 
578     counters[0] = firstCounter;
579     int value;
580     try {
581       value = parseFinderValue(counters, FINDER_PATTERNS);
582     } catch (NotFoundException ignored) {
583       return null;
584     }
585     return new FinderPattern(value, new int[] {start, end}, start, end, rowNumber);
586   }

588   DataCharacter decodeDataCharacter(BitArray row,
589                                     FinderPattern pattern,
590                                     boolean isOddPattern,
591                                     boolean leftChar) throws NotFoundException {
592     int[] counters = this.getDataCharacterCounters();
593     counters[0] = 0;
594     counters[1] = 0;
595     counters[2] = 0;
596     counters[3] = 0;
597     counters[4] = 0;
598     counters[5] = 0;
599     counters[6] = 0;
600     counters[7] = 0;
601 
602     if (leftChar) {
603       recordPatternInReverse(row, pattern.getStartEnd()[0], counters);
604     } else {
605       recordPattern(row, pattern.getStartEnd()[1], counters);
606       // reverse it
607       for (int i = 0, j = counters.length - 1; i < j; i++, j--) {
608         int temp = counters[i];
609         counters[i] = counters[j];
610         counters[j] = temp;
611       }
612     }//counters[] has the pixels of the module
613 
614     int numModules = 17; //left and right data characters have all the same length
615     float elementWidth = (float) count(counters) / (float) numModules;
616 
617     // Sanity check: element width for pattern and the character should match
618     float expectedElementWidth = (pattern.getStartEnd()[1] - pattern.getStartEnd()[0]) / 15.0f;
619     if (Math.abs(elementWidth - expectedElementWidth) / expectedElementWidth > 0.3f) {
620       throw NotFoundException.getNotFoundInstance();
621     }
622 
623     int[] oddCounts = this.getOddCounts();
624     int[] evenCounts = this.getEvenCounts();
625     float[] oddRoundingErrors = this.getOddRoundingErrors();
626     float[] evenRoundingErrors = this.getEvenRoundingErrors();
627 
628     for (int i = 0; i < counters.length; i++) {
629       float value = 1.0f * counters[i] / elementWidth;
630       int count = (int) (value + 0.5f); // Round
631       if (count < 1) {
632         if (value < 0.3f) {
633           throw NotFoundException.getNotFoundInstance();
634         }
635         count = 1;
636       } else if (count > 8) {
637         if (value > 8.7f) {
638           throw NotFoundException.getNotFoundInstance();
639         }
640         count = 8;
641       }
642       int offset = i / 2;
643       if ((i & 0x01) == 0) {
644         oddCounts[offset] = count;
645         oddRoundingErrors[offset] = value - count;
646       } else {
647         evenCounts[offset] = count;
648         evenRoundingErrors[offset] = value - count;
649       }
650     }
651 
652     adjustOddEvenCounts(numModules);
653 
654     int weightRowNumber = 4 * pattern.getValue() + (isOddPattern?0:2) + (leftChar?0:1) - 1;
655 
656     int oddSum = 0;
657     int oddChecksumPortion = 0;
658     for (int i = oddCounts.length - 1; i >= 0; i--) {
659       if(isNotA1left(pattern, isOddPattern, leftChar)){
660         int weight = WEIGHTS[weightRowNumber][2 * i];
661         oddChecksumPortion += oddCounts[i] * weight;
662       }
663       oddSum += oddCounts[i];
664     }
665     int evenChecksumPortion = 0;
666     //int evenSum = 0;
667     for (int i = evenCounts.length - 1; i >= 0; i--) {
668       if(isNotA1left(pattern, isOddPattern, leftChar)){
669         int weight = WEIGHTS[weightRowNumber][2 * i + 1];
670         evenChecksumPortion += evenCounts[i] * weight;
671       }
672       //evenSum += evenCounts[i];
673     }
674     int checksumPortion = oddChecksumPortion + evenChecksumPortion;
675 
676     if ((oddSum & 0x01) != 0 || oddSum > 13 || oddSum < 4) {
677       throw NotFoundException.getNotFoundInstance();
678     }
679 
680     int group = (13 - oddSum) / 2;
681     int oddWidest = SYMBOL_WIDEST[group];
682     int evenWidest = 9 - oddWidest;
683     int vOdd = RSSUtils.getRSSvalue(oddCounts, oddWidest, true);
684     int vEven = RSSUtils.getRSSvalue(evenCounts, evenWidest, false);
685     int tEven = EVEN_TOTAL_SUBSET[group];
686     int gSum = GSUM[group];
687     int value = vOdd * tEven + vEven + gSum;
688 
689     return new DataCharacter(value, checksumPortion);
690   }

692   private static boolean isNotA1left(FinderPattern pattern, boolean isOddPattern, boolean leftChar) {
693     // A1: pattern.getValue is 0 (A), and it's an oddPattern, and it is a left char
694     return !(pattern.getValue() == 0 && isOddPattern && leftChar);
695   }

697   private void adjustOddEvenCounts(int numModules) throws NotFoundException {
698 
699     int oddSum = count(this.getOddCounts());
700     int evenSum = count(this.getEvenCounts());
701     int mismatch = oddSum + evenSum - numModules;
702     boolean oddParityBad = (oddSum & 0x01) == 1;
703     boolean evenParityBad = (evenSum & 0x01) == 0;
704 
705     boolean incrementOdd = false;
706     boolean decrementOdd = false;
707 
708     if (oddSum > 13) {
709       decrementOdd = true;
710     } else if (oddSum < 4) {
711       incrementOdd = true;
712     }
713     boolean incrementEven = false;
714     boolean decrementEven = false;
715     if (evenSum > 13) {
716       decrementEven = true;
717     } else if (evenSum < 4) {
718       incrementEven = true;
719     }
720 
721     if (mismatch == 1) {
722       if (oddParityBad) {
723         if (evenParityBad) {
724           throw NotFoundException.getNotFoundInstance();
725         }
726         decrementOdd = true;
727       } else {
728         if (!evenParityBad) {
729           throw NotFoundException.getNotFoundInstance();
730         }
731         decrementEven = true;
732       }
733     } else if (mismatch == -1) {
734       if (oddParityBad) {
735         if (evenParityBad) {
736           throw NotFoundException.getNotFoundInstance();
737         }
738         incrementOdd = true;
739       } else {
740         if (!evenParityBad) {
741           throw NotFoundException.getNotFoundInstance();
742         }
743         incrementEven = true;
744       }
745     } else if (mismatch == 0) {
746       if (oddParityBad) {
747         if (!evenParityBad) {
748           throw NotFoundException.getNotFoundInstance();
749         }
750         // Both bad
751         if (oddSum < evenSum) {
752           incrementOdd = true;
753           decrementEven = true;
754         } else {
755           decrementOdd = true;
756           incrementEven = true;
757         }
758       } else {
759         if (evenParityBad) {
760           throw NotFoundException.getNotFoundInstance();
761         }
762         // Nothing to do!
763       }
764     } else {
765       throw NotFoundException.getNotFoundInstance();
766     }
767 
768     if (incrementOdd) {
769       if (decrementOdd) {
770         throw NotFoundException.getNotFoundInstance();
771       }
772       increment(this.getOddCounts(), this.getOddRoundingErrors());
773     }
774     if (decrementOdd) {
775       decrement(this.getOddCounts(), this.getOddRoundingErrors());
776     }
777     if (incrementEven) {
778       if (decrementEven) {
779         throw NotFoundException.getNotFoundInstance();
780       }
781       increment(this.getEvenCounts(), this.getOddRoundingErrors());
782     }
783     if (decrementEven) {
784       decrement(this.getEvenCounts(), this.getEvenRoundingErrors());
785     }
786   }
