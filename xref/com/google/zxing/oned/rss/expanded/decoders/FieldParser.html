193   static String parseFieldsInGeneralPurpose(String rawInformation) throws NotFoundException{
194     if (rawInformation.isEmpty()) {
195       return null;
196     }
197 
198     // Processing 2-digit AIs
199 
200     if(rawInformation.length() < 2) {
201       throw NotFoundException.getNotFoundInstance();
202     }
203 
204     String firstTwoDigits = rawInformation.substring(0, 2);
205 
206     for (Object[] dataLength : TWO_DIGIT_DATA_LENGTH) {
207       if (dataLength[0].equals(firstTwoDigits)) {
208         if (dataLength[1] == VARIABLE_LENGTH) {
209           return processVariableAI(2, (Integer) dataLength[2], rawInformation);
210         }
211         return processFixedAI(2, (Integer) dataLength[1], rawInformation);
212       }
213     }
214 
215     if(rawInformation.length() < 3) {
216       throw NotFoundException.getNotFoundInstance();
217     }
218 
219     String firstThreeDigits = rawInformation.substring(0, 3);
220 
221     for (Object[] dataLength : THREE_DIGIT_DATA_LENGTH) {
222       if (dataLength[0].equals(firstThreeDigits)) {
223         if (dataLength[1] == VARIABLE_LENGTH) {
224           return processVariableAI(3, (Integer) dataLength[2], rawInformation);
225         }
226         return processFixedAI(3, (Integer) dataLength[1], rawInformation);
227       }
228     }
229 
230 
231     for (Object[] dataLength : THREE_DIGIT_PLUS_DIGIT_DATA_LENGTH) {
232       if (dataLength[0].equals(firstThreeDigits)) {
233         if (dataLength[1] == VARIABLE_LENGTH) {
234           return processVariableAI(4, (Integer) dataLength[2], rawInformation);
235         }
236         return processFixedAI(4, (Integer) dataLength[1], rawInformation);
237       }
238     }
239 
240     if(rawInformation.length() < 4) {
241       throw NotFoundException.getNotFoundInstance();
242     }
243 
244     String firstFourDigits = rawInformation.substring(0, 4);
245 
246     for (Object[] dataLength : FOUR_DIGIT_DATA_LENGTH) {
247       if (dataLength[0].equals(firstFourDigits)) {
248         if (dataLength[1] == VARIABLE_LENGTH) {
249           return processVariableAI(4, (Integer) dataLength[2], rawInformation);
250         }
251         return processFixedAI(4, (Integer) dataLength[1], rawInformation);
252       }
253     }
254 
255     throw NotFoundException.getNotFoundInstance();
256   }
257 
258   private static String processFixedAI(int aiSize, int fieldSize, String rawInformation) throws NotFoundException{
259     if (rawInformation.length() < aiSize) {
260       throw NotFoundException.getNotFoundInstance();
261     }
262 
263     String ai = rawInformation.substring(0, aiSize);
264 
265     if(rawInformation.length() < aiSize + fieldSize) {
266       throw NotFoundException.getNotFoundInstance();
267     }
268 
269     String field = rawInformation.substring(aiSize, aiSize + fieldSize);
270     String remaining = rawInformation.substring(aiSize + fieldSize);
271     String result = '(' + ai + ')' + field;
272     String parsedAI = parseFieldsInGeneralPurpose(remaining);
273     return parsedAI == null ? result : result + parsedAI;
274   }
275 
276   private static String processVariableAI(int aiSize, int variableFieldSize, String rawInformation)
277       throws NotFoundException {
278     String ai = rawInformation.substring(0, aiSize);
279     int maxSize;
280     if (rawInformation.length() < aiSize + variableFieldSize) {
281       maxSize = rawInformation.length();
282     } else {
283       maxSize = aiSize + variableFieldSize;
284     }
285     String field = rawInformation.substring(aiSize, maxSize);
286     String remaining = rawInformation.substring(maxSize);
287     String result = '(' + ai + ')' + field;
288     String parsedAI = parseFieldsInGeneralPurpose(remaining);
289     return parsedAI == null ? result : result + parsedAI;
290   }
