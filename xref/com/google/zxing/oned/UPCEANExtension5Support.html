32  final class UPCEANExtension5Support {
33  
34    private static final int[] CHECK_DIGIT_ENCODINGS = {
35        0x18, 0x14, 0x12, 0x11, 0x0C, 0x06, 0x03, 0x0A, 0x09, 0x05
36    };
37  
38    private final int[] decodeMiddleCounters = new int[4];
39    private final StringBuilder decodeRowStringBuffer = new StringBuilder();
40  
41    Result decodeRow(int rowNumber, BitArray row, int[] extensionStartRange) throws NotFoundException {
42  
43      StringBuilder result = decodeRowStringBuffer;
44      result.setLength(0);
45      int end = decodeMiddle(row, extensionStartRange, result);
46  
47      String resultString = result.toString();
48      Map<ResultMetadataType,Object> extensionData = parseExtensionString(resultString);
49  
50      Result extensionResult =
51          new Result(resultString,
52                     null,
53                     new ResultPoint[] {
54                         new ResultPoint((extensionStartRange[0] + extensionStartRange[1]) / 2.0f, (float) rowNumber),
55                         new ResultPoint((float) end, (float) rowNumber),
56                     },
57                     BarcodeFormat.UPC_EAN_EXTENSION);
58      if (extensionData != null) {
59        extensionResult.putAllMetadata(extensionData);
60      }
61      return extensionResult;
62    }
63  
64    int decodeMiddle(BitArray row, int[] startRange, StringBuilder resultString) throws NotFoundException {
65      int[] counters = decodeMiddleCounters;
66      counters[0] = 0;
67      counters[1] = 0;
68      counters[2] = 0;
69      counters[3] = 0;
70      int end = row.getSize();
71      int rowOffset = startRange[1];
72  
73      int lgPatternFound = 0;
74  
75      for (int x = 0; x < 5 && rowOffset < end; x++) {
76        int bestMatch = UPCEANReader.decodeDigit(row, counters, rowOffset, UPCEANReader.L_AND_G_PATTERNS);
77        resultString.append((char) ('0' + bestMatch % 10));
78        for (int counter : counters) {
79          rowOffset += counter;
80        }
81        if (bestMatch >= 10) {
82          lgPatternFound |= 1 << (4 - x);
83        }
84        if (x != 4) {
85          // Read off separator if not last
86          rowOffset = row.getNextSet(rowOffset);
87          rowOffset = row.getNextUnset(rowOffset);
88        }
89      }
90  
91      if (resultString.length() != 5) {
92        throw NotFoundException.getNotFoundInstance();
93      }
94  
95      int checkDigit = determineCheckDigit(lgPatternFound);
96      if (extensionChecksum(resultString.toString()) != checkDigit) {
97        throw NotFoundException.getNotFoundInstance();
98      }
99      
100     return rowOffset;
101   }
102 
103   private static int extensionChecksum(CharSequence s) {
104     int length = s.length();
105     int sum = 0;
106     for (int i = length - 2; i >= 0; i -= 2) {
107       sum += (int) s.charAt(i) - (int) '0';
108     }
109     sum *= 3;
110     for (int i = length - 1; i >= 0; i -= 2) {
111       sum += (int) s.charAt(i) - (int) '0';
112     }
113     sum *= 3;
114     return sum % 10;
115   }
116 
117   private static int determineCheckDigit(int lgPatternFound)
118       throws NotFoundException {
119     for (int d = 0; d < 10; d++) {
120       if (lgPatternFound == CHECK_DIGIT_ENCODINGS[d]) {
121         return d;
122       }
123     }
124     throw NotFoundException.getNotFoundInstance();
125   }

127   /**
128    * @param raw raw content of extension
129    * @return formatted interpretation of raw content as a {@link Map} mapping
130    *  one {@link ResultMetadataType} to appropriate value, or {@code null} if not known
131    */
132   private static Map<ResultMetadataType,Object> parseExtensionString(String raw) {
133     if (raw.length() != 5) {
134       return null;
135     }
136     Object value = parseExtension5String(raw);
137     if (value == null) {
138       return null;
139     }
140     Map<ResultMetadataType,Object> result = new EnumMap<>(ResultMetadataType.class);
141     result.put(ResultMetadataType.SUGGESTED_PRICE, value);
142     return result;
143   }
144 
145   private static String parseExtension5String(String raw) {
146     String currency;
147     switch (raw.charAt(0)) {
148       case '0':
149         currency = "Â£";
150         break;
151       case '5':
152         currency = "$";
153         break;
154       case '9':
155         // Reference: http://www.jollytech.com
156         if ("90000".equals(raw)) {
157           // No suggested retail price
158           return null;
159         }
160         if ("99991".equals(raw)) {
161           // Complementary
162           return "0.00";
163         }
164         if ("99990".equals(raw)) {
165           return "Used";
166         }
167         // Otherwise... unknown currency?
168         currency = "";
169         break;
170       default:
171         currency = "";
172         break;
173     }
174     int rawAmount = Integer.parseInt(raw.substring(1));
175     String unitsString = String.valueOf(rawAmount / 100);
176     int hundredths = rawAmount % 100;
177     String hundredthsString = hundredths < 10 ? "0" + hundredths : String.valueOf(hundredths);
178     return currency + unitsString + '.' + hundredthsString;
179   }
