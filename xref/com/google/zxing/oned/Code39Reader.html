31  /**
32   * <p>Decodes Code 39 barcodes. This does not support "Full ASCII Code 39" yet.</p>
33   *
34   * @author Sean Owen
35   * @see Code93Reader
36   */
37  public final class Code39Reader extends OneDReader {
38  
39    static final String ALPHABET_STRING = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ-. *$/+%";
40    private static final char[] ALPHABET = ALPHABET_STRING.toCharArray();
41  
42    /**
43     * These represent the encodings of characters, as patterns of wide and narrow bars.
44     * The 9 least-significant bits of each int correspond to the pattern of wide and narrow,
45     * with 1s representing "wide" and 0s representing narrow.
46     */
47    static final int[] CHARACTER_ENCODINGS = {
48        0x034, 0x121, 0x061, 0x160, 0x031, 0x130, 0x070, 0x025, 0x124, 0x064, // 0-9
49        0x109, 0x049, 0x148, 0x019, 0x118, 0x058, 0x00D, 0x10C, 0x04C, 0x01C, // A-J
50        0x103, 0x043, 0x142, 0x013, 0x112, 0x052, 0x007, 0x106, 0x046, 0x016, // K-T
51        0x181, 0x0C1, 0x1C0, 0x091, 0x190, 0x0D0, 0x085, 0x184, 0x0C4, 0x094, // U-*
52        0x0A8, 0x0A2, 0x08A, 0x02A // $-%
53    };
54  
55    private static final int ASTERISK_ENCODING = CHARACTER_ENCODINGS[39];
56  
57    private final boolean usingCheckDigit;
58    private final boolean extendedMode;
59    private final StringBuilder decodeRowResult;
60    private final int[] counters;
61  
62    /**
63     * Creates a reader that assumes all encoded data is data, and does not treat the final
64     * character as a check digit. It will not decoded "extended Code 39" sequences.
65     */
66    public Code39Reader() {
67      this(false);
68    }
69  
70    /**
71     * Creates a reader that can be configured to check the last character as a check digit.
72     * It will not decoded "extended Code 39" sequences.
73     *
74     * @param usingCheckDigit if true, treat the last data character as a check digit, not
75     * data, and verify that the checksum passes.
76     */
77    public Code39Reader(boolean usingCheckDigit) {
78      this(usingCheckDigit, false);
79    }
80  
81    /**
82     * Creates a reader that can be configured to check the last character as a check digit,
83     * or optionally attempt to decode "extended Code 39" sequences that are used to encode
84     * the full ASCII character set.
85     *
86     * @param usingCheckDigit if true, treat the last data character as a check digit, not
87     * data, and verify that the checksum passes.
88     * @param extendedMode if true, will attempt to decode extended Code 39 sequences in the
89     * text.
90     */
91    public Code39Reader(boolean usingCheckDigit, boolean extendedMode) {
92      this.usingCheckDigit = usingCheckDigit;
93      this.extendedMode = extendedMode;
94      decodeRowResult = new StringBuilder(20);
95      counters = new int[9];
96    }
97  
98    @Override
99    public Result decodeRow(int rowNumber, BitArray row, Map<DecodeHintType,?> hints)
100       throws NotFoundException, ChecksumException, FormatException {
101 
102     int[] theCounters = counters;
103     Arrays.fill(theCounters, 0);
104     StringBuilder result = decodeRowResult;
105     result.setLength(0);
106 
107     int[] start = findAsteriskPattern(row, theCounters);
108     // Read off white space    
109     int nextStart = row.getNextSet(start[1]);
110     int end = row.getSize();
111 
112     char decodedChar;
113     int lastStart;
114     do {
115       recordPattern(row, nextStart, theCounters);
116       int pattern = toNarrowWidePattern(theCounters);
117       if (pattern < 0) {
118         throw NotFoundException.getNotFoundInstance();
119       }
120       decodedChar = patternToChar(pattern);
121       result.append(decodedChar);
122       lastStart = nextStart;
123       for (int counter : theCounters) {
124         nextStart += counter;
125       }
126       // Read off white space
127       nextStart = row.getNextSet(nextStart);
128     } while (decodedChar != '*');
129     result.setLength(result.length() - 1); // remove asterisk
130 
131     // Look for whitespace after pattern:
132     int lastPatternSize = 0;
133     for (int counter : theCounters) {
134       lastPatternSize += counter;
135     }
136     int whiteSpaceAfterEnd = nextStart - lastStart - lastPatternSize;
137     // If 50% of last pattern size, following last pattern, is not whitespace, fail
138     // (but if it's whitespace to the very end of the image, that's OK)
139     if (nextStart != end && (whiteSpaceAfterEnd << 1) < lastPatternSize) {
140       throw NotFoundException.getNotFoundInstance();
141     }
142 
143     if (usingCheckDigit) {
144       int max = result.length() - 1;
145       int total = 0;
146       for (int i = 0; i < max; i++) {
147         total += ALPHABET_STRING.indexOf(decodeRowResult.charAt(i));
148       }
149       if (result.charAt(max) != ALPHABET[total % 43]) {
150         throw ChecksumException.getChecksumInstance();
151       }
152       result.setLength(max);
153     }
154 
155     if (result.length() == 0) {
156       // false positive
157       throw NotFoundException.getNotFoundInstance();
158     }
159 
160     String resultString;
161     if (extendedMode) {
162       resultString = decodeExtended(result);
163     } else {
164       resultString = result.toString();
165     }
166 
167     float left = (float) (start[1] + start[0]) / 2.0f;
168     float right = lastStart + lastPatternSize / 2.0f;
169     return new Result(
170         resultString,
171         null,
172         new ResultPoint[]{
173             new ResultPoint(left, (float) rowNumber),
174             new ResultPoint(right, (float) rowNumber)},
175         BarcodeFormat.CODE_39);
176 
177   }

179   private static int[] findAsteriskPattern(BitArray row, int[] counters) throws NotFoundException {
180     int width = row.getSize();
181     int rowOffset = row.getNextSet(0);
182 
183     int counterPosition = 0;
184     int patternStart = rowOffset;
185     boolean isWhite = false;
186     int patternLength = counters.length;
187 
188     for (int i = rowOffset; i < width; i++) {
189       if (row.get(i) ^ isWhite) {
190         counters[counterPosition]++;
191       } else {
192         if (counterPosition == patternLength - 1) {
193           // Look for whitespace before start pattern, >= 50% of width of start pattern
194           if (toNarrowWidePattern(counters) == ASTERISK_ENCODING &&
195               row.isRange(Math.max(0, patternStart - ((i - patternStart) >> 1)), patternStart, false)) {
196             return new int[]{patternStart, i};
197           }
198           patternStart += counters[0] + counters[1];
199           System.arraycopy(counters, 2, counters, 0, patternLength - 2);
200           counters[patternLength - 2] = 0;
201           counters[patternLength - 1] = 0;
202           counterPosition--;
203         } else {
204           counterPosition++;
205         }
206         counters[counterPosition] = 1;
207         isWhite = !isWhite;
208       }
209     }
210     throw NotFoundException.getNotFoundInstance();
211   }

213   // For efficiency, returns -1 on failure. Not throwing here saved as many as 700 exceptions
214   // per image when using some of our blackbox images.
215   private static int toNarrowWidePattern(int[] counters) {
216     int numCounters = counters.length;
217     int maxNarrowCounter = 0;
218     int wideCounters;
219     do {
220       int minCounter = Integer.MAX_VALUE;
221       for (int counter : counters) {
222         if (counter < minCounter && counter > maxNarrowCounter) {
223           minCounter = counter;
224         }
225       }
226       maxNarrowCounter = minCounter;
227       wideCounters = 0;
228       int totalWideCountersWidth = 0;
229       int pattern = 0;
230       for (int i = 0; i < numCounters; i++) {
231         int counter = counters[i];
232         if (counter > maxNarrowCounter) {
233           pattern |= 1 << (numCounters - 1 - i);
234           wideCounters++;
235           totalWideCountersWidth += counter;
236         }
237       }
238       if (wideCounters == 3) {
239         // Found 3 wide counters, but are they close enough in width?
240         // We can perform a cheap, conservative check to see if any individual
241         // counter is more than 1.5 times the average:
242         for (int i = 0; i < numCounters && wideCounters > 0; i++) {
243           int counter = counters[i];
244           if (counter > maxNarrowCounter) {
245             wideCounters--;
246             // totalWideCountersWidth = 3 * average, so this checks if counter >= 3/2 * average
247             if ((counter << 1) >= totalWideCountersWidth) {
248               return -1;
249             }
250           }
251         }
252         return pattern;
253       }
254     } while (wideCounters > 3);
255     return -1;
256   }

258   private static char patternToChar(int pattern) throws NotFoundException {
259     for (int i = 0; i < CHARACTER_ENCODINGS.length; i++) {
260       if (CHARACTER_ENCODINGS[i] == pattern) {
261         return ALPHABET[i];
262       }
263     }
264     throw NotFoundException.getNotFoundInstance();
265   }

267   private static String decodeExtended(CharSequence encoded) throws FormatException {
268     int length = encoded.length();
269     StringBuilder decoded = new StringBuilder(length);
270     for (int i = 0; i < length; i++) {
271       char c = encoded.charAt(i);
272       if (c == '+' || c == '$' || c == '%' || c == '/') {
273         char next = encoded.charAt(i + 1);
274         char decodedChar = '\0';
275         switch (c) {
276           case '+':
277             // +A to +Z map to a to z
278             if (next >= 'A' && next <= 'Z') {
279               decodedChar = (char) (next + 32);
280             } else {
281               throw FormatException.getFormatInstance();
282             }
283             break;
284           case '$':
285             // $A to $Z map to control codes SH to SB
286             if (next >= 'A' && next <= 'Z') {
287               decodedChar = (char) (next - 64);
288             } else {
289               throw FormatException.getFormatInstance();
290             }
291             break;
292           case '%':
293             // %A to %E map to control codes ESC to US
294             if (next >= 'A' && next <= 'E') {
295               decodedChar = (char) (next - 38);
296             } else if (next >= 'F' && next <= 'W') {
297               decodedChar = (char) (next - 11);
298             } else {
299               throw FormatException.getFormatInstance();
300             }
301             break;
302           case '/':
303             // /A to /O map to ! to , and /Z maps to :
304             if (next >= 'A' && next <= 'O') {
305               decodedChar = (char) (next - 32);
306             } else if (next == 'Z') {
307               decodedChar = ':';
308             } else {
309               throw FormatException.getFormatInstance();
310             }
311             break;
312         }
313         decoded.append(decodedChar);
314         // bump up i again since we read two characters
315         i++;
316       } else {
317         decoded.append(c);
318       }
319     }
320     return decoded.toString();
321   }
