17  package com.google.zxing.oned;
18  
19  import com.google.zxing.BarcodeFormat;
20  import com.google.zxing.EncodeHintType;
21  import com.google.zxing.Writer;
22  import com.google.zxing.WriterException;
23  import com.google.zxing.common.BitMatrix;
24  
25  import java.util.Map;
26  
27  /**
28   * <p>Encapsulates functionality and implementation that is common to one-dimensional barcodes.</p>
29   *
30   * @author dsbnatut@gmail.com (Kazuki Nishiura)
31   */
32  public abstract class OneDimensionalCodeWriter implements Writer {
33  
34    @Override
35    public final BitMatrix encode(String contents, BarcodeFormat format, int width, int height)
36        throws WriterException {
37      return encode(contents, format, width, height, null);
38    }
39  
40    /**
41     * Encode the contents following specified format.
42     * {@code width} and {@code height} are required size. This method may return bigger size
43     * {@code BitMatrix} when specified size is too small. The user can set both {@code width} and
44     * {@code height} to zero to get minimum size barcode. If negative value is set to {@code width}
45     * or {@code height}, {@code IllegalArgumentException} is thrown.
46     */
47    @Override
48    public BitMatrix encode(String contents,
49                            BarcodeFormat format,
50                            int width,
51                            int height,
52                            Map<EncodeHintType,?> hints) throws WriterException {
53      if (contents.isEmpty()) {
54        throw new IllegalArgumentException("Found empty contents");
55      }
56  
57      if (width < 0 || height < 0) {
58        throw new IllegalArgumentException("Negative size is not allowed. Input: "
59                                               + width + 'x' + height);
60      }
61  
62      int sidesMargin = getDefaultMargin();
63      if (hints != null) {
64        Integer sidesMarginInt = (Integer) hints.get(EncodeHintType.MARGIN);
65        if (sidesMarginInt != null) {
66          sidesMargin = sidesMarginInt;
67        }
68      }
69  
70      boolean[] code = encode(contents);
71      return renderResult(code, width, height, sidesMargin);
72    }
73  
74    /**
75     * @return a byte array of horizontal pixels (0 = white, 1 = black)
76     */
77    private static BitMatrix renderResult(boolean[] code, int width, int height, int sidesMargin) {
78      int inputWidth = code.length;
79      // Add quiet zone on both sides.
80      int fullWidth = inputWidth + sidesMargin;
81      int outputWidth = Math.max(width, fullWidth);
82      int outputHeight = Math.max(1, height);
83  
84      int multiple = outputWidth / fullWidth;
85      int leftPadding = (outputWidth - (inputWidth * multiple)) / 2;
86  
87      BitMatrix output = new BitMatrix(outputWidth, outputHeight);
88      for (int inputX = 0, outputX = leftPadding; inputX < inputWidth; inputX++, outputX += multiple) {
89        if (code[inputX]) {
90          output.setRegion(outputX, 0, multiple, outputHeight);
91        }
92      }
93      return output;
94    }
95  
96  
97    /**
98     * @param target encode black/white pattern into this array
99     * @param pos position to start encoding at in {@code target}
100    * @param pattern lengths of black/white runs to encode
101    * @param startColor starting color - false for white, true for black
102    * @return the number of elements added to target.
103    */
104   protected static int appendPattern(boolean[] target, int pos, int[] pattern, boolean startColor) {
105     boolean color = startColor;
106     int numAdded = 0;
107     for (int len : pattern) {
108       for (int j = 0; j < len; j++) {
109         target[pos++] = color;
110       }
111       numAdded += len;
112       color = !color; // flip color after each segment
113     }
114     return numAdded;
115   }
116 
117   public int getDefaultMargin() {
118     // CodaBar spec requires a side margin to be more than ten times wider than narrow space.
119     // This seems like a decent idea for a default for all formats.
120     return 10;
121   }
122 
123   /**
124    * Encode the contents to boolean array expression of one-dimensional barcode.
125    * Start code and end code should be included in result, and side margins should not be included.
126    *
127    * @param contents barcode contents to encode
128    * @return a {@code boolean[]} of horizontal pixels (false = white, true = black)
129    */
130   public abstract boolean[] encode(String contents);
131 }
