45  /**
46   * <p>A utility which auto-translates the English-language text in a directory of HTML documents using
47   * Google Translate.</p>
48   *
49   * <p>Pass the Android client assets/ directory as first argument, and the language to translate to second
50   * as a comma-separated list. Specify "all" for language to try to translate for all existing translations.
51   * Each argument after this is the name of a file to translate; if the first one is "all", all files will
52   * be translated.</p>
53   * 
54   * <p>Usage: {@code HtmlAssetTranslator android/assets/ (all|lang1[,lang2 ...]) (all|file1.html[ file2.html ...])}</p>
55   *
56   * <p>{@code android/assets/ es all} will translate .html files in subdirectory html-en to 
57   * directory html-es, for example. Note that only text nodes in the HTML document are translated. 
58   * Any text that is a child of a node with {@code class="notranslate"} will not be translated. It will 
59   * also add a note at the end of the translated page that indicates it was automatically translated.</p>
60   *
61   * @author Sean Owen
62   */
63  public final class HtmlAssetTranslator {
64  
65    private static final Pattern COMMA = Pattern.compile(",");
66  
67    private HtmlAssetTranslator() {}
68  
69    public static void main(String[] args) throws IOException {
70      if (args.length < 3) {
71        System.err.println("Usage: HtmlAssetTranslator android/assets/ " +
72                           "(all|lang1[,lang2 ...]) (all|file1.html[ file2.html ...])");
73        return;
74      }
75      Path assetsDir = Paths.get(args[0]);
76      Collection<String> languagesToTranslate = parseLanguagesToTranslate(assetsDir, args[1]);
77      List<String> restOfArgs = Arrays.asList(args).subList(2, args.length);
78      Collection<String> fileNamesToTranslate = parseFileNamesToTranslate(assetsDir, restOfArgs);
79      for (String language : languagesToTranslate) {
80        translateOneLanguage(assetsDir, language, fileNamesToTranslate);
81      }
82    }
83  
84    private static Collection<String> parseLanguagesToTranslate(Path assetsDir,
85                                                                CharSequence languageArg) throws IOException {
86      if ("all".equals(languageArg)) {
87        Collection<String> languages = new ArrayList<>();
88        DirectoryStream.Filter<Path> fileFilter = new DirectoryStream.Filter<Path>() {
89          @Override
90          public boolean accept(Path entry) {
91            String fileName = entry.getFileName().toString();
92            return Files.isDirectory(entry) && !Files.isSymbolicLink(entry) &&
93                fileName.startsWith("html-") && !"html-en".equals(fileName);
94          }
95        };
96        try (DirectoryStream<Path> dirs = Files.newDirectoryStream(assetsDir, fileFilter)) {
97          for (Path languageDir : dirs) {
98            languages.add(languageDir.getFileName().toString().substring(5));
99          }
100       }
101       return languages;
102     } else {
103       return Arrays.asList(COMMA.split(languageArg));
104     }
105   }
106 
107   private static Collection<String> parseFileNamesToTranslate(Path assetsDir,
108                                                               List<String> restOfArgs) throws IOException {
109     if ("all".equals(restOfArgs.get(0))) {
110       Collection<String> fileNamesToTranslate = new ArrayList<>();
111       Path htmlEnAssetDir = assetsDir.resolve("html-en");
112       try (DirectoryStream<Path> files = Files.newDirectoryStream(htmlEnAssetDir, "*.html")) {
113         for (Path file : files) {
114           fileNamesToTranslate.add(file.getFileName().toString());
115         }
116       }
117       return fileNamesToTranslate;
118     } else {
119       return restOfArgs;
120     }
121   }
122 
123   private static void translateOneLanguage(Path assetsDir,
124                                            String language,
125                                            final Collection<String> filesToTranslate) throws IOException {
126     Path targetHtmlDir = assetsDir.resolve("html-" + language);
127     Files.createDirectories(targetHtmlDir);
128     Path englishHtmlDir = assetsDir.resolve("html-en");
129 
130     String translationTextTranslated =
131         StringsResourceTranslator.translateString("Translated by Google Translate.", language);
132 
133     DirectoryStream.Filter<Path> filter = new DirectoryStream.Filter<Path>() {
134       @Override
135       public boolean accept(Path entry) {
136         String name = entry.getFileName().toString();
137         return name.endsWith(".html") && (filesToTranslate.isEmpty() || filesToTranslate.contains(name));
138       }
139     };
140     try (DirectoryStream<Path> files = Files.newDirectoryStream(englishHtmlDir, filter)) {
141       for (Path sourceFile : files) {
142         translateOneFile(language, targetHtmlDir, sourceFile, translationTextTranslated);
143       }
144     }
145   }
146 
147   private static void translateOneFile(String language,
148                                        Path targetHtmlDir,
149                                        Path sourceFile,
150                                        String translationTextTranslated) throws IOException {
151 
152     Path destFile = targetHtmlDir.resolve(sourceFile.getFileName());
153 
154     DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
155     Document document;
156     try {
157       DocumentBuilder builder = factory.newDocumentBuilder();
158       document = builder.parse(sourceFile.toFile());
159     } catch (ParserConfigurationException pce) {
160       throw new IllegalStateException(pce);
161     } catch (SAXException sae) {
162       throw new IOException(sae);
163     }
164 
165     Element rootElement = document.getDocumentElement();
166     rootElement.normalize();
167 
168     Queue<Node> nodes = new LinkedList<>();
169     nodes.add(rootElement);
170 
171     while (!nodes.isEmpty()) {
172       Node node = nodes.poll();
173       if (shouldTranslate(node)) {
174         NodeList children = node.getChildNodes();
175         for (int i = 0; i < children.getLength(); i++) {
176           nodes.add(children.item(i));
177         }
178       }
179       if (node.getNodeType() == Node.TEXT_NODE) {
180         String text = node.getTextContent();
181         if (!text.trim().isEmpty()) {
182           text = StringsResourceTranslator.translateString(text, language);
183           node.setTextContent(' ' + text + ' ');
184         }
185       }
186     }
187 
188     Node translateText = document.createTextNode(translationTextTranslated);
189     Node paragraph = document.createElement("p");
190     paragraph.appendChild(translateText);
191     Node body = rootElement.getElementsByTagName("body").item(0);
192     body.appendChild(paragraph);
193 
194     DOMImplementationRegistry registry;
195     try {
196       registry = DOMImplementationRegistry.newInstance();
197     } catch (ClassNotFoundException | InstantiationException | IllegalAccessException e) {
198       throw new IllegalStateException(e);
199     }
200 
201     DOMImplementationLS impl = (DOMImplementationLS) registry.getDOMImplementation("LS");
202     LSSerializer writer = impl.createLSSerializer();
203     writer.writeToURI(document, destFile.toUri().toString());
204   }
205 
206   private static boolean shouldTranslate(Node node) {
207     // Ignore "notranslate" nodes
208     NamedNodeMap attributes = node.getAttributes();
209     if (attributes != null) {
210       Node classAttribute = attributes.getNamedItem("class");
211       if (classAttribute != null) {
212         String textContent = classAttribute.getTextContent();
213         if (textContent != null && textContent.contains("notranslate")) {
214           return false;
215         }
216       }
217     }
218     String nodeName = node.getNodeName();
219     if ("script".equalsIgnoreCase(nodeName)) {
220       return false;
221     }
222     // Ignore non-text snippets
223     String textContent = node.getTextContent();
224     if (textContent != null) {
225       for (int i = 0; i < textContent.length(); i++) {
226         if (Character.isLetter(textContent.charAt(i))) {
227           return true;
228         }
229       }
230     }
231     return false;
232   }
