17  package com.google.zxing.client.glass;
18  
19  import java.io.IOException;
20  import java.util.ResourceBundle;
21  import java.util.concurrent.ExecutorService;
22  import java.util.concurrent.Executors;
23  import javax.servlet.ServletConfig;
24  import javax.servlet.ServletContext;
25  import javax.servlet.http.HttpServlet;
26  import javax.servlet.http.HttpServletRequest;
27  import javax.servlet.http.HttpServletResponse;
28  
29  import com.google.api.client.auth.oauth2.Credential;
30  import com.google.api.client.json.JsonObjectParser;
31  import com.google.api.client.json.jackson2.JacksonFactory;
32  import com.google.api.services.mirror.model.Notification;
33  
34  /**
35   * @author Sean Owen
36   * @link https://developers.google.com/glass/develop/mirror/timeline
37   */
38  public final class NotificationServlet extends HttpServlet {
39  
40    private ExecutorService executor;
41    private final JsonObjectParser jsonParser = new JsonObjectParser(new JacksonFactory());
42    private AuthUtil authUtil;
43    private String subscriptionVerifyToken;
44    
45    @Override
46    public void init(ServletConfig servletConfig) {
47      ServletContext context = servletConfig.getServletContext();
48      authUtil = new AuthUtil(context.getInitParameter("CLIENT_ID"), context.getInitParameter("CLIENT_SECRET"));
49      executor = Executors.newFixedThreadPool(Runtime.getRuntime().availableProcessors());
50      subscriptionVerifyToken = context.getInitParameter("SUBSCRIPTION_VERIFY_TOKEN");
51    }
52    
53    @Override
54    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws IOException {
55      
56      Notification notification = jsonParser.parseAndClose(request.getReader(), Notification.class);
57      
58      if (!subscriptionVerifyToken.equals(notification.getVerifyToken())) {
59        response.sendError(HttpServletResponse.SC_BAD_REQUEST);
60        return;
61      }
62      
63      String userId = notification.getUserToken();
64      Credential credential = authUtil.newAuthorizationCodeFlow().loadCredential(userId);
65      String timelineItemID = notification.getItemId();
66      
67      ResourceBundle resources = ResourceBundle.getBundle("Strings", request.getLocale());
68      
69      executor.submit(new NotificationCallable(credential, timelineItemID, resources));
70    }
71    
72    @Override
73    public void destroy() {
74      executor.shutdownNow();
75    }
76  
77  }
