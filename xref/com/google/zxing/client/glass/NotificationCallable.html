20  import java.io.InputStream;
21  import java.text.MessageFormat;
22  import java.util.Collection;
23  import java.util.List;
24  import java.util.ResourceBundle;
25  import java.util.concurrent.Callable;

27  import com.google.api.client.auth.oauth2.Credential;
28  import com.google.api.services.mirror.model.Attachment;
29  import com.google.api.services.mirror.model.TimelineItem;
30  import com.google.zxing.Result;
31  import com.google.zxing.client.result.ParsedResult;
32  import com.google.zxing.client.result.ResultParser;
33  import com.google.zxing.client.result.URIParsedResult;
34  import com.google.zxing.client.glass.mirror.MirrorClient;

36  /**
37   * @author Sean Owen
38   */
39  final class NotificationCallable implements Callable<Void> {
40    
41    private final Credential credential;
42    private final String timelineItemID;
43    private final ResourceBundle resources;
44  
45    NotificationCallable(Credential credential, String timelineItemID, ResourceBundle resources) {
46      this.credential = credential;
47      this.timelineItemID = timelineItemID;
48      this.resources = resources;
49    }
50  
51    @Override
52    public Void call() throws Exception {
53      MirrorClient mirrorClient = MirrorClient.get();
54      TimelineItem timelineItem = mirrorClient.getTimelineItem(credential, timelineItemID);
55      if (timelineItem == null) {
56        return null;
57      }
58      List<Attachment> attachments = timelineItem.getAttachments();
59      if (attachments == null || attachments.isEmpty()) {
60        return null;
61      }
62      Attachment attachment = attachments.get(0);
63      String attachmentId = attachment.getId();
64      
65      String errorMessage = null;
66      Collection<Result> results = null;
67      try (InputStream in = mirrorClient.getAttachmentInputStream(credential, timelineItemID, attachmentId)) {
68        results = DecodeHelper.processStream(in);
69      } catch (IOException | RuntimeException e) {
70        errorMessage = e.toString();
71      }
72      
73      if (results == null || results.isEmpty()) {
74        
75        TimelineItem resultTimelineItem = new TimelineItem();
76        if (errorMessage == null) {
77          resultTimelineItem.setTitle(resources.getString("scan.result.notfound.title"));
78          resultTimelineItem.setText(resources.getString("scan.result.notfound.text"));
79        } else {
80          resultTimelineItem.setTitle(resources.getString("scan.result.error.title"));
81          resultTimelineItem.setText(MessageFormat.format(resources.getString("scan.result.error.text"), errorMessage));        
82        }
83        mirrorClient.insertTimelineItem(credential, resultTimelineItem);
84        
85      } else {    
86        
87        for (Result result : results) {
88          ParsedResult parsedResult = ResultParser.parseResult(result);
89          
90          TimelineItem resultTimelineItem = new TimelineItem();
91          resultTimelineItem.setText(parsedResult.getDisplayResult());        
92          resultTimelineItem.setTitle(
93              MessageFormat.format(resources.getString("scan.result.success.title"), parsedResult.getType()));
94          
95          addResultDetailToTimelineItem(parsedResult, resultTimelineItem);
96          
97          mirrorClient.insertTimelineItem(credential, resultTimelineItem);        
98        }
99        
100     }
101     
102     return null;
103   }
104   
105   private static void addResultDetailToTimelineItem(ParsedResult parsedResult, TimelineItem item) {
106     if (parsedResult instanceof URIParsedResult) {
107       addURIResultDetailToTimelineItem((URIParsedResult) parsedResult, item);
108     }
109   }
110   
111   private static void addURIResultDetailToTimelineItem(URIParsedResult parsedResult, TimelineItem item) {
112     String uri = parsedResult.getURI();
113     item.setHtml("<a href=\"" + uri + "\">" + parsedResult.getDisplayResult() + "</a>");
114     item.setCanonicalUrl(uri);
115   }
116   
117 }
