17  package com.google.zxing.client.glass.mirror;
18  
19  import java.io.InputStream;
20  import java.util.Collection;
21  import java.util.Collections;
22  
23  import com.google.api.client.auth.oauth2.Credential;
24  import com.google.api.services.mirror.model.Attachment;
25  import com.google.api.services.mirror.model.Contact;
26  import com.google.api.services.mirror.model.Subscription;
27  import com.google.api.services.mirror.model.TimelineItem;
28  import com.google.common.collect.Sets;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;
31  
32  final class MemoryMirrorClient extends MirrorClient {
33    
34    private static final Logger log = LoggerFactory.getLogger(MemoryMirrorClient.class);
35    
36    private static final String FAKE_ATTACHMENT_ID = "0123";
37    
38    private final Collection<Contact> contacts = 
39        Collections.synchronizedCollection(Sets.<Contact>newHashSet());
40    private final Collection<Subscription> subscriptions = 
41        Collections.synchronizedCollection(Sets.<Subscription>newHashSet());
42  
43    @Override
44    public void insertContact(Credential credential, Contact contact) {
45      log.info("Insert Contact: {}", contact);    
46      contacts.add(contact);
47    }
48    
49    @Override
50    public Collection<Contact> listContacts(Credential credential) {
51      log.info("Contacts: {}", contacts);
52      return contacts;
53    }  
54  
55    @Override
56    public void insertTimelineItem(Credential credential, TimelineItem item) {
57      log.info("Insert Timeline: {}", item);
58    }
59    
60    @Override
61    public void insertSubscription(Credential credential, Subscription subscription) {
62      log.info("Insert Subscription: {}", subscription);
63      subscriptions.add(subscription);
64    }  
65    
66    @Override
67    public Collection<Subscription> listSubscriptions(Credential credential) {
68      log.info("Subscriptions: {}", subscriptions);
69      return subscriptions;
70    }  
71    
72    @Override
73    public TimelineItem getTimelineItem(Credential credential, String id) {
74      log.info("Get Timeline {}", id);
75      TimelineItem timelineItem = new TimelineItem();
76      timelineItem.setId(id);
77      Attachment fakeAttachment = new Attachment();
78      fakeAttachment.setId(FAKE_ATTACHMENT_ID);
79      timelineItem.setAttachments(Collections.singletonList(fakeAttachment));
80      return timelineItem;
81    }
82    
83    @Override
84    public InputStream getAttachmentInputStream(Credential credential, String timelineItemId, String attachmentId) {
85      log.info("Get Attachment {} {}", timelineItemId, attachmentId);
86      if (!FAKE_ATTACHMENT_ID.equals(attachmentId)) {
87        return null;
88      }
89      return MemoryMirrorClient.class.getResourceAsStream("/test.jpg");
90    }
91  
92  }
