17  package com.google.zxing.client.j2se;
18  
19  import java.awt.image.BufferedImage;
20  
21  /**
22   * Encapsulates custom configuration used in methods of {@link MatrixToImageWriter}.
23   */
24  public final class MatrixToImageConfig {
25  
26    public static final int BLACK = 0xFF000000;
27    public static final int WHITE = 0xFFFFFFFF;
28    
29    private final int onColor;
30    private final int offColor;
31  
32    /**
33     * Creates a default config with on color {@link #BLACK} and off color {@link #WHITE}, generating normal
34     * black-on-white barcodes.
35     */
36    public MatrixToImageConfig() {
37      this(BLACK, WHITE);
38    }
39  
40    /**
41     * @param onColor pixel on color, specified as an ARGB value as an int
42     * @param offColor pixel off color, specified as an ARGB value as an int
43     */
44    public MatrixToImageConfig(int onColor, int offColor) {
45      this.onColor = onColor;
46      this.offColor = offColor;
47    }
48  
49    public int getPixelOnColor() {
50      return onColor;
51    }
52  
53    public int getPixelOffColor() {
54      return offColor;
55    }
56  
57    int getBufferedImageColorModel() {
58      // Use faster BINARY if colors match default
59      return onColor == BLACK && offColor == WHITE ? BufferedImage.TYPE_BYTE_BINARY : BufferedImage.TYPE_INT_RGB;
60    }
61  
62  }
