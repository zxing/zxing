17  package com.google.zxing.client.j2se;
18  
19  import javax.imageio.ImageIO;
20  import javax.xml.bind.DatatypeConverter;
21  import java.awt.image.BufferedImage;
22  import java.io.ByteArrayInputStream;
23  import java.io.IOException;
24  import java.net.URI;
25  import java.net.URLDecoder;
26  
27  /**
28   * Encapsulates reading URIs as images.
29   * 
30   * @author Sean Owen
31   */
32  public final class ImageReader {
33  
34    private static final String BASE64TOKEN = "base64,";
35    
36    private ImageReader() {
37    }
38  
39    public static BufferedImage readImage(URI uri) throws IOException {
40      if ("data".equals(uri.getScheme())) {
41        return readDataURIImage(uri);
42      }
43      BufferedImage result;
44      try {
45        result = ImageIO.read(uri.toURL());
46      } catch (IllegalArgumentException iae) {
47        throw new IOException("Resource not found: " + uri, iae);
48      }
49      if (result == null) {
50        throw new IOException("Could not load " + uri);
51      }
52      return result;
53    }
54    
55    public static BufferedImage readDataURIImage(URI uri) throws IOException {
56      String uriString = uri.toString();
57      if (!uriString.startsWith("data:image/")) {
58        throw new IOException("Unsupported data URI MIME type");
59      }
60      int base64Start = uriString.indexOf(BASE64TOKEN);
61      if (base64Start < 0) {
62        throw new IOException("Unsupported data URI encoding");
63      }
64      String base64DataEncoded = uriString.substring(base64Start + BASE64TOKEN.length());
65      String base64Data = URLDecoder.decode(base64DataEncoded, "UTF-8");
66      byte[] imageBytes = DatatypeConverter.parseBase64Binary(base64Data);
67      return ImageIO.read(new ByteArrayInputStream(imageBytes));
68    }
69  
70  }
