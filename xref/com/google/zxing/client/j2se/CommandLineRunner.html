39  /**
40   * This simple command line utility decodes files, directories of files, or URIs which are passed
41   * as arguments. By default it uses the normal decoding algorithms, but you can pass --try_harder
42   * to request that hint. The raw text of each barcode is printed, and when running against
43   * directories, summary statistics are also displayed.
44   *
45   * @author dswitkin@google.com (Daniel Switkin)
46   */
47  public final class CommandLineRunner {
48  
49    private static final Pattern COMMA = Pattern.compile(",");
50  
51    private CommandLineRunner() {
52    }
53  
54    public static void main(String[] args) throws Exception {
55      if (args.length == 0) {
56        printUsage();
57        return;
58      }
59  
60      Config config = new Config();
61      Queue<Path> inputs = new ConcurrentLinkedQueue<>();
62  
63      for (String arg : args) {
64        String[] argValue = arg.split("=");
65        switch (argValue[0]) {
66          case "--try_harder":
67            config.setTryHarder(true);
68            break;
69          case "--pure_barcode":
70            config.setPureBarcode(true);
71            break;
72          case "--products_only":
73            config.setProductsOnly(true);
74            break;
75          case "--dump_results":
76            config.setDumpResults(true);
77            break;
78          case "--dump_black_point":
79            config.setDumpBlackPoint(true);
80            break;
81          case "--multi":
82            config.setMulti(true);
83            break;
84          case "--brief":
85            config.setBrief(true);
86            break;
87          case "--recursive":
88            config.setRecursive(true);
89            break;
90          case "--crop":
91            int[] crop = new int[4];
92            String[] tokens = COMMA.split(argValue[1]);
93            for (int i = 0; i < crop.length; i++) {
94              crop[i] = Integer.parseInt(tokens[i]);
95            }
96            config.setCrop(crop);
97            break;
98          case "--possibleFormats":
99            config.setPossibleFormats(COMMA.split(argValue[1]));
100           break;
101         default:
102           if (arg.startsWith("-")) {
103             System.err.println("Unknown command line option " + arg);
104             printUsage();
105             return;
106           }
107           addArgumentToInputs(Paths.get(arg), config, inputs);
108           break;
109       }
110     }
111     config.setHints(buildHints(config));
112 
113     int numThreads = Math.min(inputs.size(), Runtime.getRuntime().availableProcessors());
114     int successful = 0;    
115     if (numThreads > 1) {
116       ExecutorService executor = Executors.newFixedThreadPool(numThreads);
117       Collection<Future<Integer>> futures = new ArrayList<>(numThreads);
118       for (int x = 0; x < numThreads; x++) {
119         futures.add(executor.submit(new DecodeWorker(config, inputs)));
120       }
121       executor.shutdown();
122       for (Future<Integer> future : futures) {
123         successful += future.get();
124       }
125     } else {
126       successful += new DecodeWorker(config, inputs).call();
127     }
128 
129     int total = inputs.size();
130     if (total > 1) {
131       System.out.println("\nDecoded " + successful + " files out of " + total +
132           " successfully (" + (successful * 100 / total) + "%)\n");
133     }
134   }
135 
136   // Build all the inputs up front into a single flat list, so the threads can atomically pull
137   // paths/URLs off the queue.
138   private static void addArgumentToInputs(Path inputFile, Config config, Queue<Path> inputs) throws IOException {
139     if (Files.isDirectory(inputFile)) {
140       try (DirectoryStream<Path> paths = Files.newDirectoryStream(inputFile)) {
141         for (Path singleFile : paths) {
142           String filename = singleFile.getFileName().toString().toLowerCase(Locale.ENGLISH);
143           // Skip hidden files and directories (e.g. svn stuff).
144           if (filename.startsWith(".")) {
145             continue;
146           }
147           // Recur on nested directories if requested, otherwise skip them.
148           if (Files.isDirectory(singleFile)) {
149             if (config.isRecursive()) {
150               addArgumentToInputs(singleFile, config, inputs);
151             }
152             continue;
153           }
154           // Skip text files and the results of dumping the black point.
155           if (filename.endsWith(".txt") || filename.contains(".mono.png")) {
156             continue;
157           }
158           inputs.add(singleFile);
159         }
160       }
161     } else {
162       inputs.add(inputFile);
163     }
164   }

166   // Manually turn on all formats, even those not yet considered production quality.
167   private static Map<DecodeHintType,?> buildHints(Config config) {
168     Collection<BarcodeFormat> possibleFormats = new ArrayList<>();
169     String[] possibleFormatsNames = config.getPossibleFormats();
170     if (possibleFormatsNames != null && possibleFormatsNames.length > 0) {
171       for (String format : possibleFormatsNames) {
172         possibleFormats.add(BarcodeFormat.valueOf(format));
173       }
174     } else {
175       possibleFormats.add(BarcodeFormat.UPC_A);
176       possibleFormats.add(BarcodeFormat.UPC_E);
177       possibleFormats.add(BarcodeFormat.EAN_13);
178       possibleFormats.add(BarcodeFormat.EAN_8);
179       possibleFormats.add(BarcodeFormat.RSS_14);
180       possibleFormats.add(BarcodeFormat.RSS_EXPANDED);
181       if (!config.isProductsOnly()) {
182         possibleFormats.add(BarcodeFormat.CODE_39);
183         possibleFormats.add(BarcodeFormat.CODE_93);
184         possibleFormats.add(BarcodeFormat.CODE_128);
185         possibleFormats.add(BarcodeFormat.ITF);
186         possibleFormats.add(BarcodeFormat.QR_CODE);
187         possibleFormats.add(BarcodeFormat.DATA_MATRIX);
188         possibleFormats.add(BarcodeFormat.AZTEC);
189         possibleFormats.add(BarcodeFormat.PDF_417);
190         possibleFormats.add(BarcodeFormat.CODABAR);
191         possibleFormats.add(BarcodeFormat.MAXICODE);
192       }
193     }
194     Map<DecodeHintType, Object> hints = new EnumMap<>(DecodeHintType.class);
195     hints.put(DecodeHintType.POSSIBLE_FORMATS, possibleFormats);
196     if (config.isTryHarder()) {
197       hints.put(DecodeHintType.TRY_HARDER, Boolean.TRUE);
198     }
199     if (config.isPureBarcode()) {
200       hints.put(DecodeHintType.PURE_BARCODE, Boolean.TRUE);
201     }
202     return hints;
203   }
204 
205   private static void printUsage() {
206     System.err.println("Decode barcode images using the ZXing library");
207     System.err.println();
208     System.err.println("usage: CommandLineRunner { file | dir | url } [ options ]");
209     System.err.println("  --try_harder: Use the TRY_HARDER hint, default is normal (mobile) mode");
210     System.err.println("  --pure_barcode: Input image is a pure monochrome barcode image, not a photo");
211     System.err.println("  --products_only: Only decode the UPC and EAN families of barcodes");
212     System.err.println("  --dump_results: Write the decoded contents to input.txt");
213     System.err.println("  --dump_black_point: Compare black point algorithms as input.mono.png");
214     System.err.println("  --multi: Scans image for multiple barcodes");
215     System.err.println("  --brief: Only output one line per file, omitting the contents");
216     System.err.println("  --recursive: Descend into subdirectories");
217     System.err.println("  --crop=left,top,width,height: Only examine cropped region of input image(s)");
218     StringBuilder builder = new StringBuilder();
219     builder.append("  --possibleFormats=barcodeFormat[,barcodeFormat2...] where barcodeFormat is any of: ");
220     for (BarcodeFormat format : BarcodeFormat.values()) {
221       builder.append(format).append(',');
222     }
223     builder.setLength(builder.length() - 1);
224     System.err.println(builder);
225   }
