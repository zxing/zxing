17  package com.google.zxing.client.result;
18  
19  import com.google.zxing.Result;
20  
21  import java.util.ArrayList;
22  import java.util.List;
23  
24  /**
25   * Implements KDDI AU's address book format. See
26   * <a href="http://www.au.kddi.com/ezfactory/tec/two_dimensions/index.html">
27   * http://www.au.kddi.com/ezfactory/tec/two_dimensions/index.html</a>.
28   * (Thanks to Yuzo for translating!)
29   *
30   * @author Sean Owen
31   */
32  public final class AddressBookAUResultParser extends ResultParser {
33  
34    @Override
35    public AddressBookParsedResult parse(Result result) {
36      String rawText = getMassagedText(result);
37      // MEMORY is mandatory; seems like a decent indicator, as does end-of-record separator CR/LF
38      if (!rawText.contains("MEMORY") || !rawText.contains("\r\n")) {
39        return null;
40      }
41  
42      // NAME1 and NAME2 have specific uses, namely written name and pronunciation, respectively.
43      // Therefore we treat them specially instead of as an array of names.
44      String name = matchSinglePrefixedField("NAME1:", rawText, '\r', true);
45      String pronunciation = matchSinglePrefixedField("NAME2:", rawText, '\r', true);
46  
47      String[] phoneNumbers = matchMultipleValuePrefix("TEL", 3, rawText, true);
48      String[] emails = matchMultipleValuePrefix("MAIL", 3, rawText, true);
49      String note = matchSinglePrefixedField("MEMORY:", rawText, '\r', false);
50      String address = matchSinglePrefixedField("ADD:", rawText, '\r', true);
51      String[] addresses = address == null ? null : new String[] {address};
52      return new AddressBookParsedResult(maybeWrap(name),
53                                         null,
54                                         pronunciation,
55                                         phoneNumbers,
56                                         null,
57                                         emails,
58                                         null,
59                                         null,
60                                         note,
61                                         addresses,
62                                         null,
63                                         null,
64                                         null,
65                                         null,
66                                         null,
67                                         null);
68    }
69  
70    private static String[] matchMultipleValuePrefix(String prefix,
71                                                     int max,
72                                                     String rawText,
73                                                     boolean trim) {
74      List<String> values = null;
75      for (int i = 1; i <= max; i++) {
76        String value = matchSinglePrefixedField(prefix + i + ':', rawText, '\r', trim);
77        if (value == null) {
78          break;
79        }
80        if (values == null) {
81          values = new ArrayList<>(max); // lazy init
82        }
83        values.add(value);
84      }
85      if (values == null) {
86        return null;
87      }
88      return values.toArray(new String[values.size()]);
89    }
90  
91  }
