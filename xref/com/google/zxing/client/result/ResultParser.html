29  /**
30   * <p>Abstract class representing the result of decoding a barcode, as more than
31   * a String -- as some type of structured data. This might be a subclass which represents
32   * a URL, or an e-mail address. {@link #parseResult(Result)} will turn a raw
33   * decoded string into the most appropriate type of structured representation.</p>
34   *
35   * <p>Thanks to Jeff Griffin for proposing rewrite of these classes that relies less
36   * on exception-based mechanisms during parsing.</p>
37   *
38   * @author Sean Owen
39   */

65    private static final Pattern DIGITS = Pattern.compile("\\d+");
66    private static final Pattern AMPERSAND = Pattern.compile("&");
67    private static final Pattern EQUALS = Pattern.compile("=");
68    private static final String BYTE_ORDER_MARK = "\ufeff";

70    /**
71     * Attempts to parse the raw {@link Result}'s contents as a particular type
72     * of information (email, URL, etc.) and return a {@link ParsedResult} encapsulating
73     * the result of parsing.
74     */
75    public abstract ParsedResult parse(Result theResult);

77    protected static String getMassagedText(Result result) {
78      String text = result.getText();
79      if (text.startsWith(BYTE_ORDER_MARK)) {
80        text = text.substring(1);
81      }
82      return text;
83    }

85    public static ParsedResult parseResult(Result theResult) {
86      for (ResultParser parser : PARSERS) {
87        ParsedResult result = parser.parse(theResult);
88        if (result != null) {
89          return result;
90        }
91      }
92      return new TextParsedResult(theResult.getText(), null);
93    }

115   protected static String unescapeBackslash(String escaped) {
116     int backslash = escaped.indexOf((int) '\\');
117     if (backslash < 0) {
118       return escaped;
119     }
120     int max = escaped.length();
121     StringBuilder unescaped = new StringBuilder(max - 1);
122     unescaped.append(escaped.toCharArray(), 0, backslash);
123     boolean nextIsEscaped = false;
124     for (int i = backslash; i < max; i++) {
125       char c = escaped.charAt(i);
126       if (nextIsEscaped || c != '\\') {
127         unescaped.append(c);
128         nextIsEscaped = false;
129       } else {
130         nextIsEscaped = true;
131       }
132     }
133     return unescaped.toString();
134   }

136   protected static int parseHexDigit(char c) {
137     if (c >= '0' && c <= '9') {
138       return c - '0';
139     }
140     if (c >= 'a' && c <= 'f') {
141       return 10 + (c - 'a');
142     }
143     if (c >= 'A' && c <= 'F') {
144       return 10 + (c - 'A');
145     }
146     return -1;
147   }

149   protected static boolean isStringOfDigits(CharSequence value, int length) {
150     return value != null && length > 0 && length == value.length() && DIGITS.matcher(value).matches();
151   }
152 
153   protected static boolean isSubstringOfDigits(CharSequence value, int offset, int length) {
154     if (value == null || length <= 0) {
155       return false;
156     }
157     int max = offset + length;
158     return value.length() >= max && DIGITS.matcher(value.subSequence(offset, max)).matches();
159   }

161   static Map<String,String> parseNameValuePairs(String uri) {
162     int paramStart = uri.indexOf('?');
163     if (paramStart < 0) {
164       return null;
165     }
166     Map<String,String> result = new HashMap<>(3);
167     for (String keyValue : AMPERSAND.split(uri.substring(paramStart + 1))) {
168       appendKeyValue(keyValue, result);
169     }
170     return result;
171   }
172 
173   private static void appendKeyValue(CharSequence keyValue, Map<String,String> result) {
174     String[] keyValueTokens = EQUALS.split(keyValue, 2);
175     if (keyValueTokens.length == 2) {
176       String key = keyValueTokens[0];
177       String value = keyValueTokens[1];
178       try {
179         value = urlDecode(value);
180         result.put(key, value);
181       } catch (IllegalArgumentException iae) {
182         // continue; invalid data such as an escape like %0t
183       }
184     }
185   }
186 
187   static String urlDecode(String encoded) {
188     try {
189       return URLDecoder.decode(encoded, "UTF-8");
190     } catch (UnsupportedEncodingException uee) {
191       throw new IllegalStateException(uee); // can't happen
192     }
193   }

195   static String[] matchPrefixedField(String prefix, String rawText, char endChar, boolean trim) {
196     List<String> matches = null;
197     int i = 0;
198     int max = rawText.length();
199     while (i < max) {
200       i = rawText.indexOf(prefix, i);
201       if (i < 0) {
202         break;
203       }
204       i += prefix.length(); // Skip past this prefix we found to start
205       int start = i; // Found the start of a match here
206       boolean more = true;
207       while (more) {
208         i = rawText.indexOf((int) endChar, i);
209         if (i < 0) {
210           // No terminating end character? uh, done. Set i such that loop terminates and break
211           i = rawText.length();
212           more = false;
213         } else if (rawText.charAt(i - 1) == '\\') {
214           // semicolon was escaped so continue
215           i++;
216         } else {
217           // found a match
218           if (matches == null) {
219             matches = new ArrayList<>(3); // lazy init
220           }
221           String element = unescapeBackslash(rawText.substring(start, i));
222           if (trim) {
223             element = element.trim();
224           }
225           if (!element.isEmpty()) {
226             matches.add(element);
227           }
228           i++;
229           more = false;
230         }
231       }
232     }
233     if (matches == null || matches.isEmpty()) {
234       return null;
235     }
236     return matches.toArray(new String[matches.size()]);
237   }
238 
239   static String matchSinglePrefixedField(String prefix, String rawText, char endChar, boolean trim) {
240     String[] matches = matchPrefixedField(prefix, rawText, endChar, trim);
241     return matches == null ? null : matches[0];
242   }
