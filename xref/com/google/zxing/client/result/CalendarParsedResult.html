33  public final class CalendarParsedResult extends ParsedResult {

35    private static final Pattern RFC2445_DURATION =
36        Pattern.compile("P(?:(\\d+)W)?(?:(\\d+)D)?(?:T(?:(\\d+)H)?(?:(\\d+)M)?(?:(\\d+)S)?)?");
37    private static final long[] RFC2445_DURATION_FIELD_UNITS = {
38        7 * 24 * 60 * 60 * 1000L, // 1 week
39        24 * 60 * 60 * 1000L, // 1 day
40        60 * 60 * 1000L, // 1 hour
41        60 * 1000L, // 1 minute
42        1000L, // 1 second
43    };
44  
45    private static final Pattern DATE_TIME = Pattern.compile("[0-9]{8}(T[0-9]{6}Z?)?");
46  
47    private static final DateFormat DATE_FORMAT = new SimpleDateFormat("yyyyMMdd", Locale.ENGLISH);
48    static {
49      // For dates without a time, for purposes of interacting with Android, the resulting timestamp
50      // needs to be midnight of that day in GMT. See:
51      // http://code.google.com/p/android/issues/detail?id=8330
52      DATE_FORMAT.setTimeZone(TimeZone.getTimeZone("GMT"));
53    }
54    private static final DateFormat DATE_TIME_FORMAT = new SimpleDateFormat("yyyyMMdd'T'HHmmss", Locale.ENGLISH);
55  
56    private final String summary;
57    private final Date start;
58    private final boolean startAllDay;
59    private final Date end;
60    private final boolean endAllDay;
61    private final String location;
62    private final String organizer;
63    private final String[] attendees;
64    private final String description;
65    private final double latitude;
66    private final double longitude;

68    public CalendarParsedResult(String summary,
69                                String startString,
70                                String endString,
71                                String durationString,
72                                String location,
73                                String organizer,
74                                String[] attendees,
75                                String description,
76                                double latitude,
77                                double longitude) {
78      super(ParsedResultType.CALENDAR);
79      this.summary = summary;
80  
81      try {
82        this.start = parseDate(startString);
83      } catch (ParseException pe) {
84        throw new IllegalArgumentException(pe.toString());
85      }
86  
87      if (endString == null) {
88        long durationMS = parseDurationMS(durationString);
89        end = durationMS < 0L ? null : new Date(start.getTime() + durationMS);
90      } else {
91        try {
92          this.end = parseDate(endString);
93        } catch (ParseException pe) {
94          throw new IllegalArgumentException(pe.toString());
95        }
96      }
97  
98      this.startAllDay = startString.length() == 8;
99      this.endAllDay = endString != null && endString.length() == 8;
100 
101     this.location = location;
102     this.organizer = organizer;
103     this.attendees = attendees;
104     this.description = description;
105     this.latitude = latitude;
106     this.longitude = longitude;
107   }

166   @Override
167   public String getDisplayResult() {
168     StringBuilder result = new StringBuilder(100);
169     maybeAppend(summary, result);
170     maybeAppend(format(startAllDay, start), result);
171     maybeAppend(format(endAllDay, end), result);
172     maybeAppend(location, result);
173     maybeAppend(organizer, result);
174     maybeAppend(attendees, result);
175     maybeAppend(description, result);
176     return result.toString();
177   }

179   /**
180    * Parses a string as a date. RFC 2445 allows the start and end fields to be of type DATE (e.g. 20081021)
181    * or DATE-TIME (e.g. 20081021T123000 for local time, or 20081021T123000Z for UTC).
182    *
183    * @param when The string to parse
184    * @throws ParseException if not able to parse as a date
185    */
186   private static Date parseDate(String when) throws ParseException {
187     if (!DATE_TIME.matcher(when).matches()) {
188       throw new ParseException(when, 0);
189     }
190     if (when.length() == 8) {
191       // Show only year/month/day
192       return DATE_FORMAT.parse(when);
193     } else {
194       // The when string can be local time, or UTC if it ends with a Z
195       Date date;
196       if (when.length() == 16 && when.charAt(15) == 'Z') {
197         date = DATE_TIME_FORMAT.parse(when.substring(0, 15));
198         Calendar calendar = new GregorianCalendar();
199         long milliseconds = date.getTime();
200         // Account for time zone difference
201         milliseconds += calendar.get(Calendar.ZONE_OFFSET);
202         // Might need to correct for daylight savings time, but use target time since
203         // now might be in DST but not then, or vice versa
204         calendar.setTime(new Date(milliseconds));
205         milliseconds += calendar.get(Calendar.DST_OFFSET);
206         date = new Date(milliseconds);
207       } else {
208         date = DATE_TIME_FORMAT.parse(when);
209       }
210       return date;
211     }
212   }
213 
214   private static String format(boolean allDay, Date date) {
215     if (date == null) {
216       return null;
217     }
218     DateFormat format = allDay
219         ? DateFormat.getDateInstance(DateFormat.MEDIUM)
220         : DateFormat.getDateTimeInstance(DateFormat.MEDIUM, DateFormat.MEDIUM);
221     return format.format(date);
222   }
223 
224   private static long parseDurationMS(CharSequence durationString) {
225     if (durationString == null) {
226       return -1L;
227     }
228     Matcher m = RFC2445_DURATION.matcher(durationString);
229     if (!m.matches()) {
230       return -1L;
231     }
232     long durationMS = 0L;
233     for (int i = 0; i < RFC2445_DURATION_FIELD_UNITS.length; i++) {
234       String fieldValue = m.group(i + 1);
235       if (fieldValue != null) {
236         durationMS += RFC2445_DURATION_FIELD_UNITS[i] * Integer.parseInt(fieldValue);
237       }
238     }
239     return durationMS;
240   }
