17  package com.google.zxing.client.result;
18  
19  import com.google.zxing.BarcodeFormat;
20  import com.google.zxing.Result;
21  
22  import java.util.regex.Pattern;
23  
24  /**
25   * Detects a result that is likely a vehicle identification number.
26   *
27   * @author Sean Owen
28   */
29  public final class VINResultParser extends ResultParser {
30  
31    private static final Pattern IOQ = Pattern.compile("[IOQ]");
32    private static final Pattern AZ09 = Pattern.compile("[A-Z0-9]{17}");
33  
34    @Override
35    public VINParsedResult parse(Result result) {
36      if (result.getBarcodeFormat() != BarcodeFormat.CODE_39) {
37        return null;
38      }
39      String rawText = result.getText();
40      rawText = IOQ.matcher(rawText).replaceAll("").trim();
41      if (!AZ09.matcher(rawText).matches()) {
42        return null;
43      }
44      if (!checkChecksum(rawText)) {
45        return null;
46      }
47  
48      String wmi = rawText.substring(0, 3);
49      return new VINParsedResult(rawText,
50                                 wmi,
51                                 rawText.substring(3, 9),
52                                 rawText.substring(9, 17),
53                                 countryCode(wmi),
54                                 rawText.substring(3, 8),
55                                 modelYear(rawText.charAt(9)),
56                                 rawText.charAt(10),
57                                 rawText.substring(11));
58    }
59  
60    private static boolean checkChecksum(CharSequence vin) {
61      int sum = 0;
62      for (int i = 0; i < vin.length(); i++) {
63        sum += vinPositionWeight(i + 1) * vinCharValue(vin.charAt(i));
64      }
65      char checkChar = vin.charAt(8);
66      char expectedCheckChar = checkChar(sum % 11);
67      return checkChar == expectedCheckChar;
68    }
69    
70    private static int vinCharValue(char c) {
71      if (c >= 'A' && c <= 'I') {
72        return (c - 'A') + 1;
73      }
74      if (c >= 'J' && c <= 'R') {
75        return (c - 'J') + 1;
76      }
77      if (c >= 'S' && c <= 'Z') {
78        return (c - 'S') + 2;
79      }
80      if (c >= '0' && c <= '9') {
81        return c - '0';
82      }
83      throw new IllegalArgumentException();
84    }
85    
86    private static int vinPositionWeight(int position) {
87      if (position >= 1 && position <= 7) {
88        return 9 - position;
89      }
90      if (position == 8) {
91        return 10;
92      }
93      if (position == 9) {
94        return 0;
95      }
96      if (position >= 10 && position <= 17) {
97        return 19 - position;
98      }
99      throw new IllegalArgumentException();
100   }
101 
102   private static char checkChar(int remainder) {
103     if (remainder < 10) {
104       return (char) ('0' + remainder);
105     }
106     if (remainder == 10) {
107       return 'X';
108     }
109     throw new IllegalArgumentException();
110   }
111   
112   private static int modelYear(char c) {
113     if (c >= 'E' && c <= 'H') {
114       return (c - 'E') + 1984;
115     }
116     if (c >= 'J' && c <= 'N') {
117       return (c - 'J') + 1988;
118     }
119     if (c == 'P') {
120       return 1993;
121     }
122     if (c >= 'R' && c <= 'T') {
123       return (c - 'R') + 1994;
124     }
125     if (c >= 'V' && c <= 'Y') {
126       return (c - 'V') + 1997;
127     }
128     if (c >= '1' && c <= '9') {
129       return (c - '1') + 2001;
130     }
131     if (c >= 'A' && c <= 'D') {
132       return (c - 'A') + 2010;
133     }
134     throw new IllegalArgumentException();
135   }
136 
137   private static String countryCode(CharSequence wmi) {
138     char c1 = wmi.charAt(0);
139     char c2 = wmi.charAt(1);
140     switch (c1) {
141       case '1':
142       case '4':
143       case '5':
144         return "US";
145       case '2':
146         return "CA";
147       case '3':
148         if (c2 >= 'A' && c2 <= 'W') {
149           return "MX";
150         }
151         break;
152       case '9':
153         if ((c2 >= 'A' && c2 <= 'E') || (c2 >= '3' && c2 <= '9')) {
154           return "BR";
155         }
156         break;
157       case 'J':
158         if (c2 >= 'A' && c2 <= 'T') {
159           return "JP";
160         }
161         break;
162       case 'K':
163         if (c2 >= 'L' && c2 <= 'R') {
164           return "KO";
165         }
166         break;
167       case 'L':
168         return "CN";
169       case 'M':
170         if (c2 >= 'A' && c2 <= 'E') {
171           return "IN";
172         }
173         break;
174       case 'S':
175         if (c2 >= 'A' && c2 <= 'M') {
176           return "UK";
177         }
178         if (c2 >= 'N' && c2 <= 'T') {
179           return "DE";
180         }
181         break;
182       case 'V':
183         if (c2 >= 'F' && c2 <= 'R') {
184           return "FR";
185         }
186         if (c2 >= 'S' && c2 <= 'W') {
187           return "ES";
188         }
189         break;
190       case 'W':
191         return "DE";
192       case 'X':
193         if (c2 == '0' || (c2 >= '3' && c2 <= '9')) {
194           return "RU";
195         }
196         break;
197       case 'Z':
198         if (c2 >= 'A' && c2 <= 'R') {
199           return "IT";
200         }
201         break;
202     }
203     return null;
204   }
