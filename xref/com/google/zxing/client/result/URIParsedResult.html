17  package com.google.zxing.client.result;
18  
19  import java.util.regex.Pattern;
20  
21  /**
22   * @author Sean Owen
23   */
24  public final class URIParsedResult extends ParsedResult {
25  
26    private static final Pattern USER_IN_HOST = Pattern.compile(":/*([^/@]+)@[^/]+");
27  
28    private final String uri;
29    private final String title;
30  
31    public URIParsedResult(String uri, String title) {
32      super(ParsedResultType.URI);
33      this.uri = massageURI(uri);
34      this.title = title;
35    }
36  
37    public String getURI() {
38      return uri;
39    }
40  
41    public String getTitle() {
42      return title;
43    }
44  
45    /**
46     * @return true if the URI contains suspicious patterns that may suggest it intends to
47     *  mislead the user about its true nature. At the moment this looks for the presence
48     *  of user/password syntax in the host/authority portion of a URI which may be used
49     *  in attempts to make the URI's host appear to be other than it is. Example:
50     *  http://yourbank.com@phisher.com  This URI connects to phisher.com but may appear
51     *  to connect to yourbank.com at first glance.
52     */
53    public boolean isPossiblyMaliciousURI() {
54      return USER_IN_HOST.matcher(uri).find();
55    }
56  
57    @Override
58    public String getDisplayResult() {
59      StringBuilder result = new StringBuilder(30);
60      maybeAppend(title, result);
61      maybeAppend(uri, result);
62      return result.toString();
63    }
64  
65    /**
66     * Transforms a string that represents a URI into something more proper, by adding or canonicalizing
67     * the protocol.
68     */
69    private static String massageURI(String uri) {
70      uri = uri.trim();
71      int protocolEnd = uri.indexOf(':');
72      if (protocolEnd < 0) {
73        // No protocol, assume http
74        uri = "http://" + uri;
75      } else if (isColonFollowedByPortNumber(uri, protocolEnd)) {
76        // Found a colon, but it looks like it is after the host, so the protocol is still missing
77        uri = "http://" + uri;
78      }
79      return uri;
80    }
81  
82    private static boolean isColonFollowedByPortNumber(String uri, int protocolEnd) {
83      int nextSlash = uri.indexOf('/', protocolEnd + 1);
84      if (nextSlash < 0) {
85        nextSlash = uri.length();
86      }
87      if (nextSlash <= protocolEnd + 1) {
88        return false;
89      }
90      for (int x = protocolEnd + 1; x < nextSlash; x++) {
91        if (uri.charAt(x) < '0' || uri.charAt(x) > '9') {
92          return false;
93        }
94      }
95      return true;
96    }
97  
98  
99  }
