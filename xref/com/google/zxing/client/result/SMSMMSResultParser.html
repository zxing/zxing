17  package com.google.zxing.client.result;
18  
19  import com.google.zxing.Result;
20  
21  import java.util.ArrayList;
22  import java.util.Collection;
23  import java.util.List;
24  import java.util.Map;
25  
26  /**
27   * <p>Parses an "sms:" URI result, which specifies a number to SMS.
28   * See <a href="http://tools.ietf.org/html/rfc5724"> RFC 5724</a> on this.</p>
29   *
30   * <p>This class supports "via" syntax for numbers, which is not part of the spec.
31   * For example "+12125551212;via=+12124440101" may appear as a number.
32   * It also supports a "subject" query parameter, which is not mentioned in the spec.
33   * These are included since they were mentioned in earlier IETF drafts and might be
34   * used.</p>
35   *
36   * <p>This actually also parses URIs starting with "mms:" and treats them all the same way,
37   * and effectively converts them to an "sms:" URI for purposes of forwarding to the platform.</p>
38   *
39   * @author Sean Owen
40   */
41  public final class SMSMMSResultParser extends ResultParser {
42  
43    @Override
44    public SMSParsedResult parse(Result result) {
45      String rawText = getMassagedText(result);
46      if (!(rawText.startsWith("sms:") || rawText.startsWith("SMS:") ||
47            rawText.startsWith("mms:") || rawText.startsWith("MMS:"))) {
48        return null;
49      }
50  
51      // Check up front if this is a URI syntax string with query arguments
52      Map<String,String> nameValuePairs = parseNameValuePairs(rawText);
53      String subject = null;
54      String body = null;
55      boolean querySyntax = false;
56      if (nameValuePairs != null && !nameValuePairs.isEmpty()) {
57        subject = nameValuePairs.get("subject");
58        body = nameValuePairs.get("body");
59        querySyntax = true;
60      }
61  
62      // Drop sms, query portion
63      int queryStart = rawText.indexOf('?', 4);
64      String smsURIWithoutQuery;
65      // If it's not query syntax, the question mark is part of the subject or message
66      if (queryStart < 0 || !querySyntax) {
67        smsURIWithoutQuery = rawText.substring(4);
68      } else {
69        smsURIWithoutQuery = rawText.substring(4, queryStart);
70      }
71  
72      int lastComma = -1;
73      int comma;
74      List<String> numbers = new ArrayList<>(1);
75      List<String> vias = new ArrayList<>(1);
76      while ((comma = smsURIWithoutQuery.indexOf(',', lastComma + 1)) > lastComma) {
77        String numberPart = smsURIWithoutQuery.substring(lastComma + 1, comma);
78        addNumberVia(numbers, vias, numberPart);
79        lastComma = comma;
80      }
81      addNumberVia(numbers, vias, smsURIWithoutQuery.substring(lastComma + 1));    
82  
83      return new SMSParsedResult(numbers.toArray(new String[numbers.size()]),
84                                 vias.toArray(new String[vias.size()]),
85                                 subject,
86                                 body);
87    }
88  
89    private static void addNumberVia(Collection<String> numbers,
90                                     Collection<String> vias,
91                                     String numberPart) {
92      int numberEnd = numberPart.indexOf(';');
93      if (numberEnd < 0) {
94        numbers.add(numberPart);
95        vias.add(null);
96      } else {
97        numbers.add(numberPart.substring(0, numberEnd));
98        String maybeVia = numberPart.substring(numberEnd + 1);
99        String via;
100       if (maybeVia.startsWith("via=")) {
101         via = maybeVia.substring(4);
102       } else {
103         via = null;
104       }
105       vias.add(via);
106     }
107   }
