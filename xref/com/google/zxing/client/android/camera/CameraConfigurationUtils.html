56    public static void setFocus(Camera.Parameters parameters,
57                                boolean autoFocus,
58                                boolean disableContinuous,
59                                boolean safeMode) {
60      List<String> supportedFocusModes = parameters.getSupportedFocusModes();
61      String focusMode = null;
62      if (autoFocus) {
63        if (safeMode || disableContinuous) {
64          focusMode = CameraConfigurationUtils.findSettableValue("focus mode",
65                                                                 supportedFocusModes,
66                                                                 Camera.Parameters.FOCUS_MODE_AUTO);
67        } else {
68          focusMode = CameraConfigurationUtils.findSettableValue("focus mode",
69                                                                 supportedFocusModes,
70                                                                 Camera.Parameters.FOCUS_MODE_CONTINUOUS_PICTURE,
71                                                                 Camera.Parameters.FOCUS_MODE_CONTINUOUS_VIDEO,
72                                                                 Camera.Parameters.FOCUS_MODE_AUTO);
73        }
74      }
75      // Maybe selected auto-focus but not available, so fall through here:
76      if (!safeMode && focusMode == null) {
77        focusMode = CameraConfigurationUtils.findSettableValue("focus mode",
78                                                               supportedFocusModes,
79                                                               Camera.Parameters.FOCUS_MODE_MACRO,
80                                                               Camera.Parameters.FOCUS_MODE_EDOF);
81      }
82      if (focusMode != null) {
83        if (focusMode.equals(parameters.getFocusMode())) {
84          Log.i(TAG, "Focus mode already set to " + focusMode);
85        } else {
86          parameters.setFocusMode(focusMode);
87        }
88      }
89    }

91    public static void setTorch(Camera.Parameters parameters, boolean on) {
92      List<String> supportedFlashModes = parameters.getSupportedFlashModes();
93      String flashMode;
94      if (on) {
95        flashMode = CameraConfigurationUtils.findSettableValue("flash mode",
96                                                               supportedFlashModes,
97                                                               Camera.Parameters.FLASH_MODE_TORCH,
98                                                               Camera.Parameters.FLASH_MODE_ON);
99      } else {
100       flashMode = CameraConfigurationUtils.findSettableValue("flash mode",
101                                                              supportedFlashModes,
102                                                              Camera.Parameters.FLASH_MODE_OFF);
103     }
104     if (flashMode != null) {
105       if (flashMode.equals(parameters.getFlashMode())) {
106         Log.i(TAG, "Flash mode already set to " + flashMode);
107       } else {
108         Log.i(TAG, "Setting flash mode to " + flashMode);
109         parameters.setFlashMode(flashMode);
110       }
111     }

114   public static void setBestExposure(Camera.Parameters parameters, boolean lightOn) {
115     int minExposure = parameters.getMinExposureCompensation();
116     int maxExposure = parameters.getMaxExposureCompensation();
117     float step = parameters.getExposureCompensationStep();
118     if ((minExposure != 0 || maxExposure != 0) && step > 0.0f) {
119       // Set low when light is on
120       float targetCompensation = lightOn ? MIN_EXPOSURE_COMPENSATION : MAX_EXPOSURE_COMPENSATION;
121       int compensationSteps = Math.round(targetCompensation / step);
122       float actualCompensation = step * compensationSteps;
123       // Clamp value:
124       compensationSteps = Math.max(Math.min(compensationSteps, maxExposure), minExposure);
125       if (parameters.getExposureCompensation() == compensationSteps) {
126         Log.i(TAG, "Exposure compensation already set to " + compensationSteps + " / " + actualCompensation);
127       } else {
128         Log.i(TAG, "Setting exposure compensation to " + compensationSteps + " / " + actualCompensation);
129         parameters.setExposureCompensation(compensationSteps);
130       }
131     } else {
132       Log.i(TAG, "Camera does not support exposure compensation");
133     }
134   }

140   public static void setBestPreviewFPS(Camera.Parameters parameters, int minFPS, int maxFPS) {
141     List<int[]> supportedPreviewFpsRanges = parameters.getSupportedPreviewFpsRange();
142     Log.i(TAG, "Supported FPS ranges: " + toString(supportedPreviewFpsRanges));
143     if (supportedPreviewFpsRanges != null && !supportedPreviewFpsRanges.isEmpty()) {
144       int[] suitableFPSRange = null;
145       for (int[] fpsRange : supportedPreviewFpsRanges) {
146         int thisMin = fpsRange[Camera.Parameters.PREVIEW_FPS_MIN_INDEX];
147         int thisMax = fpsRange[Camera.Parameters.PREVIEW_FPS_MAX_INDEX];
148         if (thisMin >= minFPS * 1000 && thisMax <= maxFPS * 1000) {
149           suitableFPSRange = fpsRange;
150           break;
151         }
152       }
153       if (suitableFPSRange == null) {
154         Log.i(TAG, "No suitable FPS range?");
155       } else {
156         int[] currentFpsRange = new int[2];
157         parameters.getPreviewFpsRange(currentFpsRange);
158         if (Arrays.equals(currentFpsRange, suitableFPSRange)) {
159           Log.i(TAG, "FPS range already set to " + Arrays.toString(suitableFPSRange));
160         } else {
161           Log.i(TAG, "Setting FPS range to " + Arrays.toString(suitableFPSRange));
162           parameters.setPreviewFpsRange(suitableFPSRange[Camera.Parameters.PREVIEW_FPS_MIN_INDEX],
163                                         suitableFPSRange[Camera.Parameters.PREVIEW_FPS_MAX_INDEX]);
164         }
165       }
166     }
167   }

170     if (parameters.getMaxNumFocusAreas() > 0) {
171       Log.i(TAG, "Old focus areas: " + toString(parameters.getFocusAreas()));

174       parameters.setFocusAreas(middleArea);
175     } else {
176       Log.i(TAG, "Device does not support focus areas");
177     }

222   public static void setZoom(Camera.Parameters parameters, double targetZoomRatio) {
223     if (parameters.isZoomSupported()) {
224       Integer zoom = indexOfClosestZoom(parameters, targetZoomRatio);
225       if (zoom == null) {
226         return;
227       }
228       if (parameters.getZoom() == zoom) {
229         Log.i(TAG, "Zoom is already set to " + zoom);
230       } else {
231         Log.i(TAG, "Setting zoom to " + zoom);
232         parameters.setZoom(zoom);
233       }
234     } else {
235       Log.i(TAG, "Zoom is not supported");
236     }
237   }

239   private static Integer indexOfClosestZoom(Camera.Parameters parameters, double targetZoomRatio) {
240     List<Integer> ratios = parameters.getZoomRatios();
241     Log.i(TAG, "Zoom ratios: " + ratios);
242     int maxZoom = parameters.getMaxZoom();
243     if (ratios == null || ratios.isEmpty() || ratios.size() != maxZoom + 1) {
244       Log.w(TAG, "Invalid zoom ratios!");
245       return null;
246     }
247     double target100 = 100.0 * targetZoomRatio;
248     double smallestDiff = Double.POSITIVE_INFINITY;
249     int closestIndex = 0;
250     for (int i = 0; i < ratios.size(); i++) {
251       double diff = Math.abs(ratios.get(i) - target100);
252       if (diff < smallestDiff) {
253         smallestDiff = diff;
254         closestIndex = i;
255       }
256     }
257     Log.i(TAG, "Chose zoom ratio of " + (ratios.get(closestIndex) / 100.0));
258     return closestIndex;
259   }

275   public static Point findBestPreviewSizeValue(Camera.Parameters parameters, Point screenResolution) {
276 
277     List<Camera.Size> rawSupportedSizes = parameters.getSupportedPreviewSizes();
278     if (rawSupportedSizes == null) {
279       Log.w(TAG, "Device returned no supported preview sizes; using default");
280       Camera.Size defaultSize = parameters.getPreviewSize();
281       if (defaultSize == null) {
282         throw new IllegalStateException("Parameters contained no preview size!");
283       }
284       return new Point(defaultSize.width, defaultSize.height);
285     }
286 
287     // Sort by size, descending
288     List<Camera.Size> supportedPreviewSizes = new ArrayList<>(rawSupportedSizes);
289     Collections.sort(supportedPreviewSizes, new Comparator<Camera.Size>() {
290       @Override
291       public int compare(Camera.Size a, Camera.Size b) {
292         int aPixels = a.height * a.width;
293         int bPixels = b.height * b.width;
294         if (bPixels < aPixels) {
295           return -1;
296         }
297         if (bPixels > aPixels) {
298           return 1;
299         }
300         return 0;
301       }
302     });
303 
304     if (Log.isLoggable(TAG, Log.INFO)) {
305       StringBuilder previewSizesString = new StringBuilder();
306       for (Camera.Size supportedPreviewSize : supportedPreviewSizes) {
307         previewSizesString.append(supportedPreviewSize.width).append('x')
308             .append(supportedPreviewSize.height).append(' ');
309       }
310       Log.i(TAG, "Supported preview sizes: " + previewSizesString);
311     }
312 
313     double screenAspectRatio = (double) screenResolution.x / (double) screenResolution.y;
314 
315     // Remove sizes that are unsuitable
316     Iterator<Camera.Size> it = supportedPreviewSizes.iterator();
317     while (it.hasNext()) {
318       Camera.Size supportedPreviewSize = it.next();
319       int realWidth = supportedPreviewSize.width;
320       int realHeight = supportedPreviewSize.height;
321       if (realWidth * realHeight < MIN_PREVIEW_PIXELS) {
322         it.remove();
323         continue;
324       }
325 
326       boolean isCandidatePortrait = realWidth < realHeight;
327       int maybeFlippedWidth = isCandidatePortrait ? realHeight : realWidth;
328       int maybeFlippedHeight = isCandidatePortrait ? realWidth : realHeight;
329       double aspectRatio = (double) maybeFlippedWidth / (double) maybeFlippedHeight;
330       double distortion = Math.abs(aspectRatio - screenAspectRatio);
331       if (distortion > MAX_ASPECT_DISTORTION) {
332         it.remove();
333         continue;
334       }
335 
336       if (maybeFlippedWidth == screenResolution.x && maybeFlippedHeight == screenResolution.y) {
337         Point exactPoint = new Point(realWidth, realHeight);
338         Log.i(TAG, "Found preview size exactly matching screen size: " + exactPoint);
339         return exactPoint;
340       }
341     }
342 
343     // If no exact match, use largest preview size. This was not a great idea on older devices because
344     // of the additional computation needed. We're likely to get here on newer Android 4+ devices, where
345     // the CPU is much more powerful.
346     if (!supportedPreviewSizes.isEmpty()) {
347       Camera.Size largestPreview = supportedPreviewSizes.get(0);
348       Point largestSize = new Point(largestPreview.width, largestPreview.height);
349       Log.i(TAG, "Using largest suitable preview size: " + largestSize);
350       return largestSize;
351     }
352 
353     // If there is nothing at all suitable, return current preview size
354     Camera.Size defaultPreview = parameters.getPreviewSize();
355     if (defaultPreview == null) {
356       throw new IllegalStateException("Parameters contained no preview size!");
357     }
358     Point defaultSize = new Point(defaultPreview.width, defaultPreview.height);
359     Log.i(TAG, "No suitable preview sizes, using default: " + defaultSize);
360     return defaultSize;
361   }

363   private static String findSettableValue(String name,
364                                           Collection<String> supportedValues,
365                                           String... desiredValues) {
366     Log.i(TAG, "Requesting " + name + " value from among: " + Arrays.toString(desiredValues));
367     Log.i(TAG, "Supported " + name + " values: " + supportedValues);
368     if (supportedValues != null) {
369       for (String desiredValue : desiredValues) {
370         if (supportedValues.contains(desiredValue)) {
371           Log.i(TAG, "Can set " + name + " to: " + desiredValue);
372           return desiredValue;
373         }
374       }
375     }
376     Log.i(TAG, "No supported values match");
377     return null;
378   }

380   private static String toString(Collection<int[]> arrays) {
381     if (arrays == null || arrays.isEmpty()) {
382       return "[]";
383     }
384     StringBuilder buffer = new StringBuilder();
385     buffer.append('[');
386     Iterator<int[]> it = arrays.iterator();
387     while (it.hasNext()) {
388       buffer.append(Arrays.toString(it.next()));
389       if (it.hasNext()) {
390         buffer.append(", ");
391       }
392     }
393     buffer.append(']');
394     return buffer.toString();
395   }

397   private static String toString(Iterable<Camera.Area> areas) {
398     if (areas == null) {
399       return null;
400     }
401     StringBuilder result = new StringBuilder();
402     for (Camera.Area area : areas) {
403       result.append(area.rect).append(':').append(area.weight).append(' ');
404     }
405     return result.toString();
406   }

412   public static String collectStats(CharSequence flattenedParams) {
413     StringBuilder result = new StringBuilder(1000);
414 
415     result.append("BOARD=").append(Build.BOARD).append('\n');
416     result.append("BRAND=").append(Build.BRAND).append('\n');
417     result.append("CPU_ABI=").append(Build.CPU_ABI).append('\n');
418     result.append("DEVICE=").append(Build.DEVICE).append('\n');
419     result.append("DISPLAY=").append(Build.DISPLAY).append('\n');
420     result.append("FINGERPRINT=").append(Build.FINGERPRINT).append('\n');
421     result.append("HOST=").append(Build.HOST).append('\n');
422     result.append("ID=").append(Build.ID).append('\n');
423     result.append("MANUFACTURER=").append(Build.MANUFACTURER).append('\n');
424     result.append("MODEL=").append(Build.MODEL).append('\n');
425     result.append("PRODUCT=").append(Build.PRODUCT).append('\n');
426     result.append("TAGS=").append(Build.TAGS).append('\n');
427     result.append("TIME=").append(Build.TIME).append('\n');
428     result.append("TYPE=").append(Build.TYPE).append('\n');
429     result.append("USER=").append(Build.USER).append('\n');
430     result.append("VERSION.CODENAME=").append(Build.VERSION.CODENAME).append('\n');
431     result.append("VERSION.INCREMENTAL=").append(Build.VERSION.INCREMENTAL).append('\n');
432     result.append("VERSION.RELEASE=").append(Build.VERSION.RELEASE).append('\n');
433     result.append("VERSION.SDK_INT=").append(Build.VERSION.SDK_INT).append('\n');
434 
435     if (flattenedParams != null) {
436       String[] params = SEMICOLON.split(flattenedParams);
437       Arrays.sort(params);
438       for (String param : params) {
439         result.append(param).append('\n');
440       }
441     }
442 
443     return result.toString();
444   }
