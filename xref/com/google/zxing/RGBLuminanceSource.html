17  package com.google.zxing;
18  
19  /**
20   * This class is used to help decode images from files which arrive as RGB data from
21   * an ARGB pixel array. It does not support rotation.
22   *
23   * @author dswitkin@google.com (Daniel Switkin)
24   * @author Betaminos
25   */
26  public final class RGBLuminanceSource extends LuminanceSource {
27  
28    private final byte[] luminances;
29    private final int dataWidth;
30    private final int dataHeight;
31    private final int left;
32    private final int top;
33  
34    public RGBLuminanceSource(int width, int height, int[] pixels) {
35      super(width, height);
36  
37      dataWidth = width;
38      dataHeight = height;
39      left = 0;
40      top = 0;
41  
42      // In order to measure pure decoding speed, we convert the entire image to a greyscale array
43      // up front, which is the same as the Y channel of the YUVLuminanceSource in the real app.
44      luminances = new byte[width * height];
45      for (int y = 0; y < height; y++) {
46        int offset = y * width;
47        for (int x = 0; x < width; x++) {
48          int pixel = pixels[offset + x];
49          int r = (pixel >> 16) & 0xff;
50          int g = (pixel >> 8) & 0xff;
51          int b = pixel & 0xff;
52          if (r == g && g == b) {
53            // Image is already greyscale, so pick any channel.
54            luminances[offset + x] = (byte) r;
55          } else {
56            // Calculate luminance cheaply, favoring green.
57            luminances[offset + x] = (byte) ((r + g + g + b) >> 2);
58          }
59        }
60      }
61    }
62    
63    private RGBLuminanceSource(byte[] pixels,
64                               int dataWidth,
65                               int dataHeight,
66                               int left,
67                               int top,
68                               int width,
69                               int height) {
70      super(width, height);
71      if (left + width > dataWidth || top + height > dataHeight) {
72        throw new IllegalArgumentException("Crop rectangle does not fit within image data.");
73      }
74      this.luminances = pixels;
75      this.dataWidth = dataWidth;
76      this.dataHeight = dataHeight;
77      this.left = left;
78      this.top = top;
79    }
80  
81    @Override
82    public byte[] getRow(int y, byte[] row) {
83      if (y < 0 || y >= getHeight()) {
84        throw new IllegalArgumentException("Requested row is outside the image: " + y);
85      }
86      int width = getWidth();
87      if (row == null || row.length < width) {
88        row = new byte[width];
89      }
90      int offset = (y + top) * dataWidth + left;
91      System.arraycopy(luminances, offset, row, 0, width);
92      return row;
93    }
94  
95    @Override
96    public byte[] getMatrix() {
97      int width = getWidth();
98      int height = getHeight();
99  
100     // If the caller asks for the entire underlying image, save the copy and give them the
101     // original data. The docs specifically warn that result.length must be ignored.
102     if (width == dataWidth && height == dataHeight) {
103       return luminances;
104     }
105 
106     int area = width * height;
107     byte[] matrix = new byte[area];
108     int inputOffset = top * dataWidth + left;
109 
110     // If the width matches the full width of the underlying data, perform a single copy.
111     if (width == dataWidth) {
112       System.arraycopy(luminances, inputOffset, matrix, 0, area);
113       return matrix;
114     }
115 
116     // Otherwise copy one cropped row at a time.
117     byte[] rgb = luminances;
118     for (int y = 0; y < height; y++) {
119       int outputOffset = y * width;
120       System.arraycopy(rgb, inputOffset, matrix, outputOffset, width);
121       inputOffset += dataWidth;
122     }
123     return matrix;
124   }
125   
126   @Override
127   public boolean isCropSupported() {
128     return true;
129   }
130 
131   @Override
132   public LuminanceSource crop(int left, int top, int width, int height) {
133     return new RGBLuminanceSource(luminances,
134                                   dataWidth,
135                                   dataHeight,
136                                   this.left + left,
137                                   this.top + top,
138                                   width,
139                                   height);
140   }

142 }
