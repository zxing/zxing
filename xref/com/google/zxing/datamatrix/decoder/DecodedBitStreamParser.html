29   * <p>Data Matrix Codes can encode text as bits in one of several modes, and can use multiple modes
30   * in one Data Matrix Code. This class decodes the bits back into text.</p>

41      ASCII_ENCODE,
42      C40_ENCODE,
43      TEXT_ENCODE,
44      ANSIX12_ENCODE,
45      EDIFACT_ENCODE,
46      BASE256_ENCODE
47    }

82    static DecoderResult decode(byte[] bytes) throws FormatException {
83      BitSource bits = new BitSource(bytes);
84      StringBuilder result = new StringBuilder(100);
85      StringBuilder resultTrailer = new StringBuilder(0);
86      List<byte[]> byteSegments = new ArrayList<>(1);
87      Mode mode = Mode.ASCII_ENCODE;
88      do {
89        if (mode == Mode.ASCII_ENCODE) {
90          mode = decodeAsciiSegment(bits, result, resultTrailer);
91        } else {
92          switch (mode) {
93            case C40_ENCODE:
94              decodeC40Segment(bits, result);
95              break;
96            case TEXT_ENCODE:
97              decodeTextSegment(bits, result);
98              break;
99            case ANSIX12_ENCODE:
100             decodeAnsiX12Segment(bits, result);
101             break;
102           case EDIFACT_ENCODE:
103             decodeEdifactSegment(bits, result);
104             break;
105           case BASE256_ENCODE:
106             decodeBase256Segment(bits, result, byteSegments);
107             break;
108           default:
109             throw FormatException.getFormatInstance();
110         }
111         mode = Mode.ASCII_ENCODE;
112       }
113     } while (mode != Mode.PAD_ENCODE && bits.available() > 0);
114     if (resultTrailer.length() > 0) {
115       result.append(resultTrailer);
116     }
117     return new DecoderResult(bytes, result.toString(), byteSegments.isEmpty() ? null : byteSegments, null);
118   }

120   /**
121    * See ISO 16022:2006, 5.2.3 and Annex C, Table C.2
122    */
123   private static Mode decodeAsciiSegment(BitSource bits,
124                                          StringBuilder result,
125                                          StringBuilder resultTrailer) throws FormatException {
126     boolean upperShift = false;
127     do {
128       int oneByte = bits.readBits(8);
129       if (oneByte == 0) {
130         throw FormatException.getFormatInstance();
131       } else if (oneByte <= 128) {  // ASCII data (ASCII value + 1)
132         if (upperShift) {
133           oneByte += 128;
134           //upperShift = false;
135         }
136         result.append((char) (oneByte - 1));
137         return Mode.ASCII_ENCODE;
138       } else if (oneByte == 129) {  // Pad
139         return Mode.PAD_ENCODE;
140       } else if (oneByte <= 229) {  // 2-digit data 00-99 (Numeric Value + 130)
141         int value = oneByte - 130;
142         if (value < 10) { // padd with '0' for single digit values
143           result.append('0');
144         }
145         result.append(value);
146       } else if (oneByte == 230) {  // Latch to C40 encodation
147         return Mode.C40_ENCODE;
148       } else if (oneByte == 231) {  // Latch to Base 256 encodation
149         return Mode.BASE256_ENCODE;
150       } else if (oneByte == 232) {
151         // FNC1
152         result.append((char) 29); // translate as ASCII 29
153       } else if (oneByte == 233 || oneByte == 234) {
154         // Structured Append, Reader Programming
155         // Ignore these symbols for now
156         //throw ReaderException.getInstance();
157       } else if (oneByte == 235) {  // Upper Shift (shift to Extended ASCII)
158         upperShift = true;
159       } else if (oneByte == 236) {  // 05 Macro
160         result.append("[)>\u001E05\u001D");
161         resultTrailer.insert(0, "\u001E\u0004");
162       } else if (oneByte == 237) {  // 06 Macro
163         result.append("[)>\u001E06\u001D");
164         resultTrailer.insert(0, "\u001E\u0004");
165       } else if (oneByte == 238) {  // Latch to ANSI X12 encodation
166         return Mode.ANSIX12_ENCODE;
167       } else if (oneByte == 239) {  // Latch to Text encodation
168         return Mode.TEXT_ENCODE;
169       } else if (oneByte == 240) {  // Latch to EDIFACT encodation
170         return Mode.EDIFACT_ENCODE;
171       } else if (oneByte == 241) {  // ECI Character
172         // TODO(bbrown): I think we need to support ECI
173         //throw ReaderException.getInstance();
174         // Ignore this symbol for now
175       } else if (oneByte >= 242) {  // Not to be used in ASCII encodation
176         // ... but work around encoders that end with 254, latch back to ASCII
177         if (oneByte != 254 || bits.available() != 0) {
178           throw FormatException.getFormatInstance();
179         }
180       }
181     } while (bits.available() > 0);
182     return Mode.ASCII_ENCODE;
183   }

185   /**
186    * See ISO 16022:2006, 5.2.5 and Annex C, Table C.1
187    */
188   private static void decodeC40Segment(BitSource bits, StringBuilder result) throws FormatException {
189     // Three C40 values are encoded in a 16-bit value as
190     // (1600 * C1) + (40 * C2) + C3 + 1
191     // TODO(bbrown): The Upper Shift with C40 doesn't work in the 4 value scenario all the time
192     boolean upperShift = false;
193 
194     int[] cValues = new int[3];
195     int shift = 0;
196 
197     do {
198       // If there is only one byte left then it will be encoded as ASCII
199       if (bits.available() == 8) {
200         return;
201       }
202       int firstByte = bits.readBits(8);
203       if (firstByte == 254) {  // Unlatch codeword
204         return;
205       }
206 
207       parseTwoBytes(firstByte, bits.readBits(8), cValues);
208 
209       for (int i = 0; i < 3; i++) {
210         int cValue = cValues[i];
211         switch (shift) {
212           case 0:
213             if (cValue < 3) {
214               shift = cValue + 1;
215             } else if (cValue < C40_BASIC_SET_CHARS.length) {
216               char c40char = C40_BASIC_SET_CHARS[cValue];
217               if (upperShift) {
218                 result.append((char) (c40char + 128));
219                 upperShift = false;
220               } else {
221                 result.append(c40char);
222               }
223             } else {
224               throw FormatException.getFormatInstance();
225             }
226             break;
227           case 1:
228             if (upperShift) {
229               result.append((char) (cValue + 128));
230               upperShift = false;
231             } else {
232               result.append((char) cValue);
233             }
234             shift = 0;
235             break;
236           case 2:
237             if (cValue < C40_SHIFT2_SET_CHARS.length) {
238               char c40char = C40_SHIFT2_SET_CHARS[cValue];
239               if (upperShift) {
240                 result.append((char) (c40char + 128));
241                 upperShift = false;
242               } else {
243                 result.append(c40char);
244               }
245             } else if (cValue == 27) {  // FNC1
246               result.append((char) 29); // translate as ASCII 29
247             } else if (cValue == 30) {  // Upper Shift
248               upperShift = true;
249             } else {
250               throw FormatException.getFormatInstance();
251             }
252             shift = 0;
253             break;
254           case 3:
255             if (upperShift) {
256               result.append((char) (cValue + 224));
257               upperShift = false;
258             } else {
259               result.append((char) (cValue + 96));
260             }
261             shift = 0;
262             break;
263           default:
264             throw FormatException.getFormatInstance();
265         }
266       }
267     } while (bits.available() > 0);
268   }

274     // Three Text values are encoded in a 16-bit value as
275     // (1600 * C1) + (40 * C2) + C3 + 1
276     // TODO(bbrown): The Upper Shift with Text doesn't work in the 4 value scenario all the time
277     boolean upperShift = false;
278 
279     int[] cValues = new int[3];
280     int shift = 0;
281     do {
282       // If there is only one byte left then it will be encoded as ASCII
283       if (bits.available() == 8) {
284         return;
285       }

287       if (firstByte == 254) {  // Unlatch codeword
288         return;
289       }
290 
291       parseTwoBytes(firstByte, bits.readBits(8), cValues);
292 
293       for (int i = 0; i < 3; i++) {
294         int cValue = cValues[i];
295         switch (shift) {
296           case 0:
297             if (cValue < 3) {
298               shift = cValue + 1;
299             } else if (cValue < TEXT_BASIC_SET_CHARS.length) {
300               char textChar = TEXT_BASIC_SET_CHARS[cValue];
301               if (upperShift) {
302                 result.append((char) (textChar + 128));
303                 upperShift = false;
304               } else {
305                 result.append(textChar);
306               }
307             } else {
308               throw FormatException.getFormatInstance();
309             }
310             break;
311           case 1:
312             if (upperShift) {
313               result.append((char) (cValue + 128));
314               upperShift = false;
315             } else {
316               result.append((char) cValue);
317             }
318             shift = 0;
319             break;
320           case 2:
321             // Shift 2 for Text is the same encoding as C40
322             if (cValue < C40_SHIFT2_SET_CHARS.length) {
323               char c40char = C40_SHIFT2_SET_CHARS[cValue];
324               if (upperShift) {
325                 result.append((char) (c40char + 128));

333               upperShift = true;
334             } else {
335               throw FormatException.getFormatInstance();
336             }
337             shift = 0;
338             break;
339           case 3:
340             if (cValue < TEXT_SHIFT3_SET_CHARS.length) {

343                 result.append((char) (textChar + 128));

350               throw FormatException.getFormatInstance();
351             }
352             break;
353           default:
354             throw FormatException.getFormatInstance();
355         }
356       }
357     } while (bits.available() > 0);
358   }

360   /**
361    * See ISO 16022:2006, 5.2.7
362    */
363   private static void decodeAnsiX12Segment(BitSource bits,
364                                            StringBuilder result) throws FormatException {
365     // Three ANSI X12 values are encoded in a 16-bit value as
366     // (1600 * C1) + (40 * C2) + C3 + 1
367 
368     int[] cValues = new int[3];
369     do {
370       // If there is only one byte left then it will be encoded as ASCII
371       if (bits.available() == 8) {
372         return;
373       }
374       int firstByte = bits.readBits(8);
375       if (firstByte == 254) {  // Unlatch codeword
376         return;
377       }
378 
379       parseTwoBytes(firstByte, bits.readBits(8), cValues);
380 
381       for (int i = 0; i < 3; i++) {
382         int cValue = cValues[i];
383         if (cValue == 0) {  // X12 segment terminator <CR>
384           result.append('\r');
385         } else if (cValue == 1) {  // X12 segment separator *
386           result.append('*');
387         } else if (cValue == 2) {  // X12 sub-element separator >
388           result.append('>');
389         } else if (cValue == 3) {  // space
390           result.append(' ');
391         } else if (cValue < 14) {  // 0 - 9
392           result.append((char) (cValue + 44));
393         } else if (cValue < 40) {  // A - Z
394           result.append((char) (cValue + 51));
395         } else {
396           throw FormatException.getFormatInstance();
397         }
398       }
399     } while (bits.available() > 0);
400   }

402   private static void parseTwoBytes(int firstByte, int secondByte, int[] result) {
403     int fullBitValue = (firstByte << 8) + secondByte - 1;
404     int temp = fullBitValue / 1600;
405     result[0] = temp;
406     fullBitValue -= temp * 1600;
407     temp = fullBitValue / 40;
408     result[1] = temp;
409     result[2] = fullBitValue - temp * 40;
410   }

412   /**
413    * See ISO 16022:2006, 5.2.8 and Annex C Table C.3
414    */
415   private static void decodeEdifactSegment(BitSource bits, StringBuilder result) {
416     do {
417       // If there is only two or less bytes left then it will be encoded as ASCII
418       if (bits.available() <= 16) {
419         return;
420       }
421 
422       for (int i = 0; i < 4; i++) {
423         int edifactValue = bits.readBits(6);
424 
425         // Check for the unlatch character
426         if (edifactValue == 0x1F) {  // 011111
427           // Read rest of byte, which should be 0, and stop
428           int bitsLeft = 8 - bits.getBitOffset();
429           if (bitsLeft != 8) {
430             bits.readBits(bitsLeft);
431           }
432           return;
433         }
434 
435         if ((edifactValue & 0x20) == 0) {  // no 1 in the leading (6th) bit
436           edifactValue |= 0x40;  // Add a leading 01 to the 6 bit binary value
437         }
438         result.append((char) edifactValue);
439       }
440     } while (bits.available() > 0);
441   }

443   /**
444    * See ISO 16022:2006, 5.2.9 and Annex B, B.2
445    */
446   private static void decodeBase256Segment(BitSource bits,
447                                            StringBuilder result,
448                                            Collection<byte[]> byteSegments)
449       throws FormatException {
450     // Figure out how long the Base 256 Segment is.
451     int codewordPosition = 1 + bits.getByteOffset(); // position is 1-indexed
452     int d1 = unrandomize255State(bits.readBits(8), codewordPosition++);
453     int count;
454     if (d1 == 0) {  // Read the remainder of the symbol
455       count = bits.available() / 8;
456     } else if (d1 < 250) {
457       count = d1;
458     } else {
459       count = 250 * (d1 - 249) + unrandomize255State(bits.readBits(8), codewordPosition++);
460     }
461 
462     // We're seeing NegativeArraySizeException errors from users.
463     if (count < 0) {
464       throw FormatException.getFormatInstance();
465     }
466 
467     byte[] bytes = new byte[count];
468     for (int i = 0; i < count; i++) {
469       // Have seen this particular error in the wild, such as at
470       // http://www.bcgen.com/demo/IDAutomationStreamingDataMatrix.aspx?MODE=3&D=Fred&PFMT=3&PT=F&X=0.3&O=0&LM=0.2
471       if (bits.available() < 8) {
472         throw FormatException.getFormatInstance();
473       }
474       bytes[i] = (byte) unrandomize255State(bits.readBits(8), codewordPosition++);
475     }
476     byteSegments.add(bytes);
477     try {
478       result.append(new String(bytes, "ISO8859_1"));
479     } catch (UnsupportedEncodingException uee) {
480       throw new IllegalStateException("Platform does not support required encoding: " + uee);
481     }
482   }

484   /**
485    * See ISO 16022:2006, Annex B, B.2
486    */
487   private static int unrandomize255State(int randomizedBase256Codeword,
488                                           int base256CodewordPosition) {
489     int pseudoRandomNumber = ((149 * base256CodewordPosition) % 255) + 1;
490     int tempVariable = randomizedBase256Codeword - pseudoRandomNumber;
491     return tempVariable >= 0 ? tempVariable : tempVariable + 256;
492   }
