128   private static char randomize253State(char ch, int codewordPosition) {
129     int pseudoRandom = ((149 * codewordPosition) % 253) + 1;
130     int tempVariable = ch + pseudoRandom;
131     return tempVariable <= 254 ? (char) tempVariable : (char) (tempVariable - 254);
132   }

138    * @param msg the message

145   /**
146    * Performs message encoding of a DataMatrix message using the algorithm described in annex P
147    * of ISO/IEC 16022:2000(E).
148    *
149    * @param msg     the message
150    * @param shape   requested shape. May be {@code SymbolShapeHint.FORCE_NONE},
151    *                {@code SymbolShapeHint.FORCE_SQUARE} or {@code SymbolShapeHint.FORCE_RECTANGLE}.
152    * @param minSize the minimum symbol size constraint or null for no constraint
153    * @param maxSize the maximum symbol size constraint or null for no constraint
154    * @return the encoded message (the char values range from 0 to 255)
155    */
156   public static String encodeHighLevel(String msg,
157                                        SymbolShapeHint shape, 
158                                        Dimension minSize, 
159                                        Dimension maxSize) {
160     //the codewords 0..255 are encoded as Unicode characters
161     Encoder[] encoders = {
162         new ASCIIEncoder(), new C40Encoder(), new TextEncoder(), 
163         new X12Encoder(), new EdifactEncoder(),  new Base256Encoder()
164     };
165 
166     EncoderContext context = new EncoderContext(msg);
167     context.setSymbolShape(shape);
168     context.setSizeConstraints(minSize, maxSize);
169 
170     if (msg.startsWith(MACRO_05_HEADER) && msg.endsWith(MACRO_TRAILER)) {
171       context.writeCodeword(MACRO_05);
172       context.setSkipAtEnd(2);
173       context.pos += MACRO_05_HEADER.length();
174     } else if (msg.startsWith(MACRO_06_HEADER) && msg.endsWith(MACRO_TRAILER)) {
175       context.writeCodeword(MACRO_06);
176       context.setSkipAtEnd(2);
177       context.pos += MACRO_06_HEADER.length();
178     }
179 
180     int encodingMode = ASCII_ENCODATION; //Default mode
181     while (context.hasMoreCharacters()) {
182       encoders[encodingMode].encode(context);
183       if (context.getNewEncoding() >= 0) {
184         encodingMode = context.getNewEncoding();
185         context.resetEncoderSignal();
186       }
187     }
188     int len = context.getCodewordCount();
189     context.updateSymbolInfo();
190     int capacity = context.getSymbolInfo().getDataCapacity();
191     if (len < capacity) {
192       if (encodingMode != ASCII_ENCODATION && encodingMode != BASE256_ENCODATION) {
193         context.writeCodeword('\u00fe'); //Unlatch (254)
194       }
195     }
196     //Padding
197     StringBuilder codewords = context.getCodewords();
198     if (codewords.length() < capacity) {
199       codewords.append(PAD);
200     }
201     while (codewords.length() < capacity) {
202       codewords.append(randomize253State(PAD, codewords.length() + 1));
203     }
204 
205     return context.getCodewords().toString();
206   }

208   static int lookAheadTest(CharSequence msg, int startpos, int currentMode) {
209     if (startpos >= msg.length()) {
210       return currentMode;
211     }
212     float[] charCounts;
213     //step J
214     if (currentMode == ASCII_ENCODATION) {
215       charCounts = new float[]{0, 1, 1, 1, 1, 1.25f};
216     } else {
217       charCounts = new float[]{1, 2, 2, 2, 2, 2.25f};
218       charCounts[currentMode] = 0;
219     }
220 
221     int charsProcessed = 0;
222     while (true) {
223       //step K
224       if ((startpos + charsProcessed) == msg.length()) {
225         int min = Integer.MAX_VALUE;
226         byte[] mins = new byte[6];
227         int[] intCharCounts = new int[6];
228         min = findMinimums(charCounts, intCharCounts, min, mins);
229         int minCount = getMinimumCount(mins);
230 
231         if (intCharCounts[ASCII_ENCODATION] == min) {
232           return ASCII_ENCODATION;
233         }
234         if (minCount == 1 && mins[BASE256_ENCODATION] > 0) {
235           return BASE256_ENCODATION;
236         }
237         if (minCount == 1 && mins[EDIFACT_ENCODATION] > 0) {
238           return EDIFACT_ENCODATION;
239         }
240         if (minCount == 1 && mins[TEXT_ENCODATION] > 0) {
241           return TEXT_ENCODATION;
242         }
243         if (minCount == 1 && mins[X12_ENCODATION] > 0) {
244           return X12_ENCODATION;
245         }
246         return C40_ENCODATION;
247       }
248 
249       char c = msg.charAt(startpos + charsProcessed);
250       charsProcessed++;
251 
252       //step L
253       if (isDigit(c)) {
254         charCounts[ASCII_ENCODATION] += 0.5;
255       } else if (isExtendedASCII(c)) {
256         charCounts[ASCII_ENCODATION] = (int) Math.ceil(charCounts[ASCII_ENCODATION]);
257         charCounts[ASCII_ENCODATION] += 2;
258       } else {
259         charCounts[ASCII_ENCODATION] = (int) Math.ceil(charCounts[ASCII_ENCODATION]);
260         charCounts[ASCII_ENCODATION]++;
261       }
262 
263       //step M
264       if (isNativeC40(c)) {
265         charCounts[C40_ENCODATION] += 2.0f / 3.0f;
266       } else if (isExtendedASCII(c)) {
267         charCounts[C40_ENCODATION] += 8.0f / 3.0f;
268       } else {
269         charCounts[C40_ENCODATION] += 4.0f / 3.0f;
270       }
271 
272       //step N
273       if (isNativeText(c)) {
274         charCounts[TEXT_ENCODATION] += 2.0f / 3.0f;
275       } else if (isExtendedASCII(c)) {
276         charCounts[TEXT_ENCODATION] += 8.0f / 3.0f;
277       } else {
278         charCounts[TEXT_ENCODATION] += 4.0f / 3.0f;
279       }
280 
281       //step O
282       if (isNativeX12(c)) {
283         charCounts[X12_ENCODATION] += 2.0f / 3.0f;
284       } else if (isExtendedASCII(c)) {
285         charCounts[X12_ENCODATION] += 13.0f / 3.0f;
286       } else {
287         charCounts[X12_ENCODATION] += 10.0f / 3.0f;
288       }
289 
290       //step P
291       if (isNativeEDIFACT(c)) {
292         charCounts[EDIFACT_ENCODATION] += 3.0f / 4.0f;
293       } else if (isExtendedASCII(c)) {
294         charCounts[EDIFACT_ENCODATION] += 17.0f / 4.0f;
295       } else {
296         charCounts[EDIFACT_ENCODATION] += 13.0f / 4.0f;
297       }
298 
299       // step Q
300       if (isSpecialB256(c)) {
301         charCounts[BASE256_ENCODATION] += 4;
302       } else {
303         charCounts[BASE256_ENCODATION]++;
304       }
305 
306       //step R
307       if (charsProcessed >= 4) {
308         int[] intCharCounts = new int[6];
309         byte[] mins = new byte[6];
310         findMinimums(charCounts, intCharCounts, Integer.MAX_VALUE, mins);
311         int minCount = getMinimumCount(mins);
312 
313         if (intCharCounts[ASCII_ENCODATION] < intCharCounts[BASE256_ENCODATION]
314             && intCharCounts[ASCII_ENCODATION] < intCharCounts[C40_ENCODATION]
315             && intCharCounts[ASCII_ENCODATION] < intCharCounts[TEXT_ENCODATION]
316             && intCharCounts[ASCII_ENCODATION] < intCharCounts[X12_ENCODATION]
317             && intCharCounts[ASCII_ENCODATION] < intCharCounts[EDIFACT_ENCODATION]) {
318           return ASCII_ENCODATION;
319         }
320         if (intCharCounts[BASE256_ENCODATION] < intCharCounts[ASCII_ENCODATION]
321             || (mins[C40_ENCODATION] + mins[TEXT_ENCODATION] + mins[X12_ENCODATION] + mins[EDIFACT_ENCODATION]) == 0) {
322           return BASE256_ENCODATION;
323         }
324         if (minCount == 1 && mins[EDIFACT_ENCODATION] > 0) {
325           return EDIFACT_ENCODATION;
326         }
327         if (minCount == 1 && mins[TEXT_ENCODATION] > 0) {
328           return TEXT_ENCODATION;
329         }
330         if (minCount == 1 && mins[X12_ENCODATION] > 0) {
331           return X12_ENCODATION;
332         }
333         if (intCharCounts[C40_ENCODATION] + 1 < intCharCounts[ASCII_ENCODATION]
334             && intCharCounts[C40_ENCODATION] + 1 < intCharCounts[BASE256_ENCODATION]
335             && intCharCounts[C40_ENCODATION] + 1 < intCharCounts[EDIFACT_ENCODATION]
336             && intCharCounts[C40_ENCODATION] + 1 < intCharCounts[TEXT_ENCODATION]) {
337           if (intCharCounts[C40_ENCODATION] < intCharCounts[X12_ENCODATION]) {
338             return C40_ENCODATION;
339           }
340           if (intCharCounts[C40_ENCODATION] == intCharCounts[X12_ENCODATION]) {
341             int p = startpos + charsProcessed + 1;
342             while (p < msg.length()) {
343               char tc = msg.charAt(p);
344               if (isX12TermSep(tc)) {
345                 return X12_ENCODATION;
346               }
347               if (!isNativeX12(tc)) {
348                 break;
349               }
350               p++;
351             }
352             return C40_ENCODATION;
353           }
354         }
355       }
356     }
357   }

359   private static int findMinimums(float[] charCounts, int[] intCharCounts, int min, byte[] mins) {
360     Arrays.fill(mins, (byte) 0);
361     for (int i = 0; i < 6; i++) {
362       intCharCounts[i] = (int) Math.ceil(charCounts[i]);
363       int current = intCharCounts[i];
364       if (min > current) {
365         min = current;
366         Arrays.fill(mins, (byte) 0);
367       }
368       if (min == current) {
369         mins[i]++;
370 
371       }
372     }
373     return min;
374   }

376   private static int getMinimumCount(byte[] mins) {
377     int minCount = 0;
378     for (int i = 0; i < 6; i++) {
379       minCount += mins[i];
380     }
381     return minCount;
382   }

384   static boolean isDigit(char ch) {
385     return ch >= '0' && ch <= '9';
386   }
387 
388   static boolean isExtendedASCII(char ch) {
389     return ch >= 128 && ch <= 255;
390   }
391 
392   private static boolean isNativeC40(char ch) {
393     return (ch == ' ') || (ch >= '0' && ch <= '9') || (ch >= 'A' && ch <= 'Z');
394   }
395 
396   private static boolean isNativeText(char ch) {
397     return (ch == ' ') || (ch >= '0' && ch <= '9') || (ch >= 'a' && ch <= 'z');
398   }
399 
400   private static boolean isNativeX12(char ch) {
401     return isX12TermSep(ch) || (ch == ' ') || (ch >= '0' && ch <= '9') || (ch >= 'A' && ch <= 'Z');
402   }
403 
404   private static boolean isX12TermSep(char ch) {
405     return (ch == '\r') //CR
406         || (ch == '*')
407         || (ch == '>');
408   }
409 
410   private static boolean isNativeEDIFACT(char ch) {
411     return ch >= ' ' && ch <= '^';
412   }

418   /**
419    * Determines the number of consecutive characters that are encodable using numeric compaction.
420    *
421    * @param msg      the message
422    * @param startpos the start position within the message
423    * @return the requested character count
424    */
425   public static int determineConsecutiveDigitCount(CharSequence msg, int startpos) {
426     int count = 0;
427     int len = msg.length();
428     int idx = startpos;
429     if (idx < len) {
430       char ch = msg.charAt(idx);
431       while (isDigit(ch) && idx < len) {
432         count++;
433         idx++;
434         if (idx < len) {
435           ch = msg.charAt(idx);
436         }
437       }
438     }
439     return count;
440   }

442   static void illegalCharacter(char c) {
443     String hex = Integer.toHexString(c);
444     hex = "0000".substring(0, 4 - hex.length()) + hex;
445     throw new IllegalArgumentException("Illegal character: " + c + " (0x" + hex + ')');
446   }
