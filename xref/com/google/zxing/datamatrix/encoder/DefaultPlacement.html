17  package com.google.zxing.datamatrix.encoder;
18  
19  import java.util.Arrays;
20  
21  /**
22   * Symbol Character Placement Program. Adapted from Annex M.1 in ISO/IEC 16022:2000(E).
23   */
24  public class DefaultPlacement {
25  
26    private final CharSequence codewords;
27    private final int numrows;
28    private final int numcols;
29    private final byte[] bits;
30  
31    /**
32     * Main constructor
33     *
34     * @param codewords the codewords to place
35     * @param numcols   the number of columns
36     * @param numrows   the number of rows
37     */
38    public DefaultPlacement(CharSequence codewords, int numcols, int numrows) {
39      this.codewords = codewords;
40      this.numcols = numcols;
41      this.numrows = numrows;
42      this.bits = new byte[numcols * numrows];
43      Arrays.fill(this.bits, (byte) -1); //Initialize with "not set" value
44    }
45    
46    final int getNumrows() {
47      return numrows;
48    }
49  
50    final int getNumcols() {
51      return numcols;
52    }
53  
54    final byte[] getBits() {
55      return bits;
56    }
57  
58    public final boolean getBit(int col, int row) {
59      return bits[row * numcols + col] == 1;
60    }
61  
62    final void setBit(int col, int row, boolean bit) {
63      bits[row * numcols + col] = bit ? (byte) 1 : (byte) 0;
64    }
65  
66    final boolean hasBit(int col, int row) {
67      return bits[row * numcols + col] >= 0;
68    }
69  
70    public final void place() {
71      int pos = 0;
72      int row = 4;
73      int col = 0;
74  
75      do {
76              /* repeatedly first check for one of the special corner cases, then... */
77        if ((row == numrows) && (col == 0)) {
78          corner1(pos++);
79        }
80        if ((row == numrows - 2) && (col == 0) && ((numcols % 4) != 0)) {
81          corner2(pos++);
82        }
83        if ((row == numrows - 2) && (col == 0) && (numcols % 8 == 4)) {
84          corner3(pos++);
85        }
86        if ((row == numrows + 4) && (col == 2) && ((numcols % 8) == 0)) {
87          corner4(pos++);
88        }
89              /* sweep upward diagonally, inserting successive characters... */
90        do {
91          if ((row < numrows) && (col >= 0) && !hasBit(col, row)) {
92            utah(row, col, pos++);
93          }
94          row -= 2;
95          col += 2;
96        } while (row >= 0 && (col < numcols));
97        row++;
98        col += 3;
99              
100             /* and then sweep downward diagonally, inserting successive characters, ... */
101       do {
102         if ((row >= 0) && (col < numcols) && !hasBit(col, row)) {
103           utah(row, col, pos++);
104         }
105         row += 2;
106         col -= 2;
107       } while ((row < numrows) && (col >= 0));
108       row += 3;
109       col++;
110             
111             /* ...until the entire array is scanned */
112     } while ((row < numrows) || (col < numcols));
113         
114         /* Lastly, if the lower righthand corner is untouched, fill in fixed pattern */
115     if (!hasBit(numcols - 1, numrows - 1)) {
116       setBit(numcols - 1, numrows - 1, true);
117       setBit(numcols - 2, numrows - 2, true);
118     }
119   }
120 
121   private void module(int row, int col, int pos, int bit) {
122     if (row < 0) {
123       row += numrows;
124       col += 4 - ((numrows + 4) % 8);
125     }
126     if (col < 0) {
127       col += numcols;
128       row += 4 - ((numcols + 4) % 8);
129     }
130     // Note the conversion:
131     int v = codewords.charAt(pos);
132     v &= 1 << (8 - bit);
133     setBit(col, row, v != 0);
134   }
135 
136   /**
137    * Places the 8 bits of a utah-shaped symbol character in ECC200.
138    *
139    * @param row the row
140    * @param col the column
141    * @param pos character position
142    */
143   private void utah(int row, int col, int pos) {
144     module(row - 2, col - 2, pos, 1);
145     module(row - 2, col - 1, pos, 2);
146     module(row - 1, col - 2, pos, 3);
147     module(row - 1, col - 1, pos, 4);
148     module(row - 1, col, pos, 5);
149     module(row, col - 2, pos, 6);
150     module(row, col - 1, pos, 7);
151     module(row, col, pos, 8);
152   }
153 
154   private void corner1(int pos) {
155     module(numrows - 1, 0, pos, 1);
156     module(numrows - 1, 1, pos, 2);
157     module(numrows - 1, 2, pos, 3);
158     module(0, numcols - 2, pos, 4);
159     module(0, numcols - 1, pos, 5);
160     module(1, numcols - 1, pos, 6);
161     module(2, numcols - 1, pos, 7);
162     module(3, numcols - 1, pos, 8);
163   }
164 
165   private void corner2(int pos) {
166     module(numrows - 3, 0, pos, 1);
167     module(numrows - 2, 0, pos, 2);
168     module(numrows - 1, 0, pos, 3);
169     module(0, numcols - 4, pos, 4);
170     module(0, numcols - 3, pos, 5);
171     module(0, numcols - 2, pos, 6);
172     module(0, numcols - 1, pos, 7);
173     module(1, numcols - 1, pos, 8);
174   }
175 
176   private void corner3(int pos) {
177     module(numrows - 3, 0, pos, 1);
178     module(numrows - 2, 0, pos, 2);
179     module(numrows - 1, 0, pos, 3);
180     module(0, numcols - 2, pos, 4);
181     module(0, numcols - 1, pos, 5);
182     module(1, numcols - 1, pos, 6);
183     module(2, numcols - 1, pos, 7);
184     module(3, numcols - 1, pos, 8);
185   }
186 
187   private void corner4(int pos) {
188     module(numrows - 1, 0, pos, 1);
189     module(numrows - 1, numcols - 1, pos, 2);
190     module(0, numcols - 3, pos, 3);
191     module(0, numcols - 2, pos, 4);
192     module(0, numcols - 1, pos, 5);
193     module(1, numcols - 3, pos, 6);
194     module(1, numcols - 2, pos, 7);
195     module(1, numcols - 1, pos, 8);
196   }
