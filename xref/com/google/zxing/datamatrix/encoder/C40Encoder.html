17  package com.google.zxing.datamatrix.encoder;
18  
19  class C40Encoder implements Encoder {
20  
21    @Override
22    public int getEncodingMode() {
23      return HighLevelEncoder.C40_ENCODATION;
24    }
25  
26    @Override
27    public void encode(EncoderContext context) {
28      //step C
29      StringBuilder buffer = new StringBuilder();
30      while (context.hasMoreCharacters()) {
31        char c = context.getCurrentChar();
32        context.pos++;
33  
34        int lastCharSize = encodeChar(c, buffer);
35  
36        int unwritten = (buffer.length() / 3) * 2;
37  
38        int curCodewordCount = context.getCodewordCount() + unwritten;
39        context.updateSymbolInfo(curCodewordCount);
40        int available = context.getSymbolInfo().getDataCapacity() - curCodewordCount;
41  
42        if (!context.hasMoreCharacters()) {
43          //Avoid having a single C40 value in the last triplet
44          StringBuilder removed = new StringBuilder();
45          if ((buffer.length() % 3) == 2) {
46            if (available < 2 || available > 2) {
47              lastCharSize = backtrackOneCharacter(context, buffer, removed,
48                                                   lastCharSize);
49            }
50          }
51          while ((buffer.length() % 3) == 1
52              && ((lastCharSize <= 3 && available != 1) || lastCharSize > 3)) {
53            lastCharSize = backtrackOneCharacter(context, buffer, removed, lastCharSize);
54          }
55          break;
56        }
57  
58        int count = buffer.length();
59        if ((count % 3) == 0) {
60          int newMode = HighLevelEncoder.lookAheadTest(context.getMessage(), context.pos, getEncodingMode());
61          if (newMode != getEncodingMode()) {
62            context.signalEncoderChange(newMode);
63            break;
64          }
65        }
66      }
67      handleEOD(context, buffer);
68    }
69  
70    private int backtrackOneCharacter(EncoderContext context,
71                                      StringBuilder buffer, StringBuilder removed, int lastCharSize) {
72      int count = buffer.length();
73      buffer.delete(count - lastCharSize, count);
74      context.pos--;
75      char c = context.getCurrentChar();
76      lastCharSize = encodeChar(c, removed);
77      context.resetSymbolInfo(); //Deal with possible reduction in symbol size
78      return lastCharSize;
79    }
80  
81    static void writeNextTriplet(EncoderContext context, StringBuilder buffer) {
82      context.writeCodewords(encodeToCodewords(buffer, 0));
83      buffer.delete(0, 3);
84    }
85  
86    /**
87     * Handle "end of data" situations
88     *
89     * @param context the encoder context
90     * @param buffer  the buffer with the remaining encoded characters
91     */
92    void handleEOD(EncoderContext context, StringBuilder buffer) {
93      int unwritten = (buffer.length() / 3) * 2;
94      int rest = buffer.length() % 3;
95  
96      int curCodewordCount = context.getCodewordCount() + unwritten;
97      context.updateSymbolInfo(curCodewordCount);
98      int available = context.getSymbolInfo().getDataCapacity() - curCodewordCount;
99  
100     if (rest == 2) {
101       buffer.append('\0'); //Shift 1
102       while (buffer.length() >= 3) {
103         writeNextTriplet(context, buffer);
104       }
105       if (context.hasMoreCharacters()) {
106         context.writeCodeword(HighLevelEncoder.C40_UNLATCH);
107       }
108     } else if (available == 1 && rest == 1) {
109       while (buffer.length() >= 3) {
110         writeNextTriplet(context, buffer);
111       }
112       if (context.hasMoreCharacters()) {
113         context.writeCodeword(HighLevelEncoder.C40_UNLATCH);
114       }
115       // else no unlatch
116       context.pos--;
117     } else if (rest == 0) {
118       while (buffer.length() >= 3) {
119         writeNextTriplet(context, buffer);
120       }
121       if (available > 0 || context.hasMoreCharacters()) {
122         context.writeCodeword(HighLevelEncoder.C40_UNLATCH);
123       }
124     } else {
125       throw new IllegalStateException("Unexpected case. Please report!");
126     }
127     context.signalEncoderChange(HighLevelEncoder.ASCII_ENCODATION);
128   }

130   int encodeChar(char c, StringBuilder sb) {
131     if (c == ' ') {
132       sb.append('\3');
133       return 1;
134     } else if (c >= '0' && c <= '9') {
135       sb.append((char) (c - 48 + 4));
136       return 1;
137     } else if (c >= 'A' && c <= 'Z') {
138       sb.append((char) (c - 65 + 14));
139       return 1;
140     } else if (c >= '\0' && c <= '\u001f') {
141       sb.append('\0'); //Shift 1 Set
142       sb.append(c);
143       return 2;
144     } else if (c >= '!' && c <= '/') {
145       sb.append('\1'); //Shift 2 Set
146       sb.append((char) (c - 33));
147       return 2;
148     } else if (c >= ':' && c <= '@') {
149       sb.append('\1'); //Shift 2 Set
150       sb.append((char) (c - 58 + 15));
151       return 2;
152     } else if (c >= '[' && c <= '_') {
153       sb.append('\1'); //Shift 2 Set
154       sb.append((char) (c - 91 + 22));
155       return 2;
156     } else if (c >= '\u0060' && c <= '\u007f') {
157       sb.append('\2'); //Shift 3 Set
158       sb.append((char) (c - 96));
159       return 2;
160     } else if (c >= '\u0080') {
161       sb.append("\1\u001e"); //Shift 2, Upper Shift
162       int len = 2;
163       len += encodeChar((char) (c - 128), sb);
164       return len;
165     } else {
166       throw new IllegalArgumentException("Illegal character: " + c);
167     }
168   }
169 
170   private static String encodeToCodewords(CharSequence sb, int startPos) {
171     char c1 = sb.charAt(startPos);
172     char c2 = sb.charAt(startPos + 1);
173     char c3 = sb.charAt(startPos + 2);
174     int v = (1600 * c1) + (40 * c2) + c3 + 1;
175     char cw1 = (char) (v / 256);
176     char cw2 = (char) (v % 256);
177     return new String(new char[] {cw1, cw2});
178   }
