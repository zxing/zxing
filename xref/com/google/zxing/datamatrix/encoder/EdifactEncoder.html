17  package com.google.zxing.datamatrix.encoder;
18  
19  final class EdifactEncoder implements Encoder {
20  
21    @Override
22    public int getEncodingMode() {
23      return HighLevelEncoder.EDIFACT_ENCODATION;
24    }
25  
26    @Override
27    public void encode(EncoderContext context) {
28      //step F
29      StringBuilder buffer = new StringBuilder();
30      while (context.hasMoreCharacters()) {
31        char c = context.getCurrentChar();
32        encodeChar(c, buffer);
33        context.pos++;
34  
35        int count = buffer.length();
36        if (count >= 4) {
37          context.writeCodewords(encodeToCodewords(buffer, 0));
38          buffer.delete(0, 4);
39  
40          int newMode = HighLevelEncoder.lookAheadTest(context.getMessage(), context.pos, getEncodingMode());
41          if (newMode != getEncodingMode()) {
42            context.signalEncoderChange(HighLevelEncoder.ASCII_ENCODATION);
43            break;
44          }
45        }
46      }
47      buffer.append((char) 31); //Unlatch
48      handleEOD(context, buffer);
49    }
50  
51    /**
52     * Handle "end of data" situations
53     *
54     * @param context the encoder context
55     * @param buffer  the buffer with the remaining encoded characters
56     */
57    private static void handleEOD(EncoderContext context, CharSequence buffer) {
58      try {
59        int count = buffer.length();
60        if (count == 0) {
61          return; //Already finished
62        }
63        if (count == 1) {
64          //Only an unlatch at the end
65          context.updateSymbolInfo();
66          int available = context.getSymbolInfo().getDataCapacity() - context.getCodewordCount();
67          int remaining = context.getRemainingCharacters();
68          if (remaining == 0 && available <= 2) {
69            return; //No unlatch
70          }
71        }
72  
73        if (count > 4) {
74          throw new IllegalStateException("Count must not exceed 4");
75        }
76        int restChars = count - 1;
77        String encoded = encodeToCodewords(buffer, 0);
78        boolean endOfSymbolReached = !context.hasMoreCharacters();
79        boolean restInAscii = endOfSymbolReached && restChars <= 2;
80  
81        if (restChars <= 2) {
82          context.updateSymbolInfo(context.getCodewordCount() + restChars);
83          int available = context.getSymbolInfo().getDataCapacity() - context.getCodewordCount();
84          if (available >= 3) {
85            restInAscii = false;
86            context.updateSymbolInfo(context.getCodewordCount() + encoded.length());
87            //available = context.symbolInfo.dataCapacity - context.getCodewordCount();
88          }
89        }
90  
91        if (restInAscii) {
92          context.resetSymbolInfo();
93          context.pos -= restChars;
94        } else {
95          context.writeCodewords(encoded);
96        }
97      } finally {
98        context.signalEncoderChange(HighLevelEncoder.ASCII_ENCODATION);
99      }
100   }
101 
102   private static void encodeChar(char c, StringBuilder sb) {
103     if (c >= ' ' && c <= '?') {
104       sb.append(c);
105     } else if (c >= '@' && c <= '^') {
106       sb.append((char) (c - 64));
107     } else {
108       HighLevelEncoder.illegalCharacter(c);
109     }
110   }
111 
112   private static String encodeToCodewords(CharSequence sb, int startPos) {
113     int len = sb.length() - startPos;
114     if (len == 0) {
115       throw new IllegalStateException("StringBuilder must not be empty");
116     }
117     char c1 = sb.charAt(startPos);
118     char c2 = len >= 2 ? sb.charAt(startPos + 1) : 0;
119     char c3 = len >= 3 ? sb.charAt(startPos + 2) : 0;
120     char c4 = len >= 4 ? sb.charAt(startPos + 3) : 0;
121 
122     int v = (c1 << 18) + (c2 << 12) + (c3 << 6) + c4;
123     char cw1 = (char) ((v >> 16) & 255);
124     char cw2 = (char) ((v >> 8) & 255);
125     char cw3 = (char) (v & 255);
126     StringBuilder res = new StringBuilder(3);
127     res.append(cw1);
128     if (len >= 2) {
129       res.append(cw2);
130     }
131     if (len >= 3) {
132       res.append(cw3);
133     }
134     return res.toString();
135   }
