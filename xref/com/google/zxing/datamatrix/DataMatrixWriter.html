33  /**
34   * This object renders a Data Matrix code as a BitMatrix 2D array of greyscale values.
35   *
36   * @author dswitkin@google.com (Daniel Switkin)
37   * @author Guillaume Le Biller Added to zxing lib.
38   */
39  public final class DataMatrixWriter implements Writer {
40  
41    @Override
42    public BitMatrix encode(String contents, BarcodeFormat format, int width, int height) {
43      return encode(contents, format, width, height, null);
44    }
45  
46    @Override
47    public BitMatrix encode(String contents, BarcodeFormat format, int width, int height, Map<EncodeHintType,?> hints) {
48  
49      if (contents.isEmpty()) {
50        throw new IllegalArgumentException("Found empty contents");
51      }
52      
53      if (format != BarcodeFormat.DATA_MATRIX) {
54        throw new IllegalArgumentException("Can only encode DATA_MATRIX, but got " + format);
55      }
56      
57      if (width < 0 || height < 0) {
58        throw new IllegalArgumentException("Requested dimensions are too small: " + width + 'x' + height);
59      }
60  
61      // Try to get force shape & min / max size
62      SymbolShapeHint shape = SymbolShapeHint.FORCE_NONE;
63      Dimension minSize = null;
64      Dimension maxSize = null;
65      if (hints != null) {
66        SymbolShapeHint requestedShape = (SymbolShapeHint) hints.get(EncodeHintType.DATA_MATRIX_SHAPE);
67        if (requestedShape != null) {
68          shape = requestedShape;
69        }
70        Dimension requestedMinSize = (Dimension) hints.get(EncodeHintType.MIN_SIZE);
71        if (requestedMinSize != null) {
72          minSize = requestedMinSize;
73        }
74        Dimension requestedMaxSize = (Dimension) hints.get(EncodeHintType.MAX_SIZE);
75        if (requestedMaxSize != null) {
76          maxSize = requestedMaxSize;
77        }
78      }
79  
80  
81      //1. step: Data encodation
82      String encoded = HighLevelEncoder.encodeHighLevel(contents, shape, minSize, maxSize);
83  
84      SymbolInfo symbolInfo = SymbolInfo.lookup(encoded.length(), shape, minSize, maxSize, true);
85  
86      //2. step: ECC generation
87      String codewords = ErrorCorrection.encodeECC200(encoded, symbolInfo);
88  
89      //3. step: Module placement in Matrix
90      DefaultPlacement placement =
91          new DefaultPlacement(codewords, symbolInfo.getSymbolDataWidth(), symbolInfo.getSymbolDataHeight());
92      placement.place();
93  
94      //4. step: low-level encoding
95      return encodeLowLevel(placement, symbolInfo);
96    }
97  
98    /**
99     * Encode the given symbol info to a bit matrix.
100    *
101    * @param placement  The DataMatrix placement.
102    * @param symbolInfo The symbol info to encode.
103    * @return The bit matrix generated.
104    */
105   private static BitMatrix encodeLowLevel(DefaultPlacement placement, SymbolInfo symbolInfo) {
106     int symbolWidth = symbolInfo.getSymbolDataWidth();
107     int symbolHeight = symbolInfo.getSymbolDataHeight();
108 
109     ByteMatrix matrix = new ByteMatrix(symbolInfo.getSymbolWidth(), symbolInfo.getSymbolHeight());
110 
111     int matrixY = 0;
112 
113     for (int y = 0; y < symbolHeight; y++) {
114       // Fill the top edge with alternate 0 / 1
115       int matrixX;
116       if ((y % symbolInfo.matrixHeight) == 0) {
117         matrixX = 0;
118         for (int x = 0; x < symbolInfo.getSymbolWidth(); x++) {
119           matrix.set(matrixX, matrixY, (x % 2) == 0);
120           matrixX++;
121         }
122         matrixY++;
123       }
124       matrixX = 0;
125       for (int x = 0; x < symbolWidth; x++) {
126         // Fill the right edge with full 1
127         if ((x % symbolInfo.matrixWidth) == 0) {
128           matrix.set(matrixX, matrixY, true);
129           matrixX++;
130         }
131         matrix.set(matrixX, matrixY, placement.getBit(x, y));
132         matrixX++;
133         // Fill the right edge with alternate 0 / 1
134         if ((x % symbolInfo.matrixWidth) == symbolInfo.matrixWidth - 1) {
135           matrix.set(matrixX, matrixY, (y % 2) == 0);
136           matrixX++;
137         }
138       }
139       matrixY++;
140       // Fill the bottom edge with full 1
141       if ((y % symbolInfo.matrixHeight) == symbolInfo.matrixHeight - 1) {
142         matrixX = 0;
143         for (int x = 0; x < symbolInfo.getSymbolWidth(); x++) {
144           matrix.set(matrixX, matrixY, true);
145           matrixX++;
146         }
147         matrixY++;
148       }
149     }
150 
151     return convertByteMatrixToBitMatrix(matrix);
152   }
153 
154   /**
155    * Convert the ByteMatrix to BitMatrix.
156    *
157    * @param matrix The input matrix.
158    * @return The output matrix.
159    */
160   private static BitMatrix convertByteMatrixToBitMatrix(ByteMatrix matrix) {
161     int matrixWidgth = matrix.getWidth();
162     int matrixHeight = matrix.getHeight();
163 
164     BitMatrix output = new BitMatrix(matrixWidgth, matrixHeight);
165     output.clear();
166     for (int i = 0; i < matrixWidgth; i++) {
167       for (int j = 0; j < matrixHeight; j++) {
168         // Zero is white in the bytematrix
169         if (matrix.get(i, j) == 1) {
170           output.set(i, j);
171         }
172       }
173     }
174 
175     return output;
176   }
