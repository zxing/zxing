39  /**
40   * This implementation can detect and decode PDF417 codes in an image.
41   *
42   * @author Guenther Grau
43   */
44  public final class PDF417Reader implements Reader, MultipleBarcodeReader {
45  
46    /**
47     * Locates and decodes a PDF417 code in an image.
48     *
49     * @return a String representing the content encoded by the PDF417 code
50     * @throws NotFoundException if a PDF417 code cannot be found,
51     * @throws FormatException if a PDF417 cannot be decoded
52     */
53    @Override
54    public Result decode(BinaryBitmap image) throws NotFoundException, FormatException, ChecksumException {
55      return decode(image, null);
56    }
57  
58    @Override
59    public Result decode(BinaryBitmap image, Map<DecodeHintType,?> hints) throws NotFoundException, FormatException,
60        ChecksumException {
61      Result[] result = decode(image, hints, false);
62      if (result == null || result.length == 0 || result[0] == null) {
63        throw NotFoundException.getNotFoundInstance();
64      }
65      return result[0];
66    }
67  
68    @Override
69    public Result[] decodeMultiple(BinaryBitmap image) throws NotFoundException {
70      return decodeMultiple(image, null);
71    }
72  
73    @Override
74    public Result[] decodeMultiple(BinaryBitmap image, Map<DecodeHintType,?> hints) throws NotFoundException {
75      try {
76        return decode(image, hints, true);
77      } catch (FormatException | ChecksumException ignored) {
78        throw NotFoundException.getNotFoundInstance();
79      }
80    }
81  
82    private static Result[] decode(BinaryBitmap image, Map<DecodeHintType, ?> hints, boolean multiple) 
83        throws NotFoundException, FormatException, ChecksumException {
84      List<Result> results = new ArrayList<>();
85      PDF417DetectorResult detectorResult = Detector.detect(image, hints, multiple);
86      for (ResultPoint[] points : detectorResult.getPoints()) {
87        DecoderResult decoderResult = PDF417ScanningDecoder.decode(detectorResult.getBits(), points[4], points[5],
88            points[6], points[7], getMinCodewordWidth(points), getMaxCodewordWidth(points));
89        Result result = new Result(decoderResult.getText(), decoderResult.getRawBytes(), points, BarcodeFormat.PDF_417);
90        result.putMetadata(ResultMetadataType.ERROR_CORRECTION_LEVEL, decoderResult.getECLevel());
91        PDF417ResultMetadata pdf417ResultMetadata = (PDF417ResultMetadata) decoderResult.getOther();
92        if (pdf417ResultMetadata != null) {
93          result.putMetadata(ResultMetadataType.PDF417_EXTRA_METADATA, pdf417ResultMetadata);
94        }
95        results.add(result);
96      }
97      return results.toArray(new Result[results.size()]);
98    }
99  
100   private static int getMaxWidth(ResultPoint p1, ResultPoint p2) {
101     if (p1 == null || p2 == null) {
102       return 0;
103     }
104     return (int) Math.abs(p1.getX() - p2.getX());
105   }
106 
107   private static int getMinWidth(ResultPoint p1, ResultPoint p2) {
108     if (p1 == null || p2 == null) {
109       return Integer.MAX_VALUE;
110     }
111     return (int) Math.abs(p1.getX() - p2.getX());
112   }
113 
114   private static int getMaxCodewordWidth(ResultPoint[] p) {
115     return Math.max(
116         Math.max(getMaxWidth(p[0], p[4]), getMaxWidth(p[6], p[2]) * PDF417Common.MODULES_IN_CODEWORD /
117             PDF417Common.MODULES_IN_STOP_PATTERN),
118         Math.max(getMaxWidth(p[1], p[5]), getMaxWidth(p[7], p[3]) * PDF417Common.MODULES_IN_CODEWORD /
119             PDF417Common.MODULES_IN_STOP_PATTERN));
120   }
121 
122   private static int getMinCodewordWidth(ResultPoint[] p) {
123     return Math.min(
124         Math.min(getMinWidth(p[0], p[4]), getMinWidth(p[6], p[2]) * PDF417Common.MODULES_IN_CODEWORD /
125             PDF417Common.MODULES_IN_STOP_PATTERN),
126         Math.min(getMinWidth(p[1], p[5]), getMinWidth(p[7], p[3]) * PDF417Common.MODULES_IN_CODEWORD /
127             PDF417Common.MODULES_IN_STOP_PATTERN));
128   }
129 
130   @Override
131   public void reset() {
132     // nothing needs to be reset
133   }
134 
135 }
