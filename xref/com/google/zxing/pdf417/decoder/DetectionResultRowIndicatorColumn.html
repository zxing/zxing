46    int adjustCompleteIndicatorColumnRowNumbers(BarcodeMetadata barcodeMetadata) {
47      Codeword[] codewords = getCodewords();
48      setRowNumbers();
49      removeIncorrectCodewords(codewords, barcodeMetadata);
50      BoundingBox boundingBox = getBoundingBox();
51      ResultPoint top = isLeft ? boundingBox.getTopLeft() : boundingBox.getTopRight();
52      ResultPoint bottom = isLeft ? boundingBox.getBottomLeft() : boundingBox.getBottomRight();
53      int firstRow = imageRowToCodewordIndex((int) top.getY());
54      int lastRow = imageRowToCodewordIndex((int) bottom.getY());
55      // We need to be careful using the average row height. Barcode could be skewed so that we have smaller and 
56      // taller rows
57      float averageRowHeight = (lastRow - firstRow) / (float) barcodeMetadata.getRowCount();
58      int barcodeRow = -1;
59      int maxRowHeight = 1;
60      int currentRowHeight = 0;
61      for (int codewordsRow = firstRow; codewordsRow < lastRow; codewordsRow++) {
62        if (codewords[codewordsRow] == null) {
63          continue;
64        }
65        Codeword codeword = codewords[codewordsRow];
66  
67        //      float expectedRowNumber = (codewordsRow - firstRow) / averageRowHeight;
68        //      if (Math.abs(codeword.getRowNumber() - expectedRowNumber) > 2) {
69        //        SimpleLog.log(LEVEL.WARNING,
70        //            "Removing codeword, rowNumberSkew too high, codeword[" + codewordsRow + "]: Expected Row: " +
71        //                expectedRowNumber + ", RealRow: " + codeword.getRowNumber() + ", value: " + codeword.getValue());
72        //        codewords[codewordsRow] = null;
73        //      }
74  
75        int rowDifference = codeword.getRowNumber() - barcodeRow;
76  
77        // TODO improve handling with case where first row indicator doesn't start with 0
78  
79        if (rowDifference == 0) {
80          currentRowHeight++;
81        } else if (rowDifference == 1) {
82          maxRowHeight = Math.max(maxRowHeight, currentRowHeight);
83          currentRowHeight = 1;
84          barcodeRow = codeword.getRowNumber();
85        } else if (rowDifference < 0 ||
86                   codeword.getRowNumber() >= barcodeMetadata.getRowCount() ||
87                   rowDifference > codewordsRow) {
88          codewords[codewordsRow] = null;
89        } else {
90          int checkedRows;
91          if (maxRowHeight > 2) {
92            checkedRows = (maxRowHeight - 2) * rowDifference;
93          } else {
94            checkedRows = rowDifference;
95          }
96          boolean closePreviousCodewordFound = checkedRows >= codewordsRow;
97          for (int i = 1; i <= checkedRows && !closePreviousCodewordFound; i++) {
98            // there must be (height * rowDifference) number of codewords missing. For now we assume height = 1.
99            // This should hopefully get rid of most problems already.
100           closePreviousCodewordFound = codewords[codewordsRow - i] != null;
101         }
102         if (closePreviousCodewordFound) {
103           codewords[codewordsRow] = null;
104         } else {
105           barcodeRow = codeword.getRowNumber();
106           currentRowHeight = 1;
107         }
108       }
109     }
110     return (int) (averageRowHeight + 0.5);
111   }

113   int[] getRowHeights() {
114     BarcodeMetadata barcodeMetadata = getBarcodeMetadata();
115     if (barcodeMetadata == null) {
116       return null;
117     }
118     adjustIncompleteIndicatorColumnRowNumbers(barcodeMetadata);
119     int[] result = new int[barcodeMetadata.getRowCount()];
120     for (Codeword codeword : getCodewords()) {
121       if (codeword != null) {
122         result[codeword.getRowNumber()]++;
123       }
124     }
125     return result;
126   }

134     ResultPoint bottom = isLeft ? boundingBox.getBottomLeft() : boundingBox.getBottomRight();
135     int firstRow = imageRowToCodewordIndex((int) top.getY());
136     int lastRow = imageRowToCodewordIndex((int) bottom.getY());
137     float averageRowHeight = (lastRow - firstRow) / (float) barcodeMetadata.getRowCount();
138     Codeword[] codewords = getCodewords();
139     int barcodeRow = -1;
140     int maxRowHeight = 1;
141     int currentRowHeight = 0;
142     for (int codewordsRow = firstRow; codewordsRow < lastRow; codewordsRow++) {
143       if (codewords[codewordsRow] == null) {
144         continue;
145       }
146       Codeword codeword = codewords[codewordsRow];
147 
148       codeword.setRowNumberAsRowIndicatorColumn();
149 
150       int rowDifference = codeword.getRowNumber() - barcodeRow;
151 
152       // TODO improve handling with case where first row indicator doesn't start with 0
153 
154       if (rowDifference == 0) {
155         currentRowHeight++;
156       } else if (rowDifference == 1) {
157         maxRowHeight = Math.max(maxRowHeight, currentRowHeight);
158         currentRowHeight = 1;
159         barcodeRow = codeword.getRowNumber();
160       } else if (codeword.getRowNumber() >= barcodeMetadata.getRowCount()) {
161         codewords[codewordsRow] = null;
162       } else {
163         barcodeRow = codeword.getRowNumber();
164         currentRowHeight = 1;
165       }
166     }
167     return (int) (averageRowHeight + 0.5);
168   }

170   BarcodeMetadata getBarcodeMetadata() {
171     Codeword[] codewords = getCodewords();
172     BarcodeValue barcodeColumnCount = new BarcodeValue();
173     BarcodeValue barcodeRowCountUpperPart = new BarcodeValue();
174     BarcodeValue barcodeRowCountLowerPart = new BarcodeValue();
175     BarcodeValue barcodeECLevel = new BarcodeValue();
176     for (Codeword codeword : codewords) {
177       if (codeword == null) {
178         continue;
179       }
180       codeword.setRowNumberAsRowIndicatorColumn();
181       int rowIndicatorValue = codeword.getValue() % 30;
182       int codewordRowNumber = codeword.getRowNumber();
183       if (!isLeft) {
184         codewordRowNumber += 2;
185       }
186       switch (codewordRowNumber % 3) {
187         case 0:
188           barcodeRowCountUpperPart.setValue(rowIndicatorValue * 3 + 1);
189           break;
190         case 1:
191           barcodeECLevel.setValue(rowIndicatorValue / 3);
192           barcodeRowCountLowerPart.setValue(rowIndicatorValue % 3);
193           break;
194         case 2:
195           barcodeColumnCount.setValue(rowIndicatorValue + 1);
196           break;
197       }
198     }
199     // Maybe we should check if we have ambiguous values?
200     if ((barcodeColumnCount.getValue().length == 0) ||
201         (barcodeRowCountUpperPart.getValue().length == 0) ||
202         (barcodeRowCountLowerPart.getValue().length == 0) ||
203         (barcodeECLevel.getValue().length == 0) ||
204         barcodeColumnCount.getValue()[0] < 1 ||
205         barcodeRowCountUpperPart.getValue()[0] + barcodeRowCountLowerPart.getValue()[0] < PDF417Common.MIN_ROWS_IN_BARCODE ||
206         barcodeRowCountUpperPart.getValue()[0] + barcodeRowCountLowerPart.getValue()[0] > PDF417Common.MAX_ROWS_IN_BARCODE) {
207       return null;
208     }
209     BarcodeMetadata barcodeMetadata = new BarcodeMetadata(barcodeColumnCount.getValue()[0],
210         barcodeRowCountUpperPart.getValue()[0], barcodeRowCountLowerPart.getValue()[0], barcodeECLevel.getValue()[0]);
211     removeIncorrectCodewords(codewords, barcodeMetadata);
212     return barcodeMetadata;
213   }

215   private void removeIncorrectCodewords(Codeword[] codewords, BarcodeMetadata barcodeMetadata) {
216     // Remove codewords which do not match the metadata
217     // TODO Maybe we should keep the incorrect codewords for the start and end positions?
218     for (int codewordRow = 0; codewordRow < codewords.length; codewordRow++) {
219       Codeword codeword = codewords[codewordRow];
220       if (codewords[codewordRow] == null) {
221         continue;
222       }
223       int rowIndicatorValue = codeword.getValue() % 30;
224       int codewordRowNumber = codeword.getRowNumber();
225       if (codewordRowNumber > barcodeMetadata.getRowCount()) {
226         codewords[codewordRow] = null;
227         continue;
228       }
229       if (!isLeft) {
230         codewordRowNumber += 2;
231       }
232       switch (codewordRowNumber % 3) {
233         case 0:
234           if (rowIndicatorValue * 3 + 1 != barcodeMetadata.getRowCountUpperPart()) {
235             codewords[codewordRow] = null;
236           }
237           break;
238         case 1:
239           if (rowIndicatorValue / 3 != barcodeMetadata.getErrorCorrectionLevel() ||
240               rowIndicatorValue % 3 != barcodeMetadata.getRowCountLowerPart()) {
241             codewords[codewordRow] = null;
242           }
243           break;
244         case 2:
245           if (rowIndicatorValue + 1 != barcodeMetadata.getColumnCount()) {
246             codewords[codewordRow] = null;
247           }
248           break;
249       }
250     }
251   }
