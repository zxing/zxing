28    private static final int ADJUST_ROW_NUMBER_SKIP = 2;
29  
30    private final BarcodeMetadata barcodeMetadata;
31    private final DetectionResultColumn[] detectionResultColumns;
32    private BoundingBox boundingBox;
33    private final int barcodeColumnCount;
34  
35    DetectionResult(BarcodeMetadata barcodeMetadata, BoundingBox boundingBox) {
36      this.barcodeMetadata = barcodeMetadata;
37      this.barcodeColumnCount = barcodeMetadata.getColumnCount();
38      this.boundingBox = boundingBox;
39      detectionResultColumns = new DetectionResultColumn[barcodeColumnCount + 2];
40    }
41  
42    DetectionResultColumn[] getDetectionResultColumns() {
43      adjustIndicatorColumnRowNumbers(detectionResultColumns[0]);
44      adjustIndicatorColumnRowNumbers(detectionResultColumns[barcodeColumnCount + 1]);
45      int unadjustedCodewordCount = PDF417Common.MAX_CODEWORDS_IN_BARCODE;
46      int previousUnadjustedCount;
47      do {
48        previousUnadjustedCount = unadjustedCodewordCount;
49        unadjustedCodewordCount = adjustRowNumbers();
50      } while (unadjustedCodewordCount > 0 && unadjustedCodewordCount < previousUnadjustedCount);
51      return detectionResultColumns;
52    }
53  
54    private void adjustIndicatorColumnRowNumbers(DetectionResultColumn detectionResultColumn) {
55      if (detectionResultColumn != null) {
56        ((DetectionResultRowIndicatorColumn) detectionResultColumn)
57            .adjustCompleteIndicatorColumnRowNumbers(barcodeMetadata);
58      }
59    }
60  
61    // TODO ensure that no detected codewords with unknown row number are left
62    // we should be able to estimate the row height and use it as a hint for the row number
63    // we should also fill the rows top to bottom and bottom to top
64    /**
65     * @return number of codewords which don't have a valid row number. Note that the count is not accurate as codewords 
66     * will be counted several times. It just serves as an indicator to see when we can stop adjusting row numbers 
67     */
68    private int adjustRowNumbers() {
69      int unadjustedCount = adjustRowNumbersByRow();
70      if (unadjustedCount == 0) {
71        return 0;
72      }
73      for (int barcodeColumn = 1; barcodeColumn < barcodeColumnCount + 1; barcodeColumn++) {
74        Codeword[] codewords = detectionResultColumns[barcodeColumn].getCodewords();
75        for (int codewordsRow = 0; codewordsRow < codewords.length; codewordsRow++) {
76          if (codewords[codewordsRow] == null) {
77            continue;
78          }
79          if (!codewords[codewordsRow].hasValidRowNumber()) {
80            adjustRowNumbers(barcodeColumn, codewordsRow, codewords);
81          }
82        }
83      }
84      return unadjustedCount;
85    }
86  
87    private int adjustRowNumbersByRow() {
88      adjustRowNumbersFromBothRI();
89      // TODO we should only do full row adjustments if row numbers of left and right row indicator column match.
90      // Maybe it's even better to calculated the height (in codeword rows) and divide it by the number of barcode
91      // rows. This, together with the LRI and RRI row numbers should allow us to get a good estimate where a row
92      // number starts and ends.
93      int unadjustedCount = adjustRowNumbersFromLRI();
94      return unadjustedCount + adjustRowNumbersFromRRI();
95    }
96  
97    private void adjustRowNumbersFromBothRI() {
98      if (detectionResultColumns[0] == null || detectionResultColumns[barcodeColumnCount + 1] == null) {
99        return;
100     }
101     Codeword[] LRIcodewords = detectionResultColumns[0].getCodewords();
102     Codeword[] RRIcodewords = detectionResultColumns[barcodeColumnCount + 1].getCodewords();
103     for (int codewordsRow = 0; codewordsRow < LRIcodewords.length; codewordsRow++) {
104       if (LRIcodewords[codewordsRow] != null &&
105           RRIcodewords[codewordsRow] != null &&
106           LRIcodewords[codewordsRow].getRowNumber() == RRIcodewords[codewordsRow].getRowNumber()) {
107         for (int barcodeColumn = 1; barcodeColumn <= barcodeColumnCount; barcodeColumn++) {
108           Codeword codeword = detectionResultColumns[barcodeColumn].getCodewords()[codewordsRow];
109           if (codeword == null) {
110             continue;
111           }
112           codeword.setRowNumber(LRIcodewords[codewordsRow].getRowNumber());
113           if (!codeword.hasValidRowNumber()) {
114             detectionResultColumns[barcodeColumn].getCodewords()[codewordsRow] = null;
115           }
116         }
117       }
118     }
119   }
120 
121   private int adjustRowNumbersFromRRI() {
122     if (detectionResultColumns[barcodeColumnCount + 1] == null) {
123       return 0;
124     }
125     int unadjustedCount = 0;
126     Codeword[] codewords = detectionResultColumns[barcodeColumnCount + 1].getCodewords();
127     for (int codewordsRow = 0; codewordsRow < codewords.length; codewordsRow++) {
128       if (codewords[codewordsRow] == null) {
129         continue;
130       }
131       int rowIndicatorRowNumber = codewords[codewordsRow].getRowNumber();
132       int invalidRowCounts = 0;
133       for (int barcodeColumn = barcodeColumnCount + 1; barcodeColumn > 0 && invalidRowCounts < ADJUST_ROW_NUMBER_SKIP; barcodeColumn--) {
134         Codeword codeword = detectionResultColumns[barcodeColumn].getCodewords()[codewordsRow];
135         if (codeword != null) {
136           invalidRowCounts = adjustRowNumberIfValid(rowIndicatorRowNumber, invalidRowCounts, codeword);
137           if (!codeword.hasValidRowNumber()) {
138             unadjustedCount++;
139           }
140         }
141       }
142     }
143     return unadjustedCount;
144   }
145 
146   private int adjustRowNumbersFromLRI() {
147     if (detectionResultColumns[0] == null) {
148       return 0;
149     }
150     int unadjustedCount = 0;
151     Codeword[] codewords = detectionResultColumns[0].getCodewords();
152     for (int codewordsRow = 0; codewordsRow < codewords.length; codewordsRow++) {
153       if (codewords[codewordsRow] == null) {
154         continue;
155       }
156       int rowIndicatorRowNumber = codewords[codewordsRow].getRowNumber();
157       int invalidRowCounts = 0;
158       for (int barcodeColumn = 1; barcodeColumn < barcodeColumnCount + 1 && invalidRowCounts < ADJUST_ROW_NUMBER_SKIP; barcodeColumn++) {
159         Codeword codeword = detectionResultColumns[barcodeColumn].getCodewords()[codewordsRow];
160         if (codeword != null) {
161           invalidRowCounts = adjustRowNumberIfValid(rowIndicatorRowNumber, invalidRowCounts, codeword);
162           if (!codeword.hasValidRowNumber()) {
163             unadjustedCount++;
164           }
165         }
166       }
167     }
168     return unadjustedCount;
169   }
170 
171   private static int adjustRowNumberIfValid(int rowIndicatorRowNumber, int invalidRowCounts, Codeword codeword) {
172     if (codeword == null) {
173       return invalidRowCounts;
174     }
175     if (!codeword.hasValidRowNumber()) {
176       if (codeword.isValidRowNumber(rowIndicatorRowNumber)) {
177         codeword.setRowNumber(rowIndicatorRowNumber);
178         invalidRowCounts = 0;
179       } else {
180         ++invalidRowCounts;
181       }
182     }
183     return invalidRowCounts;
184   }
185 
186   private void adjustRowNumbers(int barcodeColumn, int codewordsRow, Codeword[] codewords) {
187     Codeword codeword = codewords[codewordsRow];
188     Codeword[] previousColumnCodewords = detectionResultColumns[barcodeColumn - 1].getCodewords();
189     Codeword[] nextColumnCodewords = previousColumnCodewords;
190     if (detectionResultColumns[barcodeColumn + 1] != null) {
191       nextColumnCodewords = detectionResultColumns[barcodeColumn + 1].getCodewords();
192     }
193 
194     Codeword[] otherCodewords = new Codeword[14];
195 
196     otherCodewords[2] = previousColumnCodewords[codewordsRow];
197     otherCodewords[3] = nextColumnCodewords[codewordsRow];
198 
199     if (codewordsRow > 0) {
200       otherCodewords[0] = codewords[codewordsRow - 1];
201       otherCodewords[4] = previousColumnCodewords[codewordsRow - 1];
202       otherCodewords[5] = nextColumnCodewords[codewordsRow - 1];
203     }
204     if (codewordsRow > 1) {
205       otherCodewords[8] = codewords[codewordsRow - 2];
206       otherCodewords[10] = previousColumnCodewords[codewordsRow - 2];
207       otherCodewords[11] = nextColumnCodewords[codewordsRow - 2];
208     }
209     if (codewordsRow < codewords.length - 1) {
210       otherCodewords[1] = codewords[codewordsRow + 1];
211       otherCodewords[6] = previousColumnCodewords[codewordsRow + 1];
212       otherCodewords[7] = nextColumnCodewords[codewordsRow + 1];
213     }
214     if (codewordsRow < codewords.length - 2) {
215       otherCodewords[9] = codewords[codewordsRow + 2];
216       otherCodewords[12] = previousColumnCodewords[codewordsRow + 2];
217       otherCodewords[13] = nextColumnCodewords[codewordsRow + 2];
218     }
219     for (Codeword otherCodeword : otherCodewords) {
220       if (adjustRowNumber(codeword, otherCodeword)) {
221         return;
222       }
223     }
224   }
225 
226   /**
227    * @return true, if row number was adjusted, false otherwise
228    */
229   private static boolean adjustRowNumber(Codeword codeword, Codeword otherCodeword) {
230     if (otherCodeword == null) {
231       return false;
232     }
233     if (otherCodeword.hasValidRowNumber() && otherCodeword.getBucket() == codeword.getBucket()) {
234       codeword.setRowNumber(otherCodeword.getRowNumber());
235       return true;
236     }
237     return false;
238   }

268   @Override
269   public String toString() {
270     DetectionResultColumn rowIndicatorColumn = detectionResultColumns[0];
271     if (rowIndicatorColumn == null) {
272       rowIndicatorColumn = detectionResultColumns[barcodeColumnCount + 1];
273     }
274     Formatter formatter = new Formatter();
275     for (int codewordsRow = 0; codewordsRow < rowIndicatorColumn.getCodewords().length; codewordsRow++) {
276       formatter.format("CW %3d:", codewordsRow);
277       for (int barcodeColumn = 0; barcodeColumn < barcodeColumnCount + 2; barcodeColumn++) {
278         if (detectionResultColumns[barcodeColumn] == null) {
279           formatter.format("    |   ");
280           continue;
281         }
282         Codeword codeword = detectionResultColumns[barcodeColumn].getCodewords()[codewordsRow];
283         if (codeword == null) {
284           formatter.format("    |   ");
285           continue;
286         }
287         formatter.format(" %3d|%3d", codeword.getRowNumber(), codeword.getValue());
288       }
289       formatter.format("%n");
290     }
291     String result = formatter.toString();
292     formatter.close();
293     return result;
294   }
