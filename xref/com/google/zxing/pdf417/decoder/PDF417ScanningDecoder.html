47    // TODO don't pass in minCodewordWidth and maxCodewordWidth, pass in barcode columns for start and stop pattern
48    // columns. That way width can be deducted from the pattern column.
49    // This approach also allows to detect more details about the barcode, e.g. if a bar type (white or black) is wider 
50    // than it should be. This can happen if the scanner used a bad blackpoint.
51    public static DecoderResult decode(BitMatrix image,
52                                       ResultPoint imageTopLeft,
53                                       ResultPoint imageBottomLeft,
54                                       ResultPoint imageTopRight,
55                                       ResultPoint imageBottomRight,
56                                       int minCodewordWidth,
57                                       int maxCodewordWidth) throws NotFoundException, FormatException, ChecksumException {
58      BoundingBox boundingBox = new BoundingBox(image, imageTopLeft, imageBottomLeft, imageTopRight, imageBottomRight);
59      DetectionResultRowIndicatorColumn leftRowIndicatorColumn = null;
60      DetectionResultRowIndicatorColumn rightRowIndicatorColumn = null;
61      DetectionResult detectionResult = null;
62      for (int i = 0; i < 2; i++) {
63        if (imageTopLeft != null) {
64          leftRowIndicatorColumn = getRowIndicatorColumn(image, boundingBox, imageTopLeft, true, minCodewordWidth,
65              maxCodewordWidth);
66        }
67        if (imageTopRight != null) {
68          rightRowIndicatorColumn = getRowIndicatorColumn(image, boundingBox, imageTopRight, false, minCodewordWidth,
69              maxCodewordWidth);
70        }
71        detectionResult = merge(leftRowIndicatorColumn, rightRowIndicatorColumn);
72        if (detectionResult == null) {
73          throw NotFoundException.getNotFoundInstance();
74        }
75        if (i == 0 && detectionResult.getBoundingBox() != null &&
76            (detectionResult.getBoundingBox().getMinY() < boundingBox.getMinY() || detectionResult.getBoundingBox()
77                .getMaxY() > boundingBox.getMaxY())) {
78          boundingBox = detectionResult.getBoundingBox();
79        } else {
80          detectionResult.setBoundingBox(boundingBox);
81          break;
82        }
83      }
84      int maxBarcodeColumn = detectionResult.getBarcodeColumnCount() + 1;
85      detectionResult.setDetectionResultColumn(0, leftRowIndicatorColumn);
86      detectionResult.setDetectionResultColumn(maxBarcodeColumn, rightRowIndicatorColumn);
87  
88      boolean leftToRight = leftRowIndicatorColumn != null;
89      for (int barcodeColumnCount = 1; barcodeColumnCount <= maxBarcodeColumn; barcodeColumnCount++) {
90        int barcodeColumn = leftToRight ? barcodeColumnCount : maxBarcodeColumn - barcodeColumnCount;
91        if (detectionResult.getDetectionResultColumn(barcodeColumn) != null) {
92          // This will be the case for the opposite row indicator column, which doesn't need to be decoded again.
93          continue;
94        }
95        DetectionResultColumn detectionResultColumn;
96        if (barcodeColumn == 0 || barcodeColumn == maxBarcodeColumn) {
97          detectionResultColumn = new DetectionResultRowIndicatorColumn(boundingBox, barcodeColumn == 0);
98        } else {
99          detectionResultColumn = new DetectionResultColumn(boundingBox);
100       }
101       detectionResult.setDetectionResultColumn(barcodeColumn, detectionResultColumn);
102       int startColumn = -1;
103       int previousStartColumn = startColumn;
104       // TODO start at a row for which we know the start position, then detect upwards and downwards from there.
105       for (int imageRow = boundingBox.getMinY(); imageRow <= boundingBox.getMaxY(); imageRow++) {
106         startColumn = getStartColumn(detectionResult, barcodeColumn, imageRow, leftToRight);
107         if (startColumn < 0 || startColumn > boundingBox.getMaxX()) {
108           if (previousStartColumn == -1) {
109             continue;
110           }
111           startColumn = previousStartColumn;
112         }
113         Codeword codeword = detectCodeword(image, boundingBox.getMinX(), boundingBox.getMaxX(), leftToRight,
114             startColumn, imageRow, minCodewordWidth, maxCodewordWidth);
115         if (codeword != null) {
116           detectionResultColumn.setCodeword(imageRow, codeword);
117           previousStartColumn = startColumn;
118           minCodewordWidth = Math.min(minCodewordWidth, codeword.getWidth());
119           maxCodewordWidth = Math.max(maxCodewordWidth, codeword.getWidth());
120         }
121       }
122     }
123     return createDecoderResult(detectionResult);
124   }

126   private static DetectionResult merge(DetectionResultRowIndicatorColumn leftRowIndicatorColumn,
127                                        DetectionResultRowIndicatorColumn rightRowIndicatorColumn)
128       throws NotFoundException {
129     if (leftRowIndicatorColumn == null && rightRowIndicatorColumn == null) {
130       return null;
131     }
132     BarcodeMetadata barcodeMetadata = getBarcodeMetadata(leftRowIndicatorColumn, rightRowIndicatorColumn);
133     if (barcodeMetadata == null) {
134       return null;
135     }
136     BoundingBox boundingBox = BoundingBox.merge(adjustBoundingBox(leftRowIndicatorColumn),
137         adjustBoundingBox(rightRowIndicatorColumn));
138     return new DetectionResult(barcodeMetadata, boundingBox);
139   }

141   private static BoundingBox adjustBoundingBox(DetectionResultRowIndicatorColumn rowIndicatorColumn)
142       throws NotFoundException {
143     if (rowIndicatorColumn == null) {
144       return null;
145     }
146     int[] rowHeights = rowIndicatorColumn.getRowHeights();
147     if (rowHeights == null) {
148       return null;
149     }
150     int maxRowHeight = getMax(rowHeights);
151     int missingStartRows = 0;
152     for (int rowHeight : rowHeights) {
153       missingStartRows += maxRowHeight - rowHeight;
154       if (rowHeight > 0) {
155         break;
156       }
157     }
158     Codeword[] codewords = rowIndicatorColumn.getCodewords();
159     for (int row = 0; missingStartRows > 0 && codewords[row] == null; row++) {
160       missingStartRows--;
161     }
162     int missingEndRows = 0;
163     for (int row = rowHeights.length - 1; row >= 0; row--) {
164       missingEndRows += maxRowHeight - rowHeights[row];
165       if (rowHeights[row] > 0) {
166         break;
167       }
168     }
169     for (int row = codewords.length - 1; missingEndRows > 0 && codewords[row] == null; row--) {
170       missingEndRows--;
171     }
172     return rowIndicatorColumn.getBoundingBox().addMissingRows(missingStartRows, missingEndRows,
173         rowIndicatorColumn.isLeft());
174   }
175 
176   private static int getMax(int[] values) {
177     int maxValue = -1;
178     for (int value : values) {
179       maxValue = Math.max(maxValue, value);
180     }
181     return maxValue;
182   }

184   private static BarcodeMetadata getBarcodeMetadata(DetectionResultRowIndicatorColumn leftRowIndicatorColumn,
185                                                     DetectionResultRowIndicatorColumn rightRowIndicatorColumn) {
186     if (leftRowIndicatorColumn == null || leftRowIndicatorColumn.getBarcodeMetadata() == null) {
187       return rightRowIndicatorColumn == null ? null : rightRowIndicatorColumn.getBarcodeMetadata();
188     }
189     if (rightRowIndicatorColumn == null || rightRowIndicatorColumn.getBarcodeMetadata() == null) {
190       return leftRowIndicatorColumn == null ? null : leftRowIndicatorColumn.getBarcodeMetadata();
191     }
192     BarcodeMetadata leftBarcodeMetadata = leftRowIndicatorColumn.getBarcodeMetadata();
193     BarcodeMetadata rightBarcodeMetadata = rightRowIndicatorColumn.getBarcodeMetadata();
194 
195     if (leftBarcodeMetadata.getColumnCount() != rightBarcodeMetadata.getColumnCount() &&
196         leftBarcodeMetadata.getErrorCorrectionLevel() != rightBarcodeMetadata.getErrorCorrectionLevel() &&
197         leftBarcodeMetadata.getRowCount() != rightBarcodeMetadata.getRowCount()) {
198       return null;
199     }
200     return leftBarcodeMetadata;
201   }

203   private static DetectionResultRowIndicatorColumn getRowIndicatorColumn(BitMatrix image,
204                                                                          BoundingBox boundingBox,
205                                                                          ResultPoint startPoint,
206                                                                          boolean leftToRight,
207                                                                          int minCodewordWidth,
208                                                                          int maxCodewordWidth) {
209     DetectionResultRowIndicatorColumn rowIndicatorColumn = new DetectionResultRowIndicatorColumn(boundingBox,
210         leftToRight);
211     for (int i = 0; i < 2; i++) {
212       int increment = i == 0 ? 1 : -1;
213       int startColumn = (int) startPoint.getX();
214       for (int imageRow = (int) startPoint.getY(); imageRow <= boundingBox.getMaxY() &&
215           imageRow >= boundingBox.getMinY(); imageRow += increment) {
216         Codeword codeword = detectCodeword(image, 0, image.getWidth(), leftToRight, startColumn, imageRow,
217             minCodewordWidth, maxCodewordWidth);
218         if (codeword != null) {
219           rowIndicatorColumn.setCodeword(imageRow, codeword);
220           if (leftToRight) {
221             startColumn = codeword.getStartX();
222           } else {
223             startColumn = codeword.getEndX();
224           }
225         }
226       }
227     }
228     return rowIndicatorColumn;
229   }

231   private static void adjustCodewordCount(DetectionResult detectionResult, BarcodeValue[][] barcodeMatrix)
232       throws NotFoundException {
233     int[] numberOfCodewords = barcodeMatrix[0][1].getValue();
234     int calculatedNumberOfCodewords = detectionResult.getBarcodeColumnCount() *
235         detectionResult.getBarcodeRowCount() -
236         getNumberOfECCodeWords(detectionResult.getBarcodeECLevel());
237     if (numberOfCodewords.length == 0) {
238       if (calculatedNumberOfCodewords < 1 || calculatedNumberOfCodewords > PDF417Common.MAX_CODEWORDS_IN_BARCODE) {
239         throw NotFoundException.getNotFoundInstance();
240       }
241       barcodeMatrix[0][1].setValue(calculatedNumberOfCodewords);
242     } else if (numberOfCodewords[0] != calculatedNumberOfCodewords) {
243       // The calculated one is more reliable as it is derived from the row indicator columns
244       barcodeMatrix[0][1].setValue(calculatedNumberOfCodewords);
245     }
246   }

248   private static DecoderResult createDecoderResult(DetectionResult detectionResult) throws FormatException,
249       ChecksumException, NotFoundException {
250     BarcodeValue[][] barcodeMatrix = createBarcodeMatrix(detectionResult);
251     adjustCodewordCount(detectionResult, barcodeMatrix);
252     Collection<Integer> erasures = new ArrayList<>();
253     int[] codewords = new int[detectionResult.getBarcodeRowCount() * detectionResult.getBarcodeColumnCount()];
254     List<int[]> ambiguousIndexValuesList = new ArrayList<>();
255     List<Integer> ambiguousIndexesList = new ArrayList<>();
256     for (int row = 0; row < detectionResult.getBarcodeRowCount(); row++) {
257       for (int column = 0; column < detectionResult.getBarcodeColumnCount(); column++) {
258         int[] values = barcodeMatrix[row][column + 1].getValue();
259         int codewordIndex = row * detectionResult.getBarcodeColumnCount() + column;
260         if (values.length == 0) {
261           erasures.add(codewordIndex);
262         } else if (values.length == 1) {
263           codewords[codewordIndex] = values[0];
264         } else {
265           ambiguousIndexesList.add(codewordIndex);
266           ambiguousIndexValuesList.add(values);
267         }
268       }
269     }
270     int[][] ambiguousIndexValues = new int[ambiguousIndexValuesList.size()][];
271     for (int i = 0; i < ambiguousIndexValues.length; i++) {
272       ambiguousIndexValues[i] = ambiguousIndexValuesList.get(i);
273     }
274     return createDecoderResultFromAmbiguousValues(detectionResult.getBarcodeECLevel(), codewords,
275         PDF417Common.toIntArray(erasures), PDF417Common.toIntArray(ambiguousIndexesList), ambiguousIndexValues);
276   }

278   /**
279    * This method deals with the fact, that the decoding process doesn't always yield a single most likely value. The
280    * current error correction implementation doesn't deal with erasures very well, so it's better to provide a value
281    * for these ambiguous codewords instead of treating it as an erasure. The problem is that we don't know which of
282    * the ambiguous values to choose. We try decode using the first value, and if that fails, we use another of the
283    * ambiguous values and try to decode again. This usually only happens on very hard to read and decode barcodes,
284    * so decoding the normal barcodes is not affected by this. 
285    *
286    * @param erasureArray contains the indexes of erasures
287    * @param ambiguousIndexes array with the indexes that have more than one most likely value
288    * @param ambiguousIndexValues two dimensional array that contains the ambiguous values. The first dimension must
289    * be the same length as the ambiguousIndexes array
290    */
291   private static DecoderResult createDecoderResultFromAmbiguousValues(int ecLevel,
292                                                                       int[] codewords,
293                                                                       int[] erasureArray,
294                                                                       int[] ambiguousIndexes,
295                                                                       int[][] ambiguousIndexValues)
296       throws FormatException, ChecksumException {
297     int[] ambiguousIndexCount = new int[ambiguousIndexes.length];
298 
299     int tries = 100;
300     while (tries-- > 0) {
301       for (int i = 0; i < ambiguousIndexCount.length; i++) {
302         codewords[ambiguousIndexes[i]] = ambiguousIndexValues[i][ambiguousIndexCount[i]];
303       }
304       try {
305         return decodeCodewords(codewords, ecLevel, erasureArray);
306       } catch (ChecksumException ignored) {
307         //
308       }
309       if (ambiguousIndexCount.length == 0) {
310         throw ChecksumException.getChecksumInstance();
311       }
312       for (int i = 0; i < ambiguousIndexCount.length; i++) {
313         if (ambiguousIndexCount[i] < ambiguousIndexValues[i].length - 1) {
314           ambiguousIndexCount[i]++;
315           break;
316         } else {
317           ambiguousIndexCount[i] = 0;
318           if (i == ambiguousIndexCount.length - 1) {
319             throw ChecksumException.getChecksumInstance();
320           }
321         }
322       }
323     }
324     throw ChecksumException.getChecksumInstance();
325   }

327   private static BarcodeValue[][] createBarcodeMatrix(DetectionResult detectionResult) {
328     BarcodeValue[][] barcodeMatrix = new BarcodeValue[detectionResult.getBarcodeRowCount()][detectionResult
329         .getBarcodeColumnCount() + 2];
330     for (int row = 0; row < barcodeMatrix.length; row++) {
331       for (int column = 0; column < barcodeMatrix[row].length; column++) {
332         barcodeMatrix[row][column] = new BarcodeValue();
333       }
334     }
335 
336     int column = -1;
337     for (DetectionResultColumn detectionResultColumn : detectionResult.getDetectionResultColumns()) {
338       column++;
339       if (detectionResultColumn == null) {
340         continue;
341       }
342       for (Codeword codeword : detectionResultColumn.getCodewords()) {
343         if (codeword == null || codeword.getRowNumber() == -1) {
344           continue;
345         }
346         barcodeMatrix[codeword.getRowNumber()][column].setValue(codeword.getValue());
347       }
348     }
349     return barcodeMatrix;
350   }

356   private static int getStartColumn(DetectionResult detectionResult,
357                                     int barcodeColumn,
358                                     int imageRow,
359                                     boolean leftToRight) {
360     int offset = leftToRight ? 1 : -1;
361     Codeword codeword = null;
362     if (isValidBarcodeColumn(detectionResult, barcodeColumn - offset)) {
363       codeword = detectionResult.getDetectionResultColumn(barcodeColumn - offset).getCodeword(imageRow);
364     }
365     if (codeword != null) {
366       return leftToRight ? codeword.getEndX() : codeword.getStartX();
367     }
368     codeword = detectionResult.getDetectionResultColumn(barcodeColumn).getCodewordNearby(imageRow);
369     if (codeword != null) {
370       return leftToRight ? codeword.getStartX() : codeword.getEndX();
371     }
372     if (isValidBarcodeColumn(detectionResult, barcodeColumn - offset)) {
373       codeword = detectionResult.getDetectionResultColumn(barcodeColumn - offset).getCodewordNearby(imageRow);
374     }
375     if (codeword != null) {
376       return leftToRight ? codeword.getEndX() : codeword.getStartX();
377     }
378     int skippedColumns = 0;
379 
380     while (isValidBarcodeColumn(detectionResult, barcodeColumn - offset)) {
381       barcodeColumn -= offset;
382       for (Codeword previousRowCodeword : detectionResult.getDetectionResultColumn(barcodeColumn).getCodewords()) {
383         if (previousRowCodeword != null) {
384           return (leftToRight ? previousRowCodeword.getEndX() : previousRowCodeword.getStartX()) +
385               offset *
386               skippedColumns *
387               (previousRowCodeword.getEndX() - previousRowCodeword.getStartX());
388         }
389       }
390       skippedColumns++;
391     }
392     return leftToRight ? detectionResult.getBoundingBox().getMinX() : detectionResult.getBoundingBox().getMaxX();
393   }

395   private static Codeword detectCodeword(BitMatrix image,
396                                          int minColumn,
397                                          int maxColumn,
398                                          boolean leftToRight,
399                                          int startColumn,
400                                          int imageRow,
401                                          int minCodewordWidth,
402                                          int maxCodewordWidth) {
403     startColumn = adjustCodewordStartColumn(image, minColumn, maxColumn, leftToRight, startColumn, imageRow);
404     // we usually know fairly exact now how long a codeword is. We should provide minimum and maximum expected length
405     // and try to adjust the read pixels, e.g. remove single pixel errors or try to cut off exceeding pixels.
406     // min and maxCodewordWidth should not be used as they are calculated for the whole barcode an can be inaccurate
407     // for the current position
408     int[] moduleBitCount = getModuleBitCount(image, minColumn, maxColumn, leftToRight, startColumn, imageRow);
409     if (moduleBitCount == null) {
410       return null;
411     }
412     int endColumn;
413     int codewordBitCount = PDF417Common.getBitCountSum(moduleBitCount);
414     if (leftToRight) {
415       endColumn = startColumn + codewordBitCount;
416     } else {
417       for (int i = 0; i < moduleBitCount.length >> 1; i++) {
418         int tmpCount = moduleBitCount[i];
419         moduleBitCount[i] = moduleBitCount[moduleBitCount.length - 1 - i];
420         moduleBitCount[moduleBitCount.length - 1 - i] = tmpCount;
421       }
422       endColumn = startColumn;
423       startColumn = endColumn - codewordBitCount;
424     }
425     // TODO implement check for width and correction of black and white bars
426     // use start (and maybe stop pattern) to determine if blackbars are wider than white bars. If so, adjust.
427     // should probably done only for codewords with a lot more than 17 bits. 
428     // The following fixes 10-1.png, which has wide black bars and small white bars
429     //    for (int i = 0; i < moduleBitCount.length; i++) {
430     //      if (i % 2 == 0) {
431     //        moduleBitCount[i]--;
432     //      } else {
433     //        moduleBitCount[i]++;
434     //      }
435     //    }
436 
437     // We could also use the width of surrounding codewords for more accurate results, but this seems
438     // sufficient for now
439     if (!checkCodewordSkew(codewordBitCount, minCodewordWidth, maxCodewordWidth)) {
440       // We could try to use the startX and endX position of the codeword in the same column in the previous row,
441       // create the bit count from it and normalize it to 8. This would help with single pixel errors.
442       return null;
443     }
444 
445     int decodedValue = PDF417CodewordDecoder.getDecodedValue(moduleBitCount);
446     int codeword = PDF417Common.getCodeword(decodedValue);
447     if (codeword == -1) {
448       return null;
449     }
450     return new Codeword(startColumn, endColumn, getCodewordBucketNumber(decodedValue), codeword);
451   }

453   private static int[] getModuleBitCount(BitMatrix image,
454                                          int minColumn,
455                                          int maxColumn,
456                                          boolean leftToRight,
457                                          int startColumn,
458                                          int imageRow) {
459     int imageColumn = startColumn;
460     int[] moduleBitCount = new int[8];
461     int moduleNumber = 0;
462     int increment = leftToRight ? 1 : -1;
463     boolean previousPixelValue = leftToRight;
464     while (((leftToRight && imageColumn < maxColumn) || (!leftToRight && imageColumn >= minColumn)) &&
465         moduleNumber < moduleBitCount.length) {
466       if (image.get(imageColumn, imageRow) == previousPixelValue) {
467         moduleBitCount[moduleNumber]++;
468         imageColumn += increment;
469       } else {
470         moduleNumber++;
471         previousPixelValue = !previousPixelValue;
472       }
473     }
474     if (moduleNumber == moduleBitCount.length ||
475         (((leftToRight && imageColumn == maxColumn) || (!leftToRight && imageColumn == minColumn)) && moduleNumber == moduleBitCount.length - 1)) {
476       return moduleBitCount;
477     }
478     return null;
479   }

481   private static int getNumberOfECCodeWords(int barcodeECLevel) {
482     return 2 << barcodeECLevel;

485   private static int adjustCodewordStartColumn(BitMatrix image,
486                                                int minColumn,
487                                                int maxColumn,
488                                                boolean leftToRight,
489                                                int codewordStartColumn,
490                                                int imageRow) {
491     int correctedStartColumn = codewordStartColumn;
492     int increment = leftToRight ? -1 : 1;
493     // there should be no black pixels before the start column. If there are, then we need to start earlier.
494     for (int i = 0; i < 2; i++) {
495       while (((leftToRight && correctedStartColumn >= minColumn) || (!leftToRight && correctedStartColumn < maxColumn)) &&
496           leftToRight == image.get(correctedStartColumn, imageRow)) {
497         if (Math.abs(codewordStartColumn - correctedStartColumn) > CODEWORD_SKEW_SIZE) {
498           return codewordStartColumn;
499         }
500         correctedStartColumn += increment;
501       }
502       increment = -increment;
503       leftToRight = !leftToRight;
504     }
505     return correctedStartColumn;
506   }

508   private static boolean checkCodewordSkew(int codewordSize, int minCodewordWidth, int maxCodewordWidth) {
509     return minCodewordWidth - CODEWORD_SKEW_SIZE <= codewordSize &&
510         codewordSize <= maxCodewordWidth + CODEWORD_SKEW_SIZE;
511   }

513   private static DecoderResult decodeCodewords(int[] codewords, int ecLevel, int[] erasures) throws FormatException,
514       ChecksumException {
515     if (codewords.length == 0) {
516       throw FormatException.getFormatInstance();
517     }
518 
519     int numECCodewords = 1 << (ecLevel + 1);
520     int correctedErrorsCount = correctErrors(codewords, erasures, numECCodewords);
521     verifyCodewordCount(codewords, numECCodewords);
522 
523     // Decode the codewords
524     DecoderResult decoderResult = DecodedBitStreamParser.decode(codewords, String.valueOf(ecLevel));
525     decoderResult.setErrorsCorrected(correctedErrorsCount);
526     decoderResult.setErasures(erasures.length);
527     return decoderResult;
528   }

530   /**
531    * <p>Given data and error-correction codewords received, possibly corrupted by errors, attempts to
532    * correct the errors in-place.</p>
533    *
534    * @param codewords   data and error correction codewords
535    * @param erasures positions of any known erasures
536    * @param numECCodewords number of error correction codewords that are available in codewords
537    * @throws ChecksumException if error correction fails
538    */
539   private static int correctErrors(int[] codewords, int[] erasures, int numECCodewords) throws ChecksumException {
540     if (erasures != null &&
541         erasures.length > numECCodewords / 2 + MAX_ERRORS ||
542         numECCodewords < 0 ||
543         numECCodewords > MAX_EC_CODEWORDS) {
544       // Too many errors or EC Codewords is corrupted
545       throw ChecksumException.getChecksumInstance();
546     }
547     return errorCorrection.decode(codewords, numECCodewords, erasures);
548   }

550   /**
551    * Verify that all is OK with the codeword array.
552    */
553   private static void verifyCodewordCount(int[] codewords, int numECCodewords) throws FormatException {
554     if (codewords.length < 4) {
555       // Codeword array size should be at least 4 allowing for
556       // Count CW, At least one Data CW, Error Correction CW, Error Correction CW
557       throw FormatException.getFormatInstance();
558     }
559     // The first codeword, the Symbol Length Descriptor, shall always encode the total number of data
560     // codewords in the symbol, including the Symbol Length Descriptor itself, data codewords and pad
561     // codewords, but excluding the number of error correction codewords.
562     int numberOfCodewords = codewords[0];
563     if (numberOfCodewords > codewords.length) {
564       throw FormatException.getFormatInstance();
565     }
566     if (numberOfCodewords == 0) {
567       // Reset to the length of the array - 8 (Allow for at least level 3 Error Correction (8 Error Codewords)
568       if (numECCodewords < codewords.length) {
569         codewords[0] = codewords.length - numECCodewords;
570       } else {
571         throw FormatException.getFormatInstance();
572       }
573     }
574   }

576   private static int[] getBitCountForCodeword(int codeword) {
577     int[] result = new int[8];
578     int previousValue = 0;
579     int i = result.length - 1;
580     while (true) {
581       if ((codeword & 0x1) != previousValue) {
582         previousValue = codeword & 0x1;
583         i--;
584         if (i < 0) {
585           break;
586         }
587       }
588       result[i]++;
589       codeword >>= 1;
590     }
591     return result;
592   }

594   private static int getCodewordBucketNumber(int codeword) {
595     return getCodewordBucketNumber(getBitCountForCodeword(codeword));
596   }

598   private static int getCodewordBucketNumber(int[] moduleBitCount) {
599     return (moduleBitCount[0] - moduleBitCount[2] + moduleBitCount[4] - moduleBitCount[6] + 9) % 9;
600   }

602   public static String toString(BarcodeValue[][] barcodeMatrix) {
603     Formatter formatter = new Formatter();
604     for (int row = 0; row < barcodeMatrix.length; row++) {
605       formatter.format("Row %2d: ", row);
606       for (int column = 0; column < barcodeMatrix[row].length; column++) {
607         BarcodeValue barcodeValue = barcodeMatrix[row][column];
608         if (barcodeValue.getValue().length == 0) {
609           formatter.format("        ", (Object[]) null);
610         } else {
611           formatter.format("%4d(%2d)", barcodeValue.getValue()[0],
612               barcodeValue.getConfidence(barcodeValue.getValue()[0]));
613         }
614       }
615       formatter.format("\n");
616     }
617     String result = formatter.toString();
618     formatter.close();
619     return result;
620   }
