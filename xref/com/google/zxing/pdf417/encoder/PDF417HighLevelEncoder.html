135   static {
136     //Construct inverse lookups
137     Arrays.fill(MIXED, (byte) -1);
138     for (byte i = 0; i < TEXT_MIXED_RAW.length; i++) {
139       byte b = TEXT_MIXED_RAW[i];
140       if (b > 0) {
141         MIXED[b] = i;
142       }
143     }
144     Arrays.fill(PUNCTUATION, (byte) -1);
145     for (byte i = 0; i < TEXT_PUNCTUATION_RAW.length; i++) {
146       byte b = TEXT_PUNCTUATION_RAW[i];
147       if (b > 0) {
148         PUNCTUATION[b] = i;
149       }
150     }
151   }

153   /**
154    * Performs high-level encoding of a PDF417 message using the algorithm described in annex P
155    * of ISO/IEC 15438:2001(E). If byte compaction has been selected, then only byte compaction
156    * is used.
157    *
158    * @param msg the message
159    * @param compaction compaction mode to use
160    * @param encoding character encoding used to encode in default or byte compaction
161    *  or {@code null} for default / not applicable
162    * @return the encoded message (the char values range from 0 to 928)
163    */
164   static String encodeHighLevel(String msg, Compaction compaction, Charset encoding) throws WriterException {
165 
166     //the codewords 0..928 are encoded as Unicode characters
167     StringBuilder sb = new StringBuilder(msg.length());
168 
169     if (encoding != null && !DEFAULT_ENCODING_NAMES.contains(encoding.name())) {
170       CharacterSetECI eci = CharacterSetECI.getCharacterSetECIByName(encoding.name());
171       if (eci != null) {
172         encodingECI(eci.getValue(), sb);
173       }
174     }
175 
176     int len = msg.length();
177     int p = 0;
178     int textSubMode = SUBMODE_ALPHA;
179 
180     // User selected encoding mode
181     byte[] bytes = null; //Fill later and only if needed
182     if (compaction == Compaction.TEXT) {
183       encodeText(msg, p, len, sb, textSubMode);
184 
185     } else if (compaction == Compaction.BYTE) {
186       bytes = toBytes(msg, encoding);
187       encodeBinary(bytes, p, bytes.length, BYTE_COMPACTION, sb);
188 
189     } else if (compaction == Compaction.NUMERIC) {
190       sb.append((char) LATCH_TO_NUMERIC);
191       encodeNumeric(msg, p, len, sb);
192 
193     } else {
194       int encodingMode = TEXT_COMPACTION; //Default mode, see 4.4.2.1
195       while (p < len) {
196         int n = determineConsecutiveDigitCount(msg, p);
197         if (n >= 13) {
198           sb.append((char) LATCH_TO_NUMERIC);
199           encodingMode = NUMERIC_COMPACTION;
200           textSubMode = SUBMODE_ALPHA; //Reset after latch
201           encodeNumeric(msg, p, n, sb);
202           p += n;
203         } else {
204           int t = determineConsecutiveTextCount(msg, p);
205           if (t >= 5 || n == len) {
206             if (encodingMode != TEXT_COMPACTION) {
207               sb.append((char) LATCH_TO_TEXT);
208               encodingMode = TEXT_COMPACTION;
209               textSubMode = SUBMODE_ALPHA; //start with submode alpha after latch
210             }
211             textSubMode = encodeText(msg, p, t, sb, textSubMode);
212             p += t;
213           } else {
214             if (bytes == null) {
215               bytes = toBytes(msg, encoding);
216             }
217             int b = determineConsecutiveBinaryCount(msg, bytes, p);
218             if (b == 0) {
219               b = 1;
220             }
221             if (b == 1 && encodingMode == TEXT_COMPACTION) {
222               //Switch for one byte (instead of latch)
223               encodeBinary(bytes, p, 1, TEXT_COMPACTION, sb);
224             } else {
225               //Mode latch performed by encodeBinary()
226               encodeBinary(bytes, p, b, encodingMode, sb);
227               encodingMode = BYTE_COMPACTION;
228               textSubMode = SUBMODE_ALPHA; //Reset after latch
229             }
230             p += b;
231           }
232         }
233       }
234     }
235 
236     return sb.toString();
237   }

239   private static byte[] toBytes(String msg, Charset encoding) throws WriterException {
240     // Defer instantiating default Charset until needed, since it may be for an unsupported
241     // encoding. For example the default of Cp437 doesn't seem to exist on Android.
242     if (encoding == null) {
243       for (String encodingName : DEFAULT_ENCODING_NAMES) {
244         try {
245           encoding = Charset.forName(encodingName);
246         } catch (UnsupportedCharsetException uce) {
247           // continue
248         }
249       }
250       if (encoding == null) {
251         throw new WriterException("No support for any encoding: " + DEFAULT_ENCODING_NAMES);
252       }
253     }
254     return msg.getBytes(encoding);
255   }

257   /**
258    * Encode parts of the message using Text Compaction as described in ISO/IEC 15438:2001(E),
259    * chapter 4.4.2.
260    *
261    * @param msg            the message
262    * @param startpos       the start position within the message
263    * @param count          the number of characters to encode
264    * @param sb             receives the encoded codewords
265    * @param initialSubmode should normally be SUBMODE_ALPHA
266    * @return the text submode in which this method ends
267    */
268   private static int encodeText(CharSequence msg,
269                                 int startpos,
270                                 int count,
271                                 StringBuilder sb,
272                                 int initialSubmode) {
273     StringBuilder tmp = new StringBuilder(count);
274     int submode = initialSubmode;
275     int idx = 0;
276     while (true) {
277       char ch = msg.charAt(startpos + idx);
278       switch (submode) {
279         case SUBMODE_ALPHA:
280           if (isAlphaUpper(ch)) {
281             if (ch == ' ') {
282               tmp.append((char) 26); //space
283             } else {
284               tmp.append((char) (ch - 65));
285             }
286           } else {
287             if (isAlphaLower(ch)) {
288               submode = SUBMODE_LOWER;
289               tmp.append((char) 27); //ll
290               continue;
291             } else if (isMixed(ch)) {
292               submode = SUBMODE_MIXED;
293               tmp.append((char) 28); //ml
294               continue;
295             } else {
296               tmp.append((char) 29); //ps
297               tmp.append((char) PUNCTUATION[ch]);
298               break;
299             }
300           }
301           break;
302         case SUBMODE_LOWER:
303           if (isAlphaLower(ch)) {
304             if (ch == ' ') {
305               tmp.append((char) 26); //space
306             } else {
307               tmp.append((char) (ch - 97));
308             }
309           } else {
310             if (isAlphaUpper(ch)) {
311               tmp.append((char) 27); //as
312               tmp.append((char) (ch - 65));
313               //space cannot happen here, it is also in "Lower"
314               break;
315             } else if (isMixed(ch)) {
316               submode = SUBMODE_MIXED;
317               tmp.append((char) 28); //ml
318               continue;
319             } else {
320               tmp.append((char) 29); //ps
321               tmp.append((char) PUNCTUATION[ch]);
322               break;
323             }
324           }
325           break;
326         case SUBMODE_MIXED:
327           if (isMixed(ch)) {
328             tmp.append((char) MIXED[ch]);
329           } else {
330             if (isAlphaUpper(ch)) {
331               submode = SUBMODE_ALPHA;
332               tmp.append((char) 28); //al
333               continue;
334             } else if (isAlphaLower(ch)) {
335               submode = SUBMODE_LOWER;
336               tmp.append((char) 27); //ll
337               continue;
338             } else {
339               if (startpos + idx + 1 < count) {
340                 char next = msg.charAt(startpos + idx + 1);
341                 if (isPunctuation(next)) {
342                   submode = SUBMODE_PUNCTUATION;
343                   tmp.append((char) 25); //pl
344                   continue;
345                 }
346               }
347               tmp.append((char) 29); //ps
348               tmp.append((char) PUNCTUATION[ch]);
349             }
350           }
351           break;
352         default: //SUBMODE_PUNCTUATION
353           if (isPunctuation(ch)) {
354             tmp.append((char) PUNCTUATION[ch]);
355           } else {
356             submode = SUBMODE_ALPHA;
357             tmp.append((char) 29); //al
358             continue;
359           }
360       }
361       idx++;
362       if (idx >= count) {
363         break;
364       }
365     }
366     char h = 0;
367     int len = tmp.length();
368     for (int i = 0; i < len; i++) {
369       boolean odd = (i % 2) != 0;
370       if (odd) {
371         h = (char) ((h * 30) + tmp.charAt(i));
372         sb.append(h);
373       } else {
374         h = tmp.charAt(i);
375       }
376     }
377     if ((len % 2) != 0) {
378       sb.append((char) ((h * 30) + 29)); //ps
379     }
380     return submode;
381   }

383   /**
384    * Encode parts of the message using Byte Compaction as described in ISO/IEC 15438:2001(E),
385    * chapter 4.4.3. The Unicode characters will be converted to binary using the cp437
386    * codepage.
387    *
388    * @param bytes     the message converted to a byte array
389    * @param startpos  the start position within the message
390    * @param count     the number of bytes to encode
391    * @param startmode the mode from which this method starts
392    * @param sb        receives the encoded codewords
393    */
394   private static void encodeBinary(byte[] bytes,
395                                    int startpos,
396                                    int count,
397                                    int startmode,
398                                    StringBuilder sb) {
399     if (count == 1 && startmode == TEXT_COMPACTION) {
400       sb.append((char) SHIFT_TO_BYTE);
401     } else {
402       boolean sixpack = ((count % 6) == 0);
403       if (sixpack) {
404         sb.append((char)LATCH_TO_BYTE);
405       } else {
406         sb.append((char)LATCH_TO_BYTE_PADDED);
407       }
408     }
409 
410     int idx = startpos;
411     // Encode sixpacks
412     if (count >= 6) {
413       char[] chars = new char[5];
414       while ((startpos + count - idx) >= 6) {
415         long t = 0;
416         for (int i = 0; i < 6; i++) {
417           t <<= 8;
418           t += bytes[idx + i] & 0xff;
419         }
420         for (int i = 0; i < 5; i++) {
421           chars[i] = (char) (t % 900);
422           t /= 900;
423         }
424         for (int i = chars.length - 1; i >= 0; i--) {
425           sb.append(chars[i]);
426         }
427         idx += 6;
428       }
429     }
430     //Encode rest (remaining n<5 bytes if any)
431     for (int i = idx; i < startpos + count; i++) {
432       int ch = bytes[i] & 0xff;
433       sb.append((char) ch);
434     }
435   }

437   private static void encodeNumeric(String msg, int startpos, int count, StringBuilder sb) {
438     int idx = 0;
439     StringBuilder tmp = new StringBuilder(count / 3 + 1);
440     BigInteger num900 = BigInteger.valueOf(900);
441     BigInteger num0 = BigInteger.valueOf(0);
442     while (idx < count - 1) {
443       tmp.setLength(0);
444       int len = Math.min(44, count - idx);
445       String part = '1' + msg.substring(startpos + idx, startpos + idx + len);
446       BigInteger bigint = new BigInteger(part);
447       do {
448         tmp.append((char) bigint.mod(num900).intValue());
449         bigint = bigint.divide(num900);
450       } while (!bigint.equals(num0));
451 
452       //Reverse temporary string
453       for (int i = tmp.length() - 1; i >= 0; i--) {
454         sb.append(tmp.charAt(i));
455       }
456       idx += len;
457     }
458   }

461   private static boolean isDigit(char ch) {
462     return ch >= '0' && ch <= '9';
463   }
464 
465   private static boolean isAlphaUpper(char ch) {
466     return ch == ' ' || (ch >= 'A' && ch <= 'Z');
467   }
468 
469   private static boolean isAlphaLower(char ch) {
470     return ch == ' ' || (ch >= 'a' && ch <= 'z');
471   }
472 
473   private static boolean isMixed(char ch) {
474     return MIXED[ch] != -1;
475   }
476 
477   private static boolean isPunctuation(char ch) {
478     return PUNCTUATION[ch] != -1;
479   }
480 
481   private static boolean isText(char ch) {
482     return ch == '\t' || ch == '\n' || ch == '\r' || (ch >= 32 && ch <= 126);
483   }

485   /**
486    * Determines the number of consecutive characters that are encodable using numeric compaction.
487    *
488    * @param msg      the message
489    * @param startpos the start position within the message
490    * @return the requested character count
491    */
492   private static int determineConsecutiveDigitCount(CharSequence msg, int startpos) {
493     int count = 0;
494     int len = msg.length();
495     int idx = startpos;
496     if (idx < len) {
497       char ch = msg.charAt(idx);
498       while (isDigit(ch) && idx < len) {
499         count++;
500         idx++;
501         if (idx < len) {
502           ch = msg.charAt(idx);
503         }
504       }
505     }
506     return count;
507   }

509   /**
510    * Determines the number of consecutive characters that are encodable using text compaction.
511    *
512    * @param msg      the message
513    * @param startpos the start position within the message
514    * @return the requested character count
515    */
516   private static int determineConsecutiveTextCount(CharSequence msg, int startpos) {
517     int len = msg.length();
518     int idx = startpos;
519     while (idx < len) {
520       char ch = msg.charAt(idx);
521       int numericCount = 0;
522       while (numericCount < 13 && isDigit(ch) && idx < len) {
523         numericCount++;
524         idx++;
525         if (idx < len) {
526           ch = msg.charAt(idx);
527         }
528       }
529       if (numericCount >= 13) {
530         return idx - startpos - numericCount;
531       }
532       if (numericCount > 0) {
533         //Heuristic: All text-encodable chars or digits are binary encodable
534         continue;
535       }
536       ch = msg.charAt(idx);
537 
538       //Check if character is encodable
539       if (!isText(ch)) {
540         break;
541       }
542       idx++;
543     }
544     return idx - startpos;
545   }

547   /**
548    * Determines the number of consecutive characters that are encodable using binary compaction.
549    *
550    * @param msg      the message
551    * @param bytes    the message converted to a byte array
552    * @param startpos the start position within the message
553    * @return the requested character count
554    */
555   private static int determineConsecutiveBinaryCount(CharSequence msg, byte[] bytes, int startpos)
556       throws WriterException {
557     int len = msg.length();
558     int idx = startpos;
559     while (idx < len) {
560       char ch = msg.charAt(idx);
561       int numericCount = 0;
562 
563       while (numericCount < 13 && isDigit(ch)) {
564         numericCount++;
565         //textCount++;
566         int i = idx + numericCount;
567         if (i >= len) {
568           break;
569         }
570         ch = msg.charAt(i);
571       }
572       if (numericCount >= 13) {
573         return idx - startpos;
574       }
575       int textCount = 0;
576       while (textCount < 5 && isText(ch)) {
577         textCount++;
578         int i = idx + textCount;
579         if (i >= len) {
580           break;
581         }
582         ch = msg.charAt(i);
583       }
584       if (textCount >= 5) {
585         return idx - startpos;
586       }
587       ch = msg.charAt(idx);
588 
589       //Check if character is encodable
590       //Sun returns a ASCII 63 (?) for a character that cannot be mapped. Let's hope all
591       //other VMs do the same
592       if (bytes[idx] == 63 && ch != '?') {
593         throw new WriterException("Non-encodable character detected: " + ch + " (Unicode: " + (int) ch + ')');
594       }
595       idx++;
596     }
597     return idx - startpos;
598   }

600   private static void encodingECI(int eci, StringBuilder sb) throws WriterException {
601     if (eci >= 0 && eci < 900) {
602       sb.append((char) ECI_CHARSET);
603       sb.append((char) eci);
604     } else if (eci < 810900) {
605       sb.append((char) ECI_GENERAL_PURPOSE);
606       sb.append((char) (eci / 900 - 1));
607       sb.append((char) (eci % 900));
608     } else if (eci < 811800) {
609       sb.append((char) ECI_USER_DEFINED);
610       sb.append((char) (810900 - eci));
611     } else {
612       throw new WriterException("ECI number not in valid range from 0..811799, but was " + eci);
613     }
614   }
