37    /**
38     * default white space (margin) around the code
39     */
40    static final int WHITE_SPACE = 30;
41  
42    @Override
43    public BitMatrix encode(String contents,
44                            BarcodeFormat format,
45                            int width,
46                            int height,
47                            Map<EncodeHintType,?> hints) throws WriterException {
48      if (format != BarcodeFormat.PDF_417) {
49        throw new IllegalArgumentException("Can only encode PDF_417, but got " + format);
50      }
51  
52      PDF417 encoder = new PDF417();
53      int margin = WHITE_SPACE;
54  
55      if (hints != null) {
56        if (hints.containsKey(EncodeHintType.PDF417_COMPACT)) {
57          encoder.setCompact((Boolean) hints.get(EncodeHintType.PDF417_COMPACT));
58        }
59        if (hints.containsKey(EncodeHintType.PDF417_COMPACTION)) {
60          encoder.setCompaction((Compaction) hints.get(EncodeHintType.PDF417_COMPACTION));
61        }
62        if (hints.containsKey(EncodeHintType.PDF417_DIMENSIONS)) {
63          Dimensions dimensions = (Dimensions) hints.get(EncodeHintType.PDF417_DIMENSIONS);
64          encoder.setDimensions(dimensions.getMaxCols(),
65                                dimensions.getMinCols(),
66                                dimensions.getMaxRows(),
67                                dimensions.getMinRows());
68        }
69        if (hints.containsKey(EncodeHintType.MARGIN)) {
70          margin = ((Number) hints.get(EncodeHintType.MARGIN)).intValue();
71        }
72        if (hints.containsKey(EncodeHintType.CHARACTER_SET)) {
73          String encoding = (String) hints.get(EncodeHintType.CHARACTER_SET);
74          encoder.setEncoding(Charset.forName(encoding));
75        }
76      }
77  
78      return bitMatrixFromEncoder(encoder, contents, width, height, margin);
79    }
80  
81    @Override
82    public BitMatrix encode(String contents,
83                            BarcodeFormat format,
84                            int width,
85                            int height) throws WriterException {
86      return encode(contents, format, width, height, null);
87    }
88  
89    /**
90     * Takes encoder, accounts for width/height, and retrieves bit matrix
91     */
92    private static BitMatrix bitMatrixFromEncoder(PDF417 encoder,
93                                                  String contents,
94                                                  int width,
95                                                  int height,
96                                                  int margin) throws WriterException {
97      int errorCorrectionLevel = 2;
98      encoder.generateBarcodeLogic(contents, errorCorrectionLevel);
99  
100     int lineThickness = 2;
101     int aspectRatio = 4;
102     byte[][] originalScale = encoder.getBarcodeMatrix().getScaledMatrix(lineThickness, aspectRatio * lineThickness);
103     boolean rotated = false;
104     if ((height > width) ^ (originalScale[0].length < originalScale.length)) {
105       originalScale = rotateArray(originalScale);
106       rotated = true;
107     }
108 
109     int scaleX = width / originalScale[0].length;
110     int scaleY = height / originalScale.length;
111 
112     int scale;
113     if (scaleX < scaleY) {
114       scale = scaleX;
115     } else {
116       scale = scaleY;
117     }
118 
119     if (scale > 1) {
120       byte[][] scaledMatrix =
121           encoder.getBarcodeMatrix().getScaledMatrix(scale * lineThickness, scale * aspectRatio * lineThickness);
122       if (rotated) {
123         scaledMatrix = rotateArray(scaledMatrix);
124       }
125       return bitMatrixFrombitArray(scaledMatrix, margin);
126     }
127     return bitMatrixFrombitArray(originalScale, margin);
128   }
129 
130   /**
131    * This takes an array holding the values of the PDF 417
132    *
133    * @param input a byte array of information with 0 is black, and 1 is white
134    * @param margin border around the barcode
135    * @return BitMatrix of the input
136    */
137   private static BitMatrix bitMatrixFrombitArray(byte[][] input, int margin) {
138     // Creates the bitmatrix with extra space for whitespace
139     BitMatrix output = new BitMatrix(input[0].length + 2 * margin, input.length + 2 * margin);
140     output.clear();
141     for (int y = 0, yOutput = output.getHeight() - margin - 1; y < input.length; y++, yOutput--) {
142       for (int x = 0; x < input[0].length; x++) {
143         // Zero is white in the bytematrix
144         if (input[y][x] == 1) {
145           output.set(x + margin, yOutput);
146         }
147       }
148     }
149     return output;
150   }
151 
152   /**
153    * Takes and rotates the it 90 degrees
154    */
155   private static byte[][] rotateArray(byte[][] bitarray) {
156     byte[][] temp = new byte[bitarray[0].length][bitarray.length];
157     for (int ii = 0; ii < bitarray.length; ii++) {
158       // This makes the direction consistent on screen when rotating the
159       // screen;
160       int inverseii = bitarray.length - ii - 1;
161       for (int jj = 0; jj < bitarray[0].length; jj++) {
162         temp[jj][inverseii] = bitarray[ii][jj];
163       }
164     }
165     return temp;
166   }
