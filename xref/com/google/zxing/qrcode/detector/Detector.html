56    /**
57     * <p>Detects a QR Code in an image.</p>
58     *
59     * @return {@link DetectorResult} encapsulating results of detecting a QR Code
60     * @throws NotFoundException if QR Code cannot be found
61     * @throws FormatException if a QR Code cannot be decoded
62     */
63    public DetectorResult detect() throws NotFoundException, FormatException {
64      return detect(null);
65    }

67    /**
68     * <p>Detects a QR Code in an image.</p>
69     *
70     * @param hints optional hints to detector
71     * @return {@link DetectorResult} encapsulating results of detecting a QR Code
72     * @throws NotFoundException if QR Code cannot be found
73     * @throws FormatException if a QR Code cannot be decoded
74     */
75    public final DetectorResult detect(Map<DecodeHintType,?> hints) throws NotFoundException, FormatException {
76  
77      resultPointCallback = hints == null ? null :
78          (ResultPointCallback) hints.get(DecodeHintType.NEED_RESULT_POINT_CALLBACK);
79  
80      FinderPatternFinder finder = new FinderPatternFinder(image, resultPointCallback);
81      FinderPatternInfo info = finder.find(hints);
82  
83      return processFinderPatternInfo(info);
84    }
85  
86    protected final DetectorResult processFinderPatternInfo(FinderPatternInfo info)
87        throws NotFoundException, FormatException {
88  
89      FinderPattern topLeft = info.getTopLeft();
90      FinderPattern topRight = info.getTopRight();
91      FinderPattern bottomLeft = info.getBottomLeft();
92  
93      float moduleSize = calculateModuleSize(topLeft, topRight, bottomLeft);
94      if (moduleSize < 1.0f) {
95        throw NotFoundException.getNotFoundInstance();
96      }
97      int dimension = computeDimension(topLeft, topRight, bottomLeft, moduleSize);
98      Version provisionalVersion = Version.getProvisionalVersionForDimension(dimension);
99      int modulesBetweenFPCenters = provisionalVersion.getDimensionForVersion() - 7;
100 
101     AlignmentPattern alignmentPattern = null;
102     // Anything above version 1 has an alignment pattern
103     if (provisionalVersion.getAlignmentPatternCenters().length > 0) {
104 
105       // Guess where a "bottom right" finder pattern would have been
106       float bottomRightX = topRight.getX() - topLeft.getX() + bottomLeft.getX();
107       float bottomRightY = topRight.getY() - topLeft.getY() + bottomLeft.getY();
108 
109       // Estimate that alignment pattern is closer by 3 modules
110       // from "bottom right" to known top left location
111       float correctionToTopLeft = 1.0f - 3.0f / (float) modulesBetweenFPCenters;
112       int estAlignmentX = (int) (topLeft.getX() + correctionToTopLeft * (bottomRightX - topLeft.getX()));
113       int estAlignmentY = (int) (topLeft.getY() + correctionToTopLeft * (bottomRightY - topLeft.getY()));
114 
115       // Kind of arbitrary -- expand search radius before giving up
116       for (int i = 4; i <= 16; i <<= 1) {
117         try {
118           alignmentPattern = findAlignmentInRegion(moduleSize,
119               estAlignmentX,
120               estAlignmentY,
121               (float) i);
122           break;
123         } catch (NotFoundException re) {
124           // try next round
125         }
126       }
127       // If we didn't find alignment pattern... well try anyway without it
128     }
129 
130     PerspectiveTransform transform =
131         createTransform(topLeft, topRight, bottomLeft, alignmentPattern, dimension);
132 
133     BitMatrix bits = sampleGrid(image, transform, dimension);
134 
135     ResultPoint[] points;
136     if (alignmentPattern == null) {
137       points = new ResultPoint[]{bottomLeft, topLeft, topRight};
138     } else {
139       points = new ResultPoint[]{bottomLeft, topLeft, topRight, alignmentPattern};
140     }
141     return new DetectorResult(bits, points);
142   }

144   private static PerspectiveTransform createTransform(ResultPoint topLeft,
145                                                       ResultPoint topRight,
146                                                       ResultPoint bottomLeft,
147                                                       ResultPoint alignmentPattern,
148                                                       int dimension) {
149     float dimMinusThree = (float) dimension - 3.5f;
150     float bottomRightX;
151     float bottomRightY;
152     float sourceBottomRightX;
153     float sourceBottomRightY;
154     if (alignmentPattern != null) {
155       bottomRightX = alignmentPattern.getX();
156       bottomRightY = alignmentPattern.getY();
157       sourceBottomRightX = dimMinusThree - 3.0f;
158       sourceBottomRightY = sourceBottomRightX;
159     } else {
160       // Don't have an alignment pattern, just make up the bottom-right point
161       bottomRightX = (topRight.getX() - topLeft.getX()) + bottomLeft.getX();
162       bottomRightY = (topRight.getY() - topLeft.getY()) + bottomLeft.getY();
163       sourceBottomRightX = dimMinusThree;
164       sourceBottomRightY = dimMinusThree;
165     }
166 
167     return PerspectiveTransform.quadrilateralToQuadrilateral(
168         3.5f,
169         3.5f,
170         dimMinusThree,
171         3.5f,
172         sourceBottomRightX,
173         sourceBottomRightY,
174         3.5f,
175         dimMinusThree,
176         topLeft.getX(),
177         topLeft.getY(),
178         topRight.getX(),
179         topRight.getY(),
180         bottomRightX,
181         bottomRightY,
182         bottomLeft.getX(),
183         bottomLeft.getY());
184   }

186   private static BitMatrix sampleGrid(BitMatrix image,
187                                       PerspectiveTransform transform,
188                                       int dimension) throws NotFoundException {
189 
190     GridSampler sampler = GridSampler.getInstance();
191     return sampler.sampleGrid(image, dimension, dimension, transform);
192   }

194   /**
195    * <p>Computes the dimension (number of modules on a size) of the QR Code based on the position
196    * of the finder patterns and estimated module size.</p>
197    */
198   private static int computeDimension(ResultPoint topLeft,
199                                       ResultPoint topRight,
200                                       ResultPoint bottomLeft,
201                                       float moduleSize) throws NotFoundException {
202     int tltrCentersDimension = MathUtils.round(ResultPoint.distance(topLeft, topRight) / moduleSize);
203     int tlblCentersDimension = MathUtils.round(ResultPoint.distance(topLeft, bottomLeft) / moduleSize);
204     int dimension = ((tltrCentersDimension + tlblCentersDimension) / 2) + 7;
205     switch (dimension & 0x03) { // mod 4
206       case 0:
207         dimension++;
208         break;
209         // 1? do nothing
210       case 2:
211         dimension--;
212         break;
213       case 3:
214         throw NotFoundException.getNotFoundInstance();
215     }
216     return dimension;
217   }

219   /**
220    * <p>Computes an average estimated module size based on estimated derived from the positions
221    * of the three finder patterns.</p>
222    *
223    * @param topLeft detected top-left finder pattern center
224    * @param topRight detected top-right finder pattern center
225    * @param bottomLeft detected bottom-left finder pattern center
226    * @return estimated module size
227    */
228   protected final float calculateModuleSize(ResultPoint topLeft,
229                                             ResultPoint topRight,
230                                             ResultPoint bottomLeft) {
231     // Take the average
232     return (calculateModuleSizeOneWay(topLeft, topRight) +
233         calculateModuleSizeOneWay(topLeft, bottomLeft)) / 2.0f;
234   }
235 
236   /**
237    * <p>Estimates module size based on two finder patterns -- it uses
238    * {@link #sizeOfBlackWhiteBlackRunBothWays(int, int, int, int)} to figure the
239    * width of each, measuring along the axis between their centers.</p>
240    */
241   private float calculateModuleSizeOneWay(ResultPoint pattern, ResultPoint otherPattern) {
242     float moduleSizeEst1 = sizeOfBlackWhiteBlackRunBothWays((int) pattern.getX(),
243         (int) pattern.getY(),
244         (int) otherPattern.getX(),
245         (int) otherPattern.getY());
246     float moduleSizeEst2 = sizeOfBlackWhiteBlackRunBothWays((int) otherPattern.getX(),
247         (int) otherPattern.getY(),
248         (int) pattern.getX(),
249         (int) pattern.getY());
250     if (Float.isNaN(moduleSizeEst1)) {
251       return moduleSizeEst2 / 7.0f;
252     }
253     if (Float.isNaN(moduleSizeEst2)) {
254       return moduleSizeEst1 / 7.0f;
255     }
256     // Average them, and divide by 7 since we've counted the width of 3 black modules,
257     // and 1 white and 1 black module on either side. Ergo, divide sum by 14.
258     return (moduleSizeEst1 + moduleSizeEst2) / 14.0f;
259   }

261   /**
262    * See {@link #sizeOfBlackWhiteBlackRun(int, int, int, int)}; computes the total width of
263    * a finder pattern by looking for a black-white-black run from the center in the direction
264    * of another point (another finder pattern center), and in the opposite direction too.</p>
265    */
266   private float sizeOfBlackWhiteBlackRunBothWays(int fromX, int fromY, int toX, int toY) {
267 
268     float result = sizeOfBlackWhiteBlackRun(fromX, fromY, toX, toY);
269 
270     // Now count other way -- don't run off image though of course
271     float scale = 1.0f;
272     int otherToX = fromX - (toX - fromX);
273     if (otherToX < 0) {
274       scale = (float) fromX / (float) (fromX - otherToX);
275       otherToX = 0;
276     } else if (otherToX >= image.getWidth()) {
277       scale = (float) (image.getWidth() - 1 - fromX) / (float) (otherToX - fromX);
278       otherToX = image.getWidth() - 1;
279     }
280     int otherToY = (int) (fromY - (toY - fromY) * scale);
281 
282     scale = 1.0f;
283     if (otherToY < 0) {
284       scale = (float) fromY / (float) (fromY - otherToY);
285       otherToY = 0;
286     } else if (otherToY >= image.getHeight()) {
287       scale = (float) (image.getHeight() - 1 - fromY) / (float) (otherToY - fromY);
288       otherToY = image.getHeight() - 1;
289     }
290     otherToX = (int) (fromX + (otherToX - fromX) * scale);
291 
292     result += sizeOfBlackWhiteBlackRun(fromX, fromY, otherToX, otherToY);
293 
294     // Middle pixel is double-counted this way; subtract 1
295     return result - 1.0f;
296   }

298   /**
299    * <p>This method traces a line from a point in the image, in the direction towards another point.
300    * It begins in a black region, and keeps going until it finds white, then black, then white again.
301    * It reports the distance from the start to this point.</p>
302    *
303    * <p>This is used when figuring out how wide a finder pattern is, when the finder pattern
304    * may be skewed or rotated.</p>
305    */
306   private float sizeOfBlackWhiteBlackRun(int fromX, int fromY, int toX, int toY) {
307     // Mild variant of Bresenham's algorithm;
308     // see http://en.wikipedia.org/wiki/Bresenham's_line_algorithm
309     boolean steep = Math.abs(toY - fromY) > Math.abs(toX - fromX);
310     if (steep) {
311       int temp = fromX;
312       fromX = fromY;
313       fromY = temp;
314       temp = toX;
315       toX = toY;
316       toY = temp;
317     }
318 
319     int dx = Math.abs(toX - fromX);
320     int dy = Math.abs(toY - fromY);
321     int error = -dx / 2;
322     int xstep = fromX < toX ? 1 : -1;
323     int ystep = fromY < toY ? 1 : -1;
324 
325     // In black pixels, looking for white, first or second time.
326     int state = 0;
327     // Loop up until x == toX, but not beyond
328     int xLimit = toX + xstep;
329     for (int x = fromX, y = fromY; x != xLimit; x += xstep) {
330       int realX = steep ? y : x;
331       int realY = steep ? x : y;
332 
333       // Does current pixel mean we have moved white to black or vice versa?
334       // Scanning black in state 0,2 and white in state 1, so if we find the wrong
335       // color, advance to next state or end if we are in state 2 already
336       if ((state == 1) == image.get(realX, realY)) {
337         if (state == 2) {
338           return MathUtils.distance(x, y, fromX, fromY);
339         }
340         state++;
341       }
342 
343       error += dy;
344       if (error > 0) {
345         if (y == toY) {
346           break;
347         }
348         y += ystep;
349         error -= dx;
350       }
351     }
352     // Found black-white-black; give the benefit of the doubt that the next pixel outside the image
353     // is "white" so this last point at (toX+xStep,toY) is the right ending. This is really a
354     // small approximation; (toX+xStep,toY+yStep) might be really correct. Ignore this.
355     if (state == 2) {
356       return MathUtils.distance(toX + xstep, toY, fromX, fromY);
357     }
358     // else we didn't find even black-white-black; no estimate is really possible
359     return Float.NaN;
360   }

362   /**
363    * <p>Attempts to locate an alignment pattern in a limited region of the image, which is
364    * guessed to contain it. This method uses {@link AlignmentPattern}.</p>
365    *
366    * @param overallEstModuleSize estimated module size so far
367    * @param estAlignmentX x coordinate of center of area probably containing alignment pattern
368    * @param estAlignmentY y coordinate of above
369    * @param allowanceFactor number of pixels in all directions to search from the center
370    * @return {@link AlignmentPattern} if found, or null otherwise
371    * @throws NotFoundException if an unexpected error occurs during detection
372    */
373   protected final AlignmentPattern findAlignmentInRegion(float overallEstModuleSize,
374                                                          int estAlignmentX,
375                                                          int estAlignmentY,
376                                                          float allowanceFactor)
377       throws NotFoundException {
378     // Look for an alignment pattern (3 modules in size) around where it
379     // should be
380     int allowance = (int) (allowanceFactor * overallEstModuleSize);
381     int alignmentAreaLeftX = Math.max(0, estAlignmentX - allowance);
382     int alignmentAreaRightX = Math.min(image.getWidth() - 1, estAlignmentX + allowance);
383     if (alignmentAreaRightX - alignmentAreaLeftX < overallEstModuleSize * 3) {
384       throw NotFoundException.getNotFoundInstance();
385     }
386 
387     int alignmentAreaTopY = Math.max(0, estAlignmentY - allowance);
388     int alignmentAreaBottomY = Math.min(image.getHeight() - 1, estAlignmentY + allowance);
389     if (alignmentAreaBottomY - alignmentAreaTopY < overallEstModuleSize * 3) {
390       throw NotFoundException.getNotFoundInstance();
391     }
392 
393     AlignmentPatternFinder alignmentFinder =
394         new AlignmentPatternFinder(
395             image,
396             alignmentAreaLeftX,
397             alignmentAreaTopY,
398             alignmentAreaRightX - alignmentAreaLeftX,
399             alignmentAreaBottomY - alignmentAreaTopY,
400             overallEstModuleSize,
401             resultPointCallback);
402     return alignmentFinder.find();
403   }
