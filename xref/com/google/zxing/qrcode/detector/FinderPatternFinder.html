32  /**
33   * <p>This class attempts to find finder patterns in a QR Code. Finder patterns are the square
34   * markers at three corners of a QR Code.</p>
35   *
36   * <p>This class is thread-safe but not reentrant. Each thread must allocate its own object.
37   *
38   * @author Sean Owen
39   */
40  public class FinderPatternFinder {
41  
42    private static final int CENTER_QUORUM = 2;
43    protected static final int MIN_SKIP = 3; // 1 pixel/module times 3 modules/center
44    protected static final int MAX_MODULES = 57; // support up to version 10 for mobile clients
45    private static final int INTEGER_MATH_SHIFT = 8;
46  
47    private final BitMatrix image;
48    private final List<FinderPattern> possibleCenters;
49    private boolean hasSkipped;
50    private final int[] crossCheckStateCount;
51    private final ResultPointCallback resultPointCallback;
52  
53    /**
54     * <p>Creates a finder that will search the image for three finder patterns.</p>
55     *
56     * @param image image to search
57     */
58    public FinderPatternFinder(BitMatrix image) {
59      this(image, null);
60    }

62    public FinderPatternFinder(BitMatrix image, ResultPointCallback resultPointCallback) {
63      this.image = image;
64      this.possibleCenters = new ArrayList<>();
65      this.crossCheckStateCount = new int[5];
66      this.resultPointCallback = resultPointCallback;
67    }
68  
69    protected final BitMatrix getImage() {
70      return image;
71    }
72  
73    protected final List<FinderPattern> getPossibleCenters() {
74      return possibleCenters;
75    }

77    final FinderPatternInfo find(Map<DecodeHintType,?> hints) throws NotFoundException {
78      boolean tryHarder = hints != null && hints.containsKey(DecodeHintType.TRY_HARDER);
79      boolean pureBarcode = hints != null && hints.containsKey(DecodeHintType.PURE_BARCODE);
80      int maxI = image.getHeight();
81      int maxJ = image.getWidth();
82      // We are looking for black/white/black/white/black modules in
83      // 1:1:3:1:1 ratio; this tracks the number of such modules seen so far
84  
85      // Let's assume that the maximum version QR Code we support takes up 1/4 the height of the
86      // image, and then account for the center being 3 modules in size. This gives the smallest
87      // number of pixels the center could be, so skip this often. When trying harder, look for all
88      // QR versions regardless of how dense they are.
89      int iSkip = (3 * maxI) / (4 * MAX_MODULES);
90      if (iSkip < MIN_SKIP || tryHarder) {
91        iSkip = MIN_SKIP;
92      }
93  
94      boolean done = false;
95      int[] stateCount = new int[5];
96      for (int i = iSkip - 1; i < maxI && !done; i += iSkip) {
97        // Get a row of black/white values
98        stateCount[0] = 0;
99        stateCount[1] = 0;
100       stateCount[2] = 0;
101       stateCount[3] = 0;
102       stateCount[4] = 0;
103       int currentState = 0;
104       for (int j = 0; j < maxJ; j++) {
105         if (image.get(j, i)) {
106           // Black pixel
107           if ((currentState & 1) == 1) { // Counting white pixels
108             currentState++;
109           }
110           stateCount[currentState]++;
111         } else { // White pixel
112           if ((currentState & 1) == 0) { // Counting black pixels
113             if (currentState == 4) { // A winner?
114               if (foundPatternCross(stateCount)) { // Yes
115                 boolean confirmed = handlePossibleCenter(stateCount, i, j, pureBarcode);
116                 if (confirmed) {
117                   // Start examining every other line. Checking each line turned out to be too
118                   // expensive and didn't improve performance.
119                   iSkip = 2;
120                   if (hasSkipped) {
121                     done = haveMultiplyConfirmedCenters();
122                   } else {
123                     int rowSkip = findRowSkip();
124                     if (rowSkip > stateCount[2]) {
125                       // Skip rows between row of lower confirmed center
126                       // and top of presumed third confirmed center
127                       // but back up a bit to get a full chance of detecting
128                       // it, entire width of center of finder pattern
129 
130                       // Skip by rowSkip, but back off by stateCount[2] (size of last center
131                       // of pattern we saw) to be conservative, and also back off by iSkip which
132                       // is about to be re-added
133                       i += rowSkip - stateCount[2] - iSkip;
134                       j = maxJ - 1;
135                     }
136                   }
137                 } else {
138                   stateCount[0] = stateCount[2];
139                   stateCount[1] = stateCount[3];
140                   stateCount[2] = stateCount[4];
141                   stateCount[3] = 1;
142                   stateCount[4] = 0;
143                   currentState = 3;
144                   continue;
145                 }
146                 // Clear state to start looking again
147                 currentState = 0;
148                 stateCount[0] = 0;
149                 stateCount[1] = 0;
150                 stateCount[2] = 0;
151                 stateCount[3] = 0;
152                 stateCount[4] = 0;
153               } else { // No, shift counts back by two
154                 stateCount[0] = stateCount[2];
155                 stateCount[1] = stateCount[3];
156                 stateCount[2] = stateCount[4];
157                 stateCount[3] = 1;
158                 stateCount[4] = 0;
159                 currentState = 3;
160               }
161             } else {
162               stateCount[++currentState]++;
163             }
164           } else { // Counting white pixels
165             stateCount[currentState]++;
166           }
167         }
168       }
169       if (foundPatternCross(stateCount)) {
170         boolean confirmed = handlePossibleCenter(stateCount, i, maxJ, pureBarcode);
171         if (confirmed) {
172           iSkip = stateCount[0];
173           if (hasSkipped) {
174             // Found a third one
175             done = haveMultiplyConfirmedCenters();
176           }
177         }
178       }
179     }
180 
181     FinderPattern[] patternInfo = selectBestPatterns();
182     ResultPoint.orderBestPatterns(patternInfo);
183 
184     return new FinderPatternInfo(patternInfo);
185   }

187   /**
188    * Given a count of black/white/black/white/black pixels just seen and an end position,
189    * figures the location of the center of this run.
190    */
191   private static float centerFromEnd(int[] stateCount, int end) {
192     return (float) (end - stateCount[4] - stateCount[3]) - stateCount[2] / 2.0f;
193   }
194 
195   /**
196    * @param stateCount count of black/white/black/white/black pixels just read
197    * @return true iff the proportions of the counts is close enough to the 1/1/3/1/1 ratios
198    *         used by finder patterns to be considered a match
199    */
200   protected static boolean foundPatternCross(int[] stateCount) {
201     int totalModuleSize = 0;
202     for (int i = 0; i < 5; i++) {
203       int count = stateCount[i];
204       if (count == 0) {
205         return false;
206       }
207       totalModuleSize += count;
208     }
209     if (totalModuleSize < 7) {
210       return false;
211     }
212     int moduleSize = (totalModuleSize << INTEGER_MATH_SHIFT) / 7;
213     int maxVariance = moduleSize / 2;
214     // Allow less than 50% variance from 1-1-3-1-1 proportions
215     return Math.abs(moduleSize - (stateCount[0] << INTEGER_MATH_SHIFT)) < maxVariance &&
216         Math.abs(moduleSize - (stateCount[1] << INTEGER_MATH_SHIFT)) < maxVariance &&
217         Math.abs(3 * moduleSize - (stateCount[2] << INTEGER_MATH_SHIFT)) < 3 * maxVariance &&
218         Math.abs(moduleSize - (stateCount[3] << INTEGER_MATH_SHIFT)) < maxVariance &&
219         Math.abs(moduleSize - (stateCount[4] << INTEGER_MATH_SHIFT)) < maxVariance;
220   }

222   private int[] getCrossCheckStateCount() {
223     crossCheckStateCount[0] = 0;
224     crossCheckStateCount[1] = 0;
225     crossCheckStateCount[2] = 0;
226     crossCheckStateCount[3] = 0;
227     crossCheckStateCount[4] = 0;
228     return crossCheckStateCount;
229   }

231   /**
232    * After a vertical and horizontal scan finds a potential finder pattern, this method
233    * "cross-cross-cross-checks" by scanning down diagonally through the center of the possible
234    * finder pattern to see if the same proportion is detected.
235    * 
236    * @param startI row where a finder pattern was detected
237    * @param centerJ center of the section that appears to cross a finder pattern
238    * @param maxCount maximum reasonable number of modules that should be
239    *  observed in any reading state, based on the results of the horizontal scan
240    * @param originalStateCountTotal The original state count total.
241    * @return true if proportions are withing expected limits
242    */
243   private boolean crossCheckDiagonal(int startI, int centerJ, int maxCount, int originalStateCountTotal) {
244     int maxI = image.getHeight();
245     int maxJ = image.getWidth();
246     int[] stateCount = getCrossCheckStateCount();
247 
248     // Start counting up, left from center finding black center mass
249     int i = 0;
250     while (startI - i >= 0 && image.get(centerJ - i, startI - i)) {
251       stateCount[2]++;
252       i++;
253     }
254 
255     if ((startI - i < 0) || (centerJ - i < 0)) {
256       return false;
257     }
258 
259     // Continue up, left finding white space
260     while ((startI - i >= 0) && (centerJ - i >= 0) && !image.get(centerJ - i, startI - i) && stateCount[1] <= maxCount) {
261       stateCount[1]++;
262       i++;
263     }
264 
265     // If already too many modules in this state or ran off the edge:
266     if ((startI - i < 0) || (centerJ - i < 0) || stateCount[1] > maxCount) {
267       return false;
268     }
269 
270     // Continue up, left finding black border
271     while ((startI - i >= 0) && (centerJ - i >= 0) && image.get(centerJ - i, startI - i) && stateCount[0] <= maxCount) {
272       stateCount[0]++;
273       i++;
274     }
275     if (stateCount[0] > maxCount) {
276        return false;
277     }
278 
279     // Now also count down, right from center
280     i = 1;
281     while ((startI + i < maxI) && (centerJ + i < maxJ) && image.get(centerJ + i, startI + i)) {
282       stateCount[2]++;
283       i++;
284     }
285 
286     // Ran off the edge?
287     if ((startI + i >= maxI) || (centerJ + i >= maxJ)) {
288        return false;
289     }
290 
291     while ((startI + i < maxI) && (centerJ + i < maxJ) && !image.get(centerJ + i, startI + i) && stateCount[3] < maxCount) {
292       stateCount[3]++;
293       i++;
294     }
295 
296     if ((startI + i >= maxI) || (centerJ + i >= maxJ) || stateCount[3] >= maxCount) {
297       return false;
298     }
299 
300     while ((startI + i < maxI) && (centerJ + i < maxJ) && image.get(centerJ + i, startI + i) && stateCount[4] < maxCount) {
301       stateCount[4]++;
302       i++;
303     }
304 
305     if (stateCount[4] >= maxCount) {
306       return false;
307    }
308 
309     // If we found a finder-pattern-like section, but its size is more than 100% different than
310     // the original, assume it's a false positive
311     int stateCountTotal = stateCount[0] + stateCount[1] + stateCount[2] + stateCount[3] + stateCount[4];
312     return
313         Math.abs(stateCountTotal - originalStateCountTotal) < 2 * originalStateCountTotal &&
314         foundPatternCross(stateCount);
315   }

317   /**
318    * <p>After a horizontal scan finds a potential finder pattern, this method
319    * "cross-checks" by scanning down vertically through the center of the possible
320    * finder pattern to see if the same proportion is detected.</p>
321    *
322    * @param startI row where a finder pattern was detected
323    * @param centerJ center of the section that appears to cross a finder pattern
324    * @param maxCount maximum reasonable number of modules that should be
325    * observed in any reading state, based on the results of the horizontal scan
326    * @return vertical center of finder pattern, or {@link Float#NaN} if not found
327    */
328   private float crossCheckVertical(int startI, int centerJ, int maxCount,
329       int originalStateCountTotal) {
330     BitMatrix image = this.image;
331 
332     int maxI = image.getHeight();
333     int[] stateCount = getCrossCheckStateCount();
334 
335     // Start counting up from center
336     int i = startI;
337     while (i >= 0 && image.get(centerJ, i)) {
338       stateCount[2]++;
339       i--;
340     }
341     if (i < 0) {
342       return Float.NaN;
343     }
344     while (i >= 0 && !image.get(centerJ, i) && stateCount[1] <= maxCount) {
345       stateCount[1]++;
346       i--;
347     }
348     // If already too many modules in this state or ran off the edge:
349     if (i < 0 || stateCount[1] > maxCount) {
350       return Float.NaN;
351     }
352     while (i >= 0 && image.get(centerJ, i) && stateCount[0] <= maxCount) {
353       stateCount[0]++;
354       i--;
355     }
356     if (stateCount[0] > maxCount) {
357       return Float.NaN;
358     }
359 
360     // Now also count down from center
361     i = startI + 1;
362     while (i < maxI && image.get(centerJ, i)) {
363       stateCount[2]++;
364       i++;
365     }
366     if (i == maxI) {
367       return Float.NaN;
368     }
369     while (i < maxI && !image.get(centerJ, i) && stateCount[3] < maxCount) {
370       stateCount[3]++;
371       i++;
372     }
373     if (i == maxI || stateCount[3] >= maxCount) {
374       return Float.NaN;
375     }
376     while (i < maxI && image.get(centerJ, i) && stateCount[4] < maxCount) {
377       stateCount[4]++;
378       i++;
379     }
380     if (stateCount[4] >= maxCount) {
381       return Float.NaN;
382     }
383 
384     // If we found a finder-pattern-like section, but its size is more than 40% different than
385     // the original, assume it's a false positive
386     int stateCountTotal = stateCount[0] + stateCount[1] + stateCount[2] + stateCount[3] +
387         stateCount[4];
388     if (5 * Math.abs(stateCountTotal - originalStateCountTotal) >= 2 * originalStateCountTotal) {
389       return Float.NaN;
390     }
391 
392     return foundPatternCross(stateCount) ? centerFromEnd(stateCount, i) : Float.NaN;
393   }

395   /**
396    * <p>Like {@link #crossCheckVertical(int, int, int, int)}, and in fact is basically identical,
397    * except it reads horizontally instead of vertically. This is used to cross-cross
398    * check a vertical cross check and locate the real center of the alignment pattern.</p>
399    */
400   private float crossCheckHorizontal(int startJ, int centerI, int maxCount,
401       int originalStateCountTotal) {
402     BitMatrix image = this.image;
403 
404     int maxJ = image.getWidth();
405     int[] stateCount = getCrossCheckStateCount();
406 
407     int j = startJ;
408     while (j >= 0 && image.get(j, centerI)) {
409       stateCount[2]++;
410       j--;
411     }
412     if (j < 0) {
413       return Float.NaN;
414     }
415     while (j >= 0 && !image.get(j, centerI) && stateCount[1] <= maxCount) {
416       stateCount[1]++;
417       j--;
418     }
419     if (j < 0 || stateCount[1] > maxCount) {
420       return Float.NaN;
421     }
422     while (j >= 0 && image.get(j, centerI) && stateCount[0] <= maxCount) {
423       stateCount[0]++;
424       j--;
425     }
426     if (stateCount[0] > maxCount) {
427       return Float.NaN;
428     }
429 
430     j = startJ + 1;
431     while (j < maxJ && image.get(j, centerI)) {
432       stateCount[2]++;
433       j++;
434     }
435     if (j == maxJ) {
436       return Float.NaN;
437     }
438     while (j < maxJ && !image.get(j, centerI) && stateCount[3] < maxCount) {
439       stateCount[3]++;
440       j++;
441     }
442     if (j == maxJ || stateCount[3] >= maxCount) {
443       return Float.NaN;
444     }
445     while (j < maxJ && image.get(j, centerI) && stateCount[4] < maxCount) {
446       stateCount[4]++;
447       j++;
448     }
449     if (stateCount[4] >= maxCount) {
450       return Float.NaN;
451     }
452 
453     // If we found a finder-pattern-like section, but its size is significantly different than
454     // the original, assume it's a false positive
455     int stateCountTotal = stateCount[0] + stateCount[1] + stateCount[2] + stateCount[3] +
456         stateCount[4];
457     if (5 * Math.abs(stateCountTotal - originalStateCountTotal) >= originalStateCountTotal) {
458       return Float.NaN;
459     }
460 
461     return foundPatternCross(stateCount) ? centerFromEnd(stateCount, j) : Float.NaN;
462   }

464   /**
465    * <p>This is called when a horizontal scan finds a possible alignment pattern. It will
466    * cross check with a vertical scan, and if successful, will, ah, cross-cross-check
467    * with another horizontal scan. This is needed primarily to locate the real horizontal
468    * center of the pattern in cases of extreme skew.
469    * And then we cross-cross-cross check with another diagonal scan.</p>
470    *
471    * <p>If that succeeds the finder pattern location is added to a list that tracks
472    * the number of times each location has been nearly-matched as a finder pattern.
473    * Each additional find is more evidence that the location is in fact a finder
474    * pattern center
475    *
476    * @param stateCount reading state module counts from horizontal scan
477    * @param i row where finder pattern may be found
478    * @param j end of possible finder pattern in row
479    * @return true if a finder pattern candidate was found this time
480    */
481   protected final boolean handlePossibleCenter(int[] stateCount, int i, int j, boolean pureBarcode) {
482     int stateCountTotal = stateCount[0] + stateCount[1] + stateCount[2] + stateCount[3] +
483         stateCount[4];
484     float centerJ = centerFromEnd(stateCount, j);
485     float centerI = crossCheckVertical(i, (int) centerJ, stateCount[2], stateCountTotal);
486     if (!Float.isNaN(centerI)) {
487       // Re-cross check
488       centerJ = crossCheckHorizontal((int) centerJ, (int) centerI, stateCount[2], stateCountTotal);
489       if (!Float.isNaN(centerJ) &&
490           (!pureBarcode || crossCheckDiagonal((int) centerI, (int) centerJ, stateCount[2], stateCountTotal))) {
491         float estimatedModuleSize = (float) stateCountTotal / 7.0f;
492         boolean found = false;
493         for (int index = 0; index < possibleCenters.size(); index++) {
494           FinderPattern center = possibleCenters.get(index);
495           // Look for about the same center and module size:
496           if (center.aboutEquals(estimatedModuleSize, centerI, centerJ)) {
497             possibleCenters.set(index, center.combineEstimate(centerI, centerJ, estimatedModuleSize));
498             found = true;
499             break;
500           }
501         }
502         if (!found) {
503           FinderPattern point = new FinderPattern(centerJ, centerI, estimatedModuleSize);
504           possibleCenters.add(point);
505           if (resultPointCallback != null) {
506             resultPointCallback.foundPossibleResultPoint(point);
507           }
508         }
509         return true;
510       }
511     }
512     return false;
513   }

515   /**
516    * @return number of rows we could safely skip during scanning, based on the first
517    *         two finder patterns that have been located. In some cases their position will
518    *         allow us to infer that the third pattern must lie below a certain point farther
519    *         down in the image.
520    */
521   private int findRowSkip() {
522     int max = possibleCenters.size();
523     if (max <= 1) {
524       return 0;
525     }
526     ResultPoint firstConfirmedCenter = null;
527     for (FinderPattern center : possibleCenters) {
528       if (center.getCount() >= CENTER_QUORUM) {
529         if (firstConfirmedCenter == null) {
530           firstConfirmedCenter = center;
531         } else {
532           // We have two confirmed centers
533           // How far down can we skip before resuming looking for the next
534           // pattern? In the worst case, only the difference between the
535           // difference in the x / y coordinates of the two centers.
536           // This is the case where you find top left last.
537           hasSkipped = true;
538           return (int) (Math.abs(firstConfirmedCenter.getX() - center.getX()) -
539               Math.abs(firstConfirmedCenter.getY() - center.getY())) / 2;
540         }
541       }
542     }
543     return 0;
544   }

546   /**
547    * @return true iff we have found at least 3 finder patterns that have been detected
548    *         at least {@link #CENTER_QUORUM} times each, and, the estimated module size of the
549    *         candidates is "pretty similar"
550    */
551   private boolean haveMultiplyConfirmedCenters() {
552     int confirmedCount = 0;
553     float totalModuleSize = 0.0f;
554     int max = possibleCenters.size();
555     for (FinderPattern pattern : possibleCenters) {
556       if (pattern.getCount() >= CENTER_QUORUM) {
557         confirmedCount++;
558         totalModuleSize += pattern.getEstimatedModuleSize();
559       }
560     }
561     if (confirmedCount < 3) {
562       return false;
563     }
564     // OK, we have at least 3 confirmed centers, but, it's possible that one is a "false positive"
565     // and that we need to keep looking. We detect this by asking if the estimated module sizes
566     // vary too much. We arbitrarily say that when the total deviation from average exceeds
567     // 5% of the total module size estimates, it's too much.
568     float average = totalModuleSize / (float) max;
569     float totalDeviation = 0.0f;
570     for (FinderPattern pattern : possibleCenters) {
571       totalDeviation += Math.abs(pattern.getEstimatedModuleSize() - average);
572     }
573     return totalDeviation <= 0.05f * totalModuleSize;
574   }

576   /**
577    * @return the 3 best {@link FinderPattern}s from our list of candidates. The "best" are
578    *         those that have been detected at least {@link #CENTER_QUORUM} times, and whose module
579    *         size differs from the average among those patterns the least
580    * @throws NotFoundException if 3 such finder patterns do not exist
581    */
582   private FinderPattern[] selectBestPatterns() throws NotFoundException {
583 
584     int startSize = possibleCenters.size();
585     if (startSize < 3) {
586       // Couldn't find enough finder patterns
587       throw NotFoundException.getNotFoundInstance();
588     }
589 
590     // Filter outlier possibilities whose module size is too different
591     if (startSize > 3) {
592       // But we can only afford to do so if we have at least 4 possibilities to choose from
593       float totalModuleSize = 0.0f;
594       float square = 0.0f;
595       for (FinderPattern center : possibleCenters) {
596         float size = center.getEstimatedModuleSize();
597         totalModuleSize += size;
598         square += size * size;
599       }
600       float average = totalModuleSize / (float) startSize;
601       float stdDev = (float) Math.sqrt(square / startSize - average * average);
602 
603       Collections.sort(possibleCenters, new FurthestFromAverageComparator(average));
604 
605       float limit = Math.max(0.2f * average, stdDev);
606 
607       for (int i = 0; i < possibleCenters.size() && possibleCenters.size() > 3; i++) {
608         FinderPattern pattern = possibleCenters.get(i);
609         if (Math.abs(pattern.getEstimatedModuleSize() - average) > limit) {
610           possibleCenters.remove(i);
611           i--;
612         }
613       }
614     }
615 
616     if (possibleCenters.size() > 3) {
617       // Throw away all but those first size candidate points we found.
618 
619       float totalModuleSize = 0.0f;
620       for (FinderPattern possibleCenter : possibleCenters) {
621         totalModuleSize += possibleCenter.getEstimatedModuleSize();
622       }
623 
624       float average = totalModuleSize / (float) possibleCenters.size();
625 
626       Collections.sort(possibleCenters, new CenterComparator(average));
627 
628       possibleCenters.subList(3, possibleCenters.size()).clear();
629     }
630 
631     return new FinderPattern[]{
632         possibleCenters.get(0),
633         possibleCenters.get(1),
634         possibleCenters.get(2)
635     };
636   }

638   /**
639    * <p>Orders by furthest from average</p>
640    */
641   private static final class FurthestFromAverageComparator implements Comparator<FinderPattern>, Serializable {
642     private final float average;
643     private FurthestFromAverageComparator(float f) {
644       average = f;
645     }
646     @Override
647     public int compare(FinderPattern center1, FinderPattern center2) {
648       float dA = Math.abs(center2.getEstimatedModuleSize() - average);
649       float dB = Math.abs(center1.getEstimatedModuleSize() - average);
650       return dA < dB ? -1 : dA == dB ? 0 : 1;
651     }
652   }
653 
654   /**
655    * <p>Orders by {@link FinderPattern#getCount()}, descending.</p>
656    */
657   private static final class CenterComparator implements Comparator<FinderPattern>, Serializable {
658     private final float average;
659     private CenterComparator(float f) {
660       average = f;
661     }
662     @Override
663     public int compare(FinderPattern center1, FinderPattern center2) {
664       if (center2.getCount() == center1.getCount()) {
665         float dA = Math.abs(center2.getEstimatedModuleSize() - average);
666         float dB = Math.abs(center1.getEstimatedModuleSize() - average);
667         return dA < dB ? 1 : dA == dB ? 0 : -1;
668       } else {
669         return center2.getCount() - center1.getCount();
670       }
671     }
672   }
