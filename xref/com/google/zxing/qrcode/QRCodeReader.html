39  /**
40   * This implementation can detect and decode QR Codes in an image.
41   *
42   * @author Sean Owen
43   */
44  public class QRCodeReader implements Reader {
45  
46    private static final ResultPoint[] NO_POINTS = new ResultPoint[0];
47  
48    private final Decoder decoder = new Decoder();
49  
50    protected final Decoder getDecoder() {
51      return decoder;
52    }
53  
54    /**
55     * Locates and decodes a QR code in an image.
56     *
57     * @return a String representing the content encoded by the QR code
58     * @throws NotFoundException if a QR code cannot be found
59     * @throws FormatException if a QR code cannot be decoded
60     * @throws ChecksumException if error correction fails
61     */
62    @Override
63    public Result decode(BinaryBitmap image) throws NotFoundException, ChecksumException, FormatException {
64      return decode(image, null);
65    }
66  
67    @Override
68    public final Result decode(BinaryBitmap image, Map<DecodeHintType,?> hints)
69        throws NotFoundException, ChecksumException, FormatException {
70      DecoderResult decoderResult;
71      ResultPoint[] points;
72      if (hints != null && hints.containsKey(DecodeHintType.PURE_BARCODE)) {
73        BitMatrix bits = extractPureBits(image.getBlackMatrix());
74        decoderResult = decoder.decode(bits, hints);
75        points = NO_POINTS;
76      } else {
77        DetectorResult detectorResult = new Detector(image.getBlackMatrix()).detect(hints);
78        decoderResult = decoder.decode(detectorResult.getBits(), hints);
79        points = detectorResult.getPoints();
80      }
81  
82      // If the code was mirrored: swap the bottom-left and the top-right points.
83      if (decoderResult.getOther() instanceof QRCodeDecoderMetaData) {
84        ((QRCodeDecoderMetaData) decoderResult.getOther()).applyMirroredCorrection(points);
85      }
86  
87      Result result = new Result(decoderResult.getText(), decoderResult.getRawBytes(), points, BarcodeFormat.QR_CODE);
88      List<byte[]> byteSegments = decoderResult.getByteSegments();
89      if (byteSegments != null) {
90        result.putMetadata(ResultMetadataType.BYTE_SEGMENTS, byteSegments);
91      }
92      String ecLevel = decoderResult.getECLevel();
93      if (ecLevel != null) {
94        result.putMetadata(ResultMetadataType.ERROR_CORRECTION_LEVEL, ecLevel);
95      }
96      if (decoderResult.hasStructuredAppend()) {
97        result.putMetadata(ResultMetadataType.STRUCTURED_APPEND_SEQUENCE,
98                           decoderResult.getStructuredAppendSequenceNumber());
99        result.putMetadata(ResultMetadataType.STRUCTURED_APPEND_PARITY,
100                          decoderResult.getStructuredAppendParity());
101     }
102     return result;
103   }

105   @Override
106   public void reset() {
107     // do nothing
108   }

110   /**
111    * This method detects a code in a "pure" image -- that is, pure monochrome image
112    * which contains only an unrotated, unskewed, image of a code, with some white border
113    * around it. This is a specialized method that works exceptionally fast in this special
114    * case.
115    *
116    * @see com.google.zxing.datamatrix.DataMatrixReader#extractPureBits(BitMatrix)
117    */
118   private static BitMatrix extractPureBits(BitMatrix image) throws NotFoundException {
119 
120     int[] leftTopBlack = image.getTopLeftOnBit();
121     int[] rightBottomBlack = image.getBottomRightOnBit();
122     if (leftTopBlack == null || rightBottomBlack == null) {
123       throw NotFoundException.getNotFoundInstance();
124     }
125 
126     float moduleSize = moduleSize(leftTopBlack, image);
127 
128     int top = leftTopBlack[1];
129     int bottom = rightBottomBlack[1];
130     int left = leftTopBlack[0];
131     int right = rightBottomBlack[0];
132     
133     // Sanity check!
134     if (left >= right || top >= bottom) {
135       throw NotFoundException.getNotFoundInstance();
136     }
137 
138     if (bottom - top != right - left) {
139       // Special case, where bottom-right module wasn't black so we found something else in the last row
140       // Assume it's a square, so use height as the width
141       right = left + (bottom - top);
142     }
143 
144     int matrixWidth = Math.round((right - left + 1) / moduleSize);
145     int matrixHeight = Math.round((bottom - top + 1) / moduleSize);
146     if (matrixWidth <= 0 || matrixHeight <= 0) {
147       throw NotFoundException.getNotFoundInstance();
148     }
149     if (matrixHeight != matrixWidth) {
150       // Only possibly decode square regions
151       throw NotFoundException.getNotFoundInstance();
152     }
153 
154     // Push in the "border" by half the module width so that we start
155     // sampling in the middle of the module. Just in case the image is a
156     // little off, this will help recover.
157     int nudge = (int) (moduleSize / 2.0f);
158     top += nudge;
159     left += nudge;
160     
161     // But careful that this does not sample off the edge
162     int nudgedTooFarRight = left + (int) ((matrixWidth - 1) * moduleSize) - (right - 1);
163     if (nudgedTooFarRight > 0) {
164       if (nudgedTooFarRight > nudge) {
165         // Neither way fits; abort
166         throw NotFoundException.getNotFoundInstance();
167       }
168       left -= nudgedTooFarRight;
169     }
170     int nudgedTooFarDown = top + (int) ((matrixHeight - 1) * moduleSize) - (bottom - 1);
171     if (nudgedTooFarDown > 0) {
172       if (nudgedTooFarDown > nudge) {
173         // Neither way fits; abort
174         throw NotFoundException.getNotFoundInstance();
175       }
176       top -= nudgedTooFarDown;
177     }
178 
179     // Now just read off the bits
180     BitMatrix bits = new BitMatrix(matrixWidth, matrixHeight);
181     for (int y = 0; y < matrixHeight; y++) {
182       int iOffset = top + (int) (y * moduleSize);
183       for (int x = 0; x < matrixWidth; x++) {
184         if (image.get(left + (int) (x * moduleSize), iOffset)) {
185           bits.set(x, y);
186         }
187       }
188     }
189     return bits;
190   }
191 
192   private static float moduleSize(int[] leftTopBlack, BitMatrix image) throws NotFoundException {
193     int height = image.getHeight();
194     int width = image.getWidth();
195     int x = leftTopBlack[0];
196     int y = leftTopBlack[1];
197     boolean inBlack = true;
198     int transitions = 0;
199     while (x < width && y < height) {
200       if (inBlack != image.get(x, y)) {
201         if (++transitions == 5) {
202           break;
203         }
204         inBlack = !inBlack;
205       }
206       x++;
207       y++;
208     }
209     if (x == width || y == height) {
210       throw NotFoundException.getNotFoundInstance();
211     }
212     return (x - leftTopBlack[0]) / 7.0f;
213   }
