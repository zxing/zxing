17  package com.google.zxing.qrcode;
18  
19  import com.google.zxing.BarcodeFormat;
20  import com.google.zxing.EncodeHintType;
21  import com.google.zxing.Writer;
22  import com.google.zxing.WriterException;
23  import com.google.zxing.common.BitMatrix;
24  import com.google.zxing.qrcode.encoder.ByteMatrix;
25  import com.google.zxing.qrcode.decoder.ErrorCorrectionLevel;
26  import com.google.zxing.qrcode.encoder.Encoder;
27  import com.google.zxing.qrcode.encoder.QRCode;
28  
29  import java.util.Map;
30  
31  /**
32   * This object renders a QR Code as a BitMatrix 2D array of greyscale values.
33   *
34   * @author dswitkin@google.com (Daniel Switkin)
35   */
36  public final class QRCodeWriter implements Writer {
37  
38    private static final int QUIET_ZONE_SIZE = 4;
39  
40    @Override
41    public BitMatrix encode(String contents, BarcodeFormat format, int width, int height)
42        throws WriterException {
43  
44      return encode(contents, format, width, height, null);
45    }
46  
47    @Override
48    public BitMatrix encode(String contents,
49                            BarcodeFormat format,
50                            int width,
51                            int height,
52                            Map<EncodeHintType,?> hints) throws WriterException {
53  
54      if (contents.isEmpty()) {
55        throw new IllegalArgumentException("Found empty contents");
56      }
57  
58      if (format != BarcodeFormat.QR_CODE) {
59        throw new IllegalArgumentException("Can only encode QR_CODE, but got " + format);
60      }
61  
62      if (width < 0 || height < 0) {
63        throw new IllegalArgumentException("Requested dimensions are too small: " + width + 'x' +
64            height);
65      }
66  
67      ErrorCorrectionLevel errorCorrectionLevel = ErrorCorrectionLevel.L;
68      int quietZone = QUIET_ZONE_SIZE;
69      if (hints != null) {
70        ErrorCorrectionLevel requestedECLevel = (ErrorCorrectionLevel) hints.get(EncodeHintType.ERROR_CORRECTION);
71        if (requestedECLevel != null) {
72          errorCorrectionLevel = requestedECLevel;
73        }
74        Integer quietZoneInt = (Integer) hints.get(EncodeHintType.MARGIN);
75        if (quietZoneInt != null) {
76          quietZone = quietZoneInt;
77        }
78      }
79  
80      QRCode code = Encoder.encode(contents, errorCorrectionLevel, hints);
81      return renderResult(code, width, height, quietZone);
82    }
83  
84    // Note that the input matrix uses 0 == white, 1 == black, while the output matrix uses
85    // 0 == black, 255 == white (i.e. an 8 bit greyscale bitmap).
86    private static BitMatrix renderResult(QRCode code, int width, int height, int quietZone) {
87      ByteMatrix input = code.getMatrix();
88      if (input == null) {
89        throw new IllegalStateException();
90      }
91      int inputWidth = input.getWidth();
92      int inputHeight = input.getHeight();
93      int qrWidth = inputWidth + (quietZone << 1);
94      int qrHeight = inputHeight + (quietZone << 1);
95      int outputWidth = Math.max(width, qrWidth);
96      int outputHeight = Math.max(height, qrHeight);
97  
98      int multiple = Math.min(outputWidth / qrWidth, outputHeight / qrHeight);
99      // Padding includes both the quiet zone and the extra white pixels to accommodate the requested
100     // dimensions. For example, if input is 25x25 the QR will be 33x33 including the quiet zone.
101     // If the requested size is 200x160, the multiple will be 4, for a QR of 132x132. These will
102     // handle all the padding from 100x100 (the actual QR) up to 200x160.
103     int leftPadding = (outputWidth - (inputWidth * multiple)) / 2;
104     int topPadding = (outputHeight - (inputHeight * multiple)) / 2;
105 
106     BitMatrix output = new BitMatrix(outputWidth, outputHeight);
107 
108     for (int inputY = 0, outputY = topPadding; inputY < inputHeight; inputY++, outputY += multiple) {
109       // Write the contents of this row of the barcode
110       for (int inputX = 0, outputX = leftPadding; inputX < inputWidth; inputX++, outputX += multiple) {
111         if (input.get(inputX, inputY) == 1) {
112           output.setRegion(outputX, outputY, multiple, multiple);
113         }
114       }
115     }
116 
117     return output;
118   }
