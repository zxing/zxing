30  /**
31   * <p>The main class which implements QR Code decoding -- as opposed to locating and extracting
32   * the QR Code from an image.</p>
33   *
34   * @author Sean Owen
35   */
36  public final class Decoder {
37  
38    private final ReedSolomonDecoder rsDecoder;
39  
40    public Decoder() {
41      rsDecoder = new ReedSolomonDecoder(GenericGF.QR_CODE_FIELD_256);
42    }
43  
44    public DecoderResult decode(boolean[][] image) throws ChecksumException, FormatException {
45      return decode(image, null);
46    }
47  
48    /**
49     * <p>Convenience method that can decode a QR Code represented as a 2D array of booleans.
50     * "true" is taken to mean a black module.</p>
51     *
52     * @param image booleans representing white/black QR Code modules
53     * @param hints decoding hints that should be used to influence decoding
54     * @return text and bytes encoded within the QR Code
55     * @throws FormatException if the QR Code cannot be decoded
56     * @throws ChecksumException if error correction fails
57     */
58    public DecoderResult decode(boolean[][] image, Map<DecodeHintType,?> hints)
59        throws ChecksumException, FormatException {
60      int dimension = image.length;
61      BitMatrix bits = new BitMatrix(dimension);
62      for (int i = 0; i < dimension; i++) {
63        for (int j = 0; j < dimension; j++) {
64          if (image[i][j]) {
65            bits.set(j, i);
66          }
67        }
68      }
69      return decode(bits, hints);
70    }
71  
72    public DecoderResult decode(BitMatrix bits) throws ChecksumException, FormatException {
73      return decode(bits, null);
74    }
75  
76    /**
77     * <p>Decodes a QR Code represented as a {@link BitMatrix}. A 1 or "true" is taken to mean a black module.</p>
78     *
79     * @param bits booleans representing white/black QR Code modules
80     * @param hints decoding hints that should be used to influence decoding
81     * @return text and bytes encoded within the QR Code
82     * @throws FormatException if the QR Code cannot be decoded
83     * @throws ChecksumException if error correction fails
84     */
85    public DecoderResult decode(BitMatrix bits, Map<DecodeHintType,?> hints)
86        throws FormatException, ChecksumException {
87  
88      // Construct a parser and read version, error-correction level
89      BitMatrixParser parser = new BitMatrixParser(bits);
90      FormatException fe = null;
91      ChecksumException ce = null;
92      try {
93        return decode(parser, hints);
94      } catch (FormatException e) {
95        fe = e;
96      } catch (ChecksumException e) {
97        ce = e;
98      }
99  
100     try {
101 
102       // Revert the bit matrix
103       parser.remask();
104 
105       // Will be attempting a mirrored reading of the version and format info.
106       parser.setMirror(true);
107 
108       // Preemptively read the version.
109       parser.readVersion();
110 
111       // Preemptively read the format information.
112       parser.readFormatInformation();
113 
114       /*
115        * Since we're here, this means we have successfully detected some kind
116        * of version and format information when mirrored. This is a good sign,
117        * that the QR code may be mirrored, and we should try once more with a
118        * mirrored content.
119        */
120       // Prepare for a mirrored reading.
121       parser.mirror();
122 
123       DecoderResult result = decode(parser, hints);
124 
125       // Success! Notify the caller that the code was mirrored.
126       result.setOther(new QRCodeDecoderMetaData(true));
127 
128       return result;
129 
130     } catch (FormatException | ChecksumException e) {
131       // Throw the exception from the original reading
132       if (fe != null) {
133         throw fe;
134       }
135       if (ce != null) {
136         throw ce;
137       }
138       throw e;
139 
140     }
141   }
142 
143   private DecoderResult decode(BitMatrixParser parser, Map<DecodeHintType,?> hints)
144       throws FormatException, ChecksumException {
145     Version version = parser.readVersion();
146     ErrorCorrectionLevel ecLevel = parser.readFormatInformation().getErrorCorrectionLevel();
147 
148     // Read codewords
149     byte[] codewords = parser.readCodewords();
150     // Separate into data blocks
151     DataBlock[] dataBlocks = DataBlock.getDataBlocks(codewords, version, ecLevel);
152 
153     // Count total number of data bytes
154     int totalBytes = 0;
155     for (DataBlock dataBlock : dataBlocks) {
156       totalBytes += dataBlock.getNumDataCodewords();
157     }
158     byte[] resultBytes = new byte[totalBytes];
159     int resultOffset = 0;
160 
161     // Error-correct and copy data blocks together into a stream of bytes
162     for (DataBlock dataBlock : dataBlocks) {
163       byte[] codewordBytes = dataBlock.getCodewords();
164       int numDataCodewords = dataBlock.getNumDataCodewords();
165       correctErrors(codewordBytes, numDataCodewords);
166       for (int i = 0; i < numDataCodewords; i++) {
167         resultBytes[resultOffset++] = codewordBytes[i];
168       }
169     }
170 
171     // Decode the contents of that stream of bytes
172     return DecodedBitStreamParser.decode(resultBytes, version, ecLevel, hints);
173   }
174 
175   /**
176    * <p>Given data and error-correction codewords received, possibly corrupted by errors, attempts to
177    * correct the errors in-place using Reed-Solomon error correction.</p>
178    *
179    * @param codewordBytes data and error correction codewords
180    * @param numDataCodewords number of codewords that are data bytes
181    * @throws ChecksumException if error correction fails
182    */
183   private void correctErrors(byte[] codewordBytes, int numDataCodewords) throws ChecksumException {
184     int numCodewords = codewordBytes.length;
185     // First read into an array of ints
186     int[] codewordsInts = new int[numCodewords];
187     for (int i = 0; i < numCodewords; i++) {
188       codewordsInts[i] = codewordBytes[i] & 0xFF;
189     }
190     int numECCodewords = codewordBytes.length - numDataCodewords;
191     try {
192       rsDecoder.decode(codewordsInts, numECCodewords);
193     } catch (ReedSolomonException ignored) {
194       throw ChecksumException.getChecksumInstance();
195     }
196     // Copy back into array of bytes -- only need to worry about the bytes that were data
197     // We don't care about errors in the error-correction codewords
198     for (int i = 0; i < numDataCodewords; i++) {
199       codewordBytes[i] = (byte) codewordsInts[i];
200     }
201   }
