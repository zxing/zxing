26    // Penalty weights from section 6.8.2.1
27    private static final int N1 = 3;
28    private static final int N2 = 3;
29    private static final int N3 = 40;
30    private static final int N4 = 10;
31  
32    private MaskUtil() {
33      // do nothing
34    }
35  
36    /**
37     * Apply mask penalty rule 1 and return the penalty. Find repetitive cells with the same color and
38     * give penalty to them. Example: 00000 or 11111.
39     */
40    static int applyMaskPenaltyRule1(ByteMatrix matrix) {
41      return applyMaskPenaltyRule1Internal(matrix, true) + applyMaskPenaltyRule1Internal(matrix, false);
42    }
43  
44    /**
45     * Apply mask penalty rule 2 and return the penalty. Find 2x2 blocks with the same color and give
46     * penalty to them. This is actually equivalent to the spec's rule, which is to find MxN blocks and give a
47     * penalty proportional to (M-1)x(N-1), because this is the number of 2x2 blocks inside such a block.
48     */
49    static int applyMaskPenaltyRule2(ByteMatrix matrix) {
50      int penalty = 0;
51      byte[][] array = matrix.getArray();
52      int width = matrix.getWidth();
53      int height = matrix.getHeight();
54      for (int y = 0; y < height - 1; y++) {
55        for (int x = 0; x < width - 1; x++) {
56          int value = array[y][x];
57          if (value == array[y][x + 1] && value == array[y + 1][x] && value == array[y + 1][x + 1]) {
58            penalty++;
59          }
60        }
61      }
62      return N2 * penalty;
63    }
64  
65    /**
66     * Apply mask penalty rule 3 and return the penalty. Find consecutive runs of 1:1:3:1:1:4
67     * starting with black, or 4:1:1:3:1:1 starting with white, and give penalty to them.  If we
68     * find patterns like 000010111010000, we give penalty once.
69     */
70    static int applyMaskPenaltyRule3(ByteMatrix matrix) {
71      int numPenalties = 0;
72      byte[][] array = matrix.getArray();
73      int width = matrix.getWidth();
74      int height = matrix.getHeight();
75      for (int y = 0; y < height; y++) {
76        for (int x = 0; x < width; x++) {
77          byte[] arrayY = array[y];  // We can at least optimize this access
78          if (x + 6 < width &&
79              arrayY[x] == 1 &&
80              arrayY[x +  1] == 0 &&
81              arrayY[x +  2] == 1 &&
82              arrayY[x +  3] == 1 &&
83              arrayY[x +  4] == 1 &&
84              arrayY[x +  5] == 0 &&
85              arrayY[x +  6] == 1 &&
86              (isWhiteHorizontal(arrayY, x - 4, x) || isWhiteHorizontal(arrayY, x + 7, x + 11))) {
87            numPenalties++;
88          }
89          if (y + 6 < height &&
90              array[y][x] == 1  &&
91              array[y +  1][x] == 0  &&
92              array[y +  2][x] == 1  &&
93              array[y +  3][x] == 1  &&
94              array[y +  4][x] == 1  &&
95              array[y +  5][x] == 0  &&
96              array[y +  6][x] == 1 &&
97              (isWhiteVertical(array, x, y - 4, y) || isWhiteVertical(array, x, y + 7, y + 11))) {
98            numPenalties++;
99          }
100       }
101     }
102     return numPenalties * N3;
103   }
104 
105   private static boolean isWhiteHorizontal(byte[] rowArray, int from, int to) {
106     for (int i = from; i < to; i++) {
107       if (i >= 0 && i < rowArray.length && rowArray[i] == 1) {
108         return false;
109       }
110     }
111     return true;
112   }
113 
114   private static boolean isWhiteVertical(byte[][] array, int col, int from, int to) {
115     for (int i = from; i < to; i++) {
116       if (i >= 0 && i < array.length && array[i][col] == 1) {
117         return false;
118       }
119     }
120     return true;
121   }

123   /**
124    * Apply mask penalty rule 4 and return the penalty. Calculate the ratio of dark cells and give
125    * penalty if the ratio is far from 50%. It gives 10 penalty for 5% distance.
126    */
127   static int applyMaskPenaltyRule4(ByteMatrix matrix) {
128     int numDarkCells = 0;
129     byte[][] array = matrix.getArray();
130     int width = matrix.getWidth();
131     int height = matrix.getHeight();
132     for (int y = 0; y < height; y++) {
133       byte[] arrayY = array[y];
134       for (int x = 0; x < width; x++) {
135         if (arrayY[x] == 1) {
136           numDarkCells++;
137         }
138       }
139     }
140     int numTotalCells = matrix.getHeight() * matrix.getWidth();
141     int fivePercentVariances = Math.abs(numDarkCells * 2 - numTotalCells) * 10 / numTotalCells;
142     return fivePercentVariances * N4;
143   }

145   /**
146    * Return the mask bit for "getMaskPattern" at "x" and "y". See 8.8 of JISX0510:2004 for mask
147    * pattern conditions.
148    */
149   static boolean getDataMaskBit(int maskPattern, int x, int y) {
150     int intermediate;
151     int temp;
152     switch (maskPattern) {
153       case 0:
154         intermediate = (y + x) & 0x1;
155         break;
156       case 1:
157         intermediate = y & 0x1;
158         break;
159       case 2:
160         intermediate = x % 3;
161         break;
162       case 3:
163         intermediate = (y + x) % 3;
164         break;
165       case 4:
166         intermediate = ((y >>> 1) + (x / 3)) & 0x1;
167         break;
168       case 5:
169         temp = y * x;
170         intermediate = (temp & 0x1) + (temp % 3);
171         break;
172       case 6:
173         temp = y * x;
174         intermediate = ((temp & 0x1) + (temp % 3)) & 0x1;
175         break;
176       case 7:
177         temp = y * x;
178         intermediate = ((temp % 3) + ((y + x) & 0x1)) & 0x1;
179         break;
180       default:
181         throw new IllegalArgumentException("Invalid mask pattern: " + maskPattern);
182     }
183     return intermediate == 0;
184   }

186   /**
187    * Helper function for applyMaskPenaltyRule1. We need this for doing this calculation in both
188    * vertical and horizontal orders respectively.
189    */
190   private static int applyMaskPenaltyRule1Internal(ByteMatrix matrix, boolean isHorizontal) {
191     int penalty = 0;
192     int iLimit = isHorizontal ? matrix.getHeight() : matrix.getWidth();
193     int jLimit = isHorizontal ? matrix.getWidth() : matrix.getHeight();
194     byte[][] array = matrix.getArray();
195     for (int i = 0; i < iLimit; i++) {
196       int numSameBitCells = 0;
197       int prevBit = -1;
198       for (int j = 0; j < jLimit; j++) {
199         int bit = isHorizontal ? array[i][j] : array[j][i];
200         if (bit == prevBit) {
201           numSameBitCells++;
202         } else {
203           if (numSameBitCells >= 5) {
204             penalty += N1 + (numSameBitCells - 5);
205           }
206           numSameBitCells = 1;  // Include the cell itself.
207           prevBit = bit;
208         }
209       }
210       if (numSameBitCells >= 5) {
211         penalty += N1 + (numSameBitCells - 5);
212       }
213     }
214     return penalty;
215   }
