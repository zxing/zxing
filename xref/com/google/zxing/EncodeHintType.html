17  package com.google.zxing;
18  
19  /**
20   * These are a set of hints that you may pass to Writers to specify their behavior.
21   *
22   * @author dswitkin@google.com (Daniel Switkin)
23   */
24  public enum EncodeHintType {
25  
26    /**
27     * Specifies what degree of error correction to use, for example in QR Codes.
28     * Type depends on the encoder. For example for QR codes it's type
29     * {@link com.google.zxing.qrcode.decoder.ErrorCorrectionLevel ErrorCorrectionLevel}.
30     * For Aztec it is of type {@link Integer}, representing the minimal percentage of error correction words. 
31     * Note: an Aztec symbol should have a minimum of 25% EC words.
32     */
33    ERROR_CORRECTION,
34  
35    /**
36     * Specifies what character encoding to use where applicable (type {@link String})
37     */
38    CHARACTER_SET,
39  
40    /**
41     * Specifies the matrix shape for Data Matrix (type {@link com.google.zxing.datamatrix.encoder.SymbolShapeHint})
42     */
43    DATA_MATRIX_SHAPE,
44  
45    /**
46     * Specifies a minimum barcode size (type {@link Dimension}). Only applicable to Data Matrix now.
47     */
48    MIN_SIZE,
49  
50    /**
51     * Specifies a maximum barcode size (type {@link Dimension}). Only applicable to Data Matrix now.
52     */
53    MAX_SIZE,
54  
55    /**
56     * Specifies margin, in pixels, to use when generating the barcode. The meaning can vary
57     * by format; for example it controls margin before and after the barcode horizontally for
58     * most 1D formats. (Type {@link Integer}).
59     */
60    MARGIN,
61  
62    /**
63     * Specifies whether to use compact mode for PDF417 (type {@link Boolean}).
64     */
65    PDF417_COMPACT,
66  
67    /**
68     * Specifies what compaction mode to use for PDF417 (type
69     * {@link com.google.zxing.pdf417.encoder.Compaction Compaction}).
70     */
71    PDF417_COMPACTION,
72  
73    /**
74     * Specifies the minimum and maximum number of rows and columns for PDF417 (type
75     * {@link com.google.zxing.pdf417.encoder.Dimensions Dimensions}).
76     */
77    PDF417_DIMENSIONS,
78  
79    /**
80     * Specifies the required number of layers for an Aztec code:
81     *   a negative number (-1, -2, -3, -4) specifies a compact Aztec code
82     *   0 indicates to use the minimum number of layers (the default)
83     *   a positive number (1, 2, .. 32) specifies a normaol (non-compact) Aztec code
84     */
85     AZTEC_LAYERS,
86  }
