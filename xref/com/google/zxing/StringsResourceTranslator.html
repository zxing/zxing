43  /**
44   * <p>A utility which auto-translates English strings in Android string resources using
45   * Google Translate.</p>
46   *
47   * <p>Pass the Android client res/ directory as first argument, and optionally message keys
48   * who should be forced to retranslate.
49   * Usage: {@code StringsResourceTranslator android/res/ [key_1 ...]}</p>
50   *
51   * <p>You must set your Google Translate API key into the environment with -DtranslateAPI.key=...</p>
52   *
53   * @author Sean Owen
54   */
55  public final class StringsResourceTranslator {
56  
57    private static final String API_KEY = System.getProperty("translateAPI.key");
58    
59    private static final Charset UTF8 = Charset.forName("UTF-8");
60    private static final Pattern ENTRY_PATTERN = Pattern.compile("<string name=\"([^\"]+)\".*>([^<]+)</string>");
61    private static final Pattern STRINGS_FILE_NAME_PATTERN = Pattern.compile("values-(.+)");
62    private static final Pattern TRANSLATE_RESPONSE_PATTERN = Pattern.compile("translatedText\":\\s*\"([^\"]+)\"");
63    private static final Pattern VALUES_DIR_PATTERN = Pattern.compile("values-[a-z]{2}(-[a-zA-Z]{2,3})?");
64  
65    private static final String APACHE_2_LICENSE =
66        "<!--\n" +
67        " Copyright (C) 2013 ZXing authors\n" +
68        '\n' +
69        " Licensed under the Apache License, Version 2.0 (the \"License\");\n" +
70        " you may not use this file except in compliance with the License.\n" +
71        " You may obtain a copy of the License at\n" +
72        '\n' +
73        "      http://www.apache.org/licenses/LICENSE-2.0\n" +
74        '\n' +
75        " Unless required by applicable law or agreed to in writing, software\n" +
76        " distributed under the License is distributed on an \"AS IS\" BASIS,\n" +
77        " WITHOUT WARRANTIES OR CONDITIONS OF ANY KIND, either express or implied.\n" +
78        " See the License for the specific language governing permissions and\n" +
79        " limitations under the License.\n" +
80        " -->\n";

82    private static final Map<String,String> LANGUAGE_CODE_MASSAGINGS = new HashMap<>(3);
83    static {
84      LANGUAGE_CODE_MASSAGINGS.put("zh-rCN", "zh-cn");
85      LANGUAGE_CODE_MASSAGINGS.put("zh-rTW", "zh-tw");
86    }
87  
88    private StringsResourceTranslator() {}
89  
90    public static void main(String[] args) throws IOException {
91  
92      File resDir = new File(args[0]);
93      File valueDir = new File(resDir, "values");
94      File stringsFile = new File(valueDir, "strings.xml");
95      Collection<String> forceRetranslation = Arrays.asList(args).subList(1, args.length);
96  
97      File[] translatedValuesDirs = resDir.listFiles(new FileFilter() {
98        @Override
99        public boolean accept(File file) {
100         return file.isDirectory() && VALUES_DIR_PATTERN.matcher(file.getName()).matches();
101       }
102     });
103     for (File translatedValuesDir : translatedValuesDirs) {
104       File translatedStringsFile = new File(translatedValuesDir, "strings.xml");
105       translate(stringsFile, translatedStringsFile, forceRetranslation);
106     }
107 
108   }

110   private static void translate(File englishFile,
111                                 File translatedFile,
112                                 Collection<String> forceRetranslation) throws IOException {
113 
114     Map<String, String> english = readLines(englishFile);
115     SortedMap<String,String> translated = readLines(translatedFile);
116     String parentName = translatedFile.getParentFile().getName();
117 
118     Matcher stringsFileNameMatcher = STRINGS_FILE_NAME_PATTERN.matcher(parentName);
119     stringsFileNameMatcher.find();
120     String language = stringsFileNameMatcher.group(1);
121     String massagedLanguage = LANGUAGE_CODE_MASSAGINGS.get(language);
122     if (massagedLanguage != null) {
123       language = massagedLanguage;
124     }
125 
126     System.out.println("Translating " + language);
127 
128     File resultTempFile = File.createTempFile(parentName, ".xml");
129     resultTempFile.deleteOnExit();
130 
131     boolean anyChange = false;
132     try (Writer out = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(resultTempFile), UTF8))) {
133       out.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
134       out.write(APACHE_2_LICENSE);
135       out.write("<resources>\n");
136 
137       for (Map.Entry<String,String> englishEntry : english.entrySet()) {
138         String key = englishEntry.getKey();
139         String value = englishEntry.getValue();
140         out.write("  <string name=\"");
141         out.write(key);
142         out.write('"');
143         if (value.contains("%s") || value.contains("%f")) {
144           // Need to specify that there's a value placeholder
145           out.write(" formatted=\"false\"");
146         }
147         out.write('>');
148 
149         String translatedString = translated.get(key);
150         if (translatedString == null || forceRetranslation.contains(key)) {
151           anyChange = true;
152           translatedString = translateString(value, language);
153         }
154         out.write(translatedString);
155 
156         out.write("</string>\n");
157       }
158 
159       out.write("</resources>\n");
160       out.flush();
161     }
162 
163     if (anyChange) {
164       System.out.println("  Writing translations");
165       translatedFile.delete();
166       resultTempFile.renameTo(translatedFile);
167     } else {
168       resultTempFile.delete();
169     }
170   }
171 
172   static String translateString(String english, String language) throws IOException {
173     if ("en".equals(language)) {
174       return english;
175     }
176     String massagedLanguage = LANGUAGE_CODE_MASSAGINGS.get(language);
177     if (massagedLanguage != null) {
178       language = massagedLanguage;
179     }
180     System.out.println("  Need translation for " + english);
181 
182     if (API_KEY == null) {
183       throw new IllegalArgumentException("translateAPI.key is not specified");
184     }
185 
186     URL translateURL = new URL(
187         "https://www.googleapis.com/language/translate/v2?key=" + API_KEY + "&q=" +
188         URLEncoder.encode(english, "UTF-8") +
189         "&source=en&target=" + language);
190     CharSequence translateResult = fetch(translateURL);
191     Matcher m = TRANSLATE_RESPONSE_PATTERN.matcher(translateResult);
192     if (!m.find()) {
193       System.err.println("No translate result");
194       System.err.println(translateResult);
195       return english;
196     }
197     String translation = m.group(1);
198     System.out.println("  Got translation " + translation);
199 
200     // This is a little crude; unescape some common escapes in the raw response
201     translation = translation.replaceAll("\\\\u0026quot;", "\"");
202     translation = translation.replaceAll("\\\\u0026#39;", "'");
203     translation = translation.replaceAll("\\\\u200b", "");
204     translation = translation.replaceAll("&amp;quot;", "\"");
205     translation = translation.replaceAll("&amp;#39;", "'");
206     return translation;
207   }
208 
209   private static CharSequence fetch(URL translateURL) throws IOException {
210     URLConnection connection = translateURL.openConnection();
211     connection.connect();
212     StringBuilder translateResult = new StringBuilder(200);
213     try (Reader in = new BufferedReader(new InputStreamReader(connection.getInputStream(), UTF8))) {
214       char[] buffer = new char[1024];
215       int charsRead;
216       while ((charsRead = in.read(buffer)) > 0) {
217         translateResult.append(buffer, 0, charsRead);
218       }
219     }
220     return translateResult;
221   }
222 
223   private static SortedMap<String,String> readLines(File file) throws IOException {
224     SortedMap<String,String> entries = new TreeMap<>();
225     if (!file.exists()) {
226       return entries;
227     }
228     try (BufferedReader reader = new BufferedReader(new InputStreamReader(new FileInputStream(file), UTF8))) {
229       CharSequence line;
230       while ((line = reader.readLine()) != null) {
231         Matcher m = ENTRY_PATTERN.matcher(line);
232         if (m.find()) {
233           String key = m.group(1);
234           String value = m.group(2);
235           entries.put(key, value);
236         }
237       }
238       return entries;
239     }
240   }
