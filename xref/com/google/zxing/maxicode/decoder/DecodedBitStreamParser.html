63    static DecoderResult decode(byte[] bytes, int mode) {
64      StringBuilder result = new StringBuilder(144);
65      switch (mode) {
66        case 2:
67        case 3:
68          String postcode;
69          if (mode == 2) {
70            int pc = getPostCode2(bytes);
71            NumberFormat df = new DecimalFormat("0000000000".substring(0, getPostCode2Length(bytes)));
72            postcode = df.format(pc);
73          } else {
74            postcode = getPostCode3(bytes);
75          }
76          String country = THREE_DIGITS.format(getCountry(bytes));
77          String service = THREE_DIGITS.format(getServiceClass(bytes));
78          result.append(getMessage(bytes, 10, 84));
79          if (result.toString().startsWith("[)>"+RS+"01"+GS)) {
80            result.insert(9, postcode + GS + country + GS + service + GS);
81          } else {
82            result.insert(0, postcode + GS + country + GS + service + GS);
83          }
84          break;
85        case 4:
86          result.append(getMessage(bytes, 1, 93));
87          break;
88        case 5:
89          result.append(getMessage(bytes, 1, 77));
90          break;
91      }
92      return new DecoderResult(bytes, result.toString(), null, String.valueOf(mode));
93    }

95    private static int getBit(int bit, byte[] bytes) {
96      bit--;
97      return (bytes[bit / 6] & (1 << (5 - (bit % 6)))) == 0 ? 0 : 1;
98    }

100   private static int getInt(byte[] bytes, byte[] x) {
101     if (x.length == 0) {
102       throw new IllegalArgumentException();
103     }
104     int val = 0;
105     for (int i = 0; i < x.length; i++) {
106       val += getBit(x[i], bytes) << (x.length - i - 1);
107     }
108     return val;
109   }

111   private static int getCountry(byte[] bytes) {
112     return getInt(bytes, new byte[] {53, 54, 43, 44, 45, 46, 47, 48, 37, 38});

128   private static String getPostCode3(byte[] bytes) {
129     return String.valueOf(
130        new char[] {
131          SETS[0].charAt(getInt(bytes, new byte[] {39, 40, 41, 42, 31, 32})),
132          SETS[0].charAt(getInt(bytes, new byte[] {33, 34, 35, 36, 25, 26})),
133          SETS[0].charAt(getInt(bytes, new byte[] {27, 28, 29, 30, 19, 20})),
134          SETS[0].charAt(getInt(bytes, new byte[] {21, 22, 23, 24, 13, 14})),
135          SETS[0].charAt(getInt(bytes, new byte[] {15, 16, 17, 18,  7,  8})),
136          SETS[0].charAt(getInt(bytes, new byte[] { 9, 10, 11, 12,  1,  2})),
137        }
138     );
139   }

141   private static String getMessage(byte[] bytes, int start, int len) {
142     StringBuilder sb = new StringBuilder();
143     int shift = -1;
144     int set = 0;
145     int lastset = 0;
146     for (int i = start; i < start + len; i++) {
147       char c = SETS[set].charAt(bytes[i]);
148       switch (c) {
149         case LATCHA:
150           set = 0;
151           shift = -1;
152           break;
153         case LATCHB:
154           set = 1;
155           shift = -1;
156           break;
157         case SHIFTA:
158         case SHIFTB:
159         case SHIFTC:
160         case SHIFTD:
161         case SHIFTE:
162           lastset = set;
163           set = c - SHIFTA;
164           shift = 1;
165           break;
166         case TWOSHIFTA:
167           lastset = set;
168           set = 0;
169           shift = 2;
170           break;
171         case THREESHIFTA:
172           lastset = set;
173           set = 0;
174           shift = 3;
175           break;
176         case NS:
177           int nsval = (bytes[++i] << 24) + (bytes[++i] << 18) + (bytes[++i] << 12) + (bytes[++i] << 6) + bytes[++i];
178           sb.append(NINE_DIGITS.format(nsval));
179           break;
180         case LOCK:
181           shift = -1;
182           break;
183         default:
184           sb.append(c);
185       }
186       if (shift-- == 0) {
187         set = lastset;
188       }
189     }
190     while (sb.length() > 0 && sb.charAt(sb.length() - 1) == PAD) {
191       sb.setLength(sb.length() - 1);
192     }
193     return sb.toString();
194   }
