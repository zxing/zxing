17  package com.google.zxing.common;
18  
19  import java.util.Arrays;
20  
21  /**
22   * <p>Represents a 2D matrix of bits. In function arguments below, and throughout the common
23   * module, x is the column position, and y is the row position. The ordering is always x, y.
24   * The origin is at the top-left.</p>
25   *
26   * <p>Internally the bits are represented in a 1-D array of 32-bit ints. However, each row begins
27   * with a new int. This is done intentionally so that we can copy out a row into a BitArray very
28   * efficiently.</p>
29   *
30   * <p>The ordering of bits is row-major. Within each int, the least significant bits are used first,
31   * meaning they represent lower x values. This is compatible with BitArray's implementation.</p>
32   *
33   * @author Sean Owen
34   * @author dswitkin@google.com (Daniel Switkin)
35   */
36  public final class BitMatrix implements Cloneable {
37  
38    private final int width;
39    private final int height;
40    private final int rowSize;
41    private final int[] bits;
42  
43    // A helper to construct a square matrix.
44    public BitMatrix(int dimension) {
45      this(dimension, dimension);
46    }
47  
48    public BitMatrix(int width, int height) {
49      if (width < 1 || height < 1) {
50        throw new IllegalArgumentException("Both dimensions must be greater than 0");
51      }
52      this.width = width;
53      this.height = height;
54      this.rowSize = (width + 31) / 32;
55      bits = new int[rowSize * height];
56    }
57  
58    private BitMatrix(int width, int height, int rowSize, int[] bits) {
59      this.width = width;
60      this.height = height;
61      this.rowSize = rowSize;
62      this.bits = bits;
63    }
64  
65    /**
66     * <p>Gets the requested bit, where true means black.</p>
67     *
68     * @param x The horizontal component (i.e. which column)
69     * @param y The vertical component (i.e. which row)
70     * @return value of given bit in matrix
71     */
72    public boolean get(int x, int y) {
73      int offset = y * rowSize + (x / 32);
74      return ((bits[offset] >>> (x & 0x1f)) & 1) != 0;
75    }
76  
77    /**
78     * <p>Sets the given bit to true.</p>
79     *
80     * @param x The horizontal component (i.e. which column)
81     * @param y The vertical component (i.e. which row)
82     */
83    public void set(int x, int y) {
84      int offset = y * rowSize + (x / 32);
85      bits[offset] |= 1 << (x & 0x1f);
86    }
87  
88    /**
89     * <p>Flips the given bit.</p>
90     *
91     * @param x The horizontal component (i.e. which column)
92     * @param y The vertical component (i.e. which row)
93     */
94    public void flip(int x, int y) {
95      int offset = y * rowSize + (x / 32);
96      bits[offset] ^= 1 << (x & 0x1f);
97    }
98  
99    /**
100    * Clears all bits (sets to false).
101    */
102   public void clear() {
103     int max = bits.length;
104     for (int i = 0; i < max; i++) {
105       bits[i] = 0;
106     }
107   }
108 
109   /**
110    * <p>Sets a square region of the bit matrix to true.</p>
111    *
112    * @param left The horizontal position to begin at (inclusive)
113    * @param top The vertical position to begin at (inclusive)
114    * @param width The width of the region
115    * @param height The height of the region
116    */
117   public void setRegion(int left, int top, int width, int height) {
118     if (top < 0 || left < 0) {
119       throw new IllegalArgumentException("Left and top must be nonnegative");
120     }
121     if (height < 1 || width < 1) {
122       throw new IllegalArgumentException("Height and width must be at least 1");
123     }
124     int right = left + width;
125     int bottom = top + height;
126     if (bottom > this.height || right > this.width) {
127       throw new IllegalArgumentException("The region must fit inside the matrix");
128     }
129     for (int y = top; y < bottom; y++) {
130       int offset = y * rowSize;
131       for (int x = left; x < right; x++) {
132         bits[offset + (x / 32)] |= 1 << (x & 0x1f);
133       }
134     }
135   }
136 
137   /**
138    * A fast method to retrieve one row of data from the matrix as a BitArray.
139    *
140    * @param y The row to retrieve
141    * @param row An optional caller-allocated BitArray, will be allocated if null or too small
142    * @return The resulting BitArray - this reference should always be used even when passing
143    *         your own row
144    */
145   public BitArray getRow(int y, BitArray row) {
146     if (row == null || row.getSize() < width) {
147       row = new BitArray(width);
148     } else {
149       row.clear();
150     }
151     int offset = y * rowSize;
152     for (int x = 0; x < rowSize; x++) {
153       row.setBulk(x * 32, bits[offset + x]);
154     }
155     return row;
156   }
157 
158   /**
159    * @param y row to set
160    * @param row {@link BitArray} to copy from
161    */
162   public void setRow(int y, BitArray row) {
163     System.arraycopy(row.getBitArray(), 0, bits, y * rowSize, rowSize);
164   }
165 
166   /**
167    * Modifies this {@code BitMatrix} to represent the same but rotated 180 degrees
168    */
169   public void rotate180() {
170     int width = getWidth();
171     int height = getHeight();
172     BitArray topRow = new BitArray(width);
173     BitArray bottomRow = new BitArray(width);
174     for (int i = 0; i < (height+1) / 2; i++) {
175       topRow = getRow(i, topRow);
176       bottomRow = getRow(height - 1 - i, bottomRow);
177       topRow.reverse();
178       bottomRow.reverse();
179       setRow(i, bottomRow);
180       setRow(height - 1 - i, topRow);
181     }
182   }
183 
184   /**
185    * This is useful in detecting the enclosing rectangle of a 'pure' barcode.
186    *
187    * @return {@code left,top,width,height} enclosing rectangle of all 1 bits, or null if it is all white
188    */
189   public int[] getEnclosingRectangle() {
190     int left = width;
191     int top = height;
192     int right = -1;
193     int bottom = -1;
194 
195     for (int y = 0; y < height; y++) {
196       for (int x32 = 0; x32 < rowSize; x32++) {
197         int theBits = bits[y * rowSize + x32];
198         if (theBits != 0) {
199           if (y < top) {
200             top = y;
201           }
202           if (y > bottom) {
203             bottom = y;
204           }
205           if (x32 * 32 < left) {
206             int bit = 0;
207             while ((theBits << (31 - bit)) == 0) {
208               bit++;
209             }
210             if ((x32 * 32 + bit) < left) {
211               left = x32 * 32 + bit;
212             }
213           }
214           if (x32 * 32 + 31 > right) {
215             int bit = 31;
216             while ((theBits >>> bit) == 0) {
217               bit--;
218             }
219             if ((x32 * 32 + bit) > right) {
220               right = x32 * 32 + bit;
221             }
222           }
223         }
224       }
225     }
226 
227     int width = right - left;
228     int height = bottom - top;
229 
230     if (width < 0 || height < 0) {
231       return null;
232     }
233 
234     return new int[] {left, top, width, height};
235   }
236 
237   /**
238    * This is useful in detecting a corner of a 'pure' barcode.
239    *
240    * @return {@code x,y} coordinate of top-left-most 1 bit, or null if it is all white
241    */
242   public int[] getTopLeftOnBit() {
243     int bitsOffset = 0;
244     while (bitsOffset < bits.length && bits[bitsOffset] == 0) {
245       bitsOffset++;
246     }
247     if (bitsOffset == bits.length) {
248       return null;
249     }
250     int y = bitsOffset / rowSize;
251     int x = (bitsOffset % rowSize) * 32;
252 
253     int theBits = bits[bitsOffset];
254     int bit = 0;
255     while ((theBits << (31-bit)) == 0) {
256       bit++;
257     }
258     x += bit;
259     return new int[] {x, y};
260   }
261 
262   public int[] getBottomRightOnBit() {
263     int bitsOffset = bits.length - 1;
264     while (bitsOffset >= 0 && bits[bitsOffset] == 0) {
265       bitsOffset--;
266     }
267     if (bitsOffset < 0) {
268       return null;
269     }
270 
271     int y = bitsOffset / rowSize;
272     int x = (bitsOffset % rowSize) * 32;
273 
274     int theBits = bits[bitsOffset];
275     int bit = 31;
276     while ((theBits >>> bit) == 0) {
277       bit--;
278     }
279     x += bit;
280 
281     return new int[] {x, y};
282   }
283 
284   /**
285    * @return The width of the matrix
286    */
287   public int getWidth() {
288     return width;
289   }
290 
291   /**
292    * @return The height of the matrix
293    */
294   public int getHeight() {
295     return height;
296   }
297 
298   @Override
299   public boolean equals(Object o) {
300     if (!(o instanceof BitMatrix)) {
301       return false;
302     }
303     BitMatrix other = (BitMatrix) o;
304     return width == other.width && height == other.height && rowSize == other.rowSize &&
305     Arrays.equals(bits, other.bits);
306   }
307 
308   @Override
309   public int hashCode() {
310     int hash = width;
311     hash = 31 * hash + width;
312     hash = 31 * hash + height;
313     hash = 31 * hash + rowSize;
314      hash = 31 * hash + Arrays.hashCode(bits);
315     return hash;
316   }
317 
318   @Override
319   public String toString() {
320     StringBuilder result = new StringBuilder(height * (width + 1));
321     for (int y = 0; y < height; y++) {
322       for (int x = 0; x < width; x++) {
323         result.append(get(x, y) ? "X " : "  ");
324       }
325       result.append('\n');
326     }
327     return result.toString();
328   }
329 
330   @Override
331   public BitMatrix clone() {
332     return new BitMatrix(width, height, rowSize, bits.clone());
333   }
