17  package com.google.zxing.common.detector;
18  
19  import com.google.zxing.NotFoundException;
20  import com.google.zxing.ResultPoint;
21  import com.google.zxing.common.BitMatrix;
22  
23  /**
24   * <p>
25   * Detects a candidate barcode-like rectangular region within an image. It
26   * starts around the center of the image, increases the size of the candidate
27   * region until it finds a white rectangular region. By keeping track of the
28   * last black points it encountered, it determines the corners of the barcode.
29   * </p>
30   *
31   * @author David Olivier
32   */
33  public final class WhiteRectangleDetector {
34  
35    private static final int INIT_SIZE = 10;
36    private static final int CORR = 1;
37  
38    private final BitMatrix image;
39    private final int height;
40    private final int width;
41    private final int leftInit;
42    private final int rightInit;
43    private final int downInit;
44    private final int upInit;
45  
46    public WhiteRectangleDetector(BitMatrix image) throws NotFoundException {
47      this(image, INIT_SIZE, image.getWidth() / 2, image.getHeight() / 2);
48    }
49  
50    /**
51     * @throws NotFoundException if image is too small to accommodate {@code initSize}
52     */
53    public WhiteRectangleDetector(BitMatrix image, int initSize, int x, int y) throws NotFoundException {
54      this.image = image;
55      height = image.getHeight();
56      width = image.getWidth();
57      int halfsize = initSize / 2;
58      leftInit = x - halfsize;
59      rightInit = x + halfsize;
60      upInit = y - halfsize;
61      downInit = y + halfsize;
62      if (upInit < 0 || leftInit < 0 || downInit >= height || rightInit >= width) {
63        throw NotFoundException.getNotFoundInstance();
64      }
65    }
66  
67    /**
68     * <p>
69     * Detects a candidate barcode-like rectangular region within an image. It
70     * starts around the center of the image, increases the size of the candidate
71     * region until it finds a white rectangular region.
72     * </p>
73     *
74     * @return {@link ResultPoint}[] describing the corners of the rectangular
75     *         region. The first and last points are opposed on the diagonal, as
76     *         are the second and third. The first point will be the topmost
77     *         point and the last, the bottommost. The second point will be
78     *         leftmost and the third, the rightmost
79     * @throws NotFoundException if no Data Matrix Code can be found
80     */
81    public ResultPoint[] detect() throws NotFoundException {
82  
83      int left = leftInit;
84      int right = rightInit;
85      int up = upInit;
86      int down = downInit;
87      boolean sizeExceeded = false;
88      boolean aBlackPointFoundOnBorder = true;
89      boolean atLeastOneBlackPointFoundOnBorder = false;
90      
91      boolean atLeastOneBlackPointFoundOnRight = false;
92      boolean atLeastOneBlackPointFoundOnBottom = false;
93      boolean atLeastOneBlackPointFoundOnLeft = false;
94      boolean atLeastOneBlackPointFoundOnTop = false;
95  
96      while (aBlackPointFoundOnBorder) {
97  
98        aBlackPointFoundOnBorder = false;
99  
100       // .....
101       // .   |
102       // .....
103       boolean rightBorderNotWhite = true;
104       while ((rightBorderNotWhite || !atLeastOneBlackPointFoundOnRight) && right < width) {
105         rightBorderNotWhite = containsBlackPoint(up, down, right, false);
106         if (rightBorderNotWhite) {
107           right++;
108           aBlackPointFoundOnBorder = true;
109           atLeastOneBlackPointFoundOnRight = true;
110         } else if (!atLeastOneBlackPointFoundOnRight) {
111           right++;
112         }
113       }
114 
115       if (right >= width) {
116         sizeExceeded = true;
117         break;
118       }
119 
120       // .....
121       // .   .
122       // .___.
123       boolean bottomBorderNotWhite = true;
124       while ((bottomBorderNotWhite || !atLeastOneBlackPointFoundOnBottom) && down < height) {
125         bottomBorderNotWhite = containsBlackPoint(left, right, down, true);
126         if (bottomBorderNotWhite) {
127           down++;
128           aBlackPointFoundOnBorder = true;
129           atLeastOneBlackPointFoundOnBottom = true;
130         } else if (!atLeastOneBlackPointFoundOnBottom) {
131           down++;
132         }
133       }
134 
135       if (down >= height) {
136         sizeExceeded = true;
137         break;
138       }
139 
140       // .....
141       // |   .
142       // .....
143       boolean leftBorderNotWhite = true;
144       while ((leftBorderNotWhite || !atLeastOneBlackPointFoundOnLeft) && left >= 0) {
145         leftBorderNotWhite = containsBlackPoint(up, down, left, false);
146         if (leftBorderNotWhite) {
147           left--;
148           aBlackPointFoundOnBorder = true;
149           atLeastOneBlackPointFoundOnLeft = true;
150         } else if (!atLeastOneBlackPointFoundOnLeft) {
151           left--;
152         }
153       }
154 
155       if (left < 0) {
156         sizeExceeded = true;
157         break;
158       }
159 
160       // .___.
161       // .   .
162       // .....
163       boolean topBorderNotWhite = true;
164       while ((topBorderNotWhite  || !atLeastOneBlackPointFoundOnTop) && up >= 0) {
165         topBorderNotWhite = containsBlackPoint(left, right, up, true);
166         if (topBorderNotWhite) {
167           up--;
168           aBlackPointFoundOnBorder = true;
169           atLeastOneBlackPointFoundOnTop = true;
170         } else if (!atLeastOneBlackPointFoundOnTop) {
171           up--;
172         }
173       }
174 
175       if (up < 0) {
176         sizeExceeded = true;
177         break;
178       }
179 
180       if (aBlackPointFoundOnBorder) {
181         atLeastOneBlackPointFoundOnBorder = true;
182       }
183 
184     }
185 
186     if (!sizeExceeded && atLeastOneBlackPointFoundOnBorder) {
187 
188       int maxSize = right - left;
189 
190       ResultPoint z = null;
191       for (int i = 1; i < maxSize; i++) {
192         z = getBlackPointOnSegment(left, down - i, left + i, down);
193         if (z != null) {
194           break;
195         }
196       }
197 
198       if (z == null) {
199         throw NotFoundException.getNotFoundInstance();
200       }
201 
202       ResultPoint t = null;
203       //go down right
204       for (int i = 1; i < maxSize; i++) {
205         t = getBlackPointOnSegment(left, up + i, left + i, up);
206         if (t != null) {
207           break;
208         }
209       }
210 
211       if (t == null) {
212         throw NotFoundException.getNotFoundInstance();
213       }
214 
215       ResultPoint x = null;
216       //go down left
217       for (int i = 1; i < maxSize; i++) {
218         x = getBlackPointOnSegment(right, up + i, right - i, up);
219         if (x != null) {
220           break;
221         }
222       }
223 
224       if (x == null) {
225         throw NotFoundException.getNotFoundInstance();
226       }
227 
228       ResultPoint y = null;
229       //go up left
230       for (int i = 1; i < maxSize; i++) {
231         y = getBlackPointOnSegment(right, down - i, right - i, down);
232         if (y != null) {
233           break;
234         }
235       }
236 
237       if (y == null) {
238         throw NotFoundException.getNotFoundInstance();
239       }
240 
241       return centerEdges(y, z, x, t);
242 
243     } else {
244       throw NotFoundException.getNotFoundInstance();
245     }
246   }
247 
248   private ResultPoint getBlackPointOnSegment(float aX, float aY, float bX, float bY) {
249     int dist = MathUtils.round(MathUtils.distance(aX, aY, bX, bY));
250     float xStep = (bX - aX) / dist;
251     float yStep = (bY - aY) / dist;
252 
253     for (int i = 0; i < dist; i++) {
254       int x = MathUtils.round(aX + i * xStep);
255       int y = MathUtils.round(aY + i * yStep);
256       if (image.get(x, y)) {
257         return new ResultPoint(x, y);
258       }
259     }
260     return null;
261   }

263   /**
264    * recenters the points of a constant distance towards the center
265    *
266    * @param y bottom most point
267    * @param z left most point
268    * @param x right most point
269    * @param t top most point
270    * @return {@link ResultPoint}[] describing the corners of the rectangular
271    *         region. The first and last points are opposed on the diagonal, as
272    *         are the second and third. The first point will be the topmost
273    *         point and the last, the bottommost. The second point will be
274    *         leftmost and the third, the rightmost
275    */
276   private ResultPoint[] centerEdges(ResultPoint y, ResultPoint z,
277                                     ResultPoint x, ResultPoint t) {
278 
279     //
280     //       t            t
281     //  z                      x
282     //        x    OR    z
283     //   y                    y
284     //
285 
286     float yi = y.getX();
287     float yj = y.getY();
288     float zi = z.getX();
289     float zj = z.getY();
290     float xi = x.getX();
291     float xj = x.getY();
292     float ti = t.getX();
293     float tj = t.getY();
294 
295     if (yi < width / 2.0f) {
296       return new ResultPoint[]{
297           new ResultPoint(ti - CORR, tj + CORR),
298           new ResultPoint(zi + CORR, zj + CORR),
299           new ResultPoint(xi - CORR, xj - CORR),
300           new ResultPoint(yi + CORR, yj - CORR)};
301     } else {
302       return new ResultPoint[]{
303           new ResultPoint(ti + CORR, tj + CORR),
304           new ResultPoint(zi + CORR, zj - CORR),
305           new ResultPoint(xi - CORR, xj + CORR),
306           new ResultPoint(yi - CORR, yj - CORR)};
307     }
308   }
309 
310   /**
311    * Determines whether a segment contains a black point
312    *
313    * @param a          min value of the scanned coordinate
314    * @param b          max value of the scanned coordinate
315    * @param fixed      value of fixed coordinate
316    * @param horizontal set to true if scan must be horizontal, false if vertical
317    * @return true if a black point has been found, else false.
318    */
319   private boolean containsBlackPoint(int a, int b, int fixed, boolean horizontal) {
320 
321     if (horizontal) {
322       for (int x = a; x <= b; x++) {
323         if (image.get(x, fixed)) {
324           return true;
325         }
326       }
327     } else {
328       for (int y = a; y <= b; y++) {
329         if (image.get(fixed, y)) {
330           return true;
331         }
332       }
333     }
334 
335     return false;
336   }
