17  package com.google.zxing.common;

19  import java.util.Arrays;

21  /**
22   * <p>A simple, fast array of bits, represented compactly by an array of ints internally.</p>
23   *
24   * @author Sean Owen
25   */
26  public final class BitArray implements Cloneable {
27  
28    private int[] bits;
29    private int size;
30  
31    public BitArray() {
32      this.size = 0;
33      this.bits = new int[1];
34    }
35  
36    public BitArray(int size) {
37      this.size = size;
38      this.bits = makeArray(size);
39    }
40  
41    // For testing only
42    BitArray(int[] bits, int size) {
43      this.bits = bits;
44      this.size = size;
45    }
46  
47    public int getSize() {
48      return size;
49    }
50  
51    public int getSizeInBytes() {
52      return (size + 7) / 8;
53    }
54  
55    private void ensureCapacity(int size) {
56      if (size > bits.length * 32) {
57        int[] newBits = makeArray(size);
58        System.arraycopy(bits, 0, newBits, 0, bits.length);
59        this.bits = newBits;
60      }
61    }
62  
63    /**
64     * @param i bit to get
65     * @return true iff bit i is set
66     */
67    public boolean get(int i) {
68      return (bits[i / 32] & (1 << (i & 0x1F))) != 0;
69    }
70  
71    /**
72     * Sets bit i.
73     *
74     * @param i bit to set
75     */
76    public void set(int i) {
77      bits[i / 32] |= 1 << (i & 0x1F);
78    }
79  
80    /**
81     * Flips bit i.
82     *
83     * @param i bit to set
84     */
85    public void flip(int i) {
86      bits[i / 32] ^= 1 << (i & 0x1F);
87    }
88  
89    /**
90     * @param from first bit to check
91     * @return index of first bit that is set, starting from the given index, or size if none are set
92     *  at or beyond this given index
93     * @see #getNextUnset(int)
94     */
95    public int getNextSet(int from) {
96      if (from >= size) {
97        return size;
98      }
99      int bitsOffset = from / 32;
100     int currentBits = bits[bitsOffset];
101     // mask off lesser bits first
102     currentBits &= ~((1 << (from & 0x1F)) - 1);
103     while (currentBits == 0) {
104       if (++bitsOffset == bits.length) {
105         return size;
106       }
107       currentBits = bits[bitsOffset];
108     }
109     int result = (bitsOffset * 32) + Integer.numberOfTrailingZeros(currentBits);
110     return result > size ? size : result;
111   }
112 
113   /**
114    * @see #getNextSet(int)
115    */
116   public int getNextUnset(int from) {
117     if (from >= size) {
118       return size;
119     }
120     int bitsOffset = from / 32;
121     int currentBits = ~bits[bitsOffset];
122     // mask off lesser bits first
123     currentBits &= ~((1 << (from & 0x1F)) - 1);
124     while (currentBits == 0) {
125       if (++bitsOffset == bits.length) {
126         return size;
127       }
128       currentBits = ~bits[bitsOffset];
129     }
130     int result = (bitsOffset * 32) + Integer.numberOfTrailingZeros(currentBits);
131     return result > size ? size : result;
132   }

134   /**
135    * Sets a block of 32 bits, starting at bit i.
136    *
137    * @param i first bit to set
138    * @param newBits the new value of the next 32 bits. Note again that the least-significant bit
139    * corresponds to bit i, the next-least-significant to i+1, and so on.
140    */
141   public void setBulk(int i, int newBits) {
142     bits[i / 32] = newBits;
143   }
144 
145   /**
146    * Sets a range of bits.
147    *
148    * @param start start of range, inclusive.
149    * @param end end of range, exclusive
150    */
151   public void setRange(int start, int end) {
152     if (end < start) {
153       throw new IllegalArgumentException();
154     }
155     if (end == start) {
156       return;
157     }
158     end--; // will be easier to treat this as the last actually set bit -- inclusive
159     int firstInt = start / 32;
160     int lastInt = end / 32;
161     for (int i = firstInt; i <= lastInt; i++) {
162       int firstBit = i > firstInt ? 0 : start & 0x1F;
163       int lastBit = i < lastInt ? 31 : end & 0x1F;
164       int mask;
165       if (firstBit == 0 && lastBit == 31) {
166         mask = -1;
167       } else {
168         mask = 0;
169         for (int j = firstBit; j <= lastBit; j++) {
170           mask |= 1 << j;
171         }
172       }
173       bits[i] |= mask;
174     }
175   }
176 
177   /**
178    * Clears all bits (sets to false).
179    */
180   public void clear() {
181     int max = bits.length;
182     for (int i = 0; i < max; i++) {
183       bits[i] = 0;
184     }
185   }
186 
187   /**
188    * Efficient method to check if a range of bits is set, or not set.
189    *
190    * @param start start of range, inclusive.
191    * @param end end of range, exclusive
192    * @param value if true, checks that bits in range are set, otherwise checks that they are not set
193    * @return true iff all bits are set or not set in range, according to value argument
194    * @throws IllegalArgumentException if end is less than or equal to start
195    */
196   public boolean isRange(int start, int end, boolean value) {
197     if (end < start) {
198       throw new IllegalArgumentException();
199     }
200     if (end == start) {
201       return true; // empty range matches
202     }
203     end--; // will be easier to treat this as the last actually set bit -- inclusive
204     int firstInt = start / 32;
205     int lastInt = end / 32;
206     for (int i = firstInt; i <= lastInt; i++) {
207       int firstBit = i > firstInt ? 0 : start & 0x1F;
208       int lastBit = i < lastInt ? 31 : end & 0x1F;
209       int mask;
210       if (firstBit == 0 && lastBit == 31) {
211         mask = -1;
212       } else {
213         mask = 0;
214         for (int j = firstBit; j <= lastBit; j++) {
215           mask |= 1 << j;
216         }
217       }
218 
219       // Return false if we're looking for 1s and the masked bits[i] isn't all 1s (that is,
220       // equals the mask, or we're looking for 0s and the masked portion is not all 0s
221       if ((bits[i] & mask) != (value ? mask : 0)) {
222         return false;
223       }
224     }
225     return true;
226   }

228   public void appendBit(boolean bit) {
229     ensureCapacity(size + 1);
230     if (bit) {
231       bits[size / 32] |= 1 << (size & 0x1F);
232     }
233     size++;
234   }
235 
236   /**
237    * Appends the least-significant bits, from value, in order from most-significant to
238    * least-significant. For example, appending 6 bits from 0x000001E will append the bits
239    * 0, 1, 1, 1, 1, 0 in that order.
240    */
241   public void appendBits(int value, int numBits) {
242     if (numBits < 0 || numBits > 32) {
243       throw new IllegalArgumentException("Num bits must be between 0 and 32");
244     }
245     ensureCapacity(size + numBits);
246     for (int numBitsLeft = numBits; numBitsLeft > 0; numBitsLeft--) {
247       appendBit(((value >> (numBitsLeft - 1)) & 0x01) == 1);
248     }
249   }
250 
251   public void appendBitArray(BitArray other) {
252     int otherSize = other.size;
253     ensureCapacity(size + otherSize);
254     for (int i = 0; i < otherSize; i++) {
255       appendBit(other.get(i));
256     }
257   }
258 
259   public void xor(BitArray other) {
260     if (bits.length != other.bits.length) {
261       throw new IllegalArgumentException("Sizes don't match");
262     }
263     for (int i = 0; i < bits.length; i++) {
264       // The last byte could be incomplete (i.e. not have 8 bits in
265       // it) but there is no problem since 0 XOR 0 == 0.
266       bits[i] ^= other.bits[i];
267     }
268   }
269 
270   /**
271    *
272    * @param bitOffset first bit to start writing
273    * @param array array to write into. Bytes are written most-significant byte first. This is the opposite
274    *  of the internal representation, which is exposed by {@link #getBitArray()}
275    * @param offset position in array to start writing
276    * @param numBytes how many bytes to write
277    */
278   public void toBytes(int bitOffset, byte[] array, int offset, int numBytes) {
279     for (int i = 0; i < numBytes; i++) {
280       int theByte = 0;
281       for (int j = 0; j < 8; j++) {
282         if (get(bitOffset)) {
283           theByte |= 1 << (7 - j);
284         }
285         bitOffset++;
286       }
287       array[offset + i] = (byte) theByte;
288     }
289   }

291   /**
292    * @return underlying array of ints. The first element holds the first 32 bits, and the least
293    *         significant bit is bit 0.
294    */
295   public int[] getBitArray() {
296     return bits;
297   }

299   /**
300    * Reverses all bits in the array.
301    */
302   public void reverse() {
303     int[] newBits = new int[bits.length];
304     // reverse all int's first
305     int len = ((size-1) / 32);
306     int oldBitsLen = len + 1;
307     for (int i = 0; i < oldBitsLen; i++) {
308       long x = (long) bits[i];
309       x = ((x >>  1) & 0x55555555L) | ((x & 0x55555555L) <<  1);
310       x = ((x >>  2) & 0x33333333L) | ((x & 0x33333333L) <<  2);
311       x = ((x >>  4) & 0x0f0f0f0fL) | ((x & 0x0f0f0f0fL) <<  4);
312       x = ((x >>  8) & 0x00ff00ffL) | ((x & 0x00ff00ffL) <<  8);
313       x = ((x >> 16) & 0x0000ffffL) | ((x & 0x0000ffffL) << 16);
314       newBits[len - i] = (int) x;
315     }
316     // now correct the int's if the bit size isn't a multiple of 32
317     if (size != oldBitsLen * 32) {
318       int leftOffset = oldBitsLen * 32 - size;
319       int mask = 1;
320       for (int i = 0; i < 31 - leftOffset; i++) {
321         mask = (mask << 1) | 1;
322       }
323       int currentInt = (newBits[0] >> leftOffset) & mask;
324       for (int i = 1; i < oldBitsLen; i++) {
325         int nextInt = newBits[i];
326         currentInt |= nextInt << (32 - leftOffset);
327         newBits[i - 1] = currentInt;
328         currentInt = (nextInt >> leftOffset) & mask;
329       }
330       newBits[oldBitsLen - 1] = currentInt;
331     }
332     bits = newBits;
333   }
334 
335   private static int[] makeArray(int size) {
336     return new int[(size + 31) / 32];
337   }
338 
339   @Override
340   public boolean equals(Object o) {
341     if (!(o instanceof BitArray)) {
342       return false;
343     }
344     BitArray other = (BitArray) o;
345     return size == other.size && Arrays.equals(bits, other.bits);
346   }
347 
348   @Override
349   public int hashCode() {
350     return 31 * size + Arrays.hashCode(bits);
351   }
352 
353   @Override
354   public String toString() {
355     StringBuilder result = new StringBuilder(size);
356     for (int i = 0; i < size; i++) {
357       if ((i & 0x07) == 0) {
358         result.append(' ');
359       }
360       result.append(get(i) ? 'X' : '.');
361     }
362     return result.toString();
363   }
364 
365   @Override
366   public BitArray clone() {
367     return new BitArray(bits.clone(), size);
368   }
