17  package com.google.zxing.common;
18  
19  import java.util.List;
20  
21  /**
22   * <p>Encapsulates the result of decoding a matrix of bits. This typically
23   * applies to 2D barcode formats. For now it contains the raw bytes obtained,
24   * as well as a String interpretation of those bytes, if applicable.</p>
25   *
26   * @author Sean Owen
27   */
28  public final class DecoderResult {
29  
30    private final byte[] rawBytes;
31    private final String text;
32    private final List<byte[]> byteSegments;
33    private final String ecLevel;
34    private Integer errorsCorrected;
35    private Integer erasures;
36    private Object other;
37    private final int structuredAppendParity;
38    private final int structuredAppendSequenceNumber;
39  
40    public DecoderResult(byte[] rawBytes,
41                         String text,
42                         List<byte[]> byteSegments,
43                         String ecLevel) {
44      this(rawBytes, text, byteSegments, ecLevel, -1, -1);
45    }
46  
47    public DecoderResult(byte[] rawBytes,
48                         String text,
49                         List<byte[]> byteSegments,
50                         String ecLevel,
51                         int saSequence,
52                         int saParity) {
53      this.rawBytes = rawBytes;
54      this.text = text;
55      this.byteSegments = byteSegments;
56      this.ecLevel = ecLevel;
57      this.structuredAppendParity = saParity;
58      this.structuredAppendSequenceNumber = saSequence;
59    }
60  
61    public byte[] getRawBytes() {
62      return rawBytes;
63    }
64  
65    public String getText() {
66      return text;
67    }
68  
69    public List<byte[]> getByteSegments() {
70      return byteSegments;
71    }
72  
73    public String getECLevel() {
74      return ecLevel;
75    }
76  
77    public Integer getErrorsCorrected() {
78      return errorsCorrected;
79    }
80  
81    public void setErrorsCorrected(Integer errorsCorrected) {
82      this.errorsCorrected = errorsCorrected;
83    }
84  
85    public Integer getErasures() {
86      return erasures;
87    }
88  
89    public void setErasures(Integer erasures) {
90      this.erasures = erasures;
91    }
92    
93    public Object getOther() {
94      return other;
95    }
96  
97    public void setOther(Object other) {
98      this.other = other;
99    }
100   
101   public boolean hasStructuredAppend() {
102     return structuredAppendParity >= 0 && structuredAppendSequenceNumber >= 0;
103   }
104   
105   public int getStructuredAppendParity() {
106     return structuredAppendParity;
107   }
108   
109   public int getStructuredAppendSequenceNumber() {
110     return structuredAppendSequenceNumber;
111   }
112   
113 }
