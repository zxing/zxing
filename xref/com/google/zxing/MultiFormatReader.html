17  package com.google.zxing;
18  
19  import com.google.zxing.aztec.AztecReader;
20  import com.google.zxing.datamatrix.DataMatrixReader;
21  import com.google.zxing.maxicode.MaxiCodeReader;
22  import com.google.zxing.oned.MultiFormatOneDReader;
23  import com.google.zxing.pdf417.PDF417Reader;
24  import com.google.zxing.qrcode.QRCodeReader;
25  
26  import java.util.ArrayList;
27  import java.util.Collection;
28  import java.util.Map;
29  
30  /**
31   * MultiFormatReader is a convenience class and the main entry point into the library for most uses.
32   * By default it attempts to decode all barcode formats that the library supports. Optionally, you
33   * can provide a hints object to request different behavior, for example only decoding QR codes.
34   *
35   * @author Sean Owen
36   * @author dswitkin@google.com (Daniel Switkin)
37   */
38  public final class MultiFormatReader implements Reader {
39  
40    private Map<DecodeHintType,?> hints;
41    private Reader[] readers;
42  
43    /**
44     * This version of decode honors the intent of Reader.decode(BinaryBitmap) in that it
45     * passes null as a hint to the decoders. However, that makes it inefficient to call repeatedly.
46     * Use setHints() followed by decodeWithState() for continuous scan applications.
47     *
48     * @param image The pixel data to decode
49     * @return The contents of the image
50     * @throws NotFoundException Any errors which occurred
51     */
52    @Override
53    public Result decode(BinaryBitmap image) throws NotFoundException {
54      setHints(null);
55      return decodeInternal(image);
56    }
57  
58    /**
59     * Decode an image using the hints provided. Does not honor existing state.
60     *
61     * @param image The pixel data to decode
62     * @param hints The hints to use, clearing the previous state.
63     * @return The contents of the image
64     * @throws NotFoundException Any errors which occurred
65     */
66    @Override
67    public Result decode(BinaryBitmap image, Map<DecodeHintType,?> hints) throws NotFoundException {
68      setHints(hints);
69      return decodeInternal(image);
70    }
71  
72    /**
73     * Decode an image using the state set up by calling setHints() previously. Continuous scan
74     * clients will get a <b>large</b> speed increase by using this instead of decode().
75     *
76     * @param image The pixel data to decode
77     * @return The contents of the image
78     * @throws NotFoundException Any errors which occurred
79     */
80    public Result decodeWithState(BinaryBitmap image) throws NotFoundException {
81      // Make sure to set up the default state so we don't crash
82      if (readers == null) {
83        setHints(null);
84      }
85      return decodeInternal(image);
86    }
87  
88    /**
89     * This method adds state to the MultiFormatReader. By setting the hints once, subsequent calls
90     * to decodeWithState(image) can reuse the same set of readers without reallocating memory. This
91     * is important for performance in continuous scan clients.
92     *
93     * @param hints The set of hints to use for subsequent calls to decode(image)
94     */
95    public void setHints(Map<DecodeHintType,?> hints) {
96      this.hints = hints;
97  
98      boolean tryHarder = hints != null && hints.containsKey(DecodeHintType.TRY_HARDER);
99      @SuppressWarnings("unchecked")
100     Collection<BarcodeFormat> formats =
101         hints == null ? null : (Collection<BarcodeFormat>) hints.get(DecodeHintType.POSSIBLE_FORMATS);
102     Collection<Reader> readers = new ArrayList<>();
103     if (formats != null) {
104       boolean addOneDReader =
105           formats.contains(BarcodeFormat.UPC_A) ||
106           formats.contains(BarcodeFormat.UPC_E) ||
107           formats.contains(BarcodeFormat.EAN_13) ||
108           formats.contains(BarcodeFormat.EAN_8) ||
109           formats.contains(BarcodeFormat.CODABAR) ||
110           formats.contains(BarcodeFormat.CODE_39) ||
111           formats.contains(BarcodeFormat.CODE_93) ||
112           formats.contains(BarcodeFormat.CODE_128) ||
113           formats.contains(BarcodeFormat.ITF) ||
114           formats.contains(BarcodeFormat.RSS_14) ||
115           formats.contains(BarcodeFormat.RSS_EXPANDED);
116       // Put 1D readers upfront in "normal" mode
117       if (addOneDReader && !tryHarder) {
118         readers.add(new MultiFormatOneDReader(hints));
119       }
120       if (formats.contains(BarcodeFormat.QR_CODE)) {
121         readers.add(new QRCodeReader());
122       }
123       if (formats.contains(BarcodeFormat.DATA_MATRIX)) {
124         readers.add(new DataMatrixReader());
125       }
126       if (formats.contains(BarcodeFormat.AZTEC)) {
127         readers.add(new AztecReader());
128       }
129       if (formats.contains(BarcodeFormat.PDF_417)) {
130          readers.add(new PDF417Reader());
131       }
132       if (formats.contains(BarcodeFormat.MAXICODE)) {
133          readers.add(new MaxiCodeReader());
134       }
135       // At end in "try harder" mode
136       if (addOneDReader && tryHarder) {
137         readers.add(new MultiFormatOneDReader(hints));
138       }
139     }
140     if (readers.isEmpty()) {
141       if (!tryHarder) {
142         readers.add(new MultiFormatOneDReader(hints));
143       }
144 
145       readers.add(new QRCodeReader());
146       readers.add(new DataMatrixReader());
147       readers.add(new AztecReader());
148       readers.add(new PDF417Reader());
149       readers.add(new MaxiCodeReader());
150 
151       if (tryHarder) {
152         readers.add(new MultiFormatOneDReader(hints));
153       }
154     }
155     this.readers = readers.toArray(new Reader[readers.size()]);
156   }
157 
158   @Override
159   public void reset() {
160     if (readers != null) {
161       for (Reader reader : readers) {
162         reader.reset();
163       }
164     }
165   }
166 
167   private Result decodeInternal(BinaryBitmap image) throws NotFoundException {
168     if (readers != null) {
169       for (Reader reader : readers) {
170         try {
171           return reader.decode(image, hints);
172         } catch (ReaderException re) {
173           // continue
174         }
175       }
176     }
177     throw NotFoundException.getNotFoundInstance();
178   }
179 
180 }
