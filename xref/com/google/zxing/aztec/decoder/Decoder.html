73    public DecoderResult decode(AztecDetectorResult detectorResult) throws FormatException {
74      ddata = detectorResult;
75      BitMatrix matrix = detectorResult.getBits();
76      boolean[] rawbits = extractBits(matrix);
77      boolean[] correctedBits = correctBits(rawbits);
78      String result = getEncodedData(correctedBits);
79      return new DecoderResult(null, result, null, null);
80    }

84      return getEncodedData(correctedBits);

87    /**
88     * Gets the string encoded in the aztec code bits
89     *
90     * @return the decoded string
91     */
92    private static String getEncodedData(boolean[] correctedBits) {
93      int endIndex = correctedBits.length;
94      Table latchTable = Table.UPPER; // table most recently latched to
95      Table shiftTable = Table.UPPER; // table to use for the next read
96      StringBuilder result = new StringBuilder(20);
97      int index = 0;
98      while (index < endIndex) {
99        if (shiftTable == Table.BINARY) {
100         if (endIndex - index < 5) {
101           break;
102         }
103         int length = readCode(correctedBits, index, 5);
104         index += 5;
105         if (length == 0) {
106           if (endIndex - index < 11) {
107             break;
108           }
109           length = readCode(correctedBits, index, 11) + 31;
110           index += 11;
111         }
112         for (int charCount = 0; charCount < length; charCount++) {
113           if (endIndex - index < 8) {
114             index = endIndex;  // Force outer loop to exit
115             break;
116           }
117           int code = readCode(correctedBits, index, 8);
118           result.append((char) code);
119           index += 8;
120         }
121         // Go back to whatever mode we had been in
122         shiftTable = latchTable;
123       } else {
124         int size = shiftTable == Table.DIGIT ? 4 : 5;
125         if (endIndex - index < size) {
126           break;
127         }
128         int code = readCode(correctedBits, index, size);
129         index += size;
130         String str = getCharacter(shiftTable, code);
131         if (str.startsWith("CTRL_")) {
132           // Table changes
133           shiftTable = getTable(str.charAt(5));
134           if (str.charAt(6) == 'L') {
135             latchTable = shiftTable;
136           }
137         } else {
138           result.append(str);
139           // Go back to whatever mode we had been in
140           shiftTable = latchTable;
141         }
142       }
143     }
144     return result.toString();
145   }

147   /**
148    * gets the table corresponding to the char passed
149    */
150   private static Table getTable(char t) {
151     switch (t) {
152       case 'L':
153         return Table.LOWER;
154       case 'P':
155         return Table.PUNCT;
156       case 'M':
157         return Table.MIXED;
158       case 'D':
159         return Table.DIGIT;
160       case 'B':
161         return Table.BINARY;
162       case 'U':
163       default:
164         return Table.UPPER;
165     }
166   }

168   /**
169    * Gets the character (or string) corresponding to the passed code in the given table
170    *
171    * @param table the table used
172    * @param code the code of the character
173    */
174   private static String getCharacter(Table table, int code) {
175     switch (table) {
176       case UPPER:
177         return UPPER_TABLE[code];
178       case LOWER:
179         return LOWER_TABLE[code];
180       case MIXED:
181         return MIXED_TABLE[code];
182       case PUNCT:
183         return PUNCT_TABLE[code];
184       case DIGIT:
185         return DIGIT_TABLE[code];
186       default:
187         // Should not reach here.
188         throw new IllegalStateException("Bad table");
189     }
190   }

192   /**
193    * <p>Performs RS error correction on an array of bits.</p>
194    *
195    * @return the corrected array
196    * @throws FormatException if the input contains too many errors
197    */
198   private boolean[] correctBits(boolean[] rawbits) throws FormatException {
199     GenericGF gf;
200     int codewordSize;
201 
202     if (ddata.getNbLayers() <= 2) {
203       codewordSize = 6;
204       gf = GenericGF.AZTEC_DATA_6;
205     } else if (ddata.getNbLayers() <= 8) {
206       codewordSize = 8;
207       gf = GenericGF.AZTEC_DATA_8;
208     } else if (ddata.getNbLayers() <= 22) {
209       codewordSize = 10;
210       gf = GenericGF.AZTEC_DATA_10;
211     } else {
212       codewordSize = 12;
213       gf = GenericGF.AZTEC_DATA_12;
214     }
215 
216     int numDataCodewords = ddata.getNbDatablocks();
217     int numCodewords = rawbits.length / codewordSize;
218     if (numCodewords < numDataCodewords) {
219       throw FormatException.getFormatInstance();
220     }
221     int offset = rawbits.length % codewordSize;
222     int numECCodewords = numCodewords - numDataCodewords;
223 
224     int[] dataWords = new int[numCodewords];
225     for (int i = 0; i < numCodewords; i++, offset += codewordSize) {
226       dataWords[i] = readCode(rawbits, offset, codewordSize);
227     }
228 
229     try {
230       ReedSolomonDecoder rsDecoder = new ReedSolomonDecoder(gf);
231       rsDecoder.decode(dataWords, numECCodewords);
232     } catch (ReedSolomonException ignored) {
233       throw FormatException.getFormatInstance();
234     }
235 
236     // Now perform the unstuffing operation.
237     // First, count how many bits are going to be thrown out as stuffing
238     int mask = (1 << codewordSize) - 1;
239     int stuffedBits = 0;
240     for (int i = 0; i < numDataCodewords; i++) {
241       int dataWord = dataWords[i];
242       if (dataWord == 0 || dataWord == mask) {
243         throw FormatException.getFormatInstance();
244       } else if (dataWord == 1 || dataWord == mask - 1) {
245         stuffedBits++;
246       }
247     }
248     // Now, actually unpack the bits and remove the stuffing
249     boolean[] correctedBits = new boolean[numDataCodewords * codewordSize - stuffedBits];
250     int index = 0;
251     for (int i = 0; i < numDataCodewords; i++) {
252       int dataWord = dataWords[i];
253       if (dataWord == 1 || dataWord == mask - 1) {
254         // next codewordSize-1 bits are all zeros or all ones
255         Arrays.fill(correctedBits, index, index + codewordSize - 1, dataWord > 1);
256         index += codewordSize - 1;
257       } else {
258         for (int bit = codewordSize - 1; bit >= 0; --bit) {
259           correctedBits[index++] = (dataWord & (1 << bit)) != 0;
260         }
261       }
262     }
263     return correctedBits;
264   }

266   /**
267    * Gets the array of bits from an Aztec Code matrix
268    *
269    * @return the array of bits
270    */
271   boolean[] extractBits(BitMatrix matrix) {
272     boolean compact = ddata.isCompact();
273     int layers = ddata.getNbLayers();
274     int baseMatrixSize = compact ? 11 + layers * 4 : 14 + layers * 4; // not including alignment lines
275     int[] alignmentMap = new int[baseMatrixSize];
276     boolean[] rawbits = new boolean[totalBitsInLayer(layers, compact)];
277 
278     if (compact) {
279       for (int i = 0; i < alignmentMap.length; i++) {
280         alignmentMap[i] = i;
281       }
282     } else {
283       int matrixSize = baseMatrixSize + 1 + 2 * ((baseMatrixSize / 2 - 1) / 15);
284       int origCenter = baseMatrixSize / 2;
285       int center = matrixSize / 2;
286       for (int i = 0; i < origCenter; i++) {
287         int newOffset = i + i / 15;
288         alignmentMap[origCenter - i - 1] = center - newOffset - 1;
289         alignmentMap[origCenter + i] = center + newOffset + 1;
290       }
291     }
292     for (int i = 0, rowOffset = 0; i < layers; i++) {
293       int rowSize = compact ? (layers - i) * 4 + 9 : (layers - i) * 4 + 12;
294       // The top-left most point of this layer is <low, low> (not including alignment lines)
295       int low = i * 2;
296       // The bottom-right most point of this layer is <high, high> (not including alignment lines)
297       int high = baseMatrixSize - 1 - low;
298       // We pull bits from the two 2 x rowSize columns and two rowSize x 2 rows
299       for (int j = 0; j < rowSize; j++) {
300         int columnOffset = j * 2;
301         for (int k = 0; k < 2; k++) {
302           // left column
303           rawbits[rowOffset + columnOffset + k] =
304               matrix.get(alignmentMap[low + k], alignmentMap[low + j]);
305           // bottom row
306           rawbits[rowOffset + 2 * rowSize + columnOffset + k] =
307               matrix.get(alignmentMap[low + j], alignmentMap[high - k]);
308           // right column
309           rawbits[rowOffset + 4 * rowSize + columnOffset + k] =
310               matrix.get(alignmentMap[high - k], alignmentMap[high - j]);
311           // top row
312           rawbits[rowOffset + 6 * rowSize + columnOffset + k] =
313               matrix.get(alignmentMap[high - j], alignmentMap[low + k]);
314         }
315       }
316       rowOffset += rowSize * 8;
317     }
318     return rawbits;
319   }

321   /**
322    * Reads a code of given length and at given index in an array of bits
323    */
324   private static int readCode(boolean[] rawbits, int startIndex, int length) {
325     int res = 0;
326     for (int i = startIndex; i < startIndex + length; i++) {
327       res <<= 1;
328       if (rawbits[i]) {
329         res |= 0x01;
330       }
331     }
332     return res;
333   }
334 
335   private static int totalBitsInLayer(int layers, boolean compact) {
336     return ((compact ? 88 : 112) + 16 * layers) * layers;
337   }
