55    /**
56     * Detects an Aztec Code in an image.
57     *
58     * @param isMirror if true, image is a mirror-image of original
59     * @return {@link AztecDetectorResult} encapsulating results of detecting an Aztec Code
60     * @throws NotFoundException if no Aztec Code can be found
61     */
62     public AztecDetectorResult detect(boolean isMirror) throws NotFoundException {
63  
64      // 1. Get the center of the aztec matrix
65      Point pCenter = getMatrixCenter();
66  
67      // 2. Get the center points of the four diagonal points just outside the bull's eye
68      //  [topRight, bottomRight, bottomLeft, topLeft]
69      ResultPoint[] bullsEyeCorners = getBullsEyeCorners(pCenter);
70  
71      if (isMirror) {
72        ResultPoint temp = bullsEyeCorners[0];
73        bullsEyeCorners[0] = bullsEyeCorners[2];
74        bullsEyeCorners[2] = temp;
75      }
76  
77      // 3. Get the size of the matrix and other parameters from the bull's eye
78      extractParameters(bullsEyeCorners);
79      
80      // 4. Sample the grid
81      BitMatrix bits = sampleGrid(image,
82                                  bullsEyeCorners[shift % 4], 
83                                  bullsEyeCorners[(shift + 1) % 4],
84                                  bullsEyeCorners[(shift + 2) % 4], 
85                                  bullsEyeCorners[(shift + 3) % 4]);
86  
87      // 5. Get the corners of the matrix.
88      ResultPoint[] corners = getMatrixCornerPoints(bullsEyeCorners);
89      
90      return new AztecDetectorResult(bits, corners, compact, nbDataBlocks, nbLayers);
91    }

93    /**
94     * Extracts the number of data layers and data blocks from the layer around the bull's eye.
95     *
96     * @param bullsEyeCorners the array of bull's eye corners
97     * @throws NotFoundException in case of too many errors or invalid parameters
98     */
99    private void extractParameters(ResultPoint[] bullsEyeCorners) throws NotFoundException {
100     if (!isValid(bullsEyeCorners[0]) || !isValid(bullsEyeCorners[1]) ||
101         !isValid(bullsEyeCorners[2]) || !isValid(bullsEyeCorners[3])) {
102       throw NotFoundException.getNotFoundInstance();
103     }
104     int length = 2 * nbCenterLayers;
105     // Get the bits around the bull's eye
106     int[] sides = {
107         sampleLine(bullsEyeCorners[0], bullsEyeCorners[1], length), // Right side
108         sampleLine(bullsEyeCorners[1], bullsEyeCorners[2], length), // Bottom 
109         sampleLine(bullsEyeCorners[2], bullsEyeCorners[3], length), // Left side
110         sampleLine(bullsEyeCorners[3], bullsEyeCorners[0], length)  // Top 
111     };
112 
113     // bullsEyeCorners[shift] is the corner of the bulls'eye that has three 
114     // orientation marks.  
115     // sides[shift] is the row/column that goes from the corner with three
116     // orientation marks to the corner with two.
117     shift = getRotation(sides, length);
118 
119     // Flatten the parameter bits into a single 28- or 40-bit long
120     long parameterData = 0;
121     for (int i = 0; i < 4; i++) {
122       int side = sides[(shift + i) % 4];
123       if (compact) {
124         // Each side of the form ..XXXXXXX. where Xs are parameter data
125         parameterData <<= 7;
126         parameterData += (side >> 1) & 0x7F;
127       } else {
128         // Each side of the form ..XXXXX.XXXXX. where Xs are parameter data
129         parameterData <<= 10;
130         parameterData += ((side >> 2) & (0x1f << 5)) + ((side >> 1) & 0x1F);
131       }
132     }
133     
134     // Corrects parameter data using RS.  Returns just the data portion
135     // without the error correction.
136     int correctedData = getCorrectedParameterData(parameterData, compact);
137     
138     if (compact) {
139       // 8 bits:  2 bits layers and 6 bits data blocks
140       nbLayers = (correctedData >> 6) + 1;
141       nbDataBlocks = (correctedData & 0x3F) + 1;
142     } else {
143       // 16 bits:  5 bits layers and 11 bits data blocks
144       nbLayers = (correctedData >> 11) + 1;
145       nbDataBlocks = (correctedData & 0x7FF) + 1;
146     }
147   }

149   private static final int[] EXPECTED_CORNER_BITS = {
150       0xee0,  // 07340  XXX .XX X.. ...
151       0x1dc,  // 00734  ... XXX .XX X..
152       0x83b,  // 04073  X.. ... XXX .XX
153       0x707,  // 03407 .XX X.. ... XXX
154   };
155 
156   private static int getRotation(int[] sides, int length) throws NotFoundException {
157     // In a normal pattern, we expect to See
158     //   **    .*             D       A
159     //   *      *
160     //
161     //   .      *
162     //   ..    ..             C       B
163     //
164     // Grab the 3 bits from each of the sides the form the locator pattern and concatenate
165     // into a 12-bit integer.  Start with the bit at A
166     int cornerBits = 0;
167     for (int side : sides) {
168       // XX......X where X's are orientation marks
169       int t = ((side >> (length - 2)) << 1) + (side & 1);
170       cornerBits = (cornerBits << 3) + t;
171     }
172     // Mov the bottom bit to the top, so that the three bits of the locator pattern at A are
173     // together.  cornerBits is now:
174     //  3 orientation bits at A || 3 orientation bits at B || ... || 3 orientation bits at D
175     cornerBits = ((cornerBits & 1) << 11) + (cornerBits >> 1);
176     // The result shift indicates which element of BullsEyeCorners[] goes into the top-left
177     // corner. Since the four rotation values have a Hamming distance of 8, we
178     // can easily tolerate two errors.
179     for (int shift = 0; shift < 4; shift++) {
180       if (Integer.bitCount(cornerBits ^ EXPECTED_CORNER_BITS[shift]) <= 2) {
181         return shift;
182       }
183     }
184     throw NotFoundException.getNotFoundInstance();
185   }

187   /**
188    * Corrects the parameter bits using Reed-Solomon algorithm.
189    *
190    * @param parameterData parameter bits
191    * @param compact true if this is a compact Aztec code
192    * @throws NotFoundException if the array contains too many errors
193    */
194   private static int getCorrectedParameterData(long parameterData, boolean compact) throws NotFoundException {
195     int numCodewords;
196     int numDataCodewords;
197 
198     if (compact) {
199       numCodewords = 7;
200       numDataCodewords = 2;
201     } else {
202       numCodewords = 10;
203       numDataCodewords = 4;
204     }
205 
206     int numECCodewords = numCodewords - numDataCodewords;
207     int[] parameterWords = new int[numCodewords];
208     for (int i = numCodewords - 1; i >= 0; --i) {
209       parameterWords[i] = (int) parameterData & 0xF;
210       parameterData >>= 4;
211     }
212     try {
213       ReedSolomonDecoder rsDecoder = new ReedSolomonDecoder(GenericGF.AZTEC_PARAM);
214       rsDecoder.decode(parameterWords, numECCodewords);
215     } catch (ReedSolomonException ignored) {
216       throw NotFoundException.getNotFoundInstance();
217     }
218     // Toss the error correction.  Just return the data as an integer
219     int result = 0;
220     for (int i = 0; i < numDataCodewords; i++) {
221       result = (result << 4) + parameterWords[i];
222     }
223     return result;
224   }

226   /**
227    * Finds the corners of a bull-eye centered on the passed point.
228    * This returns the centers of the diagonal points just outside the bull's eye
229    * Returns [topRight, bottomRight, bottomLeft, topLeft]
230    * 
231    * @param pCenter Center point
232    * @return The corners of the bull-eye
233    * @throws NotFoundException If no valid bull-eye can be found
234    */
235   private ResultPoint[] getBullsEyeCorners(Point pCenter) throws NotFoundException {
236     
237     Point pina = pCenter;
238     Point pinb = pCenter;
239     Point pinc = pCenter;
240     Point pind = pCenter;
241 
242     boolean color = true;
243     
244     for (nbCenterLayers = 1; nbCenterLayers < 9; nbCenterLayers++) {
245       Point pouta = getFirstDifferent(pina, color, 1, -1);
246       Point poutb = getFirstDifferent(pinb, color, 1, 1);
247       Point poutc = getFirstDifferent(pinc, color, -1, 1);
248       Point poutd = getFirstDifferent(pind, color, -1, -1);
249 
250       //d      a
251       //
252       //c      b
253 
254       if (nbCenterLayers > 2) {
255         float q = distance(poutd, pouta) * nbCenterLayers / (distance(pind, pina) * (nbCenterLayers + 2));
256         if (q < 0.75 || q > 1.25 || !isWhiteOrBlackRectangle(pouta, poutb, poutc, poutd)) {
257           break;
258         }
259       }
260 
261       pina = pouta;
262       pinb = poutb;
263       pinc = poutc;
264       pind = poutd;
265 
266       color = !color;
267     }
268 
269     if (nbCenterLayers != 5 && nbCenterLayers != 7) {
270       throw NotFoundException.getNotFoundInstance();
271     }
272     
273     compact = nbCenterLayers == 5;
274     
275     // Expand the square by .5 pixel in each direction so that we're on the border
276     // between the white square and the black square
277     ResultPoint pinax = new ResultPoint(pina.getX() + 0.5f, pina.getY() - 0.5f);
278     ResultPoint pinbx = new ResultPoint(pinb.getX() + 0.5f, pinb.getY() + 0.5f);
279     ResultPoint pincx = new ResultPoint(pinc.getX() - 0.5f, pinc.getY() + 0.5f);
280     ResultPoint pindx = new ResultPoint(pind.getX() - 0.5f, pind.getY() - 0.5f);
281 
282     // Expand the square so that its corners are the centers of the points
283     // just outside the bull's eye.
284     return expandSquare(new ResultPoint[]{pinax, pinbx, pincx, pindx},
285                         2 * nbCenterLayers - 3,
286                         2 * nbCenterLayers);
287   }

289   /**
290    * Finds a candidate center point of an Aztec code from an image
291    *
292    * @return the center point
293    */
294   private Point getMatrixCenter() {
295 
296     ResultPoint pointA;
297     ResultPoint pointB;
298     ResultPoint pointC;
299     ResultPoint pointD;
300 
301     //Get a white rectangle that can be the border of the matrix in center bull's eye or
302     try {
303 
304       ResultPoint[] cornerPoints = new WhiteRectangleDetector(image).detect();
305       pointA = cornerPoints[0];
306       pointB = cornerPoints[1];
307       pointC = cornerPoints[2];
308       pointD = cornerPoints[3];
309 
310     } catch (NotFoundException e) {
311 
312       // This exception can be in case the initial rectangle is white
313       // In that case, surely in the bull's eye, we try to expand the rectangle.
314       int cx = image.getWidth() / 2;
315       int cy = image.getHeight() / 2;
316       pointA = getFirstDifferent(new Point(cx + 7, cy - 7), false, 1, -1).toResultPoint();
317       pointB = getFirstDifferent(new Point(cx + 7, cy + 7), false, 1, 1).toResultPoint();
318       pointC = getFirstDifferent(new Point(cx - 7, cy + 7), false, -1, 1).toResultPoint();
319       pointD = getFirstDifferent(new Point(cx - 7, cy - 7), false, -1, -1).toResultPoint();
320 
321     }
322     
323     //Compute the center of the rectangle
324     int cx = MathUtils.round((pointA.getX() + pointD.getX() + pointB.getX() + pointC.getX()) / 4.0f);
325     int cy = MathUtils.round((pointA.getY() + pointD.getY() + pointB.getY() + pointC.getY()) / 4.0f);
326 
327     // Redetermine the white rectangle starting from previously computed center.
328     // This will ensure that we end up with a white rectangle in center bull's eye
329     // in order to compute a more accurate center.
330     try {
331       ResultPoint[] cornerPoints = new WhiteRectangleDetector(image, 15, cx, cy).detect();
332       pointA = cornerPoints[0];
333       pointB = cornerPoints[1];
334       pointC = cornerPoints[2];
335       pointD = cornerPoints[3];
336     } catch (NotFoundException e) {
337       // This exception can be in case the initial rectangle is white
338       // In that case we try to expand the rectangle.
339       pointA = getFirstDifferent(new Point(cx + 7, cy - 7), false, 1, -1).toResultPoint();
340       pointB = getFirstDifferent(new Point(cx + 7, cy + 7), false, 1, 1).toResultPoint();
341       pointC = getFirstDifferent(new Point(cx - 7, cy + 7), false, -1, 1).toResultPoint();
342       pointD = getFirstDifferent(new Point(cx - 7, cy - 7), false, -1, -1).toResultPoint();
343     }
344     
345     // Recompute the center of the rectangle
346     cx = MathUtils.round((pointA.getX() + pointD.getX() + pointB.getX() + pointC.getX()) / 4.0f);
347     cy = MathUtils.round((pointA.getY() + pointD.getY() + pointB.getY() + pointC.getY()) / 4.0f);
348 
349     return new Point(cx, cy);
350   }

362   /**
363    * Creates a BitMatrix by sampling the provided image.
364    * topLeft, topRight, bottomRight, and bottomLeft are the centers of the squares on the
365    * diagonal just outside the bull's eye.
366    */
367   private BitMatrix sampleGrid(BitMatrix image,
368                                ResultPoint topLeft,
369                                ResultPoint topRight,
370                                ResultPoint bottomRight,
371                                ResultPoint bottomLeft) throws NotFoundException {
372       
373     GridSampler sampler = GridSampler.getInstance();
374     int dimension = getDimension();
375 
376     float low = dimension / 2.0f - nbCenterLayers;
377     float high = dimension / 2.0f + nbCenterLayers;
378 
379     return sampler.sampleGrid(image,
380                               dimension,
381                               dimension,
382                               low, low,   // topleft
383                               high, low,  // topright
384                               high, high, // bottomright
385                               low, high,  // bottomleft
386                               topLeft.getX(), topLeft.getY(),
387                               topRight.getX(), topRight.getY(),
388                               bottomRight.getX(), bottomRight.getY(),
389                               bottomLeft.getX(), bottomLeft.getY());
390   }

392   /**
393    * Samples a line.
394    *
395    * @param p1   start point (inclusive)
396    * @param p2   end point (exclusive)
397    * @param size number of bits
398    * @return the array of bits as an int (first bit is high-order bit of result)
399    */
400   private int sampleLine(ResultPoint p1, ResultPoint p2, int size) {
401     int result = 0;
402 
403     float d = distance(p1, p2);
404     float moduleSize = d / size;
405     float px = p1.getX();
406     float py = p1.getY();
407     float dx = moduleSize * (p2.getX() - p1.getX()) / d;
408     float dy = moduleSize * (p2.getY() - p1.getY()) / d;
409     for (int i = 0; i < size; i++) {
410       if (image.get(MathUtils.round(px + i * dx), MathUtils.round(py + i * dy))) {
411         result |= 1 << (size - i - 1);
412       }
413     }
414     return result;
415   }

417   /**
418    * @return true if the border of the rectangle passed in parameter is compound of white points only
419    *         or black points only
420    */
421   private boolean isWhiteOrBlackRectangle(Point p1,
422                                           Point p2,
423                                           Point p3,
424                                           Point p4) {
425 
426     int corr = 3;
427 
428     p1 = new Point(p1.getX() - corr, p1.getY() + corr);
429     p2 = new Point(p2.getX() - corr, p2.getY() - corr);
430     p3 = new Point(p3.getX() + corr, p3.getY() - corr);
431     p4 = new Point(p4.getX() + corr, p4.getY() + corr);
432 
433     int cInit = getColor(p4, p1);
434 
435     if (cInit == 0) {
436       return false;
437     }
438 
439     int c = getColor(p1, p2);
440 
441     if (c != cInit) {
442       return false;
443     }
444 
445     c = getColor(p2, p3);
446 
447     if (c != cInit) {
448       return false;
449     }
450 
451     c = getColor(p3, p4);
452 
453     return c == cInit;
454 
455   }

457   /**
458    * Gets the color of a segment
459    *
460    * @return 1 if segment more than 90% black, -1 if segment is more than 90% white, 0 else
461    */
462   private int getColor(Point p1, Point p2) {
463     float d = distance(p1, p2);
464     float dx = (p2.getX() - p1.getX()) / d;
465     float dy = (p2.getY() - p1.getY()) / d;
466     int error = 0;
467 
468     float px = p1.getX();
469     float py = p1.getY();
470 
471     boolean colorModel = image.get(p1.getX(), p1.getY());
472 
473     for (int i = 0; i < d; i++) {
474       px += dx;
475       py += dy;
476       if (image.get(MathUtils.round(px), MathUtils.round(py)) != colorModel) {
477         error++;
478       }
479     }
480 
481     float errRatio = error / d;
482 
483     if (errRatio > 0.1f && errRatio < 0.9f) {
484       return 0;
485     }
486 
487     return (errRatio <= 0.1f) == colorModel ? 1 : -1;
488   }

490   /**
491    * Gets the coordinate of the first point with a different color in the given direction
492    */
493   private Point getFirstDifferent(Point init, boolean color, int dx, int dy) {
494     int x = init.getX() + dx;
495     int y = init.getY() + dy;
496 
497     while (isValid(x, y) && image.get(x, y) == color) {
498       x += dx;
499       y += dy;
500     }
501 
502     x -= dx;
503     y -= dy;
504 
505     while (isValid(x, y) && image.get(x, y) == color) {
506       x += dx;
507     }
508     x -= dx;
509 
510     while (isValid(x, y) && image.get(x, y) == color) {
511       y += dy;
512     }
513     y -= dy;
514 
515     return new Point(x, y);
516   }

518   /**
519    * Expand the square represented by the corner points by pushing out equally in all directions
520    *
521    * @param cornerPoints the corners of the square, which has the bull's eye at its center
522    * @param oldSide the original length of the side of the square in the target bit matrix
523    * @param newSide the new length of the size of the square in the target bit matrix
524    * @return the corners of the expanded square
525    */
526   private static ResultPoint[] expandSquare(ResultPoint[] cornerPoints, float oldSide, float newSide) {
527     float ratio = newSide / (2 * oldSide);
528     float dx = cornerPoints[0].getX() - cornerPoints[2].getX();
529     float dy = cornerPoints[0].getY() - cornerPoints[2].getY();
530     float centerx = (cornerPoints[0].getX() + cornerPoints[2].getX()) / 2.0f;
531     float centery = (cornerPoints[0].getY() + cornerPoints[2].getY()) / 2.0f;
532 
533     ResultPoint result0 = new ResultPoint(centerx + ratio * dx, centery + ratio * dy);
534     ResultPoint result2 = new ResultPoint(centerx - ratio * dx, centery - ratio * dy);
535 
536     dx = cornerPoints[1].getX() - cornerPoints[3].getX();
537     dy = cornerPoints[1].getY() - cornerPoints[3].getY();
538     centerx = (cornerPoints[1].getX() + cornerPoints[3].getX()) / 2.0f;
539     centery = (cornerPoints[1].getY() + cornerPoints[3].getY()) / 2.0f;
540     ResultPoint result1 = new ResultPoint(centerx + ratio * dx, centery + ratio * dy);
541     ResultPoint result3 = new ResultPoint(centerx - ratio * dx, centery - ratio * dy);
542 
543     return new ResultPoint[]{result0, result1, result2, result3};
544   }

546   private boolean isValid(int x, int y) {
547     return x >= 0 && x < image.getWidth() && y > 0 && y < image.getHeight();
548   }
549 
550   private boolean isValid(ResultPoint point) {
551     int x = MathUtils.round(point.getX());
552     int y = MathUtils.round(point.getY());
553     return isValid(x, y);
554   }

556   private static float distance(Point a, Point b) {
557     return MathUtils.distance(a.getX(), a.getY(), b.getX(), b.getY());
558   }
559 
560   private static float distance(ResultPoint a, ResultPoint b) {
561     return MathUtils.distance(a.getX(), a.getY(), b.getX(), b.getY());
562   }
563 
564   private int getDimension() {
565     if (compact) {
566       return 4 * nbLayers + 11;
567     }
568     if (nbLayers <= 4) {
569       return 4 * nbLayers + 15;
570     }
571     return 4 * nbLayers + 2 * ((nbLayers - 4) / 8 + 1) + 15;
572   }

575     private final int x;
576     private final int y;
577 
578     ResultPoint toResultPoint() {
579       return new ResultPoint(getX(), getY());

583       this.x = x;
584       this.y = y;

588       return x;

591     int getY() {
592       return y;
593     }

597       return "<" + x + ' ' + y + '>';
