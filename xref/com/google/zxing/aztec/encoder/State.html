17  package com.google.zxing.aztec.encoder;
18  
19  import java.util.Deque;
20  import java.util.LinkedList;
21  
22  import com.google.zxing.common.BitArray;
23  
24  /**
25   * State represents all information about a sequence necessary to generate the current output.
26   * Note that a state is immutable.
27   */
28  final class State {
29    
30    static final State INITIAL_STATE = new State(Token.EMPTY, HighLevelEncoder.MODE_UPPER, 0, 0);
31  
32    // The current mode of the encoding (or the mode to which we'll return if
33    // we're in Binary Shift mode.
34    private final int mode;
35    // The list of tokens that we output.  If we are in Binary Shift mode, this
36    // token list does *not* yet included the token for those bytes
37    private final Token token;
38    // If non-zero, the number of most recent bytes that should be output
39    // in Binary Shift mode.
40    private final int binaryShiftByteCount;
41    // The total number of bits generated (including Binary Shift).
42    private final int bitCount;
43  
44    private State(Token token, int mode, int binaryBytes, int bitCount) {
45      this.token = token;
46      this.mode = mode;
47      this.binaryShiftByteCount = binaryBytes;
48      this.bitCount = bitCount;
49      // Make sure we match the token
50      //int binaryShiftBitCount = (binaryShiftByteCount * 8) +
51      //    (binaryShiftByteCount == 0 ? 0 :
52      //     binaryShiftByteCount <= 31 ? 10 :
53      //     binaryShiftByteCount <= 62 ? 20 : 21);
54      //assert this.bitCount == token.getTotalBitCount() + binaryShiftBitCount;
55    }
56    
57    int getMode() {
58      return mode;
59    }
60    
61    Token getToken() {
62      return token;
63    }
64    
65    int getBinaryShiftByteCount() {
66      return binaryShiftByteCount;
67    }
68    
69    int getBitCount() {
70      return bitCount;
71    }
72  
73    // Create a new state representing this state with a latch to a (not
74    // necessary different) mode, and then a code.
75    State latchAndAppend(int mode, int value) {
76      //assert binaryShiftByteCount == 0;
77      int bitCount = this.bitCount;
78      Token token = this.token;
79      if (mode != this.mode) {
80        int latch = HighLevelEncoder.LATCH_TABLE[this.mode][mode];
81        token = token.add(latch & 0xFFFF, latch >> 16);
82        bitCount += latch >> 16;
83      }
84      int latchModeBitCount = mode == HighLevelEncoder.MODE_DIGIT ? 4 : 5;
85      token = token.add(value, latchModeBitCount);
86      return new State(token, mode, 0, bitCount + latchModeBitCount);
87    }
88  
89    // Create a new state representing this state, with a temporary shift
90    // to a different mode to output a single value.
91    State shiftAndAppend(int mode, int value) {
92      //assert binaryShiftByteCount == 0 && this.mode != mode;
93      Token token = this.token;
94      int thisModeBitCount = this.mode == HighLevelEncoder.MODE_DIGIT ? 4 : 5;
95      // Shifts exist only to UPPER and PUNCT, both with tokens size 5.
96      token = token.add(HighLevelEncoder.SHIFT_TABLE[this.mode][mode], thisModeBitCount);
97      token = token.add(value, 5);
98      return new State(token, this.mode, 0, this.bitCount + thisModeBitCount + 5);
99    }
100 
101   // Create a new state representing this state, but an additional character
102   // output in Binary Shift mode.
103   State addBinaryShiftChar(int index) {
104     Token token = this.token;
105     int mode = this.mode;
106     int bitCount = this.bitCount;
107     if (this.mode == HighLevelEncoder.MODE_PUNCT || this.mode == HighLevelEncoder.MODE_DIGIT)  {
108       //assert binaryShiftByteCount == 0;
109       int latch = HighLevelEncoder.LATCH_TABLE[mode][HighLevelEncoder.MODE_UPPER];
110       token = token.add(latch & 0xFFFF, latch >> 16);
111       bitCount += latch >> 16;
112       mode = HighLevelEncoder.MODE_UPPER;
113     }
114     int deltaBitCount =
115       (binaryShiftByteCount == 0 || binaryShiftByteCount == 31) ? 18 :
116       (binaryShiftByteCount == 62) ? 9 : 8;
117     State result = new State(token, mode, binaryShiftByteCount + 1, bitCount + deltaBitCount);
118     if (result.binaryShiftByteCount == 2047 + 31) {
119       // The string is as long as it's allowed to be.  We should end it.
120       result = result.endBinaryShift(index + 1);
121     }
122     return result;
123   }
124 
125   // Create the state identical to this one, but we are no longer in
126   // Binary Shift mode.
127   State endBinaryShift(int index) {
128     if (binaryShiftByteCount == 0) {
129       return this;
130     }
131     Token token = this.token;
132     token = token.addBinaryShift(index - binaryShiftByteCount, binaryShiftByteCount);
133     //assert token.getTotalBitCount() == this.bitCount;
134     return new State(token, mode, 0, this.bitCount);
135   }
136 
137   // Returns true if "this" state is better (or equal) to be in than "that"
138   // state under all possible circumstances.
139   boolean isBetterThanOrEqualTo(State other) {
140     int mySize = this.bitCount + (HighLevelEncoder.LATCH_TABLE[this.mode][other.mode] >> 16);
141     if (other.binaryShiftByteCount > 0 &&
142         (this.binaryShiftByteCount == 0 || this.binaryShiftByteCount > other.binaryShiftByteCount)) {
143       mySize += 10;     // Cost of entering Binary Shift mode.
144     }
145     return mySize <= other.bitCount;
146   }
147 
148   BitArray toBitArray(byte[] text) {
149     // Reverse the tokens, so that they are in the order that they should
150     // be output
151     Deque<Token> symbols = new LinkedList<>();
152     for (Token token = endBinaryShift(text.length).token; token != null; token = token.getPrevious()) {
153       symbols.addFirst(token);
154     }
155     BitArray bitArray = new BitArray();
156     // Add each token to the result.
157     for (Token symbol : symbols) {
158       symbol.appendTo(bitArray, text);
159     }
160     //assert bitArray.getSize() == this.bitCount;
161     return bitArray;
162   }
163 
164   @Override
165   public String toString() {
166     return String.format("%s bits=%d bytes=%d", HighLevelEncoder.MODE_NAMES[mode], bitCount, binaryShiftByteCount);
167   }
168 
169 }
