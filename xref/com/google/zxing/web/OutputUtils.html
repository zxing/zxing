17  package com.google.zxing.web;
18  
19  /**
20   * Utility functions for {@code decoderesult.jspx}.
21   * 
22   * @author Sean Owen
23   */
24  public final class OutputUtils {
25  
26    private static final int BYTES_PER_LINE = 16;
27    private static final int HALF_BYTES_PER_LINE = BYTES_PER_LINE / 2;
28  
29    private OutputUtils() {
30    }
31  
32    public static String arrayToString(byte[] bytes) {
33      StringBuilder result = new StringBuilder(bytes.length * 4);
34      int i = 0;
35      while (i < bytes.length) {
36        int value = bytes[i] & 0xFF;
37        result.append(hexChar(value / 16));
38        result.append(hexChar(value % 16));
39        i++;
40        if (i % BYTES_PER_LINE == 0) {
41          result.append('\n');
42        } else if (i % HALF_BYTES_PER_LINE == 0) {
43          result.append("   ");
44        } else {
45          result.append(' ');        
46        }
47      }
48      return result.toString();
49    }
50    
51    private static char hexChar(int value) {
52      if (value < 0 || value > 15) {
53        throw new IllegalArgumentException();
54      }
55      return (char) (value < 10 ? ('0' + value) : ('a' + (value - 10)));
56    }
57    
58  }
