17  package com.google.zxing.web;
18  
19  import com.google.common.base.Preconditions;
20  import com.google.common.net.HttpHeaders;
21  
22  import javax.servlet.http.HttpServlet;
23  import javax.servlet.http.HttpServletRequest;
24  import javax.servlet.http.HttpServletResponse;
25  
26  public final class LegacyJavadocRedirectServlet extends HttpServlet {
27  
28    private static final String PREFIX = "/w/docs/javadoc/";
29  
30    @Override
31    protected void doGet(HttpServletRequest request, HttpServletResponse response) {
32      String requestURI = request.getRequestURI();
33      Preconditions.checkArgument(requestURI.startsWith(PREFIX));
34      String redirectURI = "http://zxing.github.io/zxing/apidocs/" + requestURI.substring(PREFIX.length());
35      response.setStatus(HttpServletResponse.SC_MOVED_PERMANENTLY);
36      response.setHeader(HttpHeaders.LOCATION, redirectURI);
37    }
38  
39  }
