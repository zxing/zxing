79  /**
80   * {@link HttpServlet} which decodes images containing barcodes. Given a URL, it will
81   * retrieve the image and decode it. It can also process image files uploaded via POST.
82   *
83   * @author Sean Owen
84   */
85  public final class DecodeServlet extends HttpServlet {
86  
87    private static final Logger log = Logger.getLogger(DecodeServlet.class.getName());
88  
89    // No real reason to let people upload more than a 4MB image
90    private static final long MAX_IMAGE_SIZE = 4000000L;
91    // No real reason to deal with more than maybe 8.3 megapixels
92    private static final int MAX_PIXELS = 1 << 23;
93    private static final byte[] REMAINDER_BUFFER = new byte[32768];
94    private static final Map<DecodeHintType,Object> HINTS;
95    private static final Map<DecodeHintType,Object> HINTS_PURE;
96  
97    static {
98      HINTS = new EnumMap<>(DecodeHintType.class);
99      HINTS.put(DecodeHintType.TRY_HARDER, Boolean.TRUE);
100     HINTS.put(DecodeHintType.POSSIBLE_FORMATS, EnumSet.allOf(BarcodeFormat.class));
101     HINTS_PURE = new EnumMap<>(HINTS);
102     HINTS_PURE.put(DecodeHintType.PURE_BARCODE, Boolean.TRUE);
103   }
104 
105   private DiskFileItemFactory diskFileItemFactory;
106   private Iterable<String> blockedURLSubstrings;
107 
108   @Override
109   public void init(ServletConfig servletConfig) throws ServletException {
110     Logger logger = Logger.getLogger("com.google.zxing");
111     ServletContext context = servletConfig.getServletContext();
112     logger.addHandler(new ServletContextLogHandler(context));
113     File repository = (File) context.getAttribute("javax.servlet.context.tempdir");
114     FileCleaningTracker fileCleaningTracker = FileCleanerCleanup.getFileCleaningTracker(context);
115     diskFileItemFactory = new DiskFileItemFactory(1 << 16, repository);
116     diskFileItemFactory.setFileCleaningTracker(fileCleaningTracker);
117     
118     try {
119       blockedURLSubstrings =
120           Resources.readLines(Resources.getResource("/private/uri-block-substrings.txt"), StandardCharsets.UTF_8);
121     } catch (IOException ioe) {
122       throw new ServletException(ioe);
123     }
124     log.info("Blocking URIs containing: " + blockedURLSubstrings);
125   }

127   @Override
128   protected void doGet(HttpServletRequest request,
129                        HttpServletResponse response) throws ServletException, IOException {
130 
131     String imageURIString = request.getParameter("u");
132     if (imageURIString == null || imageURIString.isEmpty()) {
133       log.info("URI was empty");
134       response.sendRedirect("badurl.jspx");
135       return;
136     }
137 
138     imageURIString = imageURIString.trim();
139     for (CharSequence substring : blockedURLSubstrings) {
140       if (imageURIString.contains(substring)) {
141         log.info("Disallowed URI " + imageURIString);        
142         response.sendRedirect("badurl.jspx");
143         return;
144       }
145     }
146 
147     URI imageURI;
148     try {
149       imageURI = new URI(imageURIString);
150       // Assume http: if not specified
151       if (imageURI.getScheme() == null) {
152         imageURI = new URI("http://" + imageURIString);
153       }
154     } catch (URISyntaxException urise) {
155       log.info("URI " + imageURIString + " was not valid: " + urise);
156       response.sendRedirect("badurl.jspx");
157       return;
158     }
159     
160     // Shortcut for data URI
161     if ("data".equals(imageURI.getScheme())) {
162       try {
163         BufferedImage image = ImageReader.readDataURIImage(imageURI);
164         processImage(image, request, response);
165       } catch (IOException ioe) {
166         log.info(ioe.toString());
167         response.sendRedirect("badurl.jspx");
168       }
169       return;
170     }
171     
172     URL imageURL;    
173     try {
174       imageURL = imageURI.toURL();
175     } catch (MalformedURLException ignored) {
176       log.info("URI was not valid: " + imageURIString);
177       response.sendRedirect("badurl.jspx");
178       return;
179     }
180 
181     String protocol = imageURL.getProtocol();
182     if (!"http".equalsIgnoreCase(protocol) && !"https".equalsIgnoreCase(protocol)) {
183       log.info("URI was not valid: " + imageURIString);
184       response.sendRedirect("badurl.jspx");
185       return;
186     }
187 
188     HttpURLConnection connection;
189     try {
190       connection = (HttpURLConnection) imageURL.openConnection();
191     } catch (IllegalArgumentException ignored) {
192       log.info("URI could not be opened: " + imageURL);
193       response.sendRedirect("badurl.jspx");
194       return;
195     }
196 
197     connection.setAllowUserInteraction(false);
198     connection.setReadTimeout(5000);
199     connection.setConnectTimeout(5000);
200     connection.setRequestProperty(HttpHeaders.USER_AGENT, "zxing.org");
201     connection.setRequestProperty(HttpHeaders.CONNECTION, "close");
202 
203     try {
204 
205       try {
206         connection.connect();
207       } catch (IOException ioe) {
208         // Encompasses lots of stuff, including
209         //  java.net.SocketException, java.net.UnknownHostException,
210         //  javax.net.ssl.SSLPeerUnverifiedException,
211         //  org.apache.http.NoHttpResponseException,
212         //  org.apache.http.client.ClientProtocolException,
213         log.info(ioe.toString());
214         response.sendRedirect("badurl.jspx");
215         return;
216       }
217 
218       InputStream is = null;
219       try {
220 
221         is = connection.getInputStream();
222 
223         if (connection.getResponseCode() != HttpServletResponse.SC_OK) {
224           log.info("Unsuccessful return code: " + connection.getResponseCode());
225           response.sendRedirect("badurl.jspx");
226           return;
227         }
228         if (connection.getHeaderFieldInt(HttpHeaders.CONTENT_LENGTH, 0) > MAX_IMAGE_SIZE) {
229           log.info("Too large");
230           response.sendRedirect("badimage.jspx");
231           return;
232         }
233 
234         log.info("Decoding " + imageURL);
235         processStream(is, request, response);
236 
237       } catch (IOException ioe) {
238         log.info(ioe.toString());
239         response.sendRedirect("badurl.jspx");
240       } finally {
241         if (is != null) {
242           consumeRemainder(is);
243           is.close();
244         }
245       }
246 
247     } finally {
248       connection.disconnect();
249     }
250 
251   }
252 
253   private static void consumeRemainder(InputStream is) {
254     try {
255       int available;
256       while ((available = is.available()) > 0) {
257         is.read(REMAINDER_BUFFER, 0, available); // don't care about value, or collision
258       }
259     } catch (IOException | IndexOutOfBoundsException ioe) {
260       // sun.net.www.http.ChunkedInputStream.read is throwing IndexOutOfBoundsException
261       // continue
262     }
263   }

265   @Override
266   protected void doPost(HttpServletRequest request, HttpServletResponse response)
267       throws ServletException, IOException {
268 
269     if (!ServletFileUpload.isMultipartContent(request)) {
270       log.info("File upload was not multipart");
271       response.sendRedirect("badimage.jspx");
272       return;
273     }
274 
275     ServletFileUpload upload = new ServletFileUpload(diskFileItemFactory);
276     upload.setFileSizeMax(MAX_IMAGE_SIZE);
277 
278     // Parse the request
279     try {
280       for (FileItem item : upload.parseRequest(request)) {
281         if (!item.isFormField()) {
282           if (item.getSize() <= MAX_IMAGE_SIZE) {
283             log.info("Decoding uploaded file");
284             try (InputStream is = item.getInputStream()) {
285               processStream(is, request, response);
286             }
287           } else {
288             log.info("Too large");
289             response.sendRedirect("badimage.jspx");
290           }
291           break;
292         }
293       }
294     } catch (FileUploadException fue) {
295       log.info(fue.toString());
296       response.sendRedirect("badimage.jspx");
297     }
298 
299   }

301   private static void processStream(InputStream is,
302                                     ServletRequest request,
303                                     HttpServletResponse response) throws ServletException, IOException {
304 
305     BufferedImage image;
306     try {
307       image = ImageIO.read(is);
308     } catch (IOException | CMMException | IllegalArgumentException ioe) {
309       log.info(ioe.toString());
310       // Have seen these in some logs
311       response.sendRedirect("badimage.jspx");
312       return;
313     }
314     if (image == null) {
315       response.sendRedirect("badimage.jspx");
316       return;
317     }
318     if (image.getHeight() <= 1 || image.getWidth() <= 1 ||
319         image.getHeight() * image.getWidth() > MAX_PIXELS) {
320       log.info("Dimensions out of bounds: " + image.getWidth() + 'x' + image.getHeight());
321       response.sendRedirect("badimage.jspx");
322       return;
323     }
324     
325     processImage(image, request, response);
326   }

328   private static void processImage(BufferedImage image,
329                                    ServletRequest request,
330                                    HttpServletResponse response) throws IOException, ServletException {
331 
332     LuminanceSource source = new BufferedImageLuminanceSource(image);
333     BinaryBitmap bitmap = new BinaryBitmap(new GlobalHistogramBinarizer(source));
334     Collection<Result> results = Lists.newArrayListWithCapacity(1);
335 
336     try {
337 
338       Reader reader = new MultiFormatReader();
339       ReaderException savedException = null;
340       try {
341         // Look for multiple barcodes
342         MultipleBarcodeReader multiReader = new GenericMultipleBarcodeReader(reader);
343         Result[] theResults = multiReader.decodeMultiple(bitmap, HINTS);
344         if (theResults != null) {
345           results.addAll(Arrays.asList(theResults));
346         }
347       } catch (ReaderException re) {
348         savedException = re;
349       }
350   
351       if (results.isEmpty()) {
352         try {
353           // Look for pure barcode
354           Result theResult = reader.decode(bitmap, HINTS_PURE);
355           if (theResult != null) {
356             results.add(theResult);
357           }
358         } catch (ReaderException re) {
359           savedException = re;
360         }
361       }
362   
363       if (results.isEmpty()) {
364         try {
365           // Look for normal barcode in photo
366           Result theResult = reader.decode(bitmap, HINTS);
367           if (theResult != null) {
368             results.add(theResult);
369           }
370         } catch (ReaderException re) {
371           savedException = re;
372         }
373       }
374   
375       if (results.isEmpty()) {
376         try {
377           // Try again with other binarizer
378           BinaryBitmap hybridBitmap = new BinaryBitmap(new HybridBinarizer(source));
379           Result theResult = reader.decode(hybridBitmap, HINTS);
380           if (theResult != null) {
381             results.add(theResult);
382           }
383         } catch (ReaderException re) {
384           savedException = re;
385         }
386       }
387   
388       if (results.isEmpty()) {
389         handleException(savedException, response);
390         return;
391       }
392 
393     } catch (RuntimeException re) {
394       // Call out unexpected errors in the log clearly
395       log.log(Level.WARNING, "Unexpected exception from library", re);
396       throw new ServletException(re);
397     }
398 
399     String fullParameter = request.getParameter("full");
400     boolean minimalOutput = fullParameter != null && !Boolean.parseBoolean(fullParameter);
401     if (minimalOutput) {
402       response.setContentType(MediaType.PLAIN_TEXT_UTF_8.toString());
403       response.setCharacterEncoding(StandardCharsets.UTF_8.name());
404       try (Writer out = new OutputStreamWriter(response.getOutputStream(), StandardCharsets.UTF_8)) {
405         for (Result result : results) {
406           out.write(result.getText());
407           out.write('\n');
408         }
409       }
410     } else {
411       request.setAttribute("results", results);
412       request.getRequestDispatcher("decoderesult.jspx").forward(request, response);
413     }
414   }
415 
416   private static void handleException(ReaderException re, HttpServletResponse response) throws IOException {
417     if (re instanceof NotFoundException) {
418       log.info("Not found: " + re);
419       response.sendRedirect("notfound.jspx");
420     } else if (re instanceof FormatException) {
421       log.info("Format problem: " + re);
422       response.sendRedirect("format.jspx");
423     } else if (re instanceof ChecksumException) {
424       log.info("Checksum problem: " + re);
425       response.sendRedirect("format.jspx");
426     } else {
427       log.info("Unknown problem: " + re);
428       response.sendRedirect("notfound.jspx");
429     }
430   }
