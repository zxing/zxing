42  /**
43   * A reimplementation of the
44   * <a href="https://google-developers.appspot.com/chart/infographics/docs/qr_codes">
45   *   Google Chart Server's QR code encoder</a>, which is now deprecated.
46   *
47   * @author Sean Owen
48   */
49  public final class ChartServlet extends HttpServlet {
50  
51    private static final int MAX_DIMENSION = 4096;
52    private static final Collection<Charset> SUPPORTED_OUTPUT_ENCODINGS = ImmutableSet.<Charset>builder()
53        .add(StandardCharsets.UTF_8).add(StandardCharsets.ISO_8859_1).add(Charset.forName("Shift_JIS")).build();
54  
55    @Override
56    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException {
57      doEncode(request, response, false);
58    }
59  
60    @Override
61    protected void doPost(HttpServletRequest request, HttpServletResponse response) throws IOException {
62      doEncode(request, response, true);
63    }
64  
65    private static void doEncode(ServletRequest request, HttpServletResponse response, boolean isPost)
66        throws IOException {
67  
68      ChartServletRequestParameters parameters;
69      try {
70        parameters = doParseParameters(request, isPost);
71      } catch (IllegalArgumentException | NullPointerException e) {
72        response.sendError(HttpServletResponse.SC_BAD_REQUEST, e.toString());
73        return;
74      }
75  
76      Map<EncodeHintType,Object> hints = new EnumMap<>(EncodeHintType.class);
77      hints.put(EncodeHintType.MARGIN, parameters.getMargin());
78      if (!StandardCharsets.ISO_8859_1.equals(parameters.getOutputEncoding())) {
79        // Only set if not QR code default
80        hints.put(EncodeHintType.CHARACTER_SET, parameters.getOutputEncoding().name());
81      }
82      hints.put(EncodeHintType.ERROR_CORRECTION, parameters.getEcLevel());
83  
84      BitMatrix matrix;
85      try {
86        matrix = new QRCodeWriter().encode(parameters.getText(),
87                                           BarcodeFormat.QR_CODE,
88                                           parameters.getWidth(),
89                                           parameters.getHeight(),
90                                           hints);
91      } catch (WriterException we) {
92        response.sendError(HttpServletResponse.SC_BAD_REQUEST, we.toString());
93        return;
94      }
95  
96      ByteArrayOutputStream pngOut = new ByteArrayOutputStream();
97      MatrixToImageWriter.writeToStream(matrix, "PNG", pngOut);
98      byte[] pngData = pngOut.toByteArray();
99  
100     response.setContentType("image/png");
101     response.setContentLength(pngData.length);
102     response.setHeader("Cache-Control", "public");
103     response.getOutputStream().write(pngData);
104   }
105 
106   private static ChartServletRequestParameters doParseParameters(ServletRequest request, boolean readBody)
107       throws IOException {
108 
109     Preconditions.checkArgument("qr".equals(request.getParameter("cht")), "Bad type");
110 
111     String widthXHeight = request.getParameter("chs");
112     Preconditions.checkNotNull(widthXHeight, "No size");
113     int xIndex = widthXHeight.indexOf('x');
114     Preconditions.checkArgument(xIndex >= 0, "Bad size");
115 
116     int width = Integer.parseInt(widthXHeight.substring(0, xIndex));
117     int height = Integer.parseInt(widthXHeight.substring(xIndex + 1));
118     Preconditions.checkArgument(width > 0 && height > 0, "Bad size");
119     Preconditions.checkArgument(width <= MAX_DIMENSION && height <= MAX_DIMENSION, "Bad size");
120 
121     String outputEncodingName = request.getParameter("choe");
122     Charset outputEncoding = StandardCharsets.UTF_8;
123     if (outputEncodingName != null) {
124       outputEncoding = Charset.forName(outputEncodingName);
125       Preconditions.checkArgument(SUPPORTED_OUTPUT_ENCODINGS.contains(outputEncoding), "Bad output encoding");
126     }
127 
128     ErrorCorrectionLevel ecLevel = ErrorCorrectionLevel.L;
129     int margin = 4;
130 
131     String ldString = request.getParameter("chld");
132     if (ldString != null) {
133       int pipeIndex = ldString.indexOf('|');
134       if (pipeIndex < 0) {
135         // Only an EC level
136         ecLevel = ErrorCorrectionLevel.valueOf(ldString);
137       } else {
138         ecLevel = ErrorCorrectionLevel.valueOf(ldString.substring(0, pipeIndex));
139         margin = Integer.parseInt(ldString.substring(pipeIndex + 1));
140         Preconditions.checkArgument(margin > 0, "Bad margin");
141       }
142     }
143 
144     String text;
145     if (readBody) {
146       text = CharStreams.toString(request.getReader());
147     } else {
148       text = request.getParameter("chl");
149     }
150     Preconditions.checkArgument(text != null && !text.isEmpty(), "No input");
151 
152     return new ChartServletRequestParameters(width, height, outputEncoding, ecLevel, margin, text);
153   }
