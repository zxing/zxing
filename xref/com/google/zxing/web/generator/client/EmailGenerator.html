17  package com.google.zxing.web.generator.client;
18  
19  import com.google.gwt.event.dom.client.ChangeHandler;
20  import com.google.gwt.event.dom.client.KeyPressHandler;
21  import com.google.gwt.user.client.ui.Grid;
22  import com.google.gwt.user.client.ui.TextBox;
23  import com.google.gwt.user.client.ui.Widget;
24  
25  /**
26   * Generator for email address.
27   * 
28   * @author Yohann Coppel
29   */
30  public final class EmailGenerator implements GeneratorSource {
31  
32    private Grid table;
33    private final TextBox email = new TextBox();
34    
35    public EmailGenerator(ChangeHandler handler, KeyPressHandler keyListener) {
36      email.addStyleName(StylesDefs.INPUT_FIELD_REQUIRED);
37      email.addChangeHandler(handler);
38      email.addKeyPressHandler(keyListener);
39    }
40    
41    @Override
42    public String getName() {
43      return "Email address";
44    }
45  
46    @Override
47    public String getText() throws GeneratorException {
48      return "mailto:" + getEmailField();
49    }
50  
51    private String getEmailField() throws GeneratorException {
52      String input = email.getText();
53      if (input.isEmpty()) {
54        throw new GeneratorException("Email must be present.");
55      }
56      Validators.validateEmail(input);
57      return input;
58    }
59    
60    @Override
61    public Grid getWidget() {
62      if (table != null) {
63        return table;
64      }
65      table = new Grid(1, 2);
66      table.setText(0, 0, "Address");
67      table.setWidget(0, 1, email);
68      return table;
69    }
70  
71    @Override
72    public void validate(Widget widget) throws GeneratorException {
73      if (widget == email) {
74        getEmailField();
75      }
76    }
77  
78    @Override
79    public void setFocus() {
80      email.setFocus(true);
81    }
82  }
