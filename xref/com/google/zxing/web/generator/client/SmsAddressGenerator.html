17  package com.google.zxing.web.generator.client;
18  
19  import com.google.gwt.event.dom.client.ChangeHandler;
20  import com.google.gwt.event.dom.client.KeyPressHandler;
21  import com.google.gwt.user.client.ui.Grid;
22  import com.google.gwt.user.client.ui.TextArea;
23  import com.google.gwt.user.client.ui.TextBox;
24  import com.google.gwt.user.client.ui.Widget;
25  
26  /**
27   * A generator for a sms address. Gives the option of filling up the content
28   * of the message as well.
29   * 
30   * @author Yohann Coppel
31   */
32  public final class SmsAddressGenerator implements GeneratorSource {
33  
34    private Grid table;
35    private final TextBox number = new TextBox();
36    private final TextArea message = new TextArea();
37  
38    public SmsAddressGenerator(ChangeHandler handler, KeyPressHandler keyListener) {
39      number.addStyleName(StylesDefs.INPUT_FIELD_REQUIRED);
40      number.addChangeHandler(handler);
41      number.addKeyPressHandler(keyListener);
42      message.addChangeHandler(handler);
43    }
44    
45    @Override
46    public String getName() {
47      return "SMS";
48    }
49  
50    @Override
51    public String getText() throws GeneratorException {
52      String inputNumber = getTelField();    
53      String inputMessage = getMessageField();
54      
55      String output = inputNumber;
56      // we add the text only if there actually is something in the field.
57      if (!inputMessage.isEmpty()) {
58        output += ':' + inputMessage;
59      }
60      
61      return "smsto:" + output;
62    }
63  
64    private String getTelField() throws GeneratorException {
65      String input = number.getText();
66      if (input.isEmpty()) {
67        throw new GeneratorException("Phone number must be present.");
68      }
69      input = Validators.filterNumber(input);
70      Validators.validateNumber(input);
71      return input;
72    }
73   
74    private String getMessageField() throws GeneratorException {
75      String inputMessage = message.getText();
76      if (inputMessage.length() > 150) {
77        throw new GeneratorException("Sms message can not be longer than 150 characters.");
78      }
79      return inputMessage;
80    }
81    
82    @Override
83    public Grid getWidget() {
84      if (table != null) {
85        return table;
86      }
87      table = new Grid(2, 2);
88  
89      table.setText(0, 0, "Phone number");
90      table.setWidget(0, 1, number);
91  
92      table.setText(1, 0, "Message");
93      table.setWidget(1, 1, message);
94  
95      return table;
96    }
97  
98    @Override
99    public void validate(Widget widget) throws GeneratorException {
100     if (widget == number) {
101       getTelField();
102     }
103     if (widget == message) {
104       getMessageField();
105     }
106   }
107   
108   @Override
109   public void setFocus() {
110     number.setFocus(true);
111   }
112 }
