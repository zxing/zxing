17  package com.google.zxing.web.generator.client;
18  
19  import com.google.gwt.event.dom.client.ChangeHandler;
20  import com.google.gwt.event.dom.client.KeyPressHandler;
21  import com.google.gwt.user.client.ui.Grid;
22  import com.google.gwt.user.client.ui.TextBox;
23  import com.google.gwt.user.client.ui.Widget;
24  
25  /**
26   * A generator for geo location. It also accepts a google maps links and
27   * extracts the coordinates and query from the URL.
28   * 
29   * @author Yohann Coppel
30   */
31  public final class GeoLocationGenerator implements GeneratorSource {
32  
33    private static final String LON_REGEXP = "[+-]?[0-9]+(.[0-9]+)?";
34    private static final String LAT_REGEXP = "[+-]?[0-9]+(.[0-9]+)?";
35    
36    private Grid table;
37    private final TextBox latitude = new TextBox();
38    private final TextBox longitude = new TextBox();
39    private final TextBox query = new TextBox();
40    
41    public GeoLocationGenerator(ChangeHandler handler, KeyPressHandler keyListener) {
42      latitude.addStyleName(StylesDefs.INPUT_FIELD_REQUIRED);
43      latitude.addChangeHandler(handler);
44      latitude.addKeyPressHandler(keyListener);
45      longitude.addStyleName(StylesDefs.INPUT_FIELD_REQUIRED);
46      longitude.addChangeHandler(handler);
47      longitude.addKeyPressHandler(keyListener);
48      query.addChangeHandler(handler);
49      query.addKeyPressHandler(keyListener);
50    }
51    
52    @Override
53    public String getName() {
54      return "Geo location";
55    }
56  
57    @Override
58    public String getText() throws GeneratorException {
59      String que = getQueryField();
60      if (que != null && !que.isEmpty()) {
61        if (getLatitudeField() == null) {
62          latitude.setText("0");
63        }
64        if (getLongitudeField() == null) {
65          longitude.setText("0");
66        }
67      }
68      String lat = getLatitudeField();
69      String lon = getLongitudeField();
70      
71      if (que != null && !que.isEmpty()) {
72        return "geo:" + lat + ',' + lon + "?q=" + que;
73      }
74      return "geo:" + lat + ',' + lon;
75    }
76  
77    private String getQueryField() {
78      String que = query.getText();
79      que = que.replace("&", "%26");
80      return que;
81    }
82  
83    private String getLongitudeField() throws GeneratorException {
84      String lon = longitude.getText();
85      if (!lon.matches(LON_REGEXP)) {
86        throw new GeneratorException("Longitude is not a correct value.");
87      }
88      double val = Double.parseDouble(lon);
89      if (val < -180.0 || val > 180.0) {
90        throw new GeneratorException("Longitude must be in [-180:180]");
91      }
92      return lon;
93    }
94  
95    private String getLatitudeField() throws GeneratorException {
96      String lat = latitude.getText();
97      if (!lat.matches(LAT_REGEXP)) {
98        throw new GeneratorException("Latitude is not a correct value.");
99      }
100     double val = Double.parseDouble(lat);
101     if (val < -90.0 || val > 90.0) {
102       throw new GeneratorException("Latitude must be in [-90:90]");
103     }
104     return lat;
105   }
106 
107   @Override
108   public Grid getWidget() {
109     if (table != null) {
110       return table;
111     }
112     table = new Grid(3, 2);
113     
114     table.setText(0, 0, "Latitude");
115     table.setWidget(0, 1, latitude);
116     table.setText(1, 0, "Longitude");
117     table.setWidget(1, 1, longitude);
118     table.setText(2, 0, "Query");
119     table.setWidget(2, 1, query);
120     return table;
121   }
122 
123   @Override
124   public void validate(Widget widget) throws GeneratorException {
125     if (widget == latitude) {
126       getLatitudeField();
127     }
128     if (widget == longitude) {
129       getLongitudeField();
130     }
131   }
132 
133   @Override
134   public void setFocus() {
135     latitude.setFocus(true);
136   }
137 
138 }
