46  public final class Generator implements EntryPoint {

48    private final List<GeneratorSource> generators = new ArrayList<>();
49    private final ListBox genList = new ListBox();
50    private final ListBox sizeList = new ListBox();
51    private final ListBox ecLevelList = new ListBox();
52    private final ListBox encodingList = new ListBox();
53    private final Image result = new Image("");
54    private final HTMLTable topPanel = new Grid(5, 1);
55    private GeneratorSource selectedGenerator = null;
56    private final VerticalPanel rightPanel = new VerticalPanel();
57    private final TextBox urlResult = new TextBox();
58    private final Widget downloadText =
59        new HTML("<a href=\"\" id=\"downloadlink\" >Download</a> or embed with this URL:");
60    private final TextArea rawTextResult = new TextArea();

62    @Override
63    public void onModuleLoad() {
64      loadGenerators();
65  
66      setupLeftPanel();
67      topPanel.getElement().setId("leftpanel");
68      Widget leftPanel = topPanel;
69  
70      HorizontalPanel mainPanel = new HorizontalPanel();
71      mainPanel.add(leftPanel);
72  
73      SimplePanel div2 = new SimplePanel();
74      div2.add(result);
75      div2.getElement().setId("innerresult");
76      SimplePanel div = new SimplePanel();
77      div.add(div2);
78      div.getElement().setId("imageresult");
79  
80      urlResult.getElement().setId("urlresult");
81      rawTextResult.getElement().setId("rawtextresult");
82      rawTextResult.setCharacterWidth(50);
83      rawTextResult.setVisibleLines(8);
84      downloadText.getElement().setId("downloadText");
85      rightPanel.add(div);
86      rightPanel.add(downloadText);
87      rightPanel.add(urlResult);
88      rightPanel.add(rawTextResult);
89      mainPanel.add(rightPanel);
90      mainPanel.getElement().setId("mainpanel");
91      RootPanel.get("ui").add(mainPanel);
92      setWidget(1);
93      invalidateBarcode();
94    }
95  
96    private void setWidget(int id) {
97      if (id >= 0 && id < generators.size()) {
98        topPanel.setWidget(1, 0, generators.get(id).getWidget());
99        genList.setSelectedIndex(id);
100       selectedGenerator = generators.get(id);
101       eraseErrorMessage();
102       invalidateBarcode();
103       genList.setFocus(false);
104       selectedGenerator.setFocus();
105     }
106   }
107 
108   private void loadGenerators() {
109     generators.add(new CalendarEventGenerator(changeHandler, keyPressHandler));
110     generators.add(new ContactInfoGenerator(changeHandler, keyPressHandler));
111     generators.add(new EmailGenerator(changeHandler, keyPressHandler));
112     generators.add(new GeoLocationGenerator(changeHandler, keyPressHandler));
113     generators.add(new PhoneNumberGenerator(changeHandler, keyPressHandler));
114     generators.add(new SmsAddressGenerator(changeHandler, keyPressHandler));
115     generators.add(new TextGenerator(changeHandler));
116     generators.add(new UrlGenerator(changeHandler, keyPressHandler));
117     generators.add(new WifiGenerator(changeHandler, keyPressHandler));
118   }
119 
120   void setupLeftPanel() {
121     topPanel.setHTML(2, 0,
122         "<span id=\"errorMessageID\" class=\""+StylesDefs.ERROR_MESSAGE+"\"></span>");
123 
124     // fills up the list of generators
125     for(GeneratorSource generator: generators) {
126       genList.addItem(generator.getName());
127       setGridStyle(generator.getWidget());
128     }
129 
130     sizeList.addItem("Small", "120");
131     sizeList.addItem("Medium", "230");
132     sizeList.addItem("Large", "350");
133     sizeList.setSelectedIndex(2);
134 
135     ecLevelList.addItem("L");
136     ecLevelList.addItem("M");
137     ecLevelList.addItem("Q");
138     ecLevelList.addItem("H");
139     ecLevelList.setSelectedIndex(0);
140     
141     encodingList.addItem("UTF-8");
142     encodingList.addItem("ISO-8859-1");
143     encodingList.addItem("Shift_JIS");
144     encodingList.setSelectedIndex(0);
145 
146     // updates the second row of the table with the content of the selected generator
147     genList.addChangeHandler(new ChangeHandler() {
148       @Override
149       public void onChange(ChangeEvent Event) {
150         int i = genList.getSelectedIndex();
151         setWidget(i);
152       }
153     });
154 
155     // grid for the generator picker
156     HTMLTable selectionTable = new Grid(1, 2);
157     selectionTable.setText(0, 0, "Contents");
158     selectionTable.setWidget(0, 1, genList);
159     setGridStyle(selectionTable);
160 
161     topPanel.setWidget(0, 0, selectionTable);
162 
163     // grid for the generate button
164     HTMLTable generateGrid = new Grid(1, 2);
165     setGridStyle(generateGrid);
166 
167     Button generateButton = new Button("Generate &rarr;");
168     generateButton.addClickHandler(new ClickHandler() {
169       @Override
170       public void onClick(ClickEvent event) {
171         generate();
172       }
173     });
174     generateGrid.setWidget(0, 1, generateButton);
175     topPanel.setWidget(4, 0, generateGrid);
176 
177     HTMLTable configTable = new Grid(3, 2);
178     configTable.setText(0, 0, "Barcode size");
179     configTable.setWidget(0, 1, sizeList);
180     configTable.setText(1, 0, "Error correction");
181     configTable.setWidget(1, 1, ecLevelList);
182     configTable.setText(2, 0, "Character encoding");
183     configTable.setWidget(2, 1, encodingList);
184     setGridStyle(configTable);
185     topPanel.setWidget(3, 0, configTable);
186   }

188   private static void setGridStyle(HTMLTable grid) {
189     grid.getColumnFormatter().addStyleName(0, "firstColumn");
190     grid.getColumnFormatter().addStyleName(1, "secondColumn");
191     HTMLTable.CellFormatter cellFormatter = grid.getCellFormatter();
192     for (int i = 0; i < grid.getRowCount(); ++i) {
193       cellFormatter.addStyleName(i, 0, "firstColumn");
194       cellFormatter.addStyleName(i, 1, "secondColumn");
195     }
196   }
197 
198   private static String getUrl(int sizeX, int sizeY, String ecLevel, String encoding, String content) {
199     StringBuilder result = new StringBuilder(100);
200     result.append("http://zxing.org/w/chart?cht=qr");
201     result.append("&chs=").append(sizeX).append('x').append(sizeY);
202     result.append("&chld=").append(ecLevel);
203     result.append("&choe=").append(encoding);
204     result.append("&chl=").append(URL.encodeQueryString(content));
205     return result.toString();
206   }
207 
208   private void generate() {
209     try {
210       String text = selectedGenerator.getText();
211       eraseErrorMessage();
212       int size = Integer.parseInt(sizeList.getValue(sizeList.getSelectedIndex()));
213       String ecLevel = ecLevelList.getValue(ecLevelList.getSelectedIndex());
214       String encoding = encodingList.getValue(encodingList.getSelectedIndex());
215       String url = getUrl(size, size, ecLevel, encoding, text);
216       result.setUrl(url);
217       result.setVisible(true);
218       urlResult.setText(url);
219       urlResult.setVisible(true);
220       rawTextResult.setText(text);
221       rawTextResult.setVisible(true);
222       Element linkElement = DOM.getElementById("downloadlink");
223       linkElement.setAttribute("href", url);
224       downloadText.setVisible(true);
225     } catch (GeneratorException ex) {
226       invalidateBarcode();
227       String error = ex.getMessage();
228       showErrorMessage(error);
229     }
230   }
231 
232   void invalidateBarcode() {
233     result.setVisible(false);
234     urlResult.setText("");
235     urlResult.setVisible(false);
236     rawTextResult.setText("");
237     rawTextResult.setVisible(false);
238     Element linkElement = DOM.getElementById("downloadlink");
239     linkElement.setAttribute("href", "");
240     downloadText.setVisible(false);
241   }
242 
243   private static void showErrorMessage(String error) {
244     Element errorElement = DOM.getElementById("errorMessageID");
245     errorElement.setInnerHTML(error);
246   }
247 
248   private static void eraseErrorMessage() {
249     Element errorElement = DOM.getElementById("errorMessageID");
250     errorElement.setInnerHTML("&nbsp;");
251   }
252 
253   private final ChangeHandler changeHandler = new ChangeHandler() {
254     @Override
255     public void onChange(ChangeEvent event) {
256       try {
257         selectedGenerator.validate((Widget) event.getSource());
258         eraseErrorMessage();
259       } catch (GeneratorException ex) {
260         String error = ex.getMessage();
261         showErrorMessage(error);
262         invalidateBarcode();
263       }
264     }
265   };
266 
267   private final KeyPressHandler keyPressHandler = new KeyPressHandler() {
268     @Override
269     public void onKeyPress(KeyPressEvent event) {
270       if (event.getCharCode() == '\n' || event.getCharCode() == '\r') {
271         generate();
272       }
273     }
274   };
