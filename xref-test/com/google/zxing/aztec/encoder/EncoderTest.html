130   public void testAztecWriter() throws Exception {
131     testWriter("\u20AC 1 sample data.", "ISO-8859-1", 25, true, 2);
132     testWriter("\u20AC 1 sample data.", "ISO-8859-15", 25, true, 2);
133     testWriter("\u20AC 1 sample data.", "UTF-8",  25, true, 2);
134     testWriter("\u20AC 1 sample data.", "UTF-8", 100, true, 3);
135     testWriter("\u20AC 1 sample data.", "UTF-8", 300, true, 4);
136     testWriter("\u20AC 1 sample data.", "UTF-8", 500, false, 5);
137     // Test AztecWriter defaults
138     String data = "In ut magna vel mauris malesuada";
139     AztecWriter writer = new AztecWriter();
140     BitMatrix matrix = writer.encode(data, BarcodeFormat.AZTEC, 0, 0);
141     AztecCode aztec = Encoder.encode(data.getBytes(StandardCharsets.ISO_8859_1),
142         Encoder.DEFAULT_EC_PERCENT, Encoder.DEFAULT_AZTEC_LAYERS);
143     BitMatrix expectedMatrix = aztec.getMatrix();
144     assertEquals(matrix, expectedMatrix);
145   }

309   @Test
310   public void testHighLevelEncodeBinary() throws Exception {
311     // binary short form single byte
312     testHighLevelEncodeString("N\0N",
313         // 'N'  B/S    =1   '\0'      N
314         ".XXXX XXXXX ....X ........ .XXXX");   // Encode "N" in UPPER
315 
316     testHighLevelEncodeString("N\0n",
317         // 'N'  B/S    =2   '\0'       'n'
318         ".XXXX XXXXX ...X. ........ .XX.XXX.");   // Encode "n" in BINARY
319 
320     // binary short form consecutive bytes
321     testHighLevelEncodeString("N\0\u0080 A",
322         // 'N'  B/S    =2    '\0'    \u0080   ' '  'A'
323         ".XXXX XXXXX ...X. ........ X....... ....X ...X.");
324 
325     // binary skipping over single character
326     testHighLevelEncodeString("\0a\u00FF\u0080 A",
327         // B/S  =4    '\0'      'a'     '\3ff'   '\200'   ' '   'A'
328         "XXXXX ..X.. ........ .XX....X XXXXXXXX X....... ....X ...X.");
329 
330     // getting into binary mode from digit mode
331     testHighLevelEncodeString("1234\0",
332         //D/L   '1'  '2'  '3'  '4'  U/L  B/S    =1    \0
333         "XXXX. ..XX .X.. .X.X .XX. XXX. XXXXX ....X ........"
334     );
335 
336     // Create a string in which every character requires binary
337     StringBuilder sb = new StringBuilder();
338     for (int i = 0; i <= 3000; i++) {
339       sb.append((char)(128 + (i % 30)));
340     }
341     // Test the output generated by Binary/Switch, particularly near the
342     // places where the encoding changes: 31, 62, and 2047+31=2078
343     for (int i : new int[] { 1, 2, 3, 10, 29, 30, 31, 32, 33,
344                              60, 61, 62, 63, 64, 2076, 2077, 2078, 2079, 2080, 2100 }) {
345       // This is the expected length of a binary string of length "i"
346       int expectedLength = (8 * i) +
347           ( (i <= 31) ? 10 : (i <= 62) ? 20 : (i <= 2078) ? 21 : 31);
348       // Verify that we are correct about the length.
349       testHighLevelEncodeString(sb.substring(0, i), expectedLength);
350       if (i != 1 && i != 32 && i != 2079) {
351         // The addition of an 'a' at the beginning or end gets merged into the binary code
352         // in those cases where adding another binary character only adds 8 or 9 bits to the result.
353         // So we exclude the border cases i=1,32,2079
354         // A lower case letter at the beginning will be merged into binary mode
355         testHighLevelEncodeString('a' + sb.substring(0, i - 1), expectedLength);
356         // A lower case letter at the end will also be merged into binary mode
357         testHighLevelEncodeString(sb.substring(0, i - 1) + 'a', expectedLength);
358       }
359       // A lower case letter at both ends will enough to latch us into LOWER.
360       testHighLevelEncodeString('a' + sb.substring(0, i) + 'b', expectedLength + 15);
361     }

412     // be error correction
413     String alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";

444   private static void testEncodeDecode(String data, boolean compact, int layers) throws Exception {
445     AztecCode aztec = Encoder.encode(data.getBytes(StandardCharsets.ISO_8859_1), 25, Encoder.DEFAULT_AZTEC_LAYERS);
446     assertEquals("Unexpected symbol format (compact)", compact, aztec.isCompact());
447     assertEquals("Unexpected nr. of layers", layers, aztec.getLayers());
448     BitMatrix matrix = aztec.getMatrix();
449     AztecDetectorResult r = 
450         new AztecDetectorResult(matrix, NO_POINTS, aztec.isCompact(), aztec.getCodeWords(), aztec.getLayers());
451     DecoderResult res = new Decoder().decode(r);
452     assertEquals(data, res.getText());
453     // Check error correction by introducing a few minor errors
454     Random random = getPseudoRandom();
455     matrix.flip(random.nextInt(matrix.getWidth()), random.nextInt(2));
456     matrix.flip(random.nextInt(matrix.getWidth()), matrix.getHeight() - 2 + random.nextInt(2));
457     matrix.flip(random.nextInt(2), random.nextInt(matrix.getHeight()));
458     matrix.flip(matrix.getWidth() - 2 + random.nextInt(2), random.nextInt(matrix.getHeight()));
459     r = new AztecDetectorResult(matrix, NO_POINTS, aztec.isCompact(), aztec.getCodeWords(), aztec.getLayers());
460     res = new Decoder().decode(r);
461     assertEquals(data, res.getText());
462   }

464   private static void testWriter(String data, 
465                                  String charset, 
466                                  int eccPercent, 
467                                  boolean compact, 
468                                  int layers) throws FormatException {
469     // 1. Perform an encode-decode round-trip because it can be lossy.
470     // 2. Aztec Decoder currently always decodes the data with a LATIN-1 charset:
471     String expectedData = new String(data.getBytes(Charset.forName(charset)), StandardCharsets.ISO_8859_1);
472     Map<EncodeHintType,Object> hints = new EnumMap<>(EncodeHintType.class);
473     hints.put(EncodeHintType.CHARACTER_SET, charset);
474     hints.put(EncodeHintType.ERROR_CORRECTION, eccPercent);
475     AztecWriter writer = new AztecWriter();
476     BitMatrix matrix = writer.encode(data, BarcodeFormat.AZTEC, 0, 0, hints);
477     AztecCode aztec = Encoder.encode(data.getBytes(Charset.forName(charset)), eccPercent,
478         Encoder.DEFAULT_AZTEC_LAYERS);
479     assertEquals("Unexpected symbol format (compact)", compact, aztec.isCompact());
480     assertEquals("Unexpected nr. of layers", layers, aztec.getLayers());
481     BitMatrix matrix2 = aztec.getMatrix();
482     assertEquals(matrix, matrix2);
483     AztecDetectorResult r = 
484         new AztecDetectorResult(matrix, NO_POINTS, aztec.isCompact(), aztec.getCodeWords(), aztec.getLayers());
485     DecoderResult res = new Decoder().decode(r);
486     assertEquals(expectedData, res.getText());
487     // Check error correction by introducing up to eccPercent errors
488     int ecWords = aztec.getCodeWords() * eccPercent / 100;
489     Random random = getPseudoRandom();
490     for (int i = 0; i < ecWords; i++) {
491       // don't touch the core
492       int x = random.nextBoolean() ?
493                 random.nextInt(aztec.getLayers() * 2)
494                 : matrix.getWidth() - 1 - random.nextInt(aztec.getLayers() * 2);
495       int y = random.nextBoolean() ?
496                 random.nextInt(aztec.getLayers() * 2)
497                 : matrix.getHeight() - 1 - random.nextInt(aztec.getLayers() * 2);
498       matrix.flip(x, y);
499     }
500     r = new AztecDetectorResult(matrix, NO_POINTS, aztec.isCompact(), aztec.getCodeWords(), aztec.getLayers());
501     res = new Decoder().decode(r);
502     assertEquals(expectedData, res.getText());
503   }
