48    public void testErrorInParameterLocatorZeroZero() throws Exception {
49      // Layers=1, CodeWords=1.  So the parameter info and its Reed-Solomon info
50      // will be completely zero!
51      testErrorInParameterLocator("X");
52    }

55    public void testErrorInParameterLocatorCompact() throws Exception {
56      testErrorInParameterLocator("This is an example Aztec symbol for Wikipedia.");
57    }

60    public void testErrorInParameterLocatorNotCompact() throws Exception {
61      String alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYabcdefghijklmnopqrstuvwxyz";
62      testErrorInParameterLocator(alphabet + alphabet + alphabet);
63    }

65    // Test that we can tolerate errors in the parameter locator bits
66    private static void testErrorInParameterLocator(String data) throws Exception {
67      AztecCode aztec = Encoder.encode(data.getBytes(LATIN_1), 25, Encoder.DEFAULT_AZTEC_LAYERS);
68      Random random = new Random(aztec.getMatrix().hashCode());   // pseudo-random, but deterministic
69      int layers = aztec.getLayers();
70      boolean compact = aztec.isCompact();
71      List<Point> orientationPoints = getOrientationPoints(aztec);
72      for (boolean isMirror : new boolean[] { false, true }) {
73        for (BitMatrix matrix : getRotations(aztec.getMatrix())) {
74          // Systematically try every possible 1- and 2-bit error.
75          for (int error1 = 0; error1 < orientationPoints.size(); error1++) {
76            for (int error2 = error1; error2 < orientationPoints.size(); error2++) {
77              BitMatrix copy = isMirror ? transpose(matrix) : clone(matrix);
78              copy.flip(orientationPoints.get(error1).getX(), orientationPoints.get(error1).getY());
79              if (error2 > error1) {
80                // if error2 == error1, we only test a single error
81                copy.flip(orientationPoints.get(error2).getX(), orientationPoints.get(error2).getY());
82              }
83              // The detector doesn't seem to work when matrix bits are only 1x1.  So magnify.
84              AztecDetectorResult r = new Detector(makeLarger(copy, 3)).detect(isMirror);
85              assertNotNull(r);
86              assertEquals(r.getNbLayers(), layers);
87              assertEquals(r.isCompact(), compact);
88              DecoderResult res = new Decoder().decode(r);
89              assertEquals(data, res.getText());
90            }
91          }
92          // Try a few random three-bit errors;
93          for (int i = 0; i < 5; i++) {
94            BitMatrix copy = clone(matrix);
95            Collection<Integer> errors = new TreeSet<>();
96            while (errors.size() < 3) {
97              // Quick and dirty way of getting three distinct integers between 1 and n.
98              errors.add(random.nextInt(orientationPoints.size()));
99            }
100           for (int error : errors) {
101             copy.flip(orientationPoints.get(error).getX(), orientationPoints.get(error).getY());
102           }
103           try {
104             new Detector(makeLarger(copy, 3)).detect(false);
105             fail("Should not reach here");
106           } catch (NotFoundException expected) { }
107         }
108       }
109     }
110   }

112   // Zooms a bit matrix so that each bit is factor x factor
113   private static BitMatrix makeLarger(BitMatrix input, int factor) {
114     int width = input.getWidth();
115     BitMatrix output = new BitMatrix(width * factor);
116     for (int inputY = 0; inputY < width; inputY++) {
117       for (int inputX = 0; inputX < width; inputX++) {
118         if (input.get(inputX, inputY)) {
119           output.setRegion(inputX * factor, inputY * factor, factor, factor);
120         }
121       }
122     }
123     return output;
124   }
125 
126   // Returns a list of the four rotations of the BitMatrix.
127   private static Iterable<BitMatrix> getRotations(BitMatrix matrix0) {
128     BitMatrix matrix90 = rotateRight(matrix0);
129     BitMatrix matrix180 = rotateRight(matrix90);
130     BitMatrix matrix270 = rotateRight(matrix180);
131     return Arrays.asList(matrix0, matrix90, matrix180, matrix270);
132   }
133 
134   // Rotates a square BitMatrix to the right by 90 degrees
135   private static BitMatrix rotateRight(BitMatrix input) {
136     int width = input.getWidth();
137     BitMatrix result = new BitMatrix(width);
138     for (int x = 0; x < width; x++) {
139       for (int y = 0; y < width; y++) {
140         if (input.get(x,y)) {
141           result.set(y, width - x - 1);
142         }
143       }
144     }
145     return result;
146   }
147 
148   // Returns the transpose of a bit matrix, which is equivalent to rotating the
149   // matrix to the right, and then flipping it left-to-right
150   private static BitMatrix transpose(BitMatrix input) {
151     int width = input.getWidth();
152     BitMatrix result = new BitMatrix(width);
153     for (int x = 0; x < width; x++) {
154       for (int y = 0; y < width; y++) {
155         if (input.get(x, y)) {
156           result.set(y, x);
157         }
158       }
159     }
160     return result;
161   }
162 
163   private static BitMatrix clone(BitMatrix input)  {
164     int width = input.getWidth();
165     BitMatrix result = new BitMatrix(width);
166     for (int x = 0; x < width; x++) {
167       for (int y = 0; y < width; y++) {
168         if (input.get(x,y)) {
169           result.set(x,y);
170         }
171       }
172     }
173     return result;
174   }
175 
176   private static List<Point> getOrientationPoints(AztecCode code) {
177     int center = code.getMatrix().getWidth() / 2;
178     int offset = code.isCompact() ? 5 : 7;
179     List<Point> result = new ArrayList<>();
180     for (int xSign = -1; xSign <= 1; xSign += 2) {
181       for (int ySign = -1; ySign <= 1; ySign += 2) {
182         result.add(new Point(center + xSign * offset, center + ySign * offset));
183         result.add(new Point(center + xSign * (offset - 1), center + ySign * offset));
184         result.add(new Point(center + xSign * offset, center + ySign * (offset - 1)));
185       }
186     }
187     return result;
188   }
