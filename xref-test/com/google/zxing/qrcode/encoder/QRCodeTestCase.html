31    @Test
32    public void test() {
33      QRCode qrCode = new QRCode();
34  
35      // First, test simple setters and getters.
36      // We use numbers of version 7-H.
37      qrCode.setMode(Mode.BYTE);
38      qrCode.setECLevel(ErrorCorrectionLevel.H);
39      qrCode.setVersion(Version.getVersionForNumber(7));
40      qrCode.setMaskPattern(3);
41  
42      assertSame(Mode.BYTE, qrCode.getMode());
43      assertSame(ErrorCorrectionLevel.H, qrCode.getECLevel());
44      assertEquals(7, qrCode.getVersion().getVersionNumber());
45      assertEquals(3, qrCode.getMaskPattern());
46  
47      // Prepare the matrix.
48      ByteMatrix matrix = new ByteMatrix(45, 45);
49      // Just set bogus zero/one values.
50      for (int y = 0; y < 45; ++y) {
51        for (int x = 0; x < 45; ++x) {
52          matrix.set(x, y, (y + x) % 2);
53        }
54      }
55  
56      // Set the matrix.
57      qrCode.setMatrix(matrix);
58      assertSame(matrix, qrCode.getMatrix());
59    }
