32    @Test
33    public void testToString() {
34      ByteMatrix array = new ByteMatrix(3, 3);
35      array.set(0, 0, 0);
36      array.set(1, 0, 1);
37      array.set(2, 0, 0);
38      array.set(0, 1, 1);
39      array.set(1, 1, 0);
40      array.set(2, 1, 1);
41      array.set(0, 2, -1);
42      array.set(1, 2, -1);
43      array.set(2, 2, -1);
44      String expected = " 0 1 0\n" + " 1 0 1\n" + "      \n";
45      assertEquals(expected, array.toString());
46    }

48    @Test
49    public void testClearMatrix() {
50      ByteMatrix matrix = new ByteMatrix(2, 2);
51      MatrixUtil.clearMatrix(matrix);
52      assertEquals(-1, matrix.get(0, 0));
53      assertEquals(-1, matrix.get(1, 0));
54      assertEquals(-1, matrix.get(0, 1));
55      assertEquals(-1, matrix.get(1, 1));

222   @Test
223   public void testBuildMatrix() throws WriterException {
224     // From http://www.swetake.com/qr/qr7.html
225     char[] bytes = {32, 65, 205, 69, 41, 220, 46, 128, 236,
226         42, 159, 74, 221, 244, 169, 239, 150, 138,
227         70, 237, 85, 224, 96, 74, 219 , 61};
228     BitArray bits = new BitArray();
229     for (char c: bytes) {
230       bits.appendBits(c, 8);

232     ByteMatrix matrix = new ByteMatrix(21, 21);

271   @Test
272   public void testCalculateBCHCode() {
273     // Encoding of type information.
274     // From Appendix C in JISX0510:2004 (p 65)
275     assertEquals(0xdc, MatrixUtil.calculateBCHCode(5, 0x537));
276     // From http://www.swetake.com/qr/qr6.html
277     assertEquals(0x1c2, MatrixUtil.calculateBCHCode(0x13, 0x537));
278     // From http://www.swetake.com/qr/qr11.html
279     assertEquals(0x214, MatrixUtil.calculateBCHCode(0x1b, 0x537));
280 
281     // Encofing of version information.
282     // From Appendix D in JISX0510:2004 (p 68)
283     assertEquals(0xc94, MatrixUtil.calculateBCHCode(7, 0x1f25));
284     assertEquals(0x5bc, MatrixUtil.calculateBCHCode(8, 0x1f25));
285     assertEquals(0xa99, MatrixUtil.calculateBCHCode(9, 0x1f25));
286     assertEquals(0x4d3, MatrixUtil.calculateBCHCode(10, 0x1f25));
287     assertEquals(0x9a6, MatrixUtil.calculateBCHCode(20, 0x1f25));
288     assertEquals(0xd75, MatrixUtil.calculateBCHCode(30, 0x1f25));
