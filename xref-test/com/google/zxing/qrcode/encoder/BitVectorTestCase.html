29    private static long getUnsignedInt(BitArray v, int index) {
30      long result = 0L;
31      for (int i = 0, offset = index * 8; i < 32; i++) {
32        if (v.get(offset + i)) {
33          result |= 1L << (31 - i);
34        }
35      }
36      return result;
37    }

39    @Test
40    public void testAppendBit() {
41      BitArray v = new BitArray();
42      assertEquals(0, v.getSizeInBytes());
43      // 1
44      v.appendBit(true);
45      assertEquals(1, v.getSize());
46      assertEquals(0x80000000L, getUnsignedInt(v, 0));
47      // 10
48      v.appendBit(false);
49      assertEquals(2, v.getSize());
50      assertEquals(0x80000000L, getUnsignedInt(v, 0));
51      // 101
52      v.appendBit(true);
53      assertEquals(3, v.getSize());
54      assertEquals(0xa0000000L, getUnsignedInt(v, 0));
55      // 1010
56      v.appendBit(false);
57      assertEquals(4, v.getSize());
58      assertEquals(0xa0000000L, getUnsignedInt(v, 0));
59      // 10101
60      v.appendBit(true);
61      assertEquals(5, v.getSize());
62      assertEquals(0xa8000000L, getUnsignedInt(v, 0));
63      // 101010
64      v.appendBit(false);
65      assertEquals(6, v.getSize());
66      assertEquals(0xa8000000L, getUnsignedInt(v, 0));
67      // 1010101
68      v.appendBit(true);
69      assertEquals(7, v.getSize());
70      assertEquals(0xaa000000L, getUnsignedInt(v, 0));
71      // 10101010
72      v.appendBit(false);
73      assertEquals(8, v.getSize());
74      assertEquals(0xaa000000L, getUnsignedInt(v, 0));
75      // 10101010 1
76      v.appendBit(true);
77      assertEquals(9, v.getSize());
78      assertEquals(0xaa800000L, getUnsignedInt(v, 0));
79      // 10101010 10
80      v.appendBit(false);
81      assertEquals(10, v.getSize());
82      assertEquals(0xaa800000L, getUnsignedInt(v, 0));
83    }

85    @Test
86    public void testAppendBits() {
87      BitArray v = new BitArray();
88      v.appendBits(0x1, 1);
89      assertEquals(1, v.getSize());
90      assertEquals(0x80000000L, getUnsignedInt(v, 0));
91      v = new BitArray();
92      v.appendBits(0xff, 8);
93      assertEquals(8, v.getSize());
94      assertEquals(0xff000000L, getUnsignedInt(v, 0));
95      v = new BitArray();
96      v.appendBits(0xff7, 12);
97      assertEquals(12, v.getSize());
98      assertEquals(0xff700000L, getUnsignedInt(v, 0));
99    }

101   @Test
102   public void testNumBytes() {
103     BitArray v = new BitArray();
104     assertEquals(0, v.getSizeInBytes());
105     v.appendBit(false);
106     // 1 bit was added in the vector, so 1 byte should be consumed.
107     assertEquals(1, v.getSizeInBytes());
108     v.appendBits(0, 7);
109     assertEquals(1, v.getSizeInBytes());
110     v.appendBits(0, 8);
111     assertEquals(2, v.getSizeInBytes());
112     v.appendBits(0, 1);
113     // We now have 17 bits, so 3 bytes should be consumed.
114     assertEquals(3, v.getSizeInBytes());
115   }

128   @Test
129   public void testXOR() {
130     BitArray v1 = new BitArray();
131     v1.appendBits(0x5555aaaa, 32);
132     BitArray v2 = new BitArray();
133     v2.appendBits(0xaaaa5555, 32);
134     v1.xor(v2);
135     assertEquals(0xffffffffL, getUnsignedInt(v1, 0));
136   }

138   @Test
139   public void testXOR2() {
140     BitArray v1 = new BitArray();
141     v1.appendBits(0x2a, 7);  // 010 1010
142     BitArray v2 = new BitArray();
143     v2.appendBits(0x55, 7);  // 101 0101
144     v1.xor(v2);
