17  package com.google.zxing;
18  
19  import java.awt.Graphics2D;
20  import java.awt.geom.AffineTransform;
21  import java.awt.image.BufferedImage;
22  import java.awt.image.WritableRaster;
23  
24  /**
25   * This LuminanceSource implementation is meant for J2SE clients and our blackbox unit tests.
26   *
27   * @author dswitkin@google.com (Daniel Switkin)
28   * @author Sean Owen
29   * @author code@elektrowolle.de (Wolfgang Jung)
30   */
31  public final class BufferedImageLuminanceSource extends LuminanceSource {
32  
33    private static final double MINUS_45_IN_RADIANS = -0.7853981633974483; // Math.toRadians(-45.0)
34  
35    private final BufferedImage image;
36    private final int left;
37    private final int top;
38  
39    public BufferedImageLuminanceSource(BufferedImage image) {
40      this(image, 0, 0, image.getWidth(), image.getHeight());
41    }
42  
43    public BufferedImageLuminanceSource(BufferedImage image, int left, int top, int width, int height) {
44      super(width, height);
45  
46      if (image.getType() == BufferedImage.TYPE_BYTE_GRAY) {
47        this.image = image;
48      } else {
49        int sourceWidth = image.getWidth();
50        int sourceHeight = image.getHeight();
51        if (left + width > sourceWidth || top + height > sourceHeight) {
52          throw new IllegalArgumentException("Crop rectangle does not fit within image data.");
53        }
54  
55        this.image = new BufferedImage(sourceWidth, sourceHeight, BufferedImage.TYPE_BYTE_GRAY);
56  
57        WritableRaster raster = this.image.getRaster();
58        int[] buffer = new int[width];
59        for (int y = top; y < top + height; y++) {
60          image.getRGB(left, y, width, 1, buffer, 0, sourceWidth);
61          for (int x = 0; x < width; x++) {
62            int pixel = buffer[x];
63  
64            // The color of fully-transparent pixels is irrelevant. They are often, technically, fully-transparent
65            // black (0 alpha, and then 0 RGB). They are often used, of course as the "white" area in a
66            // barcode image. Force any such pixel to be white:
67            if ((pixel & 0xFF000000) == 0) {
68              pixel = 0xFFFFFFFF; // = white
69            }
70  
71            // .229R + 0.587G + 0.114B (YUV/YIQ for PAL and NTSC)
72            buffer[x] =
73                (306 * ((pixel >> 16) & 0xFF) +
74                 601 * ((pixel >> 8) & 0xFF) +
75                 117 * (pixel & 0xFF) +
76                 0x200) >> 10;
77          }
78          raster.setPixels(left, y, width, 1, buffer);
79        }
80  
81      }
82      this.left = left;
83      this.top = top;
84    }
85  
86    @Override
87    public byte[] getRow(int y, byte[] row) {
88      if (y < 0 || y >= getHeight()) {
89        throw new IllegalArgumentException("Requested row is outside the image: " + y);
90      }
91      int width = getWidth();
92      if (row == null || row.length < width) {
93        row = new byte[width];
94      }
95      // The underlying raster of image consists of bytes with the luminance values
96      image.getRaster().getDataElements(left, top + y, width, 1, row);
97      return row;
98    }
99  
100   @Override
101   public byte[] getMatrix() {
102     int width = getWidth();
103     int height = getHeight();
104     int area = width * height;
105     byte[] matrix = new byte[area];
106     // The underlying raster of image consists of area bytes with the luminance values
107     image.getRaster().getDataElements(left, top, width, height, matrix);
108     return matrix;
109   }
110 
111   @Override
112   public boolean isCropSupported() {
113     return true;
114   }
115 
116   @Override
117   public LuminanceSource crop(int left, int top, int width, int height) {
118     return new BufferedImageLuminanceSource(image, this.left + left, this.top + top, width, height);
119   }
120 
121   /**
122    * This is always true, since the image is a gray-scale image.
123    *
124    * @return true
125    */
126   @Override
127   public boolean isRotateSupported() {
128     return true;
129   }
130 
131   @Override
132   public LuminanceSource rotateCounterClockwise() {
133     int sourceWidth = image.getWidth();
134     int sourceHeight = image.getHeight();
135 
136     // Rotate 90 degrees counterclockwise.
137     AffineTransform transform = new AffineTransform(0.0, -1.0, 1.0, 0.0, 0.0, sourceWidth);
138 
139     // Note width/height are flipped since we are rotating 90 degrees.
140     BufferedImage rotatedImage = new BufferedImage(sourceHeight, sourceWidth, BufferedImage.TYPE_BYTE_GRAY);
141 
142     // Draw the original image into rotated, via transformation
143     Graphics2D g = rotatedImage.createGraphics();
144     g.drawImage(image, transform, null);
145     g.dispose();
146 
147     // Maintain the cropped region, but rotate it too.
148     int width = getWidth();
149     return new BufferedImageLuminanceSource(rotatedImage, top, sourceWidth - (left + width), getHeight(), width);
150   }
151 
152   @Override
153   public LuminanceSource rotateCounterClockwise45() {
154     int width = getWidth();
155     int height = getHeight();
156 
157     int oldCenterX = left + width / 2;
158     int oldCenterY = top + height / 2;
159 
160     // Rotate 45 degrees counterclockwise.
161     AffineTransform transform = AffineTransform.getRotateInstance(MINUS_45_IN_RADIANS, oldCenterX, oldCenterY);
162 
163     int sourceDimension = Math.max(image.getWidth(), image.getHeight());
164     BufferedImage rotatedImage = new BufferedImage(sourceDimension, sourceDimension, BufferedImage.TYPE_BYTE_GRAY);
165 
166     // Draw the original image into rotated, via transformation
167     Graphics2D g = rotatedImage.createGraphics();
168     g.drawImage(image, transform, null);
169     g.dispose();
170 
171     int halfDimension = Math.max(width, height) / 2;
172     int newLeft = Math.max(0, oldCenterX - halfDimension);
173     int newTop = Math.max(0, oldCenterY - halfDimension);
174     int newRight = Math.min(sourceDimension - 1, oldCenterX + halfDimension);
175     int newBottom = Math.min(sourceDimension - 1, oldCenterY + halfDimension);
176 
177     return new BufferedImageLuminanceSource(rotatedImage, newLeft, newTop, newRight - newLeft, newBottom - newTop);
178   }
