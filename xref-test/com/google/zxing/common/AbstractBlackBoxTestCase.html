77    protected AbstractBlackBoxTestCase(String testBasePathSuffix,
78                                       Reader barcodeReader,
79                                       BarcodeFormat expectedFormat) {
80      // A little workaround to prevent aggravation in my IDE
81      File testBase = new File(testBasePathSuffix);
82      if (!testBase.exists()) {
83        // try starting with 'core' since the test base is often given as the project root
84        testBase = new File("core/" + testBasePathSuffix);
85      }
86      this.testBase = testBase;
87      this.barcodeReader = barcodeReader;
88      this.expectedFormat = expectedFormat;
89      testResults = new ArrayList<>();
90  
91      System.setProperty("java.util.logging.SimpleFormatter.format", "%4$s: %5$s%6$s%n");
92    }

95      addTest(mustPassCount, tryHarderCount, 0, 0, rotation);

98    /**
99     * Adds a new test for the current directory of images.
100    *
101    * @param mustPassCount The number of images which must decode for the test to pass.
102    * @param tryHarderCount The number of images which must pass using the try harder flag.
103    * @param maxMisreads Maximum number of images which can fail due to successfully reading the wrong contents
104    * @param maxTryHarderMisreads Maximum number of images which can fail due to successfully
105    *                             reading the wrong contents using the try harder flag
106    * @param rotation The rotation in degrees clockwise to use for this test.
107    */
108   protected final void addTest(int mustPassCount,
109                                int tryHarderCount,
110                                int maxMisreads,
111                                int maxTryHarderMisreads,
112                                float rotation) {
113     testResults.add(new TestResult(mustPassCount, tryHarderCount, maxMisreads, maxTryHarderMisreads, rotation));
114   }

125   // This workaround is used because AbstractNegativeBlackBoxTestCase overrides this method but does
126   // not return SummaryResults.
127   @Test
128   public void testBlackBox() throws IOException {
129     testBlackBoxCountingResults(true);
130   }

132   public final SummaryResults testBlackBoxCountingResults(boolean assertOnFailure) throws IOException {
133     assertFalse(testResults.isEmpty());
134 
135     File[] imageFiles = getImageFiles();
136     int testCount = testResults.size();
137 
138     int[] passedCounts = new int[testCount];
139     int[] misreadCounts = new int[testCount];
140     int[] tryHarderCounts = new int[testCount];
141     int[] tryHaderMisreadCounts = new int[testCount];
142 
143     for (File testImage : imageFiles) {
144       log.info(String.format("Starting %s", testImage.getAbsolutePath()));
145 
146       BufferedImage image = ImageIO.read(testImage);
147 
148       String testImageFileName = testImage.getName();
149       String fileBaseName = testImageFileName.substring(0, testImageFileName.indexOf('.'));
150       File expectedTextFile = new File(testBase, fileBaseName + ".txt");
151       String expectedText;
152       if (expectedTextFile.exists()) {
153         expectedText = readFileAsString(expectedTextFile, UTF8);
154       } else {
155         expectedTextFile = new File(testBase, fileBaseName + ".bin");
156         assertTrue(expectedTextFile.exists());
157         expectedText = readFileAsString(expectedTextFile, ISO88591);
158       }
159 
160       File expectedMetadataFile = new File(testBase, fileBaseName + ".metadata.txt");
161       Properties expectedMetadata = new Properties();
162       if (expectedMetadataFile.exists()) {
163         InputStream expectedStream = new FileInputStream(expectedMetadataFile);
164         try {
165           expectedMetadata.load(expectedStream);
166         } finally {
167           expectedStream.close();
168         }
169       }
170 
171       for (int x = 0; x < testCount; x++) {
172         float rotation = testResults.get(x).getRotation();
173         BufferedImage rotatedImage = rotateImage(image, rotation);
174         LuminanceSource source = new BufferedImageLuminanceSource(rotatedImage);
175         BinaryBitmap bitmap = new BinaryBitmap(new HybridBinarizer(source));
176         try {
177           if (decode(bitmap, rotation, expectedText, expectedMetadata, false)) {
178             passedCounts[x]++;
179           } else {
180             misreadCounts[x]++;
181           }
182         } catch (ReaderException ignored) {
183           log.fine(String.format("could not read at rotation %f", rotation));
184         }
185         try {
186           if (decode(bitmap, rotation, expectedText, expectedMetadata, true)) {
187             tryHarderCounts[x]++;
188           } else {
189             tryHaderMisreadCounts[x]++;
190           }
191         } catch (ReaderException ignored) {
192           log.fine(String.format("could not read at rotation %f w/TH", rotation));
193         }
194       }
195     }
196 
197     // Print the results of all tests first
198     int totalFound = 0;
199     int totalMustPass = 0;
200     int totalMisread = 0;
201     int totalMaxMisread = 0;
202 
203     for (int x = 0; x < testResults.size(); x++) {
204       TestResult testResult = testResults.get(x);
205       log.info(String.format("Rotation %d degrees:", (int) testResult.getRotation()));
206       log.info(String.format(" %d of %d images passed (%d required)",
207                              passedCounts[x], imageFiles.length, testResult.getMustPassCount()));
208       int failed = imageFiles.length - passedCounts[x];
209       log.info(String.format(" %d failed due to misreads, %d not detected",
210                              misreadCounts[x], failed - misreadCounts[x]));
211       log.info(String.format(" %d of %d images passed with try harder (%d required)",
212                              tryHarderCounts[x], imageFiles.length, testResult.getTryHarderCount()));
213       failed = imageFiles.length - tryHarderCounts[x];
214       log.info(String.format(" %d failed due to misreads, %d not detected",
215                              tryHaderMisreadCounts[x], failed - tryHaderMisreadCounts[x]));
216       totalFound += passedCounts[x] + tryHarderCounts[x];
217       totalMustPass += testResult.getMustPassCount() + testResult.getTryHarderCount();
218       totalMisread += misreadCounts[x] + tryHaderMisreadCounts[x];
219       totalMaxMisread += testResult.getMaxMisreads() + testResult.getMaxTryHarderMisreads();
220     }
221 
222     int totalTests = imageFiles.length * testCount * 2;
223     log.info(String.format("Decoded %d images out of %d (%d%%, %d required)",
224                            totalFound, totalTests, totalFound * 100 / totalTests, totalMustPass));
225     if (totalFound > totalMustPass) {
226       log.warning(String.format("+++ Test too lax by %d images", totalFound - totalMustPass));
227     } else if (totalFound < totalMustPass) {
228       log.warning(String.format("--- Test failed by %d images", totalMustPass - totalFound));
229     }
230 
231     if (totalMisread < totalMaxMisread) {
232       log.warning(String.format("+++ Test expects too many misreads by %d images", totalMaxMisread - totalMisread));
233     } else if (totalMisread > totalMaxMisread) {
234       log.warning(String.format("--- Test had too many misreads by %d images", totalMisread - totalMaxMisread));
235     }
236 
237     // Then run through again and assert if any failed
238     if (assertOnFailure) {
239       for (int x = 0; x < testCount; x++) {
240         TestResult testResult = testResults.get(x);
241         String label = "Rotation " + testResult.getRotation() + " degrees: Too many images failed";
242         assertTrue(label,
243                    passedCounts[x] >= testResult.getMustPassCount());
244         assertTrue("Try harder, " + label,
245                    tryHarderCounts[x] >= testResult.getTryHarderCount());
246         label = "Rotation " + testResult.getRotation() + " degrees: Too many images misread";
247         assertTrue(label,
248                    misreadCounts[x] <= testResult.getMaxMisreads());
249         assertTrue("Try harder, " + label,
250                    tryHaderMisreadCounts[x] <= testResult.getMaxTryHarderMisreads());
251       }
252     }
253     return new SummaryResults(totalFound, totalMustPass, totalTests);
254   }

256   private boolean decode(BinaryBitmap source,
257                          float rotation,
258                          String expectedText,
259                          Map<?,?> expectedMetadata,
260                          boolean tryHarder) throws ReaderException {
261 
262     String suffix = String.format(" (%srotation: %d)", tryHarder ? "try harder, " : "", (int) rotation);
263 
264     Map<DecodeHintType,Object> hints = new EnumMap<>(DecodeHintType.class);
265     if (tryHarder) {
266       hints.put(DecodeHintType.TRY_HARDER, Boolean.TRUE);
267     }
268 
269     Result result = barcodeReader.decode(source, hints);
270 
271     if (expectedFormat != result.getBarcodeFormat()) {
272       log.info(String.format("Format mismatch: expected '%s' but got '%s'%s",
273                              expectedFormat, result.getBarcodeFormat(), suffix));
274       return false;
275     }
276 
277     String resultText = result.getText();
278     if (!expectedText.equals(resultText)) {
279       log.info(String.format("Content mismatch: expected '%s' but got '%s'%s",
280                              expectedText, resultText, suffix));
281       return false;
282     }
283 
284     Map<ResultMetadataType,?> resultMetadata = result.getResultMetadata();
285     for (Map.Entry<?,?> metadatum : expectedMetadata.entrySet()) {
286       ResultMetadataType key = ResultMetadataType.valueOf(metadatum.getKey().toString());
287       Object expectedValue = metadatum.getValue();
288       Object actualValue = resultMetadata == null ? null : resultMetadata.get(key);
289       if (!expectedValue.equals(actualValue)) {
290         log.info(String.format("Metadata mismatch for key '%s': expected '%s' but got '%s'",
291                                key, expectedValue, actualValue));
292         return false;
293       }
294     }
295 
296     return true;
297   }

299   protected static String readFileAsString(File file, Charset charset) throws IOException {
300     StringBuilder result = new StringBuilder((int) file.length());
301     InputStreamReader reader = new InputStreamReader(new FileInputStream(file), charset);
302     try {
303       char[] buffer = new char[256];
304       int charsRead;
305       while ((charsRead = reader.read(buffer)) > 0) {
306         result.append(buffer, 0, charsRead);
307       }
308     } finally {
309       reader.close();
310     }
311     String stringContents = result.toString();
312     if (stringContents.endsWith("\n")) {
313       log.info("String contents of file " + file + " end with a newline. " +
314                   "This may not be intended and cause a test failure");
315     }
316     return stringContents;
317   }

319   protected static BufferedImage rotateImage(BufferedImage original, float degrees) {
320     if (degrees == 0.0f) {
321       return original;
322     }
323 
324     switch(original.getType()) {
325       case BufferedImage.TYPE_BYTE_INDEXED:
326       case BufferedImage.TYPE_BYTE_BINARY:
327         BufferedImage argb = new BufferedImage(original.getWidth(),
328                                                original.getHeight(),
329                                                BufferedImage.TYPE_INT_ARGB);
330         Graphics g = argb.createGraphics();
331         g.drawImage(original, 0, 0, null);
332         g.dispose();
333         original = argb;
334         break;
335     }
336 
337     double radians = Math.toRadians(degrees);
338 
339     // Transform simply to find out the new bounding box (don't actually run the image through it)
340     AffineTransform at = new AffineTransform();
341     at.rotate(radians, original.getWidth() / 2.0, original.getHeight() / 2.0);
342     BufferedImageOp op = new AffineTransformOp(at, AffineTransformOp.TYPE_BICUBIC);
343 
344     RectangularShape r = op.getBounds2D(original);
345     int width = (int) Math.ceil(r.getWidth());
346     int height = (int) Math.ceil(r.getHeight());
347 
348     // Real transform, now that we know the size of the new image and how to translate after we rotate
349     // to keep it centered
350     at = new AffineTransform();
351     at.rotate(radians, width / 2.0, height / 2.0);
352     at.translate((width - original.getWidth()) / 2.0,
353                  (height - original.getHeight()) / 2.0);
354     op = new AffineTransformOp(at, AffineTransformOp.TYPE_BICUBIC);
355 
356     return op.filter(original, new BufferedImage(width, height, original.getType()));
357   }
