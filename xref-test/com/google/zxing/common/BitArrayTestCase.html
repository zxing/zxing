102   @Test
103   public void testGetNextSet5() {
104     Random r = new SecureRandom(new byte[] {(byte) 0xDE, (byte) 0xAD, (byte) 0xBE, (byte) 0xEF});
105     for (int i = 0; i < 10; i++) {
106       BitArray array = new BitArray(1 + r.nextInt(100));
107       int numSet = r.nextInt(20);
108       for (int j = 0; j < numSet; j++) {
109         array.set(r.nextInt(array.getSize()));
110       }
111       int numQueries = r.nextInt(20);
112       for (int j = 0; j < numQueries; j++) {
113         int query = r.nextInt(array.getSize());
114         int expected = query;
115         while (expected < array.getSize() && !array.get(expected)) {
116           expected++;
117         }
118         int actual = array.getNextSet(query);
119         if (actual != expected) {
120           array.getNextSet(query);
121         }
122         assertEquals(expected, actual);
123       }
124     }
125   }

162   @Test
163   public void testIsRange() {
164     BitArray array = new BitArray(64);
165     assertTrue(array.isRange(0, 64, false));
166     assertFalse(array.isRange(0, 64, true));
167     array.set(32);
168     assertTrue(array.isRange(32, 33, true));
169     array.set(31);
170     assertTrue(array.isRange(31, 33, true));
171     array.set(34);
172     assertFalse(array.isRange(31, 35, true));
173     for (int i = 0; i < 31; i++) {
174       array.set(i);
175     }
176     assertTrue(array.isRange(0, 33, true));
177     for (int i = 33; i < 64; i++) {
178       array.set(i);
179     }
180     assertTrue(array.isRange(0, 64, true));
181     assertFalse(array.isRange(0, 64, false));
182   }

184   @Test
185   public void reverseAlgorithmTest() {
186     int[] oldBits = {128, 256, 512, 6453324, 50934953};
187     for (int size = 1; size < 160; size++) {
188       int[] newBitsOriginal = reverseOriginal(oldBits.clone(), size);
189       BitArray newBitArray = new BitArray(oldBits.clone(), size);
190       newBitArray.reverse();
191       int[] newBitsNew = newBitArray.getBitArray();
192       assertTrue(arraysAreEqual(newBitsOriginal, newBitsNew, size / 32 + 1));
193     }
194   }
195 
196   private static int[] reverseOriginal(int[] oldBits, int size) {
197     int[] newBits = new int[oldBits.length];
198     for (int i = 0; i < size; i++) {
199       if (bitSet(oldBits, size - i - 1)) {
200         newBits[i / 32] |= 1 << (i & 0x1F);
201       }
202     }
203     return newBits;
204   }
205 
206   private static boolean bitSet(int[] bits, int i) {
207     return (bits[i / 32] & (1 << (i & 0x1F))) != 0;
208   }
209 
210   private static boolean arraysAreEqual(int[] left, int[] right, int size) {
211     for (int i = 0; i < size; i++) {
212       if (left[i] != right[i]) {
213         return false;
214       }
215     }
216     return true;
217   }
