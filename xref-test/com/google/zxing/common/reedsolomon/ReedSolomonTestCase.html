397   private static void corrupt(int[] received, int howMany, Random random, int max) {
398     BitSet corrupted = new BitSet(received.length);
399     for (int j = 0; j < howMany; j++) {
400       int location = random.nextInt(received.length);
401       int value = random.nextInt(max);
402       if (corrupted.get(location) || received[location] == value) {
403         j--;
404       } else {
405         corrupted.set(location);
406         received[location] = value;
407       }
408     }
409   }

411   private static void testEncodeDecodeRandom(GenericGF field, int dataSize, int ecSize) {
412     assertTrue("Invalid data size for " + field, dataSize > 0 && dataSize <= field.getSize() - 3);
413     assertTrue("Invalid ECC size for " + field, ecSize > 0 && ecSize + dataSize <= field.getSize());
414     ReedSolomonEncoder encoder = new ReedSolomonEncoder(field);
415     int[] message = new int[dataSize + ecSize];
416     int[] dataWords = new int[dataSize];
417     int[] ecWords = new int[ecSize];
418     Random random = getPseudoRandom();
419     int iterations = field.getSize() > 256 ? 1 : DECODER_RANDOM_TEST_ITERATIONS;
420     for (int i = 0; i < iterations; i++) {
421       // generate random data
422       for (int k = 0; k < dataSize; k++) {
423         dataWords[k] = random.nextInt(field.getSize());
424       }
425       // generate ECC words
426       System.arraycopy(dataWords, 0, message, 0, dataWords.length);
427       encoder.encode(message, ecWords.length);
428       System.arraycopy(message, dataSize, ecWords, 0, ecSize);
429       // check to see if Decoder can fix up to ecWords/2 random errors
430       testDecoder(field, dataWords, ecWords);
431     }
432   }

439   private static void testEncoder(GenericGF field, int[] dataWords, int[] ecWords) {
440     ReedSolomonEncoder encoder = new ReedSolomonEncoder(field);
441     int[] messageExpected = new int[dataWords.length + ecWords.length];
442     int[] message = new int[dataWords.length + ecWords.length];
443     System.arraycopy(dataWords, 0, messageExpected, 0, dataWords.length);
444     System.arraycopy(ecWords, 0, messageExpected, dataWords.length, ecWords.length);
445     System.arraycopy(dataWords, 0, message, 0, dataWords.length);
446     encoder.encode(message, ecWords.length);
447     assertDataEquals("Encode in " + field + " (" + dataWords.length + ',' + ecWords.length + ") failed", 
448                      messageExpected, message);
449   }

451   private static void testDecoder(GenericGF field, int[] dataWords, int[] ecWords) {
452     ReedSolomonDecoder decoder = new ReedSolomonDecoder(field);
453     int[] message = new int[dataWords.length + ecWords.length];
454     int maxErrors = ecWords.length / 2;
455     Random random = getPseudoRandom();
456     int iterations = field.getSize() > 256 ? 1 : DECODER_TEST_ITERATIONS;
457     for (int j = 0; j < iterations; j++) {
458       for (int i = 0; i < ecWords.length; i++) {
459         if (i > 10 && i < ecWords.length / 2 - 10) {
460           // performance improvement - skip intermediate cases in long-running tests 
461           i += ecWords.length / 10;
462         }
463         System.arraycopy(dataWords, 0, message, 0, dataWords.length);
464         System.arraycopy(ecWords, 0, message, dataWords.length, ecWords.length);
465         corrupt(message, i, random, field.getSize());
466         try {
467           decoder.decode(message, ecWords.length);
468         } catch (ReedSolomonException e) {
469           // fail only if maxErrors exceeded
470           assertTrue("Decode in " + field + " (" + dataWords.length + ',' + ecWords.length + ") failed at " + 
471                          i + " errors: " + e,
472                      i > maxErrors);
473           // else stop
474           break;
475         }
476         if (i < maxErrors) {
477           assertDataEquals("Decode in " + field + " (" + dataWords.length + ',' + ecWords.length + ") failed at " + 
478                            i + " errors",
479                            dataWords, 
480                            message);
481         }
482       }
483     }
484   }

486   private static void assertDataEquals(String message, int[] expected, int[] received) {
487     for (int i = 0; i < expected.length; i++) {
488       if (expected[i] != received[i]) {
489         fail(message + ". Mismatch at " + i + ". Expected " + arrayToString(expected) + ", got " + 
490              arrayToString(Arrays.copyOf(received, expected.length)));
491       }
492     }
493   }

495   private static String arrayToString(int[] data) {
496     StringBuilder sb = new StringBuilder("{");
497     for (int i=0; i<data.length; i++) {
498       sb.append(String.format(i > 0 ? ",%X" : "%X", data[i]));
499     }
500     return sb.append('}').toString();
501   }
