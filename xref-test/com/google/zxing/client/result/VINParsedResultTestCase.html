18  package com.google.zxing.client.result;
19  
20  import com.google.zxing.BarcodeFormat;
21  import com.google.zxing.Result;
22  import org.junit.Assert;
23  import org.junit.Test;
24  
25  public final class VINParsedResultTestCase extends Assert {
26  
27    @Test
28    public void testNotVIN() {
29      Result fakeResult = new Result("1M8GDM9A1KP042788", null, null, BarcodeFormat.CODE_39);
30      ParsedResult result = ResultParser.parseResult(fakeResult);
31      assertEquals(ParsedResultType.TEXT, result.getType());
32      fakeResult = new Result("1M8GDM9AXKP042788", null, null, BarcodeFormat.CODE_128);
33      result = ResultParser.parseResult(fakeResult);
34      assertEquals(ParsedResultType.TEXT, result.getType());
35    }
36  
37    @Test
38    public void testVIN() {
39      doTest("1M8GDM9AXKP042788", "1M8", "GDM9AX", "KP042788", "US", "GDM9A", 1989, 'P', "042788");
40      doTest("I1M8GDM9AXKP042788", "1M8", "GDM9AX", "KP042788", "US", "GDM9A", 1989, 'P', "042788");
41      doTest("LJCPCBLCX11000237", "LJC", "PCBLCX", "11000237", "CN", "PCBLC", 2001, '1', "000237");
42    }
43  
44    private static void doTest(String contents,
45                               String wmi,
46                               String vds,
47                               String vis,
48                               String country,
49                               String attributes,
50                               int year,
51                               char plant,
52                               String sequential) {
53      Result fakeResult = new Result(contents, null, null, BarcodeFormat.CODE_39);
54      ParsedResult result = ResultParser.parseResult(fakeResult);
55      assertSame(ParsedResultType.VIN, result.getType());
56      VINParsedResult vinResult = (VINParsedResult) result;
57      assertEquals(wmi, vinResult.getWorldManufacturerID());
58      assertEquals(vds, vinResult.getVehicleDescriptorSection());
59      assertEquals(vis, vinResult.getVehicleIdentifierSection());
60      assertEquals(country, vinResult.getCountryCode());
61      assertEquals(attributes, vinResult.getVehicleAttributes());
62      assertEquals(year, vinResult.getModelYear());
63      assertEquals(plant, vinResult.getPlantCode());
64      assertEquals(sequential, vinResult.getSequentialNumber());
65    }
