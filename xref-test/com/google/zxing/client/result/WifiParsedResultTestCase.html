17  package com.google.zxing.client.result;
18  
19  import com.google.zxing.BarcodeFormat;
20  import com.google.zxing.Result;
21  import org.junit.Assert;
22  import org.junit.Test;
23  
24  /**
25   * Tests {@link WifiParsedResult}.
26   *
27   * @author Vikram Aggarwal
28   */
29  public final class WifiParsedResultTestCase extends Assert {
30  
31    @Test
32    public void testNoPassword() {
33      doTest("WIFI:S:NoPassword;P:;T:;;", "NoPassword", null, "nopass");
34      doTest("WIFI:S:No Password;P:;T:;;", "No Password", null, "nopass");
35    }
36  
37    @Test
38    public void testWep() {
39      doTest("WIFI:S:TenChars;P:0123456789;T:WEP;;", "TenChars", "0123456789", "WEP");
40      doTest("WIFI:S:TenChars;P:abcde56789;T:WEP;;", "TenChars", "abcde56789", "WEP");
41      // Non hex should not fail at this level
42      doTest("WIFI:S:TenChars;P:hellothere;T:WEP;;", "TenChars", "hellothere", "WEP");
43  
44      // Escaped semicolons
45      doTest("WIFI:S:Ten\\;\\;Chars;P:0123456789;T:WEP;;", "Ten;;Chars", "0123456789", "WEP");
46      // Escaped colons
47      doTest("WIFI:S:Ten\\:\\:Chars;P:0123456789;T:WEP;;", "Ten::Chars", "0123456789", "WEP");
48  
49      // TODO(vikrama) Need a test for SB as well.
50    }
51  
52    /**
53     * Put in checks for the length of the password for wep.
54     */
55    @Test
56    public void testWpa() {
57      doTest("WIFI:S:TenChars;P:wow;T:WPA;;", "TenChars", "wow", "WPA");
58      doTest("WIFI:S:TenChars;P:space is silent;T:WPA;;", "TenChars", "space is silent", "WPA");
59      doTest("WIFI:S:TenChars;P:hellothere;T:WEP;;", "TenChars", "hellothere", "WEP");
60  
61      // Escaped semicolons
62      doTest("WIFI:S:TenChars;P:hello\\;there;T:WEP;;", "TenChars", "hello;there", "WEP");
63      // Escaped colons
64      doTest("WIFI:S:TenChars;P:hello\\:there;T:WEP;;", "TenChars", "hello:there", "WEP");
65    }
66  
67    /**
68     * Given the string contents for the barcode, check that it matches our expectations
69     */
70    private static void doTest(String contents,
71                               String ssid,
72                               String password,
73                               String type) {
74      Result fakeResult = new Result(contents, null, null, BarcodeFormat.QR_CODE);
75      ParsedResult result = ResultParser.parseResult(fakeResult);
76  
77      // Ensure it is a wifi code
78      assertSame(ParsedResultType.WIFI, result.getType());
79      WifiParsedResult wifiResult = (WifiParsedResult) result;
80  
81      assertEquals(ssid, wifiResult.getSsid());
82      assertEquals(password, wifiResult.getPassword());
83      assertEquals(type, wifiResult.getNetworkEncryption());
84    }
85  }
