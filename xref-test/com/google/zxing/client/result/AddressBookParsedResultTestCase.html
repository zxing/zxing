31    @Test
32    public void testAddressBookDocomo() {
33      doTest("MECARD:N:Sean Owen;;", null, new String[] {"Sean Owen"}, null, null, null, null, null, null, null, null);
34      doTest("MECARD:NOTE:ZXing Team;N:Sean Owen;URL:google.com;EMAIL:srowen@example.org;;",
35          null, new String[] {"Sean Owen"}, null, null, new String[] {"srowen@example.org"}, null, null,
36          new String[] {"google.com"}, null, "ZXing Team");
37    }

107   @Test
108   public void testVCardEscape() {
109     doTest("BEGIN:VCARD\r\nNOTE:foo\\nbar\r\nEND:VCARD",
110            null, null, null, null, null, null, null, null, null, "foo\nbar");
111     doTest("BEGIN:VCARD\r\nNOTE:foo\\;bar\r\nEND:VCARD",
112                null, null, null, null, null, null, null, null, null, "foo;bar");
113     doTest("BEGIN:VCARD\r\nNOTE:foo\\\\bar\r\nEND:VCARD",
114                    null, null, null, null, null, null, null, null, null, "foo\\bar");
115     doTest("BEGIN:VCARD\r\nNOTE:foo\\,bar\r\nEND:VCARD",
116                        null, null, null, null, null, null, null, null, null, "foo,bar");
117   }

119   private static void doTest(String contents,
120                              String title,
121                              String[] names,
122                              String pronunciation,
123                              String[] addresses,
124                              String[] emails,
125                              String[] phoneNumbers,
126                              String org,
127                              String[] urls,
128                              String birthday,
129                              String note) {
130     Result fakeResult = new Result(contents, null, null, BarcodeFormat.QR_CODE);
131     ParsedResult result = ResultParser.parseResult(fakeResult);
132     assertSame(ParsedResultType.ADDRESSBOOK, result.getType());
133     AddressBookParsedResult addressResult = (AddressBookParsedResult) result;
134     assertEquals(title, addressResult.getTitle());
135     assertArrayEquals(names, addressResult.getNames());
136     assertEquals(pronunciation, addressResult.getPronunciation());
137     assertArrayEquals(addresses, addressResult.getAddresses());
138     assertArrayEquals(emails, addressResult.getEmails());
139     assertArrayEquals(phoneNumbers, addressResult.getPhoneNumbers());
140     assertEquals(org, addressResult.getOrg());
141     assertArrayEquals(urls, addressResult.getURLs());
142     assertEquals(birthday, addressResult.getBirthday());
143     assertEquals(note, addressResult.getNote());
144   }
