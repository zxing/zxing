109   private static void doTest(String contents, String uri, String title) {
110     Result fakeResult = new Result(contents, null, null, BarcodeFormat.QR_CODE);
111     ParsedResult result = ResultParser.parseResult(fakeResult);
112     assertSame(ParsedResultType.URI, result.getType());
113     URIParsedResult uriResult = (URIParsedResult) result;
114     assertEquals(uri, uriResult.getURI());
115     assertEquals(title, uriResult.getTitle());
116   }
117   
118   private static void doTestNotUri(String text) {
119     Result fakeResult = new Result(text, null, null, BarcodeFormat.QR_CODE);
120     ParsedResult result = ResultParser.parseResult(fakeResult);
121     assertSame(ParsedResultType.TEXT, result.getType());
122     assertEquals(text, result.getDisplayResult());
123   }
124 
125   private static void doTestIsPossiblyMalicious(String uri, boolean expected) {
126     URIParsedResult result = new URIParsedResult(uri, null);
127     assertEquals(expected, result.isPossiblyMaliciousURI());
128   }
