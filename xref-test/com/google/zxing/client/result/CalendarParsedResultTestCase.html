206   private static void doTest(String contents,
207                              String description,
208                              String summary,
209                              String location,
210                              String startString,
211                              String endString) {
212     doTest(contents, description, summary, location, startString, endString, null, null, Double.NaN, Double.NaN);
213   }
214 
215   private static void doTest(String contents,
216                              String description,
217                              String summary,
218                              String location,
219                              String startString,
220                              String endString,
221                              String organizer,
222                              String[] attendees,
223                              double latitude,
224                              double longitude) {
225     Result fakeResult = new Result(contents, null, null, BarcodeFormat.QR_CODE);
226     ParsedResult result = ResultParser.parseResult(fakeResult);
227     assertSame(ParsedResultType.CALENDAR, result.getType());
228     CalendarParsedResult calResult = (CalendarParsedResult) result;
229     assertEquals(description, calResult.getDescription());
230     assertEquals(summary, calResult.getSummary());
231     assertEquals(location, calResult.getLocation());
232     assertEquals(startString, DATE_TIME_FORMAT.format(calResult.getStart()));
233     assertEquals(endString, calResult.getEnd() == null ? null : DATE_TIME_FORMAT.format(calResult.getEnd()));
234     assertEquals(organizer, calResult.getOrganizer());
235     assertArrayEquals(attendees, calResult.getAttendees());
236     assertEqualOrNaN(latitude, calResult.getLatitude());
237     assertEqualOrNaN(longitude, calResult.getLongitude());
238   }
239 
240   private static void assertEqualOrNaN(double expected, double actual) {
241     if (Double.isNaN(expected)) {
242       assertTrue(Double.isNaN(actual));
243     } else {
244       assertEquals(expected, actual, EPSILON);
245     }
246   }
