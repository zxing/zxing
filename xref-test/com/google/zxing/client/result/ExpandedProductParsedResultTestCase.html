41  public final class ExpandedProductParsedResultTestCase extends Assert {
42  
43    @Test
44    public void test_RSSExpanded() {
45      Map<String,String> uncommonAIs = new HashMap<>();
46      uncommonAIs.put("123", "544654");
47      Result result =
48          new Result("(01)66546(13)001205(3932)4455(3102)6544(123)544654", null, null, BarcodeFormat.RSS_EXPANDED);
49      ExpandedProductParsedResult o = new ExpandedProductResultParser().parse(result);
50      assertNotNull(o);
51      assertEquals("66546", o.getProductID());
52      assertNull(o.getSscc());
53      assertNull(o.getLotNumber());
54      assertNull(o.getProductionDate());
55      assertEquals("001205", o.getPackagingDate());
56      assertNull(o.getBestBeforeDate());
57      assertNull(o.getExpirationDate());
58      assertEquals("6544", o.getWeight());
59      assertEquals("KG", o.getWeightType());
60      assertEquals("2", o.getWeightIncrement());
61      assertEquals("5", o.getPrice());
62      assertEquals("2", o.getPriceIncrement());
63      assertEquals("445", o.getPriceCurrency());
64      assertEquals(uncommonAIs, o.getUncommonAIs());
65    }
66  }
