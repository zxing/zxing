56    @Test
57    public void testFindFinderPatterns() throws Exception {
58      BufferedImage image = readImage("2.png");
59      BinaryBitmap binaryMap = new BinaryBitmap(new GlobalHistogramBinarizer(new BufferedImageLuminanceSource(image)));
60      int rowNumber = binaryMap.getHeight() / 2;
61      BitArray row = binaryMap.getBlackRow(rowNumber, null);
62      List<ExpandedPair> previousPairs = new ArrayList<>();
63  
64      RSSExpandedReader rssExpandedReader = new RSSExpandedReader();
65      ExpandedPair pair1 = rssExpandedReader.retrieveNextPair(row, previousPairs, rowNumber);
66      previousPairs.add(pair1);
67      FinderPattern finderPattern = pair1.getFinderPattern();
68      assertNotNull(finderPattern);
69      assertEquals(0, finderPattern.getValue());
70  
71      ExpandedPair pair2 = rssExpandedReader.retrieveNextPair(row, previousPairs, rowNumber);
72      previousPairs.add(pair2);
73      finderPattern = pair2.getFinderPattern();
74      assertNotNull(finderPattern);
75      assertEquals(1, finderPattern.getValue());
76  
77      ExpandedPair pair3 = rssExpandedReader.retrieveNextPair(row, previousPairs, rowNumber);
78      previousPairs.add(pair3);
79      finderPattern = pair3.getFinderPattern();
80      assertNotNull(finderPattern);
81      assertEquals(1, finderPattern.getValue());
82  
83      try{
84        rssExpandedReader.retrieveNextPair(row, previousPairs, rowNumber);
85        //   the previous was the last pair
86        fail(NotFoundException.class.getName() + " expected");
87      }catch(NotFoundException nfe){
88        // ok
89      }
90    }

114   @Test
115   public void testDecodeCheckCharacter() throws Exception {
116     BufferedImage image = readImage("3.png");
117     BinaryBitmap binaryMap = new BinaryBitmap(new GlobalHistogramBinarizer(new BufferedImageLuminanceSource(image)));
118     BitArray row = binaryMap.getBlackRow(binaryMap.getHeight() / 2, null);
119 
120     int[] startEnd = {145, 243};//image pixels where the A1 pattern starts (at 124) and ends (at 214)
121     int value = 0;// A
122     FinderPattern finderPatternA1 = new FinderPattern(value, startEnd, startEnd[0], startEnd[1], image.getHeight() / 2);
123     //{1, 8, 4, 1, 1};
124     RSSExpandedReader rssExpandedReader = new RSSExpandedReader();
125     DataCharacter dataCharacter = rssExpandedReader.decodeDataCharacter(row, finderPatternA1, true, true);
126 
127     assertEquals(98, dataCharacter.getValue());
128   }

130   @Test
131   public void testDecodeDataCharacter() throws Exception {
132     BufferedImage image = readImage("3.png");
133     BinaryBitmap binaryMap = new BinaryBitmap(new GlobalHistogramBinarizer(new BufferedImageLuminanceSource(image)));
134     BitArray row = binaryMap.getBlackRow(binaryMap.getHeight() / 2, null);
135 
136     int[] startEnd = {145, 243};//image pixels where the A1 pattern starts (at 124) and ends (at 214)
137     int value = 0; // A
138     FinderPattern finderPatternA1 = new FinderPattern(value, startEnd, startEnd[0], startEnd[1], image.getHeight() / 2);
139     //{1, 8, 4, 1, 1};
140     RSSExpandedReader rssExpandedReader = new RSSExpandedReader();
141     DataCharacter dataCharacter = rssExpandedReader.decodeDataCharacter(row, finderPatternA1, true, false);
142 
143     assertEquals(19, dataCharacter.getValue());
144     assertEquals(1007, dataCharacter.getChecksumPortion());

147   private static BufferedImage readImage(String fileName) throws IOException {
148     Path path = Paths.get("src/test/resources/blackbox/rssexpanded-1/").resolve(fileName);
149     if (!Files.exists(path)) {
150       // Support running from project root too
151       path = Paths.get("core").resolve(path);
152     }
153 
154     return ImageIO.read(path.toFile());
155   }
