27  package com.google.zxing.oned.rss.expanded.decoders;

29  import org.junit.Test;
30  
31  /**
32   * @author Pablo Ordu√±a, University of Deusto (pablo.orduna@deusto.es)
33   */
34  public final class AnyAIDecoderTest extends AbstractDecoderTest {
35  
36    private static final String header           = ".....";
37  
38    @Test
39    public void testAnyAIDecoder_1() throws Exception {
40      CharSequence data = header + numeric_10 + numeric_12 + numeric2alpha + alpha_A + alpha2numeric + numeric_12;
41      String expected = "(10)12A12";
42  
43      assertCorrectBinaryString(data, expected);
44    }
45  
46    @Test
47    public void testAnyAIDecoder_2() throws Exception {
48      CharSequence data = header + numeric_10 + numeric_12 + numeric2alpha + alpha_A + alpha2isoiec646 + i646_B;
49      String expected = "(10)12AB";
50  
51      assertCorrectBinaryString(data, expected);
52    }
53  
54    @Test
55    public void testAnyAIDecoder_3() throws Exception {
56      CharSequence data = header + numeric_10 + numeric2alpha + alpha2isoiec646 + i646_B + i646_C + isoiec646_2alpha + alpha_A + alpha2numeric + numeric_10;
57      String expected = "(10)BCA10";
58  
59      assertCorrectBinaryString(data, expected);
60    }
61  
62    @Test
63    public void testAnyAIDecoder_numericFNC1_secondDigit() throws Exception {
64      CharSequence data = header + numeric_10 + numeric_1FNC1;
65      String expected = "(10)1";
66  
67      assertCorrectBinaryString(data, expected);
68    }
69  
70    @Test
71    public void testAnyAIDecoder_alphaFNC1() throws Exception {
72      CharSequence data = header + numeric_10 + numeric2alpha + alpha_A + alpha_FNC1;
73      String expected = "(10)A";
74  
75      assertCorrectBinaryString(data, expected);
76    }
77  
78    @Test
79    public void testAnyAIDecoder_646FNC1() throws Exception {
80      CharSequence data = header + numeric_10 + numeric2alpha + alpha_A + isoiec646_2alpha + i646_B + i646_FNC1;
81      String expected = "(10)AB";
82  
83      assertCorrectBinaryString(data, expected);
84    }
85  }
