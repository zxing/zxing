268   private static void assertCorrectImage2string(String path, String expected) throws IOException, NotFoundException {
269 
270     File file = new File(path);
271     if (!file.exists()) {
272       // Support running from project root too
273       file = new File("core", path);
274     }
275 
276     BufferedImage image = ImageIO.read(file);
277     BinaryBitmap binaryMap = new BinaryBitmap(new GlobalHistogramBinarizer(new BufferedImageLuminanceSource(image)));
278     int rowNumber = binaryMap.getHeight() / 2;
279     BitArray row = binaryMap.getBlackRow(rowNumber, null);
280 
281     Result result;
282     try {
283       RSSExpandedReader rssExpandedReader = new RSSExpandedReader();
284       result = rssExpandedReader.decodeRow(rowNumber, row, null);
285     } catch (ReaderException re) {
286       fail(re.toString());
287       return;
288     }
289 
290     assertSame(BarcodeFormat.RSS_EXPANDED, result.getBarcodeFormat());
291     assertEquals(expected, result.getText());
292   }
