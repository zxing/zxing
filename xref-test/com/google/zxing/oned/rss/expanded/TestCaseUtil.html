27  package com.google.zxing.oned.rss.expanded;
28  
29  import java.awt.image.BufferedImage;
30  import java.io.File;
31  import java.io.IOException;
32  
33  import javax.imageio.ImageIO;
34  
35  import com.google.zxing.BinaryBitmap;
36  import com.google.zxing.BufferedImageLuminanceSource;
37  import com.google.zxing.common.GlobalHistogramBinarizer;
38  
39  final class TestCaseUtil {
40  
41    private TestCaseUtil() {
42    }
43  
44    private static BufferedImage getBufferedImage(String path) throws IOException {
45      File file = new File(path);
46      if (!file.exists()) {
47        // Support running from project root too
48        file = new File("core", path);
49      }
50      return ImageIO.read(file);
51    }
52  
53    static BinaryBitmap getBinaryBitmap(String path) throws IOException {
54      BufferedImage bufferedImage = getBufferedImage(path);
55      BufferedImageLuminanceSource luminanceSource = new BufferedImageLuminanceSource(bufferedImage);
56      return new BinaryBitmap(new GlobalHistogramBinarizer(luminanceSource));
57    }
58  
59  }
