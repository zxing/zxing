17  package com.google.zxing.datamatrix;
18  
19  import com.google.zxing.BarcodeFormat;
20  import com.google.zxing.EncodeHintType;
21  import com.google.zxing.common.BitMatrix;
22  import com.google.zxing.datamatrix.encoder.SymbolShapeHint;
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  import java.util.EnumMap;
27  import java.util.Map;
28  
29  /**
30   * @author satorux@google.com (Satoru Takabayashi) - creator
31   * @author dswitkin@google.com (Daniel Switkin) - ported and expanded from C++
32   */
33  public final class DataMatrixWriterTestCase extends Assert {
34  
35    @Test
36    public void testDataMatrixImageWriter() {
37  
38      Map<EncodeHintType,Object> hints = new EnumMap<>(EncodeHintType.class);
39      hints.put(EncodeHintType.DATA_MATRIX_SHAPE, SymbolShapeHint.FORCE_SQUARE);
40  
41      int bigEnough = 64;
42      DataMatrixWriter writer = new DataMatrixWriter();
43      BitMatrix matrix = writer.encode("Hello Google", BarcodeFormat.DATA_MATRIX, bigEnough, bigEnough, hints);
44      assertNotNull(matrix);
45      assertTrue(bigEnough >= matrix.getWidth());
46      assertTrue(bigEnough >= matrix.getHeight());
47    }
48  
49    @Test
50    public void testDataMatrixWriter() {
51  
52      Map<EncodeHintType,Object> hints = new EnumMap<>(EncodeHintType.class);
53      hints.put(EncodeHintType.DATA_MATRIX_SHAPE, SymbolShapeHint.FORCE_SQUARE);
54  
55      int bigEnough = 14;
56      DataMatrixWriter writer = new DataMatrixWriter();
57      BitMatrix matrix = writer.encode("Hello Me", BarcodeFormat.DATA_MATRIX, bigEnough, bigEnough, hints);
58      assertNotNull(matrix);
59      assertEquals(bigEnough, matrix.getWidth());
60      assertEquals(bigEnough, matrix.getHeight());
61    }
62  
63    @Test
64    public void testDataMatrixTooSmall() {
65      // The DataMatrix will not fit in this size, so the matrix should come back bigger
66      int tooSmall = 8;
67      DataMatrixWriter writer = new DataMatrixWriter();
68      BitMatrix matrix = writer.encode("http://www.google.com/", BarcodeFormat.DATA_MATRIX, tooSmall, tooSmall, null);
69      assertNotNull(matrix);
70      assertTrue(tooSmall < matrix.getWidth());
71      assertTrue(tooSmall < matrix.getHeight());
72    }

74  }
