68    @Test  
69    public void testC40EncodationBasic2() {
70  
71      String visualized = encodeHighLevel("AIMAIAB");
72      assertEquals("230 91 11 90 255 254 67 129", visualized);
73      //"B" is normally encoded as "15" (one C40 value)
74      //"else" case: "B" is encoded as ASCII
75  
76      visualized = encodeHighLevel("AIMAIAb");
77      assertEquals("66 74 78 66 74 66 99 129", visualized); //Encoded as ASCII
78      //Alternative solution:
79      //assertEquals("230 91 11 90 255 254 99 129", visualized);
80      //"b" is normally encoded as "Shift 3, 2" (two C40 values)
81      //"else" case: "b" is encoded as ASCII
82  
83      visualized = encodeHighLevel("AIMAIMAIMË");
84      assertEquals("230 91 11 91 11 91 11 254 235 76", visualized);
85      //Alternative solution:
86      //assertEquals("230 91 11 91 11 91 11 11 9 254", visualized);
87      //Expl: 230 = shift to C40, "91 11" = "AIM",
88      //"11 9" = "�" = "Shift 2, UpperShift, <char>
89      //"else" case
90  
91      visualized = encodeHighLevel("AIMAIMAIMë");
92      assertEquals("230 91 11 91 11 91 11 254 235 108", visualized); //Activate when additional rectangulars are available
93      //Expl: 230 = shift to C40, "91 11" = "AIM",
94      //"�" in C40 encodes to: 1 30 2 11 which doesn't fit into a triplet
95      //"10 243" =
96      //254 = unlatch, 235 = Upper Shift, 108 = � = 0xEB/235 - 128 + 1
97      //"else" case
98    }

223   @Test  
224   public void testBase256Encodation() {
225 
226     //231 shifts to Base256 encodation
227 
228     String visualized = encodeHighLevel("\u00ABäöüé\u00BB");
229     assertEquals("231 44 108 59 226 126 1 104", visualized);
230     visualized = encodeHighLevel("\u00ABäöüéà\u00BB");
231     assertEquals("231 51 108 59 226 126 1 141 254 129", visualized);
232     visualized = encodeHighLevel("\u00ABäöüéàá\u00BB");
233     assertEquals("231 44 108 59 226 126 1 141 36 147", visualized);
234 
235     visualized = encodeHighLevel(" 23£"); //ASCII only (for reference)
236     assertEquals("33 153 235 36 129", visualized);
237 
238     visualized = encodeHighLevel("\u00ABäöüé\u00BB 234"); //Mixed Base256 + ASCII
239     assertEquals("231 51 108 59 226 126 1 104 99 153 53 129", visualized);
240 
241     visualized = encodeHighLevel("\u00ABäöüé\u00BB 23£ 1234567890123456789");
242     assertEquals("231 55 108 59 226 126 1 104 99 10 161 167 185 142 164 186 208"
243                      + " 220 142 164 186 208 58 129 59 209 104 254 150 45", visualized);
244 
245     visualized = encodeHighLevel(createBinaryMessage(20));
246     assertEquals("231 44 108 59 226 126 1 141 36 5 37 187 80 230 123 17 166 60 210 103 253 150",
247                  visualized);
248     visualized = encodeHighLevel(createBinaryMessage(19)); //padding necessary at the end
249     assertEquals("231 63 108 59 226 126 1 141 36 5 37 187 80 230 123 17 166 60 210 103 1 129",
250                  visualized);
251 
252     visualized = encodeHighLevel(createBinaryMessage(276));
253     assertStartsWith("231 38 219 2 208 120 20 150 35", visualized);
254     assertEndsWith("146 40 194 129", visualized);
255 
256     visualized = encodeHighLevel(createBinaryMessage(277));
257     assertStartsWith("231 38 220 2 208 120 20 150 35", visualized);
258     assertEndsWith("146 40 190 87", visualized);
259   }

261   private static String createBinaryMessage(int len) {
262     StringBuilder sb = new StringBuilder();
263     sb.append("\u00ABäöüéàá-");
264     for (int i = 0; i < len - 9; i++) {
265       sb.append('\u00B7');
266     }
267     sb.append('\u00BB');
268     return sb.toString();
269   }
270 
271   private static void assertStartsWith(String expected, String actual) {
272     if (!actual.startsWith(expected)) {
273       throw new ComparisonFailure(null, expected, actual.substring(0, expected.length()));
274     }
275   }
276 
277   private static void assertEndsWith(String expected, String actual) {
278     if (!actual.endsWith(expected)) {
279       throw new ComparisonFailure(null, expected, actual.substring(actual.length() - expected.length()));
280     }
281   }

355   private static String encodeHighLevel(String msg) {
356     CharSequence encoded = HighLevelEncoder.encodeHighLevel(msg);
357     //DecodeHighLevel.decode(encoded);
358     return visualize(encoded);
359   }
360   
361   /**
362    * Convert a string of char codewords into a different string which lists each character
363    * using its decimal value.
364    *
365    * @param codewords the codewords
366    * @return the visualized codewords
367    */
368   static String visualize(CharSequence codewords) {
369     StringBuilder sb = new StringBuilder();
370     for (int i = 0; i < codewords.length(); i++) {
371       if (i > 0) {
372         sb.append(' ');
373       }
374       sb.append((int) codewords.charAt(i));
375     }
376     return sb.toString();
377   }
