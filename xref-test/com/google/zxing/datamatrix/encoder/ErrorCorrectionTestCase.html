17  package com.google.zxing.datamatrix.encoder;
18  
19  import org.junit.Assert;
20  import org.junit.Test;
21  
22  /**
23   * Tests for the ECC200 error correction.
24   */
25  public final class ErrorCorrectionTestCase extends Assert {
26  
27    @Test
28    public void testRS() {
29      //Sample from Annexe R in ISO/IEC 16022:2000(E)
30      char[] cw = {142, 164, 186};
31      SymbolInfo symbolInfo = SymbolInfo.lookup(3);
32      CharSequence s = ErrorCorrection.encodeECC200(String.valueOf(cw), symbolInfo);
33      assertEquals("142 164 186 114 25 5 88 102", HighLevelEncodeTestCase.visualize(s));
34  
35      //"A" encoded (ASCII encoding + 2 padding characters)
36      cw = new char[]{66, 129, 70};
37      s = ErrorCorrection.encodeECC200(String.valueOf(cw), symbolInfo);
38      assertEquals("66 129 70 138 234 82 82 95", HighLevelEncodeTestCase.visualize(s));
39    }
40  
41  }
