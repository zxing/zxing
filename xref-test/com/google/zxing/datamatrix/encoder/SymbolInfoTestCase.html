17  package com.google.zxing.datamatrix.encoder;
18  
19  import com.google.zxing.Dimension;
20  import org.junit.Assert;
21  import org.junit.Test;
22  
23  /**
24   * Tests the SymbolInfo class.
25   */
26  public final class SymbolInfoTestCase extends Assert {
27  
28    @Test
29    public void testSymbolInfo() {
30      SymbolInfo info = SymbolInfo.lookup(3);
31      assertEquals(5, info.getErrorCodewords());
32      assertEquals(8, info.matrixWidth);
33      assertEquals(8, info.matrixHeight);
34      assertEquals(10, info.getSymbolWidth());
35      assertEquals(10, info.getSymbolHeight());
36  
37      info = SymbolInfo.lookup(3, SymbolShapeHint.FORCE_RECTANGLE);
38      assertEquals(7, info.getErrorCodewords());
39      assertEquals(16, info.matrixWidth);
40      assertEquals(6, info.matrixHeight);
41      assertEquals(18, info.getSymbolWidth());
42      assertEquals(8, info.getSymbolHeight());
43  
44      info = SymbolInfo.lookup(9);
45      assertEquals(11, info.getErrorCodewords());
46      assertEquals(14, info.matrixWidth);
47      assertEquals(6, info.matrixHeight);
48      assertEquals(32, info.getSymbolWidth());
49      assertEquals(8, info.getSymbolHeight());
50  
51      info = SymbolInfo.lookup(9, SymbolShapeHint.FORCE_SQUARE);
52      assertEquals(12, info.getErrorCodewords());
53      assertEquals(14, info.matrixWidth);
54      assertEquals(14, info.matrixHeight);
55      assertEquals(16, info.getSymbolWidth());
56      assertEquals(16, info.getSymbolHeight());
57  
58      try {
59        SymbolInfo.lookup(1559);
60        fail("There's no rectangular symbol for more than 1558 data codewords");
61      } catch (IllegalArgumentException iae) {
62        //expected
63      }
64      try {
65        SymbolInfo.lookup(50, SymbolShapeHint.FORCE_RECTANGLE);
66        fail("There's no rectangular symbol for 50 data codewords");
67      } catch (IllegalArgumentException iae) {
68        //expected
69      }
70  
71      info = SymbolInfo.lookup(35);
72      assertEquals(24, info.getSymbolWidth());
73      assertEquals(24, info.getSymbolHeight());
74  
75      Dimension fixedSize = new Dimension(26, 26);
76      info = SymbolInfo.lookup(35,
77                               SymbolShapeHint.FORCE_NONE, fixedSize, fixedSize, false);
78      assertEquals(26, info.getSymbolWidth());
79      assertEquals(26, info.getSymbolHeight());
80  
81      info = SymbolInfo.lookup(45,
82                               SymbolShapeHint.FORCE_NONE, fixedSize, fixedSize, false);
83      assertNull(info);
84  
85      Dimension minSize = fixedSize;
86      Dimension maxSize = new Dimension(32, 32);
87  
88      info = SymbolInfo.lookup(35,
89                               SymbolShapeHint.FORCE_NONE, minSize, maxSize, false);
90      assertEquals(26, info.getSymbolWidth());
91      assertEquals(26, info.getSymbolHeight());
92  
93      info = SymbolInfo.lookup(40,
94                               SymbolShapeHint.FORCE_NONE, minSize, maxSize, false);
95      assertEquals(26, info.getSymbolWidth());
96      assertEquals(26, info.getSymbolHeight());
97  
98      info = SymbolInfo.lookup(45,
99                               SymbolShapeHint.FORCE_NONE, minSize, maxSize, false);
100     assertEquals(32, info.getSymbolWidth());
101     assertEquals(32, info.getSymbolHeight());
102 
103     info = SymbolInfo.lookup(63,
104                              SymbolShapeHint.FORCE_NONE, minSize, maxSize, false);
105     assertNull(info);
106   }
107 
108 }
