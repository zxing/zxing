47    private static final int ECC_BYTES = PDF417_TEST_WITH_EC.length - PDF417_TEST.length;
48    // Example is EC level 1 (s=1). The number of erasures (l) and substitutions (f) must obey:
49    // l + 2f <= 2^(s+1) - 3
50    private static final int EC_LEVEL = 5;
51    private static final int ERROR_LIMIT = (1 << (EC_LEVEL + 1)) - 3;
52    private static final int MAX_ERRORS = ERROR_LIMIT / 2;
53    private static final int MAX_ERASURES = ERROR_LIMIT;
54  
55    private final ErrorCorrection ec = new ErrorCorrection();
56  
57    @Test
58    public void testNoError() throws ChecksumException {
59      int[] received = PDF417_TEST_WITH_EC.clone();
60      // no errors
61      checkDecode(received);
62    }
63  
64    @Test
65    public void testOneError() throws ChecksumException {
66      Random random = getRandom();
67      for (int i = 0; i < PDF417_TEST_WITH_EC.length; i++) {
68        int[] received = PDF417_TEST_WITH_EC.clone();
69        received[i] = random.nextInt(256);
70        checkDecode(received);
71      }
72    }
73  
74    @Test
75    public void testMaxErrors() throws ChecksumException {
76      Random random = getRandom();
77      for (int testIterations = 0; testIterations < 100; testIterations++) { // # iterations is kind of arbitrary
78        int[] received = PDF417_TEST_WITH_EC.clone();
79        corrupt(received, MAX_ERRORS, random);
80        checkDecode(received);
81      }
82    }

124   private void checkDecode(int[] received) throws ChecksumException {
125     checkDecode(received, new int[0]);
126   }
127 
128   private void checkDecode(int[] received, int[] erasures) throws ChecksumException {
129     ec.decode(received, ECC_BYTES, erasures);
130     for (int i = 0; i < PDF417_TEST.length; i++) {
131       assertEquals(received[i], PDF417_TEST[i]);
132     }
133   }
